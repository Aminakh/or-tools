\begin{algorithmic}
\State $x \gets x_0$
\While{stop criteria not met}
    \State 1. Find neighborhood $\mathcal{N}_{x}$ by \emph{destroying} and \emph{repairing} $x$
    \State 2. Find "best" solution in $\mathcal{N}_{x}$: $x_{\text{best}}$
    \State 3. $x \gets x_{\text{best}}$
\EndWhile
\end{algorithmic}
