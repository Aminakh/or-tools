00040 // A scoped_ptr<T> is like a T*, except that the destructor of scoped_ptr<T>
00041 // automatically deletes the pointer it holds (if any).
00042 // That is, scoped_ptr<T> owns the T object that it points to.
00043 // Like a T*, a scoped_ptr<T> may hold either NULL or a pointer to a T object.
00044 //
00045 // The size of a scoped_ptr is small:
00046 // sizeof(scoped_ptr<C>) == sizeof(C*)
00047 template <class C>
00048 class scoped_ptr {
00049  public:
00050 
00051   // The element type
00052   typedef C element_type;
00053 
00054   // Constructor.  Defaults to intializing with NULL.
00055   // There is no way to create an uninitialized scoped_ptr.
00056   // The input parameter must be allocated with new.
00057   explicit scoped_ptr(C* p = NULL) : ptr_(p) { }
00058 
00059   // Destructor.  If there is a C object, delete it.
00060   // We don't need to test ptr_ == NULL because C++ does that for us.
00061   ~scoped_ptr() {
00062     enum { type_must_be_complete = sizeof(C) };
00063     delete ptr_;
00064   }
00065 
00066   // Reset.  Deletes the current owned object, if any.
00067   // Then takes ownership of a new object, if given.
00068   // this->reset(this->get()) works.
00069   void reset(C* p = NULL) {
00070     if (p != ptr_) {
00071       enum { type_must_be_complete = sizeof(C) };
00072       delete ptr_;
00073       ptr_ = p;
00074     }
00075   }
00076 
00077   // Accessors to get the owned object.
00078   // operator* and operator-> will assert() if there is no current object.
00079   C& operator*() const {
00080     assert(ptr_ != NULL);
00081     return *ptr_;
00082   }
00083   C* operator->() const  {
00084     assert(ptr_ != NULL);
00085     return ptr_;
00086   }
00087   C* get() const { return ptr_; }
00088 
00089   // Comparison operators.
00090   // These return whether two scoped_ptr refer to the same object, not just to
00091   // two different but equal objects.
00092   bool operator==(C* p) const { return ptr_ == p; }
00093   bool operator!=(C* p) const { return ptr_ != p; }
00094 
00095   // Swap two scoped pointers.
00096   void swap(scoped_ptr& p2) {
00097     C* tmp = ptr_;
00098     ptr_ = p2.ptr_;
00099     p2.ptr_ = tmp;
00100   }
00101 
00102   // Release a pointer.
00103   // The return value is the current pointer held by this object.
00104   // If this object holds a NULL pointer, the return value is NULL.
00105   // After this operation, this object will hold a NULL pointer,
00106   // and will not own the object any more.
00107   C* release() {
00108     C* retVal = ptr_;
00109     ptr_ = NULL;
00110     return retVal;
00111   }
00112 
00113  private:
00114   C* ptr_;
00115 
00116   // friend class that can access copy ctor (although if it actually
00117   // calls a copy ctor, there will be a problem) see below
00118   friend scoped_ptr<C> make_scoped_ptr<C>(C *p);
00119 
00120   // Forbid comparison of scoped_ptr types.  If C2 != C, it totally doesn't
00121   // make sense, and if C2 == C, it still doesn't make sense because you should
00122   // never have the same object owned by two different scoped_ptrs.
00123   template <class C2> bool operator==(scoped_ptr<C2> const& p2) const;
00124   template <class C2> bool operator!=(scoped_ptr<C2> const& p2) const;
00125 
00126   DISALLOW_COPY_AND_ASSIGN(scoped_ptr);
00127 };

00145 template <class C>
00146 scoped_ptr<C> make_scoped_ptr(C *p) {
00147   // This does nothing but to return a scoped_ptr of the type that the passed
00148   // pointer is of.  (This eliminates the need to specify the name of T when
00149   // making a scoped_ptr that is used anonymously/temporarily.)  From an
00150   // access control point of view, we construct an unnamed scoped_ptr here
00151   // which we return and thus copy-construct.  Hence, we need to have access
00152   // to scoped_ptr::scoped_ptr(scoped_ptr const &).  However, it is guaranteed
00153   // that we never actually call the copy constructor, which is a good thing
00154   // as we would call the temporary's object destructor (and thus delete p)
00155   // if we actually did copy some object, here.
00156   return scoped_ptr<C>(p);
00157 }
