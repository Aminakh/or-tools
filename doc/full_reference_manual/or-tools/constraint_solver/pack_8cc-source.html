00203 void Pack::InitialPropagate() {
00204   ClearAll();
00205   Solver* const s = solver();
00206   in_process_ = true;
00207   InitialPropagateData* data = s->RevAlloc(new InitialPropagateData(bins_));
00208   for (int var_index = 0; var_index < vsize_; ++var_index) {
00209     IntVar* const var = vars_[var_index];
00210     var->SetRange(0, bins_);  // bins_ -> item is not assigned to a bin.
00211     if (var->Bound()) {
00212       const int64 value = var->Min();
00213       if (value < bins_) {
00214         forced_[value].push_back(var_index);
00215         data->PushAssigned(var_index);
00216       } else {
00217         data->PushUnassigned(var_index);
00218       }
00219     } else {
00220             DCHECK_GT(bins_, var->Min());
00221       if (var->Max() < bins_) {
00222         data->PushAssigned(var_index);
00223       }
00224       scoped_ptr<IntVarIterator> it(var->MakeDomainIterator(false));
00225       for (it->Init(); it->Ok(); it->Next()) {
00226         const int64 value = it->Value();
00227         if (value >= 0 && value <= bins_) {
00228           unprocessed_.SetToOne(s, value, var_index);
00229           if (value != bins_) {
00230             data->PushUndecided(value, var_index);
00231           }
00232         }
00233       }
00234     }
00235   }
00236   for (int bin_index = 0; bin_index < bins_; ++bin_index) {
00237     for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00238       dims_[dim_index]->InitialPropagate(bin_index,
00239                                          forced_[bin_index],
00240                                          data->undecided(bin_index));
00241     }
00242   }
00243   for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00244     dims_[dim_index]->InitialPropagateUnassigned(data->assigned(),
00245                                                  data->unassigned());
00246     dims_[dim_index]->EndInitialPropagate();
00247   }

00249   PropagateDelayed();

00253 void Pack::Propagate() {
00254   in_process_ = true;
00255   DCHECK_EQ(stamp_, solver()->fail_stamp());
00256   for (int bin_index = 0; bin_index < bins_; ++bin_index) {
00257     if (!removed_[bin_index].empty() || !forced_[bin_index].empty()) {
00258       for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {

00260                                     forced_[bin_index],

00278 void Pack::OneDomain(int var_index) {
00279   // TODO(user): We know var ranges from 0 to bins_. There are lots
00280   // of simplifications possible.
00281   Solver* const s = solver();
00282   const uint64 current_stamp = s->fail_stamp();
00283   if (stamp_ < current_stamp) {
00284     stamp_ = current_stamp;
00285     ClearAll();
00286   }
00287   IntVar* const var = vars_[var_index];
00288   bool bound = var->Bound();
00289   const int64 oldmin = var->OldMin();
00290   const int64 oldmax = var->OldMax();
00291   const int64 vmin = var->Min();
00292   const int64 vmax = var->Max();
00293   for (int64 value = std::max(oldmin, 0LL);
00294        value < std::min(vmin, bins_ + 1);
00295        ++value) {
00296     if (unprocessed_.IsSet(value, var_index)) {
00297       unprocessed_.SetToZero(s, value, var_index);
00298       removed_[value].push_back(var_index);
00299     }
00300   }
00301   if (!bound) {
00302     IntVarIterator* const holes = holes_[var_index];
00303     for (holes->Init(); holes->Ok(); holes->Next()) {
00304       const int64 value = holes->Value();
00305       if (value >= std::max(0LL, vmin) &&  value <= std::min(bins_, vmax)) {
00306         DCHECK(unprocessed_.IsSet(value, var_index));
00307         unprocessed_.SetToZero(s, value, var_index);
00308         removed_[value].push_back(var_index);
00309       }
00310     }
00311   }
00312   for (int64 value = std::max(vmax + 1, 0LL);
00313        value <= std::min(oldmax, bins_);
00314        ++value) {
00315     if (unprocessed_.IsSet(value, var_index)) {
00316       unprocessed_.SetToZero(s, value, var_index);
00317       removed_[value].push_back(var_index);
00318     }
00319   }
00320   if (bound) {
00321     unprocessed_.SetToZero(s, var->Min(), var_index);
00322     forced_[var->Min()].push_back(var_index);
00323   }
00324   Enqueue(demon_);

00655   void PushFromTop(int64 bin_index) {
00656     IntVar* const load = loads_[bin_index];
00657     const int64 sum_min = sum_of_bound_variables_vector_[bin_index].Value();
00658     const int64 sum_max = sum_of_all_variables_vector_[bin_index].Value();
00659     load->SetRange(sum_min, sum_max);
00660     const int64 slack_up = load->Max() - sum_min;
00661     const int64 slack_down = sum_max - load->Min();
00662     DCHECK_GE(slack_down, 0);
00663     DCHECK_GE(slack_up, 0);
00664     int64 last_unbound = first_unbound_backward_vector_[bin_index].Value();
00665     for (; last_unbound >= 0; --last_unbound) {
00666       const int64 var_index = ranked_[last_unbound];
00667       const int64 weight = weights_[var_index];
00668       if (IsUndecided(var_index, bin_index)) {
00669         if (weight > slack_up) {
00670           SetImpossible(var_index, bin_index);
00671         } else if (weight > slack_down) {
00672           Assign(var_index, bin_index);
00673         } else {
00674           break;
00675         }
00676       }
00677     }
00678     first_unbound_backward_vector_[bin_index].SetValue(solver(), last_unbound);
00679   }
