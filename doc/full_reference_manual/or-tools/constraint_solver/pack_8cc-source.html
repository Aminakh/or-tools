00207 void Pack::InitialPropagate() {
00208   ClearAll();
00209   Solver* const s = solver();
00210   in_process_ = true;
00211   InitialPropagateData* data = s->RevAlloc(new InitialPropagateData(bins_));
00212   for (int var_index = 0; var_index < vsize_; ++var_index) {
00213     IntVar* const var = vars_[var_index];
00214     var->SetRange(0, bins_);  // bins_ -> item is not assigned to a bin.
00215     if (var->Bound()) {
00216       const int64 value = var->Min();
00217       if (value < bins_) {
00218         forced_[value].push_back(var_index);
00219         data->PushAssigned(var_index);
00220       } else {
00221         data->PushUnassigned(var_index);
00222       }
00223     } else {
00224             DCHECK_GT(bins_, var->Min());
00225       if (var->Max() < bins_) {
00226         data->PushAssigned(var_index);
00227       }
00228       scoped_ptr<IntVarIterator> it(var->MakeDomainIterator(false));
00229       for (it->Init(); it->Ok(); it->Next()) {
00230         const int64 value = it->Value();
00231         if (value >= 0 && value <= bins_) {
00232           unprocessed_.SetToOne(s, value, var_index);
00233           if (value != bins_) {
00234             data->PushUndecided(value, var_index);
00235           }
00236         }
00237       }
00238     }
00239   }
00240   for (int bin_index = 0; bin_index < bins_; ++bin_index) {
00241     for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00242       dims_[dim_index]->InitialPropagate(bin_index,
00243                                          forced_[bin_index],
00244                                          data->undecided(bin_index));
00245     }
00246   }
00247   for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00248     dims_[dim_index]->InitialPropagateUnassigned(data->assigned(),
00249                                                  data->unassigned());
00250     dims_[dim_index]->EndInitialPropagate();
00251   }
00252 
00253   PropagateDelayed();
00254   ClearAll();
00255 }

00257 void Pack::Propagate() {
00258   in_process_ = true;
00259   DCHECK_EQ(stamp_, solver()->fail_stamp());
00260   for (int bin_index = 0; bin_index < bins_; ++bin_index) {
00261     if (!removed_[bin_index].empty() || !forced_[bin_index].empty()) {
00262       for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00263         dims_[dim_index]->Propagate(bin_index,
00264                                     forced_[bin_index],
00265                                     removed_[bin_index]);
00266       }
00267     }
00268   }
00269   if (!removed_[bins_].empty() || !forced_[bins_].empty()) {
00270     for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00271       dims_[dim_index]->PropagateUnassigned(removed_[bins_], forced_[bins_]);
00272     }
00273   }
00274   for (int dim_index = 0; dim_index < dims_.size(); ++dim_index) {
00275     dims_[dim_index]->EndPropagate();
00276   }

00282 void Pack::OneDomain(int var_index) {
00283   // TODO(user): We know var ranges from 0 to bins_. There are lots
00284   // of simplifications possible.
00285   Solver* const s = solver();
00286   const uint64 current_stamp = s->fail_stamp();
00287   if (stamp_ < current_stamp) {
00288     stamp_ = current_stamp;
00289     ClearAll();
00290   }
00291   IntVar* const var = vars_[var_index];
00292   bool bound = var->Bound();
00293   const int64 oldmin = var->OldMin();
00294   const int64 oldmax = var->OldMax();
00295   const int64 vmin = var->Min();
00296   const int64 vmax = var->Max();
00297   for (int64 value = std::max(oldmin, 0LL);
00298        value < std::min(vmin, bins_ + 1);
00299        ++value) {
00300     if (unprocessed_.IsSet(value, var_index)) {
00301       unprocessed_.SetToZero(s, value, var_index);
00302       removed_[value].push_back(var_index);
00303     }
00304   }
00305   if (!bound) {
00306     IntVarIterator* const holes = holes_[var_index];
00307     for (holes->Init(); holes->Ok(); holes->Next()) {
00308       const int64 value = holes->Value();
00309       if (value >= std::max(0LL, vmin) &&  value <= std::min(bins_, vmax)) {
00310         DCHECK(unprocessed_.IsSet(value, var_index));
00311         unprocessed_.SetToZero(s, value, var_index);
00312         removed_[value].push_back(var_index);
00313       }
00314     }
00315   }
00316   for (int64 value = std::max(vmax + 1, 0LL);
00317        value <= std::min(oldmax, bins_);
00318        ++value) {
00319     if (unprocessed_.IsSet(value, var_index)) {
00320       unprocessed_.SetToZero(s, value, var_index);
00321       removed_[value].push_back(var_index);
00322     }
00323   }
00324   if (bound) {
00325     unprocessed_.SetToZero(s, var->Min(), var_index);
00326     forced_[var->Min()].push_back(var_index);
00327   }
00328   Enqueue(demon_);
00329 }

00463 int SortIndexByWeight(int64* const indices,
00464                       const int64* const weights,
00465                       int size) {

00659   void PushFromTop(int64 bin_index) {
00660     IntVar* const load = loads_[bin_index];
00661     const int64 sum_min = sum_of_bound_variables_vector_[bin_index].Value();
00662     const int64 sum_max = sum_of_all_variables_vector_[bin_index].Value();

00665     const int64 slack_down = sum_max - load->Min();

00667     DCHECK_GE(slack_up, 0);
00668     int64 last_unbound = first_unbound_backward_vector_[bin_index].Value();
00669     for (; last_unbound >= 0; --last_unbound) {
00670       const int64 var_index = ranked_[last_unbound];
00671       const int64 weight = weights_[var_index];
00672       if (IsUndecided(var_index, bin_index)) {
00673         if (weight > slack_up) {
00674           SetImpossible(var_index, bin_index);
00675         } else if (weight > slack_down) {
00676           Assign(var_index, bin_index);
00677         } else {
00678           break;
00679         }
00680       }
00681     }
00682     first_unbound_backward_vector_[bin_index].SetValue(solver(), last_unbound);
00683   }
