00171 // Complexity: O(Sum(Length(paths modified)) + #paths modified ^ 2).
00172 // (#paths modified is usually very small).
00173 bool PathCumulFilter::Accept(const Assignment* delta,
00174                              const Assignment* deltadelta) {
00175   const Assignment::IntContainer& container = delta->IntVarContainer();
00176   const int delta_size = container.Size();
00177   // Determining touched paths.
00178   std::vector<int64> touched_paths;
00179   for (int i = 0; i < delta_size; ++i) {
00180     const IntVarElement& new_element = container.Element(i);
00181     const IntVar* const var = new_element.Var();
00182     int64 index = kUnassigned;
00183     if (FindIndex(var, &index)) {
00184       const int64 start = node_path_starts_[index];
00185       if (start != kUnassigned
00186           && find(touched_paths.begin(), touched_paths.end(), start)
00187           == touched_paths.end()) {
00188         touched_paths.push_back(start);
00189       }
00190     }
00191   }
00192   // Checking feasibility of touched paths.
00193   const int touched_paths_size = touched_paths.size();
00194   for (int i = 0; i < touched_paths_size; ++i) {
00195     int64 node = touched_paths[i];
00196     int64 cumul = cumuls_[node]->Min();
00197     while (node < Size()) {
00198       const IntVar* const next_var = Var(node);
00199       int64 next = saved_nexts_[node];
00200       if (container.Contains(next_var)) {
00201         const IntVarElement& element = container.Element(next_var);
00202         if (element.Bound()) {
00203           next = element.Value();
00204         } else {
00205           // LNS detected, return true since path was ok up to now.
00206           return true;
00207         }
00208       }
00209       cumul += evaluator_->Run(node, next);
00210       if (cumul > cumuls_[next]->Max()) {
00211         return false;
00212       }
00213       cumul = std::max(cumuls_[next]->Min(), cumul);
00214       node = next;
00215     }
00216   }
00217   return true;
00218 }

00220 void PathCumulFilter::OnSynchronize() {
00221   const int nexts_size = Size();
00222   // Detecting path starts, used to track which node belongs to which path.
00223   std::vector<int64> path_starts;
00224   Bitmap has_prevs(nexts_size, false);
00225   for (int i = 0; i < nexts_size; ++i) {
00226     const int next = Value(i);
00227     if (next < nexts_size) {
00228       has_prevs.Set(next, true);
00229     }
00230   }
00231   for (int i = 0; i < nexts_size; ++i) {
00232     if (!has_prevs.Get(i)) {
00233       path_starts.push_back(i);
00234     }
00235   }
00236   // Marking unactive nodes (which are not on a path).
00237   node_path_starts_.assign(cumuls_size_, kUnassigned);
00238   // Marking nodes on a path and storing next values.
00239   for (int i = 0; i < path_starts.size(); ++i) {
00240     const int64 start = path_starts[i];
00241     int node = start;
00242     node_path_starts_[node] = start;
00243     int next = Value(node);
00244     saved_nexts_[node] = next;
00245     while (next < nexts_size) {
00246       node = next;
00247       node_path_starts_[node] = start;
00248       next = Value(node);
00249       saved_nexts_[node] = next;
00250     }
00251     saved_nexts_[node] = next;
00252     node_path_starts_[next] = start;
00253   }
00254 }

00608 void RoutingModel::AddDisjunctionInternal(const std::vector<int64>& nodes,
00609                                           int64 penalty) {
00610   const int size = disjunctions_.size();
00611   disjunctions_.resize(size + 1);
00612   std::vector<int64>& disjunction_nodes = disjunctions_[size].nodes;
00613   disjunction_nodes.resize(nodes.size());

00615     CHECK_NE(kUnassigned, node_to_index_[nodes[i]]);
00616     disjunction_nodes[i] = node_to_index_[nodes[i]];
00617   }
00618   disjunctions_[size].penalty = penalty;
00619   for (int i = 0; i < nodes.size(); ++i) {
00620     // TODO(user): support multiple disjunction per node
00621     node_to_disjunction_[node_to_index_[nodes[i]]] = size;

00625 IntVar* RoutingModel::CreateDisjunction(int disjunction) {
00626   const std::vector<int64>& nodes = disjunctions_[disjunction].nodes;
00627   const int nodes_size = nodes.size();
00628   std::vector<IntVar*> disjunction_vars(nodes_size + 1);
00629   for (int i = 0; i < nodes_size; ++i) {
00630     const int64 node = nodes[i];
00631     CHECK_LT(node, Size());
00632     disjunction_vars[i] = ActiveVar(node);
00633   }
00634   IntVar* no_active_var = solver_->MakeBoolVar();
00635   disjunction_vars[nodes_size] = no_active_var;
00636   solver_->AddConstraint(solver_->MakeSumEquality(disjunction_vars, 1));
00637   const int64 penalty = disjunctions_[disjunction].penalty;
00638   if (penalty < 0) {
00639     no_active_var->SetMax(0);
00640     return NULL;
00641   } else {
00642     return solver_->MakeProd(no_active_var, penalty)->Var();
00643   }
00644 }

00656 void RoutingModel::SetStartEnd(const std::vector<std::pair<int, int> >& start_end) {
00657   if (is_depot_set_) {
00658     LOG(WARNING) << "A depot has already been specified, ignoring new ones";
00659     return;
00660   }
00661   CHECK_EQ(start_end.size(), vehicles_);
00662   const int size = Size();
00663   hash_set<int> starts;
00664   hash_set<int> ends;
00665   for (int i = 0; i < vehicles_; ++i) {
00666     int start = start_end[i].first;
00667     int end = start_end[i].second;
00668     CHECK_GE(start, 0);
00669     CHECK_GE(end, 0);
00670     CHECK_LE(start, nodes_);
00671     CHECK_LE(end, nodes_);
00672     starts.insert(start);
00673     ends.insert(end);
00674   }
00675   index_to_node_.resize(size + vehicles_);
00676   node_to_index_.resize(nodes_, kUnassigned);
00677   int index = 0;
00678   for (int i = 0; i < nodes_; ++i) {
00679     if (starts.count(i) != 0 || ends.count(i) == 0) {
00680       index_to_node_[index] = i;
00681       node_to_index_[i] = index;
00682       ++index;
00683     }
00684   }
00685   hash_set<int> node_set;
00686   index_to_vehicle_.resize(size + vehicles_, kUnassigned);
00687   for (int i = 0; i < vehicles_; ++i) {
00688     int start = start_end[i].first;
00689     if (node_set.count(start) == 0) {
00690       node_set.insert(start);
00691       const int start_index = node_to_index_[start];
00692       starts_[i] = start_index;
00693       CHECK_NE(kUnassigned, start_index);
00694       index_to_vehicle_[start_index] = i;
00695     } else {
00696       starts_[i] = index;
00697       index_to_node_[index] = start;
00698       index_to_vehicle_[index] = i;
00699       ++index;
00700     }
00701   }
00702   for (int i = 0; i < vehicles_; ++i) {
00703     int end = start_end[i].second;
00704     index_to_node_[index] = end;
00705     ends_[i] = index;
00706     CHECK_LE(size, index);
00707     index_to_vehicle_[index] = i;
00708     ++index;
00709   }
00710   for (int i = 0; i < size; ++i) {
00711     for (int j = 0; j < vehicles_; ++j) {
00712       // "start" node: nexts_[i] != start
00713       solver_->AddConstraint(solver_->MakeNonEquality(nexts_[i], starts_[j]));
00714     }
00715     // Extra constraint to state a node can't point to itself
00716     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(nexts_[i],
00717                                                          i,
00718                                                          active_[i]));
00719   }
00720   is_depot_set_ = true;
00721 
00722   // Logging model information.
00723   VLOG(1) << "Number of nodes: " << nodes_;
00724   VLOG(1) << "Number of vehicles: " << vehicles_;
00725   for (int index = 0; index < index_to_node_.size(); ++index) {
00726     VLOG(1) << "Variable index " << index
00727             << " -> Node index " << index_to_node_[index];
00728   }
00729   for (int node = 0; node < node_to_index_.size(); ++node) {
00730     VLOG(2) << "Node index " << node
00731             << " -> Variable index " << node_to_index_[node];
00732   }
00733 }

00735 void RoutingModel::CloseModel() {
00736   if (closed_) {
00737     LOG(WARNING) << "Model already closed";
00738     return;
00739   }
00740   closed_ = true;
00741 
00742   CheckDepot();
00743 
00744   AddNoCycleConstraintInternal();
00745 
00746   const int size = Size();
00747 
00748   // Vehicle variable constraints
00749   for (int i = 0; i < vehicles_; ++i) {
00750     solver_->AddConstraint(solver_->MakeEquality(
00751         vehicle_vars_[starts_[i]], solver_->MakeIntConst(i)));
00752     solver_->AddConstraint(solver_->MakeEquality(
00753         vehicle_vars_[ends_[i]], solver_->MakeIntConst(i)));
00754   }
00755   std::vector<IntVar*> zero_transit(size, solver_->MakeIntConst(Zero()));
00756   solver_->AddConstraint(solver_->MakePathCumul(nexts_.get(),
00757                                                 active_.get(),
00758                                                 vehicle_vars_.get(),
00759                                                 zero_transit.data(),
00760                                                 size,
00761                                                 size + vehicles_));
00762 
00763   // Add constraints to bind vehicle_vars_[i] to -1 in case that node i is not
00764   // active.
00765   for (int i = 0; i < size; ++i) {
00766     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(vehicle_vars_[i], -1,
00767                                                          active_[i]));
00768   }
00769 
00770   // Set all active unless there are disjunctions
00771   if (disjunctions_.size() == 0) {
00772     AddAllActive();
00773   }
00774 
00775   // Associate first and "logical" last nodes
00776   for (int i = 0; i < vehicles_; ++i) {
00777     for (int j = 0; j < vehicles_; ++j) {
00778       if (i != j) {
00779         nexts_[starts_[i]]->RemoveValue(ends_[j]);
00780       }
00781     }
00782   }
00783 
00784   std::vector<IntVar*> cost_elements;
00785   // Arc costs: the cost of an arc (i, nexts_[i], vehicle_vars_[i]) is
00786   // costs_(nexts_[i], vehicle_vars_[i]); the total cost is the sum of arc
00787   // costs.
00788   for (int i = 0; i < size; ++i) {
00789     IntExpr* expr = NULL;
00790     if (homogeneous_costs_) {
00791       expr = solver_->MakeElement(
00792           NewPermanentCallback(this,
00793                                &RoutingModel::GetHomogeneousCost,
00794                                static_cast<int64>(i)),
00795           nexts_[i]);
00796     } else {
00797       expr = solver_->MakeElement(
00798           NewPermanentCallback(this,
00799                                &RoutingModel::GetCost,
00800                                static_cast<int64>(i)),
00801           nexts_[i],
00802           vehicle_vars_[i]);
00803     }
00804     IntVar* var = solver_->MakeProd(expr, active_[i])->Var();
00805     cost_elements.push_back(var);
00806   }
00807   // Penalty costs
00808   for (int i = 0; i < disjunctions_.size(); ++i) {
00809     IntVar* penalty_var = CreateDisjunction(i);
00810     if (penalty_var != NULL) {
00811       cost_elements.push_back(penalty_var);
00812     }
00813   }
00814   cost_ = solver_->MakeSum(cost_elements)->Var();
00815   cost_->set_name("Cost");
00816 
00817   SetUpSearch();
00818 }

00849 const Assignment* RoutingModel::Solve(const Assignment* assignment) {
00850   if (!closed_) {
00851     CloseModel();
00852   }
00853   const int64 start_time_ms = solver_->wall_time();
00854   if (NULL == assignment) {
00855     solver_->Solve(solve_db_, monitors_);
00856   } else {
00857     assignment_->Copy(assignment);
00858     solver_->Solve(improve_db_, monitors_);
00859   }
00860   const int64 elapsed_time_ms = solver_->wall_time() - start_time_ms;
00861   if (collect_assignments_->solution_count() == 1) {
00862     status_ = ROUTING_SUCCESS;
00863     return collect_assignments_->solution(0);
00864   } else {
00865     if (elapsed_time_ms >= time_limit_ms_) {
00866       status_ = ROUTING_FAIL_TIMEOUT;
00867     } else {
00868       status_ = ROUTING_FAIL;
00869     }
00870     return NULL;
00871   }
00872 }

00884 IntVar* RoutingModel::ApplyLocks(const std::vector<int>& locks) {
00885   preassignment_->Clear();
00886   IntVar* next_var = NULL;
00887   int lock_index = FindNextActive(-1, locks);
00888   const int size = locks.size();
00889   if (lock_index < size) {
00890     next_var = NextVar(locks[lock_index]);
00891     preassignment_->Add(next_var);
00892     for (lock_index = FindNextActive(lock_index, locks);
00893          lock_index < size;
00894          lock_index = FindNextActive(lock_index, locks)) {
00895       preassignment_->SetValue(next_var, locks[lock_index]);
00896       next_var = NextVar(locks[lock_index]);
00897       preassignment_->Add(next_var);
00898     }
00899   }
00900   return next_var;
00901 }

00949 int64 RoutingModel::GetArcCost(int64 i, int64 j, int64 vehicle) {
00950   if (vehicle < 0) {
00951     // If a node is inactive, vehicle will be set to -1; handle this situation
00952     // correctly.
00953     // Even though the constraint model does not allow vehice == -1 when i != j,
00954     // this method gets called with such parameters when the model is created,
00955     // because the constraing solver needs to find the minimal and maximal arc
00956     // costs and calls the method with all possible combinations of parameters.
00957     // Zero should be a safe value here.
00958     return 0;
00959   }
00960   CostCacheElement& cache = cost_cache_[i];
00961   if (cache.node == j && cache.vehicle == vehicle) {
00962     return cache.cost;
00963   }
00964   const int64 node_i = IndexToNode(i);
00965   const int64 node_j = IndexToNode(j);
00966   int64 cost = 0;
00967   if (!IsStart(i)) {
00968     cost = costs_[vehicle]->Run(node_i, node_j);
00969   } else if (!IsEnd(j)) {
00970     // Apply route fixed cost on first non-first/last node, in other words on
00971     // the arc from the first node to its next node if it's not the last node.
00972     cost = costs_[vehicle]->Run(node_i, node_j)
00973       + fixed_costs_[index_to_vehicle_[i]];
00974   } else {
00975     // If there's only the first and last nodes on the route, it is considered
00976     // as an empty route thus the cost of 0.
00977     cost = 0;
00978   }
00979   cache.node = j;
00980   cache.vehicle = vehicle;
00981   cache.cost = cost;
00982   return cost;
00983 }

00993       return 0;
00994     }
00995   } else {
00996     return 0;
00997   }

01063 void RoutingModel::SetUpSearch() {
01064   const int size = Size();
01065   assignment_ = solver_->RevAlloc(new Assignment(solver_.get()));
01066   assignment_->Add(nexts_.get(), size);
01067   if (!homogeneous_costs_) {
01068     assignment_->Add(vehicle_vars_.get(), size + vehicles_);
01069   }
01070   assignment_->AddObjective(cost_);
01071 
01072   Assignment* full_assignment =
01073       solver_->RevAlloc(new Assignment(solver_.get()));
01074   for (ConstIter<VarMap> it(cumuls_); !it.at_end(); ++it) {
01075     full_assignment->Add(it->second, size + vehicles_);
01076   }
01077   for (int i = 0; i < extra_vars_.size(); ++i) {
01078     full_assignment->Add(extra_vars_[i]);
01079   }
01080   full_assignment->Add(nexts_.get(), size);
01081   full_assignment->Add(active_.get(), size);
01082   full_assignment->Add(vehicle_vars_.get(), size + vehicles_);
01083   full_assignment->AddObjective(cost_);
01084 
01085   collect_assignments_ =
01086       solver_->MakeBestValueSolutionCollector(full_assignment, false);
01087   monitors_.push_back(collect_assignments_);
01088 
01089   SearchMonitor* optimize;
01090   switch (GetSelectedMetaheuristic()) {
01091     case ROUTING_GUIDED_LOCAL_SEARCH:
01092       LG << "Using Guided Local Search";
01093       if (homogeneous_costs_) {
01094         optimize = solver_->MakeGuidedLocalSearch(
01095             false,
01096             cost_,
01097             NewPermanentCallback(this, &RoutingModel::GetHomogeneousCost),
01098             FLAGS_routing_optimization_step,
01099             nexts_.get(), size,
01100             FLAGS_routing_guided_local_search_lamda_coefficient);
01101       } else {
01102         optimize = solver_->MakeGuidedLocalSearch(
01103             false,
01104             cost_,
01105             BuildCostCallback(),
01106             FLAGS_routing_optimization_step,
01107             nexts_.get(), vehicle_vars_.get(), size,
01108             FLAGS_routing_guided_local_search_lamda_coefficient);
01109       }
01110       break;
01111     case ROUTING_SIMULATED_ANNEALING:
01112       LG << "Using Simulated Annealing";
01113       optimize =
01114           solver_->MakeSimulatedAnnealing(false, cost_,
01115                                           FLAGS_routing_optimization_step,
01116                                           100);
01117       break;
01118     case ROUTING_TABU_SEARCH:
01119       LG << "Using Tabu Search";
01120       optimize = solver_->MakeTabuSearch(false, cost_,
01121                                          FLAGS_routing_optimization_step,
01122                                          nexts_.get(), size,
01123                                          10, 10, .8);
01124       break;
01125     default:
01126       LG << "Using greedy descent";
01127       optimize = solver_->MakeMinimize(cost_, FLAGS_routing_optimization_step);
01128   }
01129   monitors_.push_back(optimize);
01130 
01131   limit_ = solver_->MakeLimit(time_limit_ms_,
01132                               kint64max,
01133                               kint64max,
01134                               FLAGS_routing_solution_limit,
01135                               true);
01136   monitors_.push_back(limit_);
01137 
01138   ls_limit_ = solver_->MakeLimit(time_limit_ms_,
01139                                  kint64max,
01140                                  kint64max,
01141                                  1,
01142                                  true);
01143 
01144   lns_limit_ = solver_->MakeLimit(lns_time_limit_ms_,
01145                                   kint64max,
01146                                   kint64max,
01147                                   kint64max);
01148 
01149   std::vector<LocalSearchOperator*> operators = extra_operators_;
01150   if (vehicles_ > 1) {
01151     if (!FLAGS_routing_no_relocate) {
01152       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::RELOCATE);
01153     }
01154     if (!FLAGS_routing_no_exchange) {
01155       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::EXCHANGE);
01156     }
01157     if (!FLAGS_routing_no_cross) {
01158       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::CROSS);
01159     }
01160   }
01161   if (!FLAGS_routing_no_lkh
01162       && !FLAGS_routing_tabu_search
01163       && !FLAGS_routing_simulated_annealing) {
01164     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::LK);
01165   }
01166   if (!FLAGS_routing_no_2opt) {
01167     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::TWOOPT);
01168   }
01169   if (!FLAGS_routing_no_oropt) {
01170     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::OROPT);
01171   }
01172   if (!FLAGS_routing_no_make_active && disjunctions_.size() != 0) {
01173     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEINACTIVE);
01174     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEACTIVE);
01175     if (!FLAGS_routing_use_extended_swap_active) {
01176       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::SWAPACTIVE);
01177     } else {
01178       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::EXTENDEDSWAPACTIVE);
01179     }
01180   }
01181   // TODO(user): move the following operators to a second local search loop.
01182   if (!FLAGS_routing_no_tsp
01183       && !FLAGS_routing_tabu_search
01184       && !FLAGS_routing_simulated_annealing) {
01185     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPOPT);
01186   }
01187   if (!FLAGS_routing_no_tsplns
01188       && !FLAGS_routing_tabu_search
01189       && !FLAGS_routing_simulated_annealing) {
01190     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPLNS);
01191   }
01192   if (!FLAGS_routing_no_lns) {
01193     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::PATHLNS);
01194     if (disjunctions_.size() != 0) {
01195       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::UNACTIVELNS);
01196     }
01197   }
01198   LocalSearchOperator* local_search_operator =
01199       solver_->ConcatenateOperators(operators);
01200 
01201   DecisionBuilder* finalize_solution =
01202       solver_->MakePhase(nexts_.get(), size,
01203                          Solver::CHOOSE_FIRST_UNBOUND,
01204                          Solver::ASSIGN_MIN_VALUE);
01205 
01206   DecisionBuilder* first_solution = finalize_solution;
01207   switch (GetSelectedFirstSolutionStrategy()) {
01208     case ROUTING_GLOBAL_CHEAPEST_ARC:
01209       LG << "Using ROUTING_GLOBAL_CHEAPEST_ARC";
01210       first_solution =
01211           solver_->MakePhase(nexts_.get(), size,
01212                              NewPermanentCallback(
01213                                  this,
01214                                  &RoutingModel::GetFirstSolutionCost),
01215                              Solver::CHOOSE_STATIC_GLOBAL_BEST);
01216       break;
01217     case ROUTING_LOCAL_CHEAPEST_ARC:
01218       LG << "Using ROUTING_LOCAL_CHEAPEST_ARC";
01219       first_solution =
01220           solver_->MakePhase(nexts_.get(), size,
01221                              Solver::CHOOSE_FIRST_UNBOUND,
01222                              NewPermanentCallback(
01223                                  this,
01224                                  &RoutingModel::GetFirstSolutionCost));
01225       break;
01226     case ROUTING_PATH_CHEAPEST_ARC:
01227       LG << "Using ROUTING_PATH_CHEAPEST_ARC";
01228       first_solution =
01229           solver_->MakePhase(nexts_.get(), size,
01230                              Solver::CHOOSE_PATH,
01231                              NewPermanentCallback(
01232                                  this,
01233                                  &RoutingModel::GetFirstSolutionCost));
01234       break;
01235     case ROUTING_EVALUATOR_STRATEGY:
01236       LG << "Using ROUTING_EVALUATOR_STRATEGY";
01237       CHECK(first_solution_evaluator_ != NULL);
01238       first_solution =
01239           solver_->MakePhase(nexts_.get(), size,
01240                              Solver::CHOOSE_PATH,
01241                              NewPermanentCallback(
01242                                  first_solution_evaluator_.get(),
01243                                  &Solver::IndexEvaluator2::Run));
01244       break;
01245     case ROUTING_DEFAULT_STRATEGY:
01246       LG << "Using DEFAULT";
01247       break;
01248     default:
01249       LOG(WARNING) << "Unknown argument for routing_first_solution, "
01250           "using default";
01251   }
01252   if (FLAGS_routing_use_first_solution_dive) {
01253     DecisionBuilder* apply =
01254         solver_->MakeApplyBranchSelector(NewPermanentCallback(&LeftDive));
01255     first_solution = solver_->Compose(apply, first_solution);
01256   }
01257 
01258   std::vector<LocalSearchFilter*> filters;
01259   if (FLAGS_routing_use_objective_filter) {
01260     if (homogeneous_costs_) {
01261       LocalSearchFilter* filter =
01262           solver_->MakeLocalSearchObjectiveFilter(
01263               nexts_.get(),
01264               size,
01265               NewPermanentCallback(this,
01266                                    &RoutingModel::GetHomogeneousFilterCost),
01267               cost_,
01268               Solver::EQ,
01269               Solver::SUM);
01270       filters.push_back(filter);
01271     } else {
01272       LocalSearchFilter* filter =
01273           solver_->MakeLocalSearchObjectiveFilter(
01274               nexts_.get(),
01275               vehicle_vars_.get(),
01276               size,
01277               NewPermanentCallback(this, &RoutingModel::GetFilterCost),
01278               cost_,
01279               Solver::EQ,
01280               Solver::SUM);
01281       filters.push_back(filter);
01282     }
01283   }
01284   if (FLAGS_routing_use_path_cumul_filter) {
01285     for (ConstIter<VarMap> iter(cumuls_); !iter.at_end(); ++iter) {
01286       const string& name = iter->first;
01287       filters.push_back(solver_->RevAlloc(
01288           new PathCumulFilter(nexts_.get(),
01289                               Size(),
01290                               iter->second,
01291                               Size() + vehicles_,
01292                               transit_evaluators_[name],
01293                               name)));
01294     }
01295   }
01296 
01297   LocalSearchPhaseParameters* parameters =
01298       solver_->MakeLocalSearchPhaseParameters(
01299           local_search_operator,
01300           solver_->MakeSolveOnce(finalize_solution, lns_limit_),
01301           ls_limit_,
01302           filters);
01303 
01304   if (FLAGS_routing_dfs) {
01305     solve_db_ = finalize_solution;
01306   } else {
01307     if (homogeneous_costs_) {
01308       solve_db_ = solver_->MakeLocalSearchPhase(nexts_.get(),
01309                                                 size,
01310                                                 first_solution,
01311                                                 parameters);
01312     } else {
01313       const int all_size = size + size + vehicles_;
01314       scoped_array<IntVar*> all_vars(new IntVar*[all_size]);
01315       for (int i = 0; i < size; ++i) {
01316         all_vars[i] = nexts_[i];
01317       }
01318       for (int i = size; i < all_size; ++i) {
01319         all_vars[i] = vehicle_vars_[i - size];
01320       }
01321       solve_db_ = solver_->MakeLocalSearchPhase(all_vars.get(),
01322                                                 all_size,
01323                                                 first_solution,
01324                                                 parameters);
01325     }
01326   }
01327   DecisionBuilder* restore_preassignment =
01328       solver_->MakeRestoreAssignment(preassignment_);
01329   solve_db_ = solver_->Compose(restore_preassignment, solve_db_);
01330   improve_db_ =
01331       solver_->Compose(restore_preassignment,
01332                        solver_->MakeLocalSearchPhase(assignment_, parameters));
01333   restore_assignment_ =
01334       solver_->Compose(solver_->MakeRestoreAssignment(assignment_),
01335                        finalize_solution);
01336 
01337   if (FLAGS_routing_trace) {
01338     const int kLogPeriod = 10000;
01339     SearchMonitor* trace = solver_->MakeSearchLog(kLogPeriod, cost_);
01340     monitors_.push_back(trace);
01341   }
01342   if (FLAGS_routing_search_trace) {
01343     SearchMonitor* trace = solver_->MakeSearchTrace("Routing ");
01344     monitors_.push_back(trace);
01345   }
01346 }

01412 IntVar** RoutingModel::GetOrMakeTransits(Solver::IndexEvaluator2* evaluator,
01413                                          int64 slack_max,
01414                                          int64 capacity,
01415                                          const string& name) {
01416   evaluator->CheckIsRepeatable();
01417   IntVar** named_transits = FindPtrOrNull(transits_, name);
01418   if (named_transits == NULL) {
01419     std::vector<IntVar*> transits;
01420     const int size = Size();
01421     IntVar** transit_array = solver_->RevAlloc(new IntVar*[size]);
01422     for (int i = 0; i < size; ++i) {
01423       IntVar* fixed_transit =
01424           solver_->MakeElement(NewPermanentCallback(
01425               this,
01426               &RoutingModel::WrappedEvaluator,
01427               evaluator,
01428               static_cast<int64>(i)),
01429                                nexts_[i])->Var();
01430       if (slack_max == 0) {
01431         transit_array[i] = fixed_transit;
01432       } else {
01433         IntVar* slack_var = solver_->MakeIntVar(0, slack_max, "slack");
01434         transit_array[i] = solver_->MakeSum(slack_var, fixed_transit)->Var();
01435       }
01436       transit_array[i]->SetMin(0);
01437       transit_array[i]->SetMax(capacity);
01438     }
01439     transits_[name] = transit_array;
01440     transit_evaluators_[name] =
01441         NewPermanentCallback(this,
01442                              &RoutingModel::WrappedEvaluator,
01443                              evaluator);
01444     owned_callbacks_.insert(transit_evaluators_[name]);
01445     owned_callbacks_.insert(evaluator);
01446     return transit_array;
01447   }
01448   return named_transits;
01449 }
