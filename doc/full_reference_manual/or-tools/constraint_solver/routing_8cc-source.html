00172 // Complexity: O(Sum(Length(paths modified)) + #paths modified ^ 2).
00173 // (#paths modified is usually very small).
00174 bool PathCumulFilter::Accept(const Assignment* delta,
00175                              const Assignment* deltadelta) {
00176   const Assignment::IntContainer& container = delta->IntVarContainer();
00177   const int delta_size = container.Size();
00178   // Determining touched paths.
00179   std::vector<int64> touched_paths;
00180   for (int i = 0; i < delta_size; ++i) {
00181     const IntVarElement& new_element = container.Element(i);
00182     const IntVar* const var = new_element.Var();
00183     int64 index = kUnassigned;
00184     if (FindIndex(var, &index)) {
00185       const int64 start = node_path_starts_[index];
00186       if (start != kUnassigned
00187           && find(touched_paths.begin(), touched_paths.end(), start)
00188           == touched_paths.end()) {
00189         touched_paths.push_back(start);
00190       }
00191     }
00192   }
00193   // Checking feasibility of touched paths.
00194   const int touched_paths_size = touched_paths.size();
00195   for (int i = 0; i < touched_paths_size; ++i) {
00196     int64 node = touched_paths[i];
00197     int64 cumul = cumuls_[node]->Min();
00198     while (node < Size()) {
00199       const IntVar* const next_var = Var(node);
00200       int64 next = saved_nexts_[node];
00201       if (container.Contains(next_var)) {
00202         const IntVarElement& element = container.Element(next_var);
00203         if (element.Bound()) {
00204           next = element.Value();
00205         } else {
00206           // LNS detected, return true since path was ok up to now.
00207           return true;
00208         }
00209       }
00210       cumul += evaluator_->Run(node, next);
00211       if (cumul > cumuls_[next]->Max()) {
00212         return false;
00213       }
00214       cumul = std::max(cumuls_[next]->Min(), cumul);
00215       node = next;
00216     }
00217   }
00218   return true;
00219 }

00221 void PathCumulFilter::OnSynchronize() {
00222   const int nexts_size = Size();
00223   // Detecting path starts, used to track which node belongs to which path.
00224   std::vector<int64> path_starts;
00225   Bitmap has_prevs(nexts_size, false);
00226   for (int i = 0; i < nexts_size; ++i) {
00227     const int next = Value(i);
00228     if (next < nexts_size) {
00229       has_prevs.Set(next, true);
00230     }
00231   }
00232   for (int i = 0; i < nexts_size; ++i) {
00233     if (!has_prevs.Get(i)) {
00234       path_starts.push_back(i);
00235     }
00236   }
00237   // Marking unactive nodes (which are not on a path).
00238   node_path_starts_.assign(cumuls_size_, kUnassigned);
00239   // Marking nodes on a path and storing next values.
00240   for (int i = 0; i < path_starts.size(); ++i) {
00241     const int64 start = path_starts[i];
00242     int node = start;
00243     node_path_starts_[node] = start;
00244     int next = Value(node);
00245     saved_nexts_[node] = next;
00246     while (next < nexts_size) {
00247       node = next;
00248       node_path_starts_[node] = start;
00249       next = Value(node);
00250       saved_nexts_[node] = next;
00251     }
00252     saved_nexts_[node] = next;
00253     node_path_starts_[next] = start;
00254   }
00255 }

00664 void RoutingModel::SetStartEnd(
00665     const std::vector<std::pair<NodeIndex, NodeIndex> >& start_end) {
00666   if (is_depot_set_) {
00667     LOG(WARNING) << "A depot has already been specified, ignoring new ones";
00668     return;
00669   }
00670   CHECK_EQ(start_end.size(), vehicles_);
00671   const int size = Size();
00672   hash_set<NodeIndex> starts;
00673   hash_set<NodeIndex> ends;
00674   for (int i = 0; i < vehicles_; ++i) {
00675     const NodeIndex start = start_end[i].first;
00676     const NodeIndex end = start_end[i].second;
00677     CHECK_GE(start, 0);
00678     CHECK_GE(end, 0);
00679     CHECK_LE(start, nodes_);
00680     CHECK_LE(end, nodes_);
00681     starts.insert(start);
00682     ends.insert(end);
00683   }
00684   index_to_node_.resize(size + vehicles_);
00685   node_to_index_.resize(nodes_, kUnassigned);
00686   int index = 0;
00687   for (NodeIndex i = kFirstNode; i < nodes_; ++i) {
00688     if (starts.count(i) != 0 || ends.count(i) == 0) {
00689       index_to_node_[index] = i;
00690       node_to_index_[i] = index;
00691       ++index;
00692     }
00693   }
00694   hash_set<NodeIndex> node_set;
00695   index_to_vehicle_.resize(size + vehicles_, kUnassigned);
00696   for (int i = 0; i < vehicles_; ++i) {
00697     const NodeIndex start = start_end[i].first;
00698     if (node_set.count(start) == 0) {
00699       node_set.insert(start);
00700       const int start_index = node_to_index_[start];
00701       starts_[i] = start_index;
00702       CHECK_NE(kUnassigned, start_index);
00703       index_to_vehicle_[start_index] = i;
00704     } else {
00705       starts_[i] = index;
00706       index_to_node_[index] = start;
00707       index_to_vehicle_[index] = i;
00708       ++index;
00709     }
00710   }
00711   for (int i = 0; i < vehicles_; ++i) {
00712     NodeIndex end = start_end[i].second;
00713     index_to_node_[index] = end;
00714     ends_[i] = index;
00715     CHECK_LE(size, index);
00716     index_to_vehicle_[index] = i;
00717     ++index;
00718   }
00719   for (int i = 0; i < size; ++i) {
00720     for (int j = 0; j < vehicles_; ++j) {
00721       // "start" node: nexts_[i] != start
00722       solver_->AddConstraint(solver_->MakeNonEquality(nexts_[i], starts_[j]));
00723     }
00724     // Extra constraint to state a node can't point to itself
00725     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(nexts_[i],
00726                                                          i,
00727                                                          active_[i]));
00728   }

00736             << " -> Node index " << index_to_node_[index];

00744 void RoutingModel::CloseModel() {
00745   if (closed_) {
00746     LOG(WARNING) << "Model already closed";
00747     return;
00748   }
00749   closed_ = true;
00750 
00751   CheckDepot();
00752 
00753   AddNoCycleConstraintInternal();
00754 
00755   const int size = Size();
00756 
00757   // Vehicle variable constraints
00758   for (int i = 0; i < vehicles_; ++i) {
00759     solver_->AddConstraint(solver_->MakeEquality(
00760         vehicle_vars_[starts_[i]], solver_->MakeIntConst(i)));
00761     solver_->AddConstraint(solver_->MakeEquality(
00762         vehicle_vars_[ends_[i]], solver_->MakeIntConst(i)));
00763   }
00764   std::vector<IntVar*> zero_transit(size, solver_->MakeIntConst(Zero()));
00765   solver_->AddConstraint(solver_->MakePathCumul(nexts_.get(),
00766                                                 active_.get(),
00767                                                 vehicle_vars_.get(),
00768                                                 zero_transit.data(),
00769                                                 size,
00770                                                 size + vehicles_));
00771 
00772   // Add constraints to bind vehicle_vars_[i] to -1 in case that node i is not
00773   // active.
00774   for (int i = 0; i < size; ++i) {
00775     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(vehicle_vars_[i], -1,
00776                                                          active_[i]));
00777   }
00778 
00779   // Set all active unless there are disjunctions
00780   if (disjunctions_.size() == 0) {
00781     AddAllActive();
00782   }
00783 
00784   // Associate first and "logical" last nodes
00785   for (int i = 0; i < vehicles_; ++i) {
00786     for (int j = 0; j < vehicles_; ++j) {
00787       if (i != j) {
00788         nexts_[starts_[i]]->RemoveValue(ends_[j]);
00789       }
00790     }
00791   }
00792 
00793   std::vector<IntVar*> cost_elements;
00794   // Arc costs: the cost of an arc (i, nexts_[i], vehicle_vars_[i]) is
00795   // costs_(nexts_[i], vehicle_vars_[i]); the total cost is the sum of arc
00796   // costs.
00797   for (int i = 0; i < size; ++i) {
00798     IntExpr* expr = NULL;
00799     if (homogeneous_costs_) {
00800       expr = solver_->MakeElement(
00801           NewPermanentCallback(this,
00802                                &RoutingModel::GetHomogeneousCost,
00803                                static_cast<int64>(i)),
00804           nexts_[i]);
00805     } else {
00806       expr = solver_->MakeElement(
00807           NewPermanentCallback(this,
00808                                &RoutingModel::GetCost,
00809                                static_cast<int64>(i)),
00810           nexts_[i],
00811           vehicle_vars_[i]);
00812     }
00813     IntVar* var = solver_->MakeProd(expr, active_[i])->Var();
00814     cost_elements.push_back(var);
00815   }
00816   // Penalty costs
00817   for (int i = 0; i < disjunctions_.size(); ++i) {
00818     IntVar* penalty_var = CreateDisjunction(i);
00819     if (penalty_var != NULL) {
00820       cost_elements.push_back(penalty_var);
00821     }
00822   }
00823   cost_ = solver_->MakeSum(cost_elements)->Var();
00824   cost_->set_name("Cost");
00825 
00826   SetUpSearch();
00827 }

00899     next_var = NextVar(locks[lock_index]);

00902          lock_index < size;
00903          lock_index = FindNextActive(lock_index, locks)) {

00958 int64 RoutingModel::GetArcCost(int64 i, int64 j, int64 vehicle) {
00959   if (vehicle < 0) {
00960     // If a node is inactive, vehicle will be set to -1; handle this situation
00961     // correctly.
00962     // Even though the constraint model does not allow vehice == -1 when i != j,
00963     // this method gets called with such parameters when the model is created,
00964     // because the constraing solver needs to find the minimal and maximal arc
00965     // costs and calls the method with all possible combinations of parameters.
00966     // Zero should be a safe value here.
00967     return 0;
00968   }
00969   CostCacheElement& cache = cost_cache_[i];
00970   if (cache.node == j && cache.vehicle == vehicle) {
00971     return cache.cost;
00972   }
00973   const NodeIndex node_i = IndexToNode(i);
00974   const NodeIndex node_j = IndexToNode(j);
00975   int64 cost = 0;
00976   if (!IsStart(i)) {
00977     cost = costs_[vehicle]->Run(node_i, node_j);
00978   } else if (!IsEnd(j)) {
00979     // Apply route fixed cost on first non-first/last node, in other words on
00980     // the arc from the first node to its next node if it's not the last node.
00981     cost = costs_[vehicle]->Run(node_i, node_j)
00982       + fixed_costs_[index_to_vehicle_[i]];
00983   } else {
00984     // If there's only the first and last nodes on the route, it is considered
00985     // as an empty route thus the cost of 0.
00986     cost = 0;
00987   }
00988   cache.node = j;
00989   cache.vehicle = vehicle;
00990   cache.cost = cost;
00991   return cost;
00992 }

01072 void RoutingModel::SetUpSearch() {
01073   const int size = Size();
01074   assignment_ = solver_->RevAlloc(new Assignment(solver_.get()));
01075   assignment_->Add(nexts_.get(), size);
01076   if (!homogeneous_costs_) {
01077     assignment_->Add(vehicle_vars_.get(), size + vehicles_);
01078   }
01079   assignment_->AddObjective(cost_);
01080 
01081   Assignment* full_assignment =
01082       solver_->RevAlloc(new Assignment(solver_.get()));
01083   for (ConstIter<VarMap> it(cumuls_); !it.at_end(); ++it) {
01084     full_assignment->Add(it->second, size + vehicles_);
01085   }
01086   for (int i = 0; i < extra_vars_.size(); ++i) {
01087     full_assignment->Add(extra_vars_[i]);
01088   }
01089   full_assignment->Add(nexts_.get(), size);
01090   full_assignment->Add(active_.get(), size);
01091   full_assignment->Add(vehicle_vars_.get(), size + vehicles_);
01092   full_assignment->AddObjective(cost_);
01093 
01094   collect_assignments_ =
01095       solver_->MakeBestValueSolutionCollector(full_assignment, false);
01096   monitors_.push_back(collect_assignments_);
01097 
01098   SearchMonitor* optimize;
01099   switch (GetSelectedMetaheuristic()) {
01100     case ROUTING_GUIDED_LOCAL_SEARCH:
01101       LG << "Using Guided Local Search";
01102       if (homogeneous_costs_) {
01103         optimize = solver_->MakeGuidedLocalSearch(
01104             false,
01105             cost_,
01106             NewPermanentCallback(this, &RoutingModel::GetHomogeneousCost),
01107             FLAGS_routing_optimization_step,
01108             nexts_.get(), size,
01109             FLAGS_routing_guided_local_search_lamda_coefficient);
01110       } else {
01111         optimize = solver_->MakeGuidedLocalSearch(
01112             false,
01113             cost_,
01114             BuildCostCallback(),
01115             FLAGS_routing_optimization_step,
01116             nexts_.get(), vehicle_vars_.get(), size,
01117             FLAGS_routing_guided_local_search_lamda_coefficient);
01118       }
01119       break;
01120     case ROUTING_SIMULATED_ANNEALING:
01121       LG << "Using Simulated Annealing";
01122       optimize =
01123           solver_->MakeSimulatedAnnealing(false, cost_,
01124                                           FLAGS_routing_optimization_step,
01125                                           100);
01126       break;
01127     case ROUTING_TABU_SEARCH:
01128       LG << "Using Tabu Search";
01129       optimize = solver_->MakeTabuSearch(false, cost_,
01130                                          FLAGS_routing_optimization_step,
01131                                          nexts_.get(), size,
01132                                          10, 10, .8);
01133       break;
01134     default:
01135       LG << "Using greedy descent";
01136       optimize = solver_->MakeMinimize(cost_, FLAGS_routing_optimization_step);
01137   }

01139 
01140   limit_ = solver_->MakeLimit(time_limit_ms_,

01147   ls_limit_ = solver_->MakeLimit(time_limit_ms_,
01148                                  kint64max,
01149                                  kint64max,
01150                                  1,
01151                                  true);
01152 
01153   lns_limit_ = solver_->MakeLimit(lns_time_limit_ms_,
01154                                   kint64max,
01155                                   kint64max,
01156                                   kint64max);
01157 
01158   std::vector<LocalSearchOperator*> operators = extra_operators_;
01159   if (vehicles_ > 1) {
01160     if (!FLAGS_routing_no_relocate) {
01161       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::RELOCATE);
01162     }
01163     if (!FLAGS_routing_no_exchange) {
01164       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::EXCHANGE);
01165     }
01166     if (!FLAGS_routing_no_cross) {
01167       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::CROSS);
01168     }
01169   }
01170   if (!FLAGS_routing_no_lkh
01171       && !FLAGS_routing_tabu_search
01172       && !FLAGS_routing_simulated_annealing) {
01173     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::LK);
01174   }
01175   if (!FLAGS_routing_no_2opt) {
01176     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::TWOOPT);
01177   }
01178   if (!FLAGS_routing_no_oropt) {
01179     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::OROPT);
01180   }
01181   if (!FLAGS_routing_no_make_active && disjunctions_.size() != 0) {
01182     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEINACTIVE);
01183     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEACTIVE);
01184     if (!FLAGS_routing_use_extended_swap_active) {
01185       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::SWAPACTIVE);
01186     } else {

01188     }

01191   if (!FLAGS_routing_no_tsp
01192       && !FLAGS_routing_tabu_search
01193       && !FLAGS_routing_simulated_annealing) {
01194     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPOPT);
01195   }
01196   if (!FLAGS_routing_no_tsplns
01197       && !FLAGS_routing_tabu_search
01198       && !FLAGS_routing_simulated_annealing) {
01199     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPLNS);
01200   }
01201   if (!FLAGS_routing_no_lns) {
01202     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::PATHLNS);
01203     if (disjunctions_.size() != 0) {
01204       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::UNACTIVELNS);
01205     }
01206   }
01207   LocalSearchOperator* local_search_operator =
01208       solver_->ConcatenateOperators(operators);
01209 
01210   DecisionBuilder* finalize_solution =
01211       solver_->MakePhase(nexts_.get(), size,
01212                          Solver::CHOOSE_FIRST_UNBOUND,
01213                          Solver::ASSIGN_MIN_VALUE);
01214 
01215   DecisionBuilder* first_solution = finalize_solution;
01216   switch (GetSelectedFirstSolutionStrategy()) {
01217     case ROUTING_GLOBAL_CHEAPEST_ARC:
01218       LG << "Using ROUTING_GLOBAL_CHEAPEST_ARC";
01219       first_solution =
01220           solver_->MakePhase(nexts_.get(), size,
01221                              NewPermanentCallback(
01222                                  this,
01223                                  &RoutingModel::GetFirstSolutionCost),
01224                              Solver::CHOOSE_STATIC_GLOBAL_BEST);
01225       break;
01226     case ROUTING_LOCAL_CHEAPEST_ARC:
01227       LG << "Using ROUTING_LOCAL_CHEAPEST_ARC";
01228       first_solution =
01229           solver_->MakePhase(nexts_.get(), size,
01230                              Solver::CHOOSE_FIRST_UNBOUND,
01231                              NewPermanentCallback(
01232                                  this,
01233                                  &RoutingModel::GetFirstSolutionCost));
01234       break;
01235     case ROUTING_PATH_CHEAPEST_ARC:
01236       LG << "Using ROUTING_PATH_CHEAPEST_ARC";
01237       first_solution =
01238           solver_->MakePhase(nexts_.get(), size,
01239                              Solver::CHOOSE_PATH,
01240                              NewPermanentCallback(
01241                                  this,
01242                                  &RoutingModel::GetFirstSolutionCost));
01243       break;
01244     case ROUTING_EVALUATOR_STRATEGY:
01245       LG << "Using ROUTING_EVALUATOR_STRATEGY";
01246       CHECK(first_solution_evaluator_ != NULL);
01247       first_solution =
01248           solver_->MakePhase(nexts_.get(), size,
01249                              Solver::CHOOSE_PATH,
01250                              NewPermanentCallback(
01251                                  first_solution_evaluator_.get(),
01252                                  &Solver::IndexEvaluator2::Run));
01253       break;
01254     case ROUTING_DEFAULT_STRATEGY:
01255       LG << "Using DEFAULT";
01256       break;
01257     default:
01258       LOG(WARNING) << "Unknown argument for routing_first_solution, "
01259           "using default";
01260   }
01261   if (FLAGS_routing_use_first_solution_dive) {
01262     DecisionBuilder* apply =
01263         solver_->MakeApplyBranchSelector(NewPermanentCallback(&LeftDive));
01264     first_solution = solver_->Compose(apply, first_solution);
01265   }
01266 
01267   std::vector<LocalSearchFilter*> filters;
01268   if (FLAGS_routing_use_objective_filter) {
01269     if (homogeneous_costs_) {
01270       LocalSearchFilter* filter =
01271           solver_->MakeLocalSearchObjectiveFilter(
01272               nexts_.get(),
01273               size,
01274               NewPermanentCallback(this,
01275                                    &RoutingModel::GetHomogeneousFilterCost),
01276               cost_,
01277               Solver::EQ,
01278               Solver::SUM);
01279       filters.push_back(filter);
01280     } else {
01281       LocalSearchFilter* filter =
01282           solver_->MakeLocalSearchObjectiveFilter(
01283               nexts_.get(),
01284               vehicle_vars_.get(),
01285               size,
01286               NewPermanentCallback(this, &RoutingModel::GetFilterCost),
01287               cost_,
01288               Solver::EQ,
01289               Solver::SUM);
01290       filters.push_back(filter);
01291     }
01292   }
01293   if (FLAGS_routing_use_path_cumul_filter) {
01294     for (ConstIter<VarMap> iter(cumuls_); !iter.at_end(); ++iter) {
01295       const string& name = iter->first;
01296       filters.push_back(solver_->RevAlloc(
01297           new PathCumulFilter(nexts_.get(),
01298                               Size(),
01299                               iter->second,
01300                               Size() + vehicles_,
01301                               transit_evaluators_[name],
01302                               name)));
01303     }
01304   }
01305 
01306   LocalSearchPhaseParameters* parameters =
01307       solver_->MakeLocalSearchPhaseParameters(
01308           local_search_operator,
01309           solver_->MakeSolveOnce(finalize_solution, lns_limit_),
01310           ls_limit_,
01311           filters);
01312 
01313   if (FLAGS_routing_dfs) {
01314     solve_db_ = finalize_solution;
01315   } else {
01316     if (homogeneous_costs_) {
01317       solve_db_ = solver_->MakeLocalSearchPhase(nexts_.get(),
01318                                                 size,
01319                                                 first_solution,
01320                                                 parameters);
01321     } else {
01322       const int all_size = size + size + vehicles_;
01323       scoped_array<IntVar*> all_vars(new IntVar*[all_size]);
01324       for (int i = 0; i < size; ++i) {
01325         all_vars[i] = nexts_[i];
01326       }
01327       for (int i = size; i < all_size; ++i) {
01328         all_vars[i] = vehicle_vars_[i - size];
01329       }
01330       solve_db_ = solver_->MakeLocalSearchPhase(all_vars.get(),
01331                                                 all_size,
01332                                                 first_solution,
01333                                                 parameters);
01334     }
01335   }
01336   DecisionBuilder* restore_preassignment =
01337       solver_->MakeRestoreAssignment(preassignment_);
01338   solve_db_ = solver_->Compose(restore_preassignment, solve_db_);
01339   improve_db_ =
01340       solver_->Compose(restore_preassignment,
01341                        solver_->MakeLocalSearchPhase(assignment_, parameters));
01342   restore_assignment_ =
01343       solver_->Compose(solver_->MakeRestoreAssignment(assignment_),
01344                        finalize_solution);
01345 
01346   if (FLAGS_routing_trace) {
01347     const int kLogPeriod = 10000;
01348     SearchMonitor* trace = solver_->MakeSearchLog(kLogPeriod, cost_);
01349     monitors_.push_back(trace);
01350   }
01351   if (FLAGS_routing_search_trace) {
01352     SearchMonitor* trace = solver_->MakeSearchTrace("Routing ");
01353     monitors_.push_back(trace);
01354   }
01355 }

01421 IntVar** RoutingModel::GetOrMakeTransits(NodeEvaluator2* evaluator,
01422                                          int64 slack_max,
01423                                          int64 capacity,
01424                                          const string& name) {
01425   evaluator->CheckIsRepeatable();
01426   IntVar** named_transits = FindPtrOrNull(transits_, name);
01427   if (named_transits == NULL) {
01428     std::vector<IntVar*> transits;
01429     const int size = Size();
01430     IntVar** transit_array = solver_->RevAlloc(new IntVar*[size]);
01431     for (int i = 0; i < size; ++i) {
01432       IntVar* fixed_transit =
01433           solver_->MakeElement(NewPermanentCallback(
01434               this,
01435               &RoutingModel::WrappedEvaluator,
01436               evaluator,
01437               static_cast<int64>(i)),
01438                                nexts_[i])->Var();
01439       if (slack_max == 0) {
01440         transit_array[i] = fixed_transit;
01441       } else {
01442         IntVar* slack_var = solver_->MakeIntVar(0, slack_max, "slack");
01443         transit_array[i] = solver_->MakeSum(slack_var, fixed_transit)->Var();
01444       }
01445       transit_array[i]->SetMin(0);
01446       transit_array[i]->SetMax(capacity);
01447     }
01448     transits_[name] = transit_array;
01449     transit_evaluators_[name] =
01450         NewPermanentCallback(this,
01451                              &RoutingModel::WrappedEvaluator,
01452                              evaluator);
01453     owned_index_callbacks_.insert(transit_evaluators_[name]);
01454     owned_node_callbacks_.insert(evaluator);
01455     return transit_array;
01456   }
01457   return named_transits;
01458 }
