00134       demon_run->add_end_time(CurrentTime());
00135       demon_run->set_failures(demon_run->failures() + 1);

00282   void ExportInformation(DemonRuns* const demon_runs,
00283                          int64* const demon_invocations,
00284                          int64* const fails,
00285                          int64* const total_demon_runtime,
00286                          double* const mean_demon_runtime,
00287                          double* const median_demon_runtime,
00288                          double* const stddev_demon_runtime) {
00289     CHECK_NOTNULL(demon_runs);
00290     CHECK_EQ(demon_runs->start_time_size(), demon_runs->end_time_size());
00291 
00292     const int runs = demon_runs->start_time_size();
00293     *demon_invocations = runs;
00294     *fails = demon_runs->failures();
00295     *total_demon_runtime = 0;
00296     *mean_demon_runtime = 0.0;
00297     *median_demon_runtime = 0.0;
00298     *stddev_demon_runtime = 0.0;
00299     std::vector<double> runtimes;
00300     for (int run_index = 0; run_index < runs; ++run_index) {
00301       const int64 demon_time = demon_runs->end_time(run_index) -
00302           demon_runs->start_time(run_index);
00303       *total_demon_runtime += demon_time;
00304       runtimes.push_back(demon_time);
00305     }
00306     // Compute mean.
00307     if (runtimes.size()) {
00308       *mean_demon_runtime = (1.0L * *total_demon_runtime) / runtimes.size();
00309 
00310       // Compute median.
00311       sort(runtimes.begin(), runtimes.end());
00312       const int pivot = runtimes.size() / 2;
00313 
00314       if (runtimes.size() == 1) {
00315         *median_demon_runtime = runtimes[0];
00316       } else {
00317         *median_demon_runtime = runtimes.size() % 2 == 1 ?
00318             runtimes[pivot] :
00319             (runtimes[pivot - 1] + runtimes[pivot]) / 2.0;
00320       }
00321 
00322       // Compute standard deviation.
00323       double total_deviation = 0.0f;
00324 
00325       for (int i = 0; i < runtimes.size(); ++i) {
00326         total_deviation += pow(runtimes[i] - *mean_demon_runtime, 2);
00327       }
00328 
00329       *stddev_demon_runtime = sqrt(total_deviation / runtimes.size());
00330     }
00331   }
