00213   void ExportInformation(const Constraint* const constraint,
00214                          int64* const fails,
00215                          int64* const initial_propagation_runtime,
00216                          int64* const demon_invocations,
00217                          int64* const total_demon_runtime,
00218                          int* demons) {
00219     CHECK_NOTNULL(constraint);
00220     ConstraintRuns* const ct_run = constraint_map_[constraint];
00221     CHECK_NOTNULL(ct_run);
00222     *demon_invocations = 0;
00223     *fails = ct_run->failures();
00224     *initial_propagation_runtime = ct_run->initial_propagation_end_time() -
00225         ct_run->initial_propagation_start_time();
00226     *total_demon_runtime = 0;
00227 
00228     // Gather information.
00229     *demons = ct_run->demons_size();
00230     CHECK_EQ(*demons, demons_per_constraint_[constraint].size());
00231     for (int demon_index = 0; demon_index < *demons; ++demon_index) {
00232       const DemonRuns& demon_runs = ct_run->demons(demon_index);
00233       *fails += demon_runs.failures();
00234       CHECK_EQ(demon_runs.start_time_size(), demon_runs.end_time_size());
00235       const int runs = demon_runs.start_time_size();
00236       *demon_invocations += runs;
00237       for (int run_index = 0; run_index < runs; ++run_index) {
00238         const int64 demon_time = demon_runs.end_time(run_index) -
00239             demon_runs.start_time(run_index);

00245   void ExportInformation(DemonRuns* const demon_runs,
00246                          int64* const demon_invocations,
00247                          int64* const fails,
00248                          int64* const total_demon_runtime,
00249                          double* const mean_demon_runtime,
00250                          double* const median_demon_runtime,
00251                          double* const stddev_demon_runtime) {
00252     CHECK_NOTNULL(demon_runs);
00253     CHECK_EQ(demon_runs->start_time_size(), demon_runs->end_time_size());
00254 
00255     const int runs = demon_runs->start_time_size();
00256     *demon_invocations = runs;
00257     *fails = demon_runs->failures();
00258     *total_demon_runtime = 0;
00259     *mean_demon_runtime = 0.0;
00260     *median_demon_runtime = 0.0;
00261     *stddev_demon_runtime = 0.0;
00262     std::vector<double> runtimes;
00263     for (int run_index = 0; run_index < runs; ++run_index) {
00264       const int64 demon_time = demon_runs->end_time(run_index) -
00265           demon_runs->start_time(run_index);
00266       *total_demon_runtime += demon_time;
00267       runtimes.push_back(demon_time);
00268     }
00269     // Compute mean.
00270     if (runtimes.size()) {
00271       *mean_demon_runtime = (1.0L * *total_demon_runtime) / runtimes.size();
00272 
00273       // Compute median.
00274       sort(runtimes.begin(), runtimes.end());
00275       const int pivot = runtimes.size() / 2;
00276 
00277       if (runtimes.size() == 1) {
00278         *median_demon_runtime = runtimes[0];
00279       } else {
00280         *median_demon_runtime = runtimes.size() % 2 == 1 ?
00281             runtimes[pivot] :
00282             (runtimes[pivot - 1] + runtimes[pivot]) / 2.0;
00283       }
00284 
00285       // Compute standard deviation.
00286       double total_deviation = 0.0f;
00287 
00288       for (int i = 0; i < runtimes.size(); ++i) {
00289         total_deviation += pow(runtimes[i] - *mean_demon_runtime, 2);
00290       }
00291 
00292       *stddev_demon_runtime = sqrt(total_deviation / runtimes.size());
00293     }
00294   }
