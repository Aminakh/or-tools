00361   void FirstRun(Solver* const solver, int64 splits) {
00362     current_log_space_ = domain_watcher_.LogSearchSpaceSize();
00363     if (display_level_ != DefaultPhaseParameters::NONE) {
00364       LOG(INFO) << "  - initial log2(SearchSpace) = " << current_log_space_;
00365     }
00366     const int64 init_time = solver->wall_time();
00367     int64 removed_counter = 0;
00368     FirstRunVariableContainers* container = solver->RevAlloc(
00369         new FirstRunVariableContainers(this, splits));
00370     // Loop on the variables, scan domains and initialize impacts.
00371     for (int var_index = 0; var_index < size_; ++var_index) {
00372       IntVar* const var = vars_[var_index];
00373       if (var->Bound()) {
00374         continue;
00375       }
00376       IntVarIterator* const iterator = domain_iterators_[var_index];
00377       DecisionBuilder* init_decision_builder = NULL;
00378       if (var->Max() - var->Min() < splits) {
00379         // The domain is small enough, we scan it completely.
00380         container->without_split()->set_update_impact_callback(
00381             container->update_impact_callback());
00382         container->without_split()->Init(var, iterator, var_index);
00383         init_decision_builder = container->without_split();
00384       } else {
00385         // The domain is too big, we scan it in initialization_splits
00386         // intervals.
00387         container->with_splits()->set_update_impact_callback(
00388             container->update_impact_callback());
00389         container->with_splits()->Init(var, iterator, var_index);
00390         init_decision_builder = container->with_splits();
00391       }
00392       // Reset the number of impacts initialized.
00393       init_count_ = 0;
00394       // Use NestedSolve() to scan all values of one variable.
00395       solver->NestedSolve(init_decision_builder, true);
00396 
00397       // If we have not initialized all values, then they can be removed.
00398       // As the iterator is not stable w.r.t. deletion, we need to store
00399       // removed values in an intermediate vector.
00400       if (init_count_ != var->Size()) {
00401         container->ClearRemovedValues();
00402         for (iterator->Init(); iterator->Ok(); iterator->Next()) {
00403           const int64 value = iterator->Value();
00404           const int64 value_index = value - original_min_[var_index];
00405           if (impacts_[var_index][value_index] == kInitFailureImpact) {
00406             container->PushBackRemovedValue(value);
00407           }
00408         }
00409         CHECK(container->HasRemovedValues()) << var->DebugString();
00410         removed_counter += container->NumRemovedValues();
00411         const double old_log = domain_watcher_.Log2(var->Size());
00412         var->RemoveValues(container->removed_values());
00413         current_log_space_ += domain_watcher_.Log2(var->Size()) - old_log;
00414       }
00415     }
00416     if (display_level_ != DefaultPhaseParameters::NONE) {
00417       if (removed_counter) {
00418         LOG(INFO) << "  - init done, time = " << solver->wall_time() - init_time
00419                   << " ms, " << removed_counter
00420                   << " values removed, log2(SearchSpace) = "
00421                   << current_log_space_;
00422       } else {
00423         LOG(INFO) << "  - init done, time = " << solver->wall_time() - init_time
00424                   << " ms";
00425       }
00426     }
00427   }

00443   // of the impacts of all values in its domain, along with the value

00447                       double* const var_impacts,
00448                       DefaultPhaseParameters::VariableSelection var_select,
00449                       DefaultPhaseParameters::ValueSelection value_select) {

00455     int64 min_impact_value = -1;

00458     for (it->Init(); it->Ok(); it->Next()) {

00462       DCHECK_LT(value_index, impacts_[var_index].size());
00463       const double current_impact = impacts_[var_index][value_index];
00464       sum_var_impact += current_impact;
00465       if (current_impact > max_impact) {
00466         max_impact = current_impact;
00467         max_impact_value = value;
00468       }
00469       if (current_impact < min_impact) {
00470         min_impact = current_impact;
00471         min_impact_value = value;
00472       }
00473     }

00476       case DefaultPhaseParameters::CHOOSE_MAX_AVERAGE_IMPACT: {
00477         *var_impacts = sum_var_impact / vars_[var_index]->Size();
00478         break;
00479       }
00480       case DefaultPhaseParameters::CHOOSE_MAX_VALUE_IMPACT: {
00481         *var_impacts = max_impact;
00482         break;
00483       }
00484       default: {
00485         *var_impacts = sum_var_impact;
00486         break;
00487       }
00488     }
00489 
00490     switch (value_select) {
00491       case DefaultPhaseParameters::SELECT_MIN_IMPACT: {
00492         *best_impact_value = min_impact_value;

00616       if (fail_stamp_ != kUninitializedFailStamp) {
00617         if (solver->fail_stamp() == fail_stamp_) {
00618           impact_recorder_.UpdateAfterAssignment(current_var_index_,
00619                                                  current_value_);
00620         } else {
00621           impact_recorder_.UpdateAfterFailure(current_var_index_,
00622                                               current_value_);
00623         }
00624       }
00625     }
00626     fail_stamp_ = solver->fail_stamp();
00627 
00628     ++heuristic_branch_count_;
00629     if (heuristic_branch_count_ % parameters_.heuristic_period == 0) {
00630       current_var_index_ = kUninitializedVarIndex;
00631       return &runner_;
00632     }
00633     current_var_index_ = kUninitializedVarIndex;
00634     current_value_ = 0;
00635     if (FindVarValue(&current_var_index_, &current_value_)) {
00636       return solver->MakeAssignVariableValue(vars_[current_var_index_],
00637                                              current_value_);
00638     } else {
00639       return NULL;
00640     }
00641   }

00801   // Called  before applying the refutation of the decision.
00802   bool CheckRestart(Solver* const solver) {
00803     if (parameters_.restart_log_size >= 0) {
00804       const double log_search_space_size =
00805           impact_recorder_.LogSearchSpaceSize();
00806       const int search_depth = solver->SearchDepth();
00807       if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00808         LOG(INFO) << "search_depth = " << search_depth
00809                   << ", log_search_space_size = " << log_search_space_size
00810                   << ", min_log_search_space = " << min_log_search_space_;
00811       }
00812       if (min_log_search_space_ > log_search_space_size) {
00813         min_log_search_space_ = log_search_space_size;
00814       } else if (min_log_search_space_ + parameters_.restart_log_size
00815                  < log_search_space_size) {
00816         if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00817           LOG(INFO) << "Restarting ";
00818         }
00819         min_log_search_space_ = std::numeric_limits<double>::infinity();
00820         return true;
00821       }
00822     }
00823     return false;
00824   }

00826   // This method will do an exhaustive scan of all domains of all
00827   // variables to select the variable with the maximal sum of impacts
00828   // per value in its domain, and then select the value with the
00829   // minimal impact.
00830   bool FindVarValue(int* const var_index, int64* const value) {
00831     CHECK_NOTNULL(var_index);
00832     CHECK_NOTNULL(value);
00833     *var_index = -1;
00834     *value = 0;
00835     double best_var_impact = -std::numeric_limits<double>::max();
00836     for (int i = 0; i < size_; ++i) {
00837       if (!vars_[i]->Bound()) {
00838         int64 current_value = 0;
00839         double current_var_impact = 0.0;
00840         impact_recorder_.ScanVarImpacts(i,
00841                                         &current_value,
00842                                         &current_var_impact,
00843                                         parameters_.var_selection_schema,
00844                                         parameters_.value_selection_schema);
00845         if (current_var_impact > best_var_impact) {
00846           *var_index = i;
00847           *value = current_value;
00848           best_var_impact = current_var_impact;
00849         }
00850       }
00851     }
00852     return (*var_index != -1);
00853   }

00855   bool RunOneHeuristic(Solver* const solver, int index) {
00856     HeuristicWrapper* const wrapper = heuristics_[index];
00857 
00858     const bool result =
00859         solver->NestedSolve(wrapper->phase, false, heuristic_limit_);
00860     if (result && parameters_.display_level != DefaultPhaseParameters::NONE) {
00861       LOG(INFO) << "Solution found by heuristic " << wrapper->name;
00862     }
00863     return result;
00864   }

00866   bool RunHeuristics(Solver* const solver) {
00867     if (parameters_.run_all_heuristics) {
00868       for (int index = 0; index < heuristics_.size(); ++index) {
00869         for (int run = 0; run < heuristics_[index]->runs; ++run) {
00870           if (RunOneHeuristic(solver, index)) {
00871             return true;
00872           }
00873         }
00874       }
00875       return false;
00876     } else {
00877       const int index = random_.Uniform(heuristics_.size());
00878       return RunOneHeuristic(solver, index);
00879     }
00880   }
