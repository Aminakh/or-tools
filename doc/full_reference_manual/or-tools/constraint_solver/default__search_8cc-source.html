00357   void FirstRun(Solver* const solver, int64 splits) {
00358     current_log_space_ = domain_watcher_.LogSearchSpaceSize();
00359     if (display_level_ != DefaultPhaseParameters::NONE) {
00360       LOG(INFO) << "  - initial log2(SearchSpace) = " << current_log_space_;
00361     }
00362     const int64 init_time = solver->wall_time();
00363     int64 removed_counter = 0;
00364     FirstRunVariableContainers* container = solver->RevAlloc(
00365         new FirstRunVariableContainers(this, splits));
00366     // Loop on the variables, scan domains and initialize impacts.
00367     for (int var_index = 0; var_index < size_; ++var_index) {
00368       IntVar* const var = vars_[var_index];
00369       if (var->Bound()) {
00370         continue;
00371       }
00372       IntVarIterator* const iterator = domain_iterators_[var_index];
00373       DecisionBuilder* init_decision_builder = NULL;
00374       if (var->Max() - var->Min() < splits) {
00375         // The domain is small enough, we scan it completely.
00376         container->without_split()->set_update_impact_callback(
00377             container->update_impact_callback());
00378         container->without_split()->Init(var, iterator, var_index);
00379         init_decision_builder = container->without_split();
00380       } else {
00381         // The domain is too big, we scan it in initialization_splits
00382         // intervals.
00383         container->with_splits()->set_update_impact_callback(
00384             container->update_impact_callback());
00385         container->with_splits()->Init(var, iterator, var_index);
00386         init_decision_builder = container->with_splits();
00387       }
00388       // Reset the number of impacts initialized.
00389       init_count_ = 0;
00390       // Use NestedSolve() to scan all values of one variable.
00391       solver->NestedSolve(init_decision_builder, true);
00392 
00393       // If we have not initialized all values, then they can be removed.
00394       // As the iterator is not stable w.r.t. deletion, we need to store
00395       // removed values in an intermediate vector.
00396       if (init_count_ != var->Size()) {
00397         container->ClearRemovedValues();

00399           const int64 value = iterator->Value();
00400           const int64 value_index = value - original_min_[var_index];
00401           if (impacts_[var_index][value_index] == kInitFailureImpact) {
00402             container->PushBackRemovedValue(value);
00403           }
00404         }

00406         removed_counter += container->NumRemovedValues();

00408         var->RemoveValues(container->removed_values());

00410       }
00411     }

00413       if (removed_counter) {
00414         LOG(INFO) << "  - init done, time = " << solver->wall_time() - init_time

00416                   << " values removed, log2(SearchSpace) = "

00418       } else {
00419         LOG(INFO) << "  - init done, time = " << solver->wall_time() - init_time
00420                   << " ms";
00421       }
00422     }
00423   }

00624         new HeuristicWrapper(solver,
00625                              vars_.get(),
00626                              size_,
00627                              Solver::CHOOSE_RANDOM,
00628                              Solver::ASSIGN_MIN_VALUE,
00629                              "AssignMinValueToRandomVariable",
00630                              kRunMore));
00631 
00632     heuristics_.push_back(
00633         new HeuristicWrapper(solver,
00634                              vars_.get(),
00635                              size_,
00636                              Solver::CHOOSE_RANDOM,
00637                              Solver::ASSIGN_MAX_VALUE,
00638                              "AssignMaxValueToRandomVariable",
00639                              kRunMore));

00641     heuristics_.push_back(
00642         new HeuristicWrapper(solver,
00643                              vars_.get(),
00644                              size_,
00645                              Solver::CHOOSE_RANDOM,
00646                              Solver::ASSIGN_RANDOM_VALUE,
00647                              "AssignRandomValueToRandomVariable",
00648                              kRunMore));
00649 
00650     heuristic_limit_ =
00651         solver->MakeLimit(kint64max,  // time.
00652                           kint64max,  // branches.
00653                           parameters_.heuristic_num_failures_limit,  // fails.
00654                           kint64max);  // solutions.
00655   }

00658   // This method will do an exhaustive scan of all domains of all
00659   // variables to select the variable with the maximal sum of impacts
00660   // per value in its domain, and then select the value with the
00661   // minimal impact.
00662   bool FindVarValue(int* const var_index, int64* const value) {
00663     CHECK_NOTNULL(var_index);
00664     CHECK_NOTNULL(value);
00665     *var_index = -1;
00666     *value = 0;
00667     double best_var_impact = -std::numeric_limits<double>::max();
00668     for (int i = 0; i < size_; ++i) {
00669       if (!vars_[i]->Bound()) {
00670         int64 current_value = 0;
00671         double current_var_impact = 0.0;
00672         impact_recorder_.ScanVarImpacts(i,
00673                                         &current_value,
00674                                         &current_var_impact,
00675                                         parameters_.var_selection_schema,
00676                                         parameters_.value_selection_schema);
00677         if (current_var_impact > best_var_impact) {
00678           *var_index = i;
00679           *value = current_value;
00680           best_var_impact = current_var_impact;
00681         }
00682       }
00683     }
00684     return (*var_index != -1);
00685   }

00687   bool RunOneHeuristic(Solver* const solver, int index) {
00688     HeuristicWrapper* const wrapper = heuristics_[index];
00689 
00690     const bool result =
00691         solver->NestedSolve(wrapper->phase, false, heuristic_limit_);
00692     if (result && parameters_.display_level != DefaultPhaseParameters::NONE) {
00693       LOG(INFO) << "Solution found by heuristic " << wrapper->name;
00694     }
00695     return result;
00696   }

00698   bool RunHeuristics(Solver* const solver) {
00699     if (parameters_.run_all_heuristics) {
00700       for (int index = 0; index < heuristics_.size(); ++index) {
00701         for (int run = 0; run < heuristics_[index]->runs; ++run) {
00702           if (RunOneHeuristic(solver, index)) {
00703             return true;
00704           }
00705         }
00706       }
00707       return false;
00708     } else {
00709       const int index = random_.Uniform(heuristics_.size());
00710       return RunOneHeuristic(solver, index);
00711     }
00712   }

00714   virtual Decision* Next(Solver* const solver) {
00715     if (!init_done_) {
00716       if (parameters_.display_level != DefaultPhaseParameters::NONE) {
00717         LOG(INFO) << "Init impact based search phase on " << size_
00718                   << " variables, initialization splits = "
00719                   << parameters_.initialization_splits
00720                   << ", heuristic_period = " << parameters_.heuristic_period
00721                   << ", run_all_heuristics = " << parameters_.run_all_heuristics
00722                   << ", restart_log_size = " << parameters_.restart_log_size;
00723       }
00724       // We need to reset the impacts because FirstRun calls RemoveValues
00725       // which can result in a Fail() therefore calling this method again.
00726       impact_recorder_.ResetImpacts();
00727       impact_recorder_.FirstRun(solver, parameters_.initialization_splits);
00728       if (parameters_.persistent_impact) {
00729         init_done_ = true;
00730       } else {
00731         solver->SaveAndSetValue(&init_done_, true);
00732       }
00733     }
00734 
00735     if (current_var_index_ == -1 && fail_stamp_ != 0) {
00736       // After solution or after heuristics.
00737       impact_recorder_.RecordLogSearchSpace();
00738     } else {
00739       if (fail_stamp_ != 0) {
00740         if (solver->fail_stamp() == fail_stamp_) {
00741           impact_recorder_.UpdateAfterAssignment(current_var_index_,
00742                                                  current_value_);
00743         } else {
00744           impact_recorder_.UpdateAfterFailure(current_var_index_,
00745                                               current_value_);
00746         }
00747       }
00748     }
00749     fail_stamp_ = solver->fail_stamp();
00750 
00751     if (parameters_.restart_log_size >= 0) {
00752       const double log_search_space_size =
00753           impact_recorder_.LogSearchSpaceSize();
00754       const int search_depth = solver->SearchDepth();
00755       if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00756         LOG(INFO) << "search_depth = " << search_depth
00757                   << ", log_search_space_size = " << log_search_space_size
00758                   << ", min_log_search_space = " << min_log_search_space_;
00759       }
00760       if (min_log_search_space_ > log_search_space_size) {
00761         min_log_search_space_ = log_search_space_size;
00762       } else if (min_log_search_space_ + parameters_.restart_log_size
00763                  < log_search_space_size) {
00764         if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00765           LOG(INFO) << "Restarting ";
00766         }
00767         min_log_search_space_ = std::numeric_limits<double>::infinity();
00768         solver->RestartSearch();
00769       }
00770     }
00771 
00772     ++heuristic_branch_count_;
00773     if (heuristic_branch_count_ % parameters_.heuristic_period == 0) {
00774       current_var_index_ = -1;
00775       return &runner_;
00776     }
00777     current_var_index_ = -1;
00778     current_value_ = 0;
00779     if (FindVarValue(&current_var_index_, &current_value_)) {
00780       return solver->MakeAssignVariableValue(vars_[current_var_index_],
00781                                              current_value_);
00782     } else {
00783       return NULL;
00784     }
00785   }
