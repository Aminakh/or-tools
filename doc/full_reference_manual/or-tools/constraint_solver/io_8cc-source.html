01621   // Add dimensions. They are stored as extensions in the proto.

01623     const CPExtensionProto& dimension_proto = proto.extensions(i);

01630       VERIFY(AddCountUsedBinDimension(pack, builder, dimension_proto));
01631     } else if (IS_TYPE(type_index, builder, kUsageEqualVariableExtension)) {
01632       VERIFY(AddUsageEqualVariableDimension(pack, builder, dimension_proto));
01633     } else if (IS_TYPE(type_index,
01634                        builder,

01640                        builder,
01641                        kWeightedSumOfAssignedEqualVariableExtension)) {

02092 bool Solver::LoadModel(const CPModelProto& model_proto,
02093                        std::vector<SearchMonitor*>* monitors) {
02094   if (model_proto.version() > kModelVersion) {
02095     LOG(ERROR) << "Model protocol buffer version is greater than"
02096                << " the one compiled in the reader ("
02097                << model_proto.version() << " vs " << kModelVersion << ")";
02098     return false;
02099   }
02100   CPModelBuilder builder(this);
02101   for (int i = 0; i < model_proto.tags_size(); ++i) {
02102     builder.AddTag(model_proto.tags(i));
02103   }
02104   for (int i = 0; i < model_proto.intervals_size(); ++i) {
02105     if (!builder.BuildFromProto(model_proto.intervals(i))) {
02106       LOG(ERROR) << "Interval variable proto "
02107                  << model_proto.intervals(i).DebugString()
02108                  << " was not parsed correctly";
02109       return false;
02110     }
02111   }
02112   for (int i = 0; i < model_proto.expressions_size(); ++i) {
02113     if (!builder.BuildFromProto(model_proto.expressions(i))) {
02114       LOG(ERROR) << "Integer expression proto "
02115                  << model_proto.expressions(i).DebugString()
02116                  << " was not parsed correctly";
02117       return false;
02118     }
02119   }
02120   for (int i = 0; i < model_proto.constraints_size(); ++i) {
02121     Constraint* const constraint =
02122         builder.BuildFromProto(model_proto.constraints(i));
02123     if (constraint == NULL) {
02124       LOG(ERROR) << "Constraint proto "
02125                  << model_proto.constraints(i).DebugString()
02126                  << " was not parsed correctly";
02127       return false;
02128     }
02129     AddConstraint(constraint);
02130   }
02131   if (monitors != NULL) {
02132     if (model_proto.has_search_limit()) {
02133       monitors->push_back(MakeLimit(model_proto.search_limit()));
02134     }
02135     if (model_proto.has_objective()) {
02136       const CPObjectiveProto& objective_proto = model_proto.objective();
02137       IntVar* const objective_var =
02138           builder.IntegerExpression(objective_proto.objective_index())->Var();
02139       const bool maximize = objective_proto.maximize();
02140       const int64 step = objective_proto.step();
02141       OptimizeVar* const objective =
02142           MakeOptimize(maximize, objective_var, step);
02143       monitors->push_back(objective);
02144     }
02145   }
02146   return true;
02147 }
