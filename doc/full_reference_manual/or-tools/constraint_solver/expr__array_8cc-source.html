00295   void PushDown(int depth, int position, int64 new_min, int64 new_max) {
00296     // Nothing to do?
00297     if (new_min <= Min(depth, position) && new_max >= Max(depth, position)) {
00298       return;
00299     }
00300 
00301     // Leaf node -> push to leaf var.
00302     if (IsLeaf(depth)) {
00303       vars_[position]->SetRange(new_min, new_max);
00304       return;
00305     }
00306 
00307     // Standard propagation from the bounds of the sum to the
00308     // individuals terms.
00309 
00310     // These are maintained automatically in the tree structure.
00311     const int64 sum_min = Min(depth, position);
00312     const int64 sum_max = Max(depth, position);
00313 
00314     // Intersect the new bounds with the computed bounds.
00315     new_max = std::min(sum_max, new_max);
00316     new_min = std::max(sum_min, new_min);
00317 
00318     // Detect failure early.
00319     if (new_max < sum_min || new_min > sum_max) {
00320       solver()->Fail();
00321     }
00322 
00323     // Push to children nodes.
00324     const int block_start = ChildStart(position);
00325     const int block_end = ChildEnd(depth, position);
00326     for (int i = block_start; i <= block_end; ++i) {
00327       const int64 var_min = Min(depth + 1, i);
00328       const int64 var_max = Max(depth + 1, i);
00329       const int64 residual_min = sum_min - var_min;
00330       const int64 residual_max = sum_max - var_max;
00331       PushDown(depth + 1, i, new_min - residual_max, new_max - residual_min);
00332     }
00333     // TODO(user) : Is the diameter optimization (see reference
00334     // above, rule 5) useful?
00335   }

01333 IntExpr* BuildLogSplitArray(Solver* const s,
01334                             IntVar* const* vars,
01335                             int size,
01336                             BuildOp op) {
01337   const int split_size = s->parameters().array_split_size;
01338   if (size == 0) {
01339     return s->MakeIntConst(0LL);
01340   } else if (size == 1) {
01341     return vars[0];
01342   } else if (size == 2) {
01343     switch (op) {
01344       case MIN_OP:
01345         return s->MakeMin(vars[0], vars[1]);
01346       case MAX_OP:
01347         return s->MakeMax(vars[0], vars[1]);
01348     };
01349   } else if (size > split_size) {
01350     const int nb_blocks = (size - 1) / split_size + 1;
01351     const int block_size = (size + nb_blocks - 1) / nb_blocks;
01352     std::vector<IntVar*> top_vector;
01353     int start = 0;
01354     while (start < size) {
01355       int real_size = (start + block_size > size ? size - start : block_size);
01356       IntVar* intermediate = NULL;
01357       switch (op) {
01358         case MIN_OP:
01359           intermediate = s->MakeMin(vars + start, real_size)->Var();
01360           break;
01361         case MAX_OP:
01362           intermediate = s->MakeMax(vars + start, real_size)->Var();
01363           break;
01364       }
01365       top_vector.push_back(intermediate);
01366       start += real_size;
01367     }
01368     switch (op) {
01369       case MIN_OP:
01370         return s->MakeMin(top_vector);
01371       case MAX_OP:

01907 struct Container {
01908   IntVar* var;
01909   int64 coef;
01910   Container(IntVar* v, int64 c) : var(v), coef(c) {}
01911   bool operator<(const Container& c) const { return (coef < c.coef); }
01912 };
01913 
01914 // This method will sort both vars and coefficients in increasing
01915 // coefficient order. Vars with null coefficients will be
01916 // removed. Bound vars will be collected and the sum of the
01917 // corresponding products (when the var is bound to 1) is returned by
01918 // this method.
01919 // If keep_inside is true, the constant will be added back into the
01920 // scalprod as IntConst(1) * constant.
01921 int64 SortBothChangeConstant(IntVar** const vars,
01922                              int64* const coefs,
01923                              int* const size,
01924                              bool keep_inside) {
01925   CHECK_NOTNULL(vars);
01926   CHECK_NOTNULL(coefs);
01927   CHECK_NOTNULL(size);
01928   if (*size == 0) {
01929     return 0;
01930   }
01931   int64 cst = 0;
01932   std::vector<Container> to_sort;
01933   for (int index = 0; index < *size; ++index) {
01934     if (vars[index]->Bound()) {
01935       cst += coefs[index] * vars[index]->Min();
01936     } else if (coefs[index] != 0) {
01937       to_sort.push_back(Container(vars[index], coefs[index]));
01938     }
01939   }
01940   if (keep_inside && cst != 0) {
01941     CHECK_LT(to_sort.size(), *size);
01942     Solver* const solver = vars[0]->solver();
01943     to_sort.push_back(Container(solver->MakeIntConst(1), cst));
01944     cst = 0;
01945   }
01946   std::sort(to_sort.begin(), to_sort.end());
01947   *size = to_sort.size();
01948   for (int index = 0; index < *size; ++index) {
01949     vars[index] = to_sort[index].var;
01950     coefs[index] = to_sort[index].coef;
01951   }
01952   return cst;
01953 }

02842 namespace {
02843 template<class T> IntExpr* MakeScalProdFct(Solver* solver,
02844                                            IntVar* const * vars,
02845                                            const T* const coefs,
02846                                            int size) {
02847   if (size == 0 || AreAllNull<T>(coefs, size)) {
02848     return solver->MakeIntConst(0LL);
02849   }
02850   if (AreAllBoundOrNull(vars, coefs, size)) {
02851     int64 cst = 0;
02852     for (int i = 0; i < size; ++i) {
02853       cst += vars[i]->Min() * coefs[i];
02854     }
02855     return solver->MakeIntConst(cst);
02856   }
02857   if (AreAllBooleans(vars, size)) {
02858     if (AreAllPositive<T>(coefs, size)) {
02859       return solver->RevAlloc(
02860           new PositiveBooleanScalProd(solver, vars, size, coefs));
02861     } else {
02862       // If some coefficients are non-positive, partition coefficients in two
02863       // sets, one for the positive coefficients P and one for the negative
02864       // ones N.
02865       // Create two PositiveBooleanScalProd expressions, one on P (s1), the
02866       // other on Opposite(N) (s2).
02867       // The final expression is then s1 - s2.
02868       // If P is empty, the expression is Opposite(s2).
02869       std::vector<T> positive_coefs;
02870       std::vector<T> negative_coefs;
02871       std::vector<IntVar*> positive_coef_vars;
02872       std::vector<IntVar*> negative_coef_vars;
02873       for (int i = 0; i < size; ++i) {
02874         const T coef = coefs[i];
02875         if (coef > 0) {
02876           positive_coefs.push_back(coef);
02877           positive_coef_vars.push_back(vars[i]);
02878         } else if (coef < 0) {
02879           negative_coefs.push_back(-coef);
02880           negative_coef_vars.push_back(vars[i]);
02881         }
02882       }
02883       CHECK_GT(negative_coef_vars.size(), 0);
02884       IntExpr* negatives =
02885           solver->RevAlloc(
02886               new PositiveBooleanScalProd(solver,
02887                                           negative_coef_vars.data(),
02888                                           negative_coef_vars.size(),
02889                                           negative_coefs.data()));
02890       if (!positive_coefs.empty()) {
02891         IntExpr* positives =
02892             solver->RevAlloc(
02893                 new PositiveBooleanScalProd(solver,
02894                                             positive_coef_vars.data(),
02895                                             positive_coef_vars.size(),
02896                                             positive_coefs.data()));
02897         // Cast to var to avoid slow propagation; all operations on the expr are
02898         // O(n)!
02899         return solver->MakeDifference(positives->Var(), negatives->Var());
02900       } else {
02901         return solver->MakeOpposite(negatives);
02902       }
02903     }
02904   }
02905   std::vector<IntVar*> terms;
02906   for (int i = 0; i < size; ++i) {
02907     terms.push_back(solver->MakeProd(vars[i], coefs[i])->Var());
02908   }
02909   return solver->MakeSum(terms);
02910 }
