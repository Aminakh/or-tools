00289   void PushDown(int depth, int position, int64 new_min, int64 new_max) {
00290     // Nothing to do?
00291     if (new_min <= Min(depth, position) && new_max >= Max(depth, position)) {
00292       return;
00293     }
00294 
00295     // Leaf node -> push to leaf var.
00296     if (IsLeaf(depth)) {
00297       vars_[position]->SetRange(new_min, new_max);
00298       return;
00299     }
00300 
00301     // Standard propagation from the bounds of the sum to the
00302     // individuals terms.
00303 
00304     // These are maintained automatically in the tree structure.
00305     const int64 sum_min = Min(depth, position);
00306     const int64 sum_max = Max(depth, position);
00307 
00308     // Intersect the new bounds with the computed bounds.
00309     new_max = std::min(sum_max, new_max);
00310     new_min = std::max(sum_min, new_min);
00311 
00312     // Detect failure early.
00313     if (new_max < sum_min || new_min > sum_max) {
00314       solver()->Fail();
00315     }
00316 
00317     // Push to children nodes.
00318     const int block_start = ChildStart(position);
00319     const int block_end = ChildEnd(depth, position);
00320     for (int i = block_start; i <= block_end; ++i) {
00321       const int64 var_min = Min(depth + 1, i);
00322       const int64 var_max = Max(depth + 1, i);
00323       const int64 residual_min = sum_min - var_min;
00324       const int64 residual_max = sum_max - var_max;
00325       PushDown(depth + 1, i, new_min - residual_max, new_max - residual_min);
00326     }
00327     // TODO(user) : Is the diameter optimization (see reference
00328     // above, rule 5) useful?
00329   }

01327 IntExpr* BuildLogSplitArray(Solver* const s,
01328                             IntVar* const* vars,
01329                             int size,

01331   const int split_size = s->parameters().array_split_size;
01332   if (size == 0) {
01333     return s->MakeIntConst(0LL);
01334   } else if (size == 1) {
01335     return vars[0];
01336   } else if (size == 2) {
01337     switch (op) {
01338       case MIN_OP:
01339         return s->MakeMin(vars[0], vars[1]);
01340       case MAX_OP:
01341         return s->MakeMax(vars[0], vars[1]);
01342     };
01343   } else if (size > split_size) {
01344     const int nb_blocks = (size - 1) / split_size + 1;
01345     const int block_size = (size + nb_blocks - 1) / nb_blocks;
01346     std::vector<IntVar*> top_vector;
01347     int start = 0;
01348     while (start < size) {
01349       int real_size = (start + block_size > size ? size - start : block_size);
01350       IntVar* intermediate = NULL;
01351       switch (op) {
01352         case MIN_OP:
01353           intermediate = s->MakeMin(vars + start, real_size)->Var();
01354           break;
01355         case MAX_OP:
01356           intermediate = s->MakeMax(vars + start, real_size)->Var();
01357           break;
01358       }
01359       top_vector.push_back(intermediate);
01360       start += real_size;
01361     }
01362     switch (op) {
01363       case MIN_OP:
01364         return s->MakeMin(top_vector);
01365       case MAX_OP:
01366         return s->MakeMax(top_vector);
01367     };
01368   } else {
01369     for (int i = 0; i < size; ++i) {
01370       CHECK_EQ(s, vars[i]->solver());
01371     }

01901 struct Container {
01902   IntVar* var;
01903   int64 coef;
01904   Container(IntVar* v, int64 c) : var(v), coef(c) {}
01905   bool operator<(const Container& c) const { return (coef < c.coef); }
01906 };
01907 
01908 // This method will sort both vars and coefficients in increasing
01909 // coefficient order. Vars with null coefficients will be
01910 // removed. Bound vars will be collected and the sum of the
01911 // corresponding products (when the var is bound to 1) is returned by
01912 // this method.
01913 // If keep_inside is true, the constant will be added back into the
01914 // scalprod as IntConst(1) * constant.
01915 int64 SortBothChangeConstant(IntVar** const vars,
01916                              int64* const coefs,
01917                              int* const size,
01918                              bool keep_inside) {
01919   CHECK_NOTNULL(vars);
01920   CHECK_NOTNULL(coefs);
01921   CHECK_NOTNULL(size);
01922   if (*size == 0) {
01923     return 0;
01924   }
01925   int64 cst = 0;
01926   std::vector<Container> to_sort;
01927   for (int index = 0; index < *size; ++index) {
01928     if (vars[index]->Bound()) {
01929       cst += coefs[index] * vars[index]->Min();
01930     } else if (coefs[index] != 0) {
01931       to_sort.push_back(Container(vars[index], coefs[index]));
01932     }
01933   }
01934   if (keep_inside && cst != 0) {
01935     CHECK_LT(to_sort.size(), *size);
01936     Solver* const solver = vars[0]->solver();
01937     to_sort.push_back(Container(solver->MakeIntConst(1), cst));
01938     cst = 0;
01939   }
01940   std::sort(to_sort.begin(), to_sort.end());
01941   *size = to_sort.size();
01942   for (int index = 0; index < *size; ++index) {
01943     vars[index] = to_sort[index].var;
01944     coefs[index] = to_sort[index].coef;
01945   }
01946   return cst;
01947 }

02836 namespace {
02837 template<class T> IntExpr* MakeScalProdFct(Solver* solver,
02838                                            IntVar* const * vars,
02839                                            const T* const coefs,
02840                                            int size) {
02841   if (size == 0 || AreAllNull<T>(coefs, size)) {
02842     return solver->MakeIntConst(0LL);
02843   }
02844   if (AreAllBoundOrNull(vars, coefs, size)) {
02845     int64 cst = 0;
02846     for (int i = 0; i < size; ++i) {
02847       cst += vars[i]->Min() * coefs[i];
02848     }
02849     return solver->MakeIntConst(cst);
02850   }
02851   if (AreAllBooleans(vars, size)) {
02852     if (AreAllPositive<T>(coefs, size)) {
02853       return solver->RevAlloc(
02854           new PositiveBooleanScalProd(solver, vars, size, coefs));
02855     } else {
02856       // If some coefficients are non-positive, partition coefficients in two
02857       // sets, one for the positive coefficients P and one for the negative
02858       // ones N.
02859       // Create two PositiveBooleanScalProd expressions, one on P (s1), the
02860       // other on Opposite(N) (s2).
02861       // The final expression is then s1 - s2.
02862       // If P is empty, the expression is Opposite(s2).
02863       std::vector<T> positive_coefs;
02864       std::vector<T> negative_coefs;
02865       std::vector<IntVar*> positive_coef_vars;
02866       std::vector<IntVar*> negative_coef_vars;
02867       for (int i = 0; i < size; ++i) {
02868         const T coef = coefs[i];
02869         if (coef > 0) {
02870           positive_coefs.push_back(coef);
02871           positive_coef_vars.push_back(vars[i]);
02872         } else if (coef < 0) {
02873           negative_coefs.push_back(-coef);
02874           negative_coef_vars.push_back(vars[i]);
02875         }
02876       }
02877       CHECK_GT(negative_coef_vars.size(), 0);
02878       IntExpr* negatives =
02879           solver->RevAlloc(
02880               new PositiveBooleanScalProd(solver,
02881                                           negative_coef_vars.data(),
02882                                           negative_coef_vars.size(),
02883                                           negative_coefs.data()));
02884       if (!positive_coefs.empty()) {
02885         IntExpr* positives =
02886             solver->RevAlloc(
02887                 new PositiveBooleanScalProd(solver,
02888                                             positive_coef_vars.data(),
02889                                             positive_coef_vars.size(),
02890                                             positive_coefs.data()));
02891         // Cast to var to avoid slow propagation; all operations on the expr are
02892         // O(n)!
02893         return solver->MakeDifference(positives->Var(), negatives->Var());
02894       } else {
02895         return solver->MakeOpposite(negatives);
02896       }
02897     }
02898   }
02899   std::vector<IntVar*> terms;
02900   for (int i = 0; i < size; ++i) {
02901     terms.push_back(solver->MakeProd(vars[i], coefs[i])->Var());
02902   }
02903   return solver->MakeSum(terms);
02904 }
