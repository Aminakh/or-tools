00058 // This class represent a reversible FIFO structure.
00059 // The main diffence w.r.t a standart FIFO structure is that a Solver is
00060 // given as parameter to the modifiers such that the solver can store the
00061 // backtrack information
00062 // Iterator's traversing order should not be changed, as some algorithm
00063 // depend on it to be .
00064 #ifndef SWIG
00065 template <class T> class SimpleRevFIFO {
00066  private:
00067   enum { CHUNK_SIZE = 16 };  // TODO(user): could be an extra template param
00068   struct Chunk {
00069     T data_[CHUNK_SIZE];
00070     const Chunk* const next_;
00071     explicit Chunk(const Chunk* next) : next_(next) {}
00072   };
00073 
00074  public:
00075   // This iterator is non stable w.r.t. deletion.
00076   class Iterator {
00077    public:
00078     explicit Iterator(const SimpleRevFIFO<T>* l)
00079         : chunk_(l->chunks_), data_(l->Last()) {}
00080     bool ok() const { return (data_ != NULL); }
00081     T operator*() const { return *data_; }
00082     void operator++() {
00083       ++data_;
00084       if (data_ == chunk_->data_ + CHUNK_SIZE) {
00085         chunk_ = chunk_->next_;
00086         data_ = chunk_ ? chunk_->data_ : NULL;
00087       }
00088     }
00089 
00090    private:
00091     const Chunk* chunk_;
00092     const T* data_;
00093   };
00094 
00095   SimpleRevFIFO() : chunks_(NULL), pos_(0) {}
00096 
00097   void Push(Solver* const s, T val) {
00098     if (pos_ == 0) {
00099       Chunk* const chunk = s->UnsafeRevAlloc(new Chunk(chunks_));
00100       s->SaveAndSetValue(reinterpret_cast<void**>(&chunks_),
00101                          reinterpret_cast<void*>(chunk));
00102       s->SaveAndSetValue(&pos_, CHUNK_SIZE - 1);
00103     } else {
00104       s->SaveAndAdd(&pos_, -1);
00105     }
00106     chunks_->data_[pos_] = val;
00107   }
00108 
00109   void PushIfNotTop(Solver* const s, T val) {
00110     if (chunks_ == NULL || LastValue() != val) {
00111       Push(s, val);
00112     }
00113   }
00114   const T* Last() const { return chunks_ ? &chunks_->data_[pos_] : NULL; }
00115   const T& LastValue() const {
00116     DCHECK(chunks_);
00117     return chunks_->data_[pos_];
00118   }
00119   void SetLastValue(const T& v) {
00120     DCHECK(Last());
00121     chunks_->data_[pos_] = v;
00122   }
00123 
00124  private:
00125   Chunk *chunks_;
00126   int pos_;
00127 };

00387 // ---------- Local search operators ----------
00388 
00389 // A local search operator is an object which defines the neighborhood of a
00390 // solution; in other words, a neighborhood is the set of solutions which can
00391 // be reached from a given solution using an operator.
00392 // The behavior of the LocalSearchOperator class is similar to the one of an
00393 // iterator. The operator is synchronized with an assignment (gives the
00394 // current values of the variables); this is done in the Start() method.
00395 // Then one can iterate over the neighbors using the MakeNextNeighbor method.
00396 // This method returns an assignment which represents the incremental changes
00397 // to the curent solution. It also returns a second assignment representing the
00398 // changes to the last solution defined by the neighborhood operator; this
00399 // assignment is empty is the neighborhood operator cannot track this
00400 // information.
00401 // TODO(user): rename Start to Synchronize ?
00402 // TODO(user): decouple the iterating from the defining of a neighbor
00403 
00404 // Mother of all local search operators

00406  public:
00407   LocalSearchOperator() {}
00408   virtual ~LocalSearchOperator() {}
00409   virtual bool MakeNextNeighbor(Assignment* delta, Assignment* deltadelta) = 0;
00410   virtual void Start(const Assignment* assignment) = 0;
00411 };

00663  public:
00664   // Accepts a "delta" given the assignment with which the filter has been
00665   // synchronized; the delta holds the variables which have been modified and
00666   // their new value.
00667   // Sample: supposing one wants to maintain a[0,1] + b[0,1] <= 1,
00668   // for the assignment (a,1), (b,0), the delta (b,1) will be rejected
00669   // but the delta (a,0) will be accepted.
00670   virtual bool Accept(const Assignment* delta,
00671                       const Assignment* deltadelta) = 0;
00672 
00673   // Synchronizes the filter with the current solution
00674   virtual void Synchronize(const Assignment* assignment) = 0;
00675   virtual bool IsIncremental() const { return false; }
00676 };
