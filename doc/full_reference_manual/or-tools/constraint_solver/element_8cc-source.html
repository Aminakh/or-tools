00453 IntExpr* BuildElement(Solver* const solver,
00454                       ConstIntArray* values,
00455                       IntVar* const index) {
00456   // Various checks.
00457   // Is array constant?
00458   if (values->HasProperty(ConstIntArray::IS_CONSTANT)) {
00459     solver->AddConstraint(solver->MakeBetweenCt(index, 0, values->size() - 1));
00460     return solver->MakeIntConst(values->get(0));
00461   }
00462   // Is array built with booleans only?
00463   // TODO(user): We could maintain the index of the first one.
00464   if (values->HasProperty(ConstIntArray::IS_BOOLEAN)) {
00465     std::vector<int64> ones;
00466     int first_zero = -1;
00467     for (int i = 0; i < values->size(); ++i) {
00468       if (values->get(i) == 1LL) {
00469         ones.push_back(i);
00470       } else {
00471         first_zero = i;
00472       }
00473     }
00474     if (ones.size() == 1) {
00475       DCHECK_EQ(1LL, values->get(ones.back()));
00476       solver->AddConstraint(
00477           solver->MakeBetweenCt(index, 0, values->size() - 1));
00478       return solver->MakeIsEqualCstVar(index, ones.back());
00479     } else if (ones.size() == values->size() - 1) {
00480       solver->AddConstraint(
00481           solver->MakeBetweenCt(index, 0, values->size() - 1));
00482       return solver->MakeIsDifferentCstVar(index, first_zero);
00483     } else if (ones.size() == ones.back() - ones.front() + 1) {  // contiguous.
00484       solver->AddConstraint(
00485           solver->MakeBetweenCt(index, 0, values->size() - 1));
00486       IntVar* const b = solver->MakeBoolVar("ContiguousBooleanElementVar");
00487       solver->AddConstraint(
00488           solver->MakeIsBetweenCt(index, ones.front(), ones.back(), b));
00489       return b;
00490     } else {
00491       IntVar* const b = solver->MakeBoolVar("NonContiguousBooleanElementVar");
00492       solver->AddConstraint(
00493           solver->MakeBetweenCt(index, 0, values->size() - 1));
00494       solver->AddConstraint(solver->MakeIsMemberCt(index, ones, b));
00495       return b;
00496     }
00497   }
00498   // Is Array increasing
00499   if (values->HasProperty(ConstIntArray::IS_INCREASING)) {
00500     return solver->RevAlloc(
00501         new IncreasingIntExprElement(solver, values->Release(), index));
00502   }
00503   return solver->RevAlloc(new IntExprElement(solver, values->Release(), index));

01285     while (nmin <= emax && vars_[nmin]->Min() > m) {
01286       nmin++;
01287     }
01288     if (nmin > emax) {
01289       solver()->Fail();
01290     }
01291     int64 nmax = emax;
01292     while (nmax >= nmin && vars_[nmax]->Min() > m) {
01293       nmax--;
01294     }
01295     var_->SetRange(nmin, nmax);
01296     if (var_->Bound()) {
01297       vars_[var_->Min()]->SetMax(m);
01298     }
01299   }
01300 }

01302 void IntExprArrayElement::SetRange(int64 mi, int64 ma) {
01303   if (mi > ma) {
01304     solver()->Fail();
01305   }
01306   const int64 emin = std::max(0LL, var_->Min());
01307   const int64 emax = std::min(size_ - 1LL, var_->Max());
01308   if (emin == emax) {
01309     var_->SetValue(emin);  // in case it was reduced by the above min/max.
01310     vars_[emin]->SetRange(mi, ma);
01311   } else {
01312     int64 nmin = emin;
01313     while (nmin <= emax && (vars_[nmin]->Min() > ma ||
01314                             vars_[nmin]->Max() < mi)) {
01315       nmin++;
01316     }
01317     if (nmin > emax) {
01318       solver()->Fail();
01319     }
01320     int64 nmax = emax;
01321     while (nmax >= nmin && (vars_[nmax]->Max() < mi ||
01322                             vars_[nmax]->Min() > ma)) {
01323       nmax--;
01324     }
01325     if (nmax < emin) {
01326       solver()->Fail();
01327     }
01328     var_->SetRange(nmin, nmax);
01329     if (var_->Bound()) {
01330       vars_[var_->Min()]->SetRange(mi, ma);
01331     }
01332   }
01333 }

01335 bool IntExprArrayElement::Bound() const {
01336   const int64 emin = std::max(0LL, var_->Min());
01337   const int64 emax = std::min(size_ - 1LL, var_->Max());
01338   const int64 v = vars_[emin]->Min();
01339   for (int i = emin; i <= emax; ++i) {
01340     if (var_->Contains(i) && (!vars_[i]->Bound() || vars_[i]->Value() != v)) {
01341       return false;
01342     }
01343   }
01344   return true;
01345 }
