00974 class IsMemberCt : public Constraint {
00975  public:
00976   IsMemberCt(Solver* const s,
00977              IntVar* const v,
00978              std::vector<int64>* const sorted_values,
00979              IntVar* const b)
00980       : Constraint(s),
00981         var_(v),
00982         values_(sorted_values),
00983         boolvar_(b),
00984         support_pos_(0),
00985         demon_(NULL) {
00986     DCHECK(v != NULL);
00987     DCHECK(s != NULL);
00988     DCHECK(b != NULL);
00989     DCHECK(sorted_values != NULL);
00990   }
00991 
00992   virtual void Post() {
00993     demon_ = solver()->MakeConstraintInitialPropagateCallback(this);
00994     if (!var_->Bound()) {
00995       var_->WhenDomain(demon_);
00996     }
00997     if (!boolvar_->Bound()) {
00998       boolvar_->WhenBound(demon_);
00999     }
01000   }
01001 
01002   virtual void InitialPropagate() {
01003     if (boolvar_->Min() == 1LL) {
01004       demon_->inhibit(solver());
01005       var_->SetValues(values_.RawData(), values_.size());
01006     } else if (boolvar_->Max() == 1LL) {
01007       int support = support_pos_.Value();
01008       const int64 vmin = var_->Min();
01009       const int64 vmax = var_->Max();
01010       while (support < values_.size() &&
01011              (values_[support] < vmin ||
01012               !var_->Contains(values_[support]))) {
01013         if (values_[support] <= vmax) {
01014           support++;
01015         } else {
01016           support = values_.size();
01017         }
01018       }
01019       support_pos_.SetValue(solver(), support);
01020       if (support >= values_.size()) {
01021         demon_->inhibit(solver());
01022         boolvar_->SetValue(0LL);
01023       } else if (var_->Bound()) {
01024         demon_->inhibit(solver());
01025         boolvar_->SetValue(1LL);
01026       }
01027     } else {  // boolvar_ set to 0.
01028       demon_->inhibit(solver());
01029       var_->RemoveValues(values_.RawData(), values_.size());
01030     }
01031   }
01032 
01033   virtual string DebugString() const {
01034     return StringPrintf("IsMemberCt(%s, %s, %s)",
01035                         var_->DebugString().c_str(),
01036                         values_.DebugString().c_str(),
01037                         boolvar_->DebugString().c_str());
01038   }
01039 
01040   virtual void Accept(ModelVisitor* const visitor) const {
01041     visitor->BeginVisitConstraint(ModelVisitor::kIsMember, this);
01042     visitor->VisitIntegerExpressionArgument(ModelVisitor::kExpressionArgument,
01043                                             var_);
01044     visitor->VisitConstIntArrayArgument(ModelVisitor::kValuesArgument,
01045                                         values_);
01046     visitor->VisitIntegerExpressionArgument(ModelVisitor::kTargetArgument,
01047                                             boolvar_);
01048     visitor->EndVisitConstraint(ModelVisitor::kIsMember, this);
01049   }
01050 
01051  private:
01052   IntVar* const var_;
01053   ConstIntArray values_;
01054   IntVar* const boolvar_;
01055   Rev<int> support_pos_;
01056   Demon* demon_;
01057 };
