00120   if (checked_ || size_ == 0) {
00121     return true;

00310 void BoundsAllDifferent::SortArray() {
00311   std::sort(min_sorted_.get(), min_sorted_.get() + size_, CompareIntervalMin());
00312   std::sort(max_sorted_.get(), max_sorted_.get() + size_, CompareIntervalMax());
00313 
00314   int64 min = min_sorted_[0]->min;
00315   int64 max = max_sorted_[0]->max + 1;
00316   int64 last = min - 2;
00317   bounds_[0] = last;
00318 
00319   int i = 0;
00320   int j = 0;
00321   int nb = 0;
00322   for (;;) {  // merge min_sorted_[] and max_sorted_[] into bounds_[].
00323     if (i < size_ && min <= max) {  // make sure min_sorted_ exhausted first.
00324       if (min != last) {
00325         last = min;
00326         bounds_[++nb] = last;
00327       }
00328       min_sorted_[i]->min_rank = nb;
00329       if (++i < size_) {
00330         min = min_sorted_[i]->min;
00331       }
00332     } else {
00333       if (max != last) {
00334         last = max;
00335         bounds_[++nb] = last;
00336       }
00337       max_sorted_[j]->max_rank = nb;
00338       if (++j == size_) {
00339         break;
00340       }
00341       max = max_sorted_[j]->max + 1;
00342     }
00343   }
00344   active_size_ = nb;
00345   bounds_[nb + 1] = bounds_[nb] + 2;
00346 }

00348 bool BoundsAllDifferent::PropagateMin() {
00349   bool modified = false;
00350 
00351   for (int i = 1; i <= active_size_ + 1; ++i) {
00352     hall_[i] = i - 1;
00353     tree_[i] = i - 1;
00354     diff_[i] = bounds_[i] - bounds_[i - 1];
00355   }
00356   for (int i = 0; i < size_; ++i) {  // visit intervals in increasing max order
00357     const int x = max_sorted_[i]->min_rank;
00358     const int y = max_sorted_[i]->max_rank;
00359     int z = PathMax(tree_.get(), x + 1);
00360     int j = tree_[z];
00361     if (--diff_[z] == 0) {
00362       tree_[z] = z + 1;
00363       z = PathMax(tree_.get(), z + 1);
00364       tree_[z] = j;
00365     }
00366     PathSet(x + 1, z, z, tree_.get());  // path compression
00367     if (diff_[z] < bounds_[z] - bounds_[y]) {
00368       solver()->Fail();
00369     }
00370     if (hall_[x] > x) {
00371       int w = PathMax(hall_.get(), hall_[x]);
00372       max_sorted_[i]->min = bounds_[w];
00373       PathSet(x, w, w, hall_.get());  // path compression
00374       modified = true;
00375     }
00376     if (diff_[z] == bounds_[z] - bounds_[y]) {
00377       PathSet(hall_[y], j - 1, y, hall_.get());  // mark hall interval
00378       hall_[y] = j - 1;
00379     }
00380   }
00381   return modified;
00382 }

00402       tree_[z] = j;

00414     }
