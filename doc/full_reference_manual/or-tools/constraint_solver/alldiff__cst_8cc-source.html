00305 void BoundsAllDifferent::SortArray() {
00306   std::sort(min_sorted_.get(), min_sorted_.get() + size_, CompareIntervalMin());
00307   std::sort(max_sorted_.get(), max_sorted_.get() + size_, CompareIntervalMax());
00308 
00309   int64 min = min_sorted_[0]->min;
00310   int64 max = max_sorted_[0]->max + 1;
00311   int64 last = min - 2;
00312   bounds_[0] = last;
00313 
00314   int i = 0;
00315   int j = 0;
00316   int nb = 0;
00317   for (;;) {  // merge min_sorted_[] and max_sorted_[] into bounds_[].
00318     if (i < size_ && min <= max) {  // make sure min_sorted_ exhausted first.
00319       if (min != last) {
00320         last = min;
00321         bounds_[++nb] = last;
00322       }
00323       min_sorted_[i]->min_rank = nb;
00324       if (++i < size_) {
00325         min = min_sorted_[i]->min;
00326       }
00327     } else {
00328       if (max != last) {
00329         last = max;
00330         bounds_[++nb] = last;
00331       }
00332       max_sorted_[j]->max_rank = nb;
00333       if (++j == size_) {
00334         break;
00335       }
00336       max = max_sorted_[j]->max + 1;
00337     }
00338   }
00339   active_size_ = nb;
00340   bounds_[nb + 1] = bounds_[nb] + 2;
00341 }

00343 bool BoundsAllDifferent::PropagateMin() {
00344   bool modified = false;
00345 
00346   for (int i = 1; i <= active_size_ + 1; ++i) {
00347     hall_[i] = i - 1;
00348     tree_[i] = i - 1;
00349     diff_[i] = bounds_[i] - bounds_[i - 1];
00350   }
00351   for (int i = 0; i < size_; ++i) {  // visit intervals in increasing max order
00352     const int x = max_sorted_[i]->min_rank;
00353     const int y = max_sorted_[i]->max_rank;
00354     int z = PathMax(tree_.get(), x + 1);
00355     int j = tree_[z];
00356     if (--diff_[z] == 0) {
00357       tree_[z] = z + 1;
00358       z = PathMax(tree_.get(), z + 1);
00359       tree_[z] = j;
00360     }
00361     PathSet(x + 1, z, z, tree_.get());  // path compression
00362     if (diff_[z] < bounds_[z] - bounds_[y]) {
00363       solver()->Fail();
00364     }
00365     if (hall_[x] > x) {
00366       int w = PathMax(hall_.get(), hall_[x]);
00367       max_sorted_[i]->min = bounds_[w];
00368       PathSet(x, w, w, hall_.get());  // path compression
00369       modified = true;
00370     }
00371     if (diff_[z] == bounds_[z] - bounds_[y]) {
00372       PathSet(hall_[y], j - 1, y, hall_.get());  // mark hall interval
00373       hall_[y] = j - 1;
00374     }
00375   }
00376   return modified;
00377 }
