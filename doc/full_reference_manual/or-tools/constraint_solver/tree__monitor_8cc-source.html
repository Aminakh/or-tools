00038 // String comparator that compares strings naturally, even those
00039 // including integer numbers.
00040 struct NaturalLess {
00041   bool operator()(const string& s1, const string& s2) const {
00042     int start = 0;
00043     int length = std::min(s1.length(), s2.length());
00044 
00045     while (start < length) {
00046       // Ignore common characters at the beginning.
00047       while (start < length && s1[start] == s2[start] &&
00048           (s1[start] < '0' || s1[start] > '9')) {
00049         ++start;
00050       }
00051 
00052       // If one string is the substring of another, then the shorter string is
00053       // smaller.
00054       if (start == length) {
00055         return s1.length() < s2.length();
00056       }
00057 
00058       int number_s1 = 0;
00059       int number_s2 = 0;
00060 
00061       // Extract a number if we have one.
00062       for (int i = start;
00063            i < s1.length() && s1[i] >= '0' && s1[i] <= '9';
00064            ++i) {
00065         number_s1 = number_s1 * 10 + (s1[i] - '0');
00066       }
00067 
00068       for (; start < s2.length() && s2[start] >= '0' && s2[start] <= '9';
00069            ++start) {
00070         number_s2 = number_s2 * 10 + (s2[start] - '0');
00071       }
00072 
00073       // Do a numerical comparison only if there are two numbers.
00074       if (number_s1 && number_s2) {
00075         // If we have similar numbers followed by other characters, we have to
00076         // check the rest of the string.
00077         if (number_s1 != number_s2) {
00078           return number_s1 < number_s2;
00079         }
00080       } else {
00081         return s1.compare(s2) < 0;
00082       }
00083     }
00084 
00085     return s1.length() < s2.length();
00086   }
00087 };

00627 void TreeMonitor::RefuteDecision(Decision* const decision) {
00628   // Called when the solver goes up one level in the tree and undos a
00629   // change in the tree. As we have added multiple levels for both 'fail'
00630   // and 'success', we have to go up two levels in some cases.
00631   CHECK_NOTNULL(decision);
00632 
00633   if (current_node_->node_type() == TreeNode::SOLUTION) {
00634     // Solver calls RefuteDecision even on success if it looks for
00635     // more than one solution.
00636     // Just go back to the previous decision.
00637     current_node_ = current_node_->Parent();
00638   } else if (current_node_->node_type() == TreeNode::TRY) {
00639     // Add an extra node in case of a failure, so we can see the failed
00640     // decision.
00641     current_node_->set_node_type(TreeNode::TRY);
00642 
00643     if (current_node_->AddChild(id_counter_, last_variable_,
00644                                 last_value_, true, vars_, &current_node_)) {
00645       ++id_counter_;
00646     }
00647 
00648     current_node_->set_node_type(TreeNode::FAIL);
00649     current_node_ = current_node_->Parent();
00650   }
00651 
00652   current_node_ = current_node_->Parent();
00653 }

00706 void TreeMonitor::ExitSearch() {
00707   --search_level_;
00708 
00709   VLOG(1) << "Current search level " << search_level_;
00710 
00711   if (!search_level_) {
00712     // If search has compeleted and a filename is specified, automatically
00713     // output the XML to these files.
00714     if (!filename_tree_.empty()) {
00715       std::ofstream file_tree_(filename_tree_.c_str());
00716 
00717       if (file_tree_.is_open()) {
00718         file_tree_ << GenerateTreeXML().c_str();
00719         file_tree_.close();
00720       } else {
00721         LG << "Failed to gain write access to file: " << filename_tree_;
00722       }
00723 
00724       std::ofstream file_visualizer_(filename_visualizer_.c_str());
00725 
00726       if (file_visualizer_.is_open()) {
00727         file_visualizer_ << GenerateVisualizationXML().c_str();
00728         file_visualizer_.close();
00729       } else {
00730         LG << "Failed to gain write access to file: " << filename_tree_;
00731       }
00732 
00733       if(!filename_config_.empty()) {
00734         std::ofstream file_config_(filename_config_.c_str());
00735 
00736         if (file_config_.is_open()) {
00737           file_config_ << kConfigXml;
00738           file_config_.close();
00739         } else {
00740           LG << "Failed to gain write access to file: " << filename_config_;
00741         }
00742       }
00743     } else {
00744       CHECK_NOTNULL(tree_xml_);
00745       *tree_xml_ = GenerateTreeXML();
00746 
00747       CHECK_NOTNULL(visualization_xml_);
00748       *visualization_xml_ = GenerateVisualizationXML();
00749 
00750       if (config_xml_) {
00751         *config_xml_ = kConfigXml;
00752       }
00753     }
00754   }

00757 string TreeMonitor::StripSpecialCharacters(string attribute) {
00758   // Numbers, characters, dashes, underscored, brackets, colons, slashes,
00759   // periods, question marks, and parentheses are allowed
00760   const char* kAllowedCharacters = "0123456789abcdefghijklmnopqrstuvwxyz"
00761                                    "ABCDEFGHIJKLMNOPQRSTUVWXYZ_-[]:/.?()";
00762 
00763   std::set<char> character_set;
00764 
00765   char* allowed = const_cast<char*>(kAllowedCharacters);
00766 
00767   while (*allowed) {
00768     character_set.insert(*(allowed++));
00769   }
00770 
00771   for (int i = 0; i < attribute.length(); ++i) {
00772     if (character_set.find(attribute[i]) == character_set.end()) {
00773       attribute.replace(i,1,"_");
00774     }
00775   }
00776 
00777   return attribute;
00778 }
