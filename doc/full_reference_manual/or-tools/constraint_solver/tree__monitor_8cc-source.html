00026 // String comparator that compares strings naturally, even those
00027 // including integer numbers.
00028 struct NaturalLess {
00029   bool operator()(const string& s1, const string& s2) const {
00030     int start = 0;
00031     int length = std::min(s1.length(), s2.length());
00032 
00033     while (start < length) {
00034       // Ignore common characters at the beginning.
00035       while (start < length && s1[start] == s2[start] &&
00036           (s1[start] < '0' || s1[start] > '9')) {
00037         ++start;
00038       }
00039 
00040       // If one string is the substring of another, then the shorter string is
00041       // smaller.
00042       if (start == length) {
00043         return s1.length() < s2.length();
00044       }
00045 
00046       int number_s1 = 0;
00047       int number_s2 = 0;
00048 
00049       // Extract a number if we have one.
00050       for (int i = start;
00051            i < s1.length() && s1[i] >= '0' && s1[i] <= '9';
00052            ++i) {
00053         number_s1 = number_s1 * 10 + (s1[i] - '0');
00054       }
00055 
00056       for (; start < s2.length() && s2[start] >= '0' && s2[start] <= '9';
00057            ++start) {
00058         number_s2 = number_s2 * 10 + (s2[start] - '0');
00059       }
00060 
00061       // Do a numerical comparison only if there are two numbers.
00062       if (number_s1 && number_s2) {
00063         // If we have similar numbers followed by other characters, we have to
00064         // check the rest of the string.
00065         if (number_s1 != number_s2) {
00066           return number_s1 < number_s2;
00067         }
00068       } else {
00069         return s1.compare(s2) < 0;
00070       }
00071     }
00072 
00073     return s1.length() < s2.length();
00074   }
00075 };

00611 void TreeMonitor::RefuteDecision(Decision* const decision) {
00612   // Called when the solver goes up one level in the tree and undos a
00613   // change in the tree. As we have added multiple levels for both 'fail'
00614   // and 'success', we have to go up two levels in some cases.
00615   CHECK_NOTNULL(decision);
00616 
00617   if (current_node_->node_type() == TreeNode::SOLUTION) {
00618     // Solver calls RefuteDecision even on success if it looks for
00619     // more than one solution.
00620     // Just go back to the previous decision.
00621     current_node_ = current_node_->Parent();
00622   } else if (current_node_->node_type() == TreeNode::TRY) {
00623     // Add an extra node in case of a failure, so we can see the failed
00624     // decision.
00625     current_node_->set_node_type(TreeNode::TRY);
00626 
00627     if (current_node_->AddChild(id_counter_, last_variable_,
00628                                 last_value_, true, vars_, &current_node_)) {
00629       ++id_counter_;
00630     }
00631 
00632     current_node_->set_node_type(TreeNode::FAIL);
00633     current_node_ = current_node_->Parent();
00634   }
00635 
00636   current_node_ = current_node_->Parent();

00741 string TreeMonitor::StripSpecialCharacters(string attribute) {
00742   // Numbers, characters, dashes, underscored, brackets, colons, slashes,
00743   // periods, question marks, and parentheses are allowed
00744   const char* kAllowedCharacters = "0123456789abcdefghijklmnopqrstuvwxyz"
00745                                    "ABCDEFGHIJKLMNOPQRSTUVWXYZ_-[]:/.?()";
00746 
00747   std::set<char> character_set;
00748 
00749   char* allowed = const_cast<char*>(kAllowedCharacters);
00750 
00751   while (*allowed) {
00752     character_set.insert(*(allowed++));
00753   }
00754 
00755   for (int i = 0; i < attribute.length(); ++i) {
00756     if (character_set.find(attribute[i]) == character_set.end()) {
00757       attribute.replace(i,1,"_");
00758     }
00759   }
00760 
00761   return attribute;
00762 }
