00083   virtual Decision* Next(Solver* const s) {
00084     int64 best_est = kint64max;
00085     int64 best_lct = kint64max;
00086     int support = -1;
00087     int refuted = 0;
00088     for (int i = 0; i < size_; ++i) {
00089       IntervalVar* const v = vars_[i];
00090       if (v->MayBePerformed() && v->StartMax() > v->StartMin()) {
00091         if (v->StartMin() >= markers_[i] &&
00092                   (v->StartMin() < best_est ||
00093                    (v->StartMin() == best_est && v->EndMax() < best_lct))) {
00094           best_est = v->StartMin();
00095           best_lct = v->EndMax();
00096           support = i;
00097         } else {
00098           refuted++;
00099         }
00100       }
00101     }
00102     // TODO(user) : remove this crude quadratic loop with
00103     // reversibles range reduction.
00104     if (support == -1) {
00105       if (refuted == 0) {
00106         return NULL;
00107       } else {
00108         s->Fail();
00109       }
00110     }
00111     return s->RevAlloc(new ScheduleOrPostpone(vars_[support],
00112                                               vars_[support]->StartMin(),
00113                                               &markers_[support]));

00180  public:

00187   virtual Decision* Next(Solver* const s) {
00188     Sequence* seq = NULL;
00189     int64 slack = kint64max;
00190     int64 best_hmin = kint64max;
00191     for (int i = 0; i < size_; ++i) {
00192       Sequence* const se = sequences_[i];
00193       if (se->NotRanked() > 0) {
00194         int64 hmin, hmax, dmin, dmax;
00195         se->HorizonRange(&hmin, &hmax);
00196         se->DurationRange(&dmin, &dmax);
00197         const int64 current_slack = (hmax - hmin - dmax);
00198         if (current_slack < slack) {
00199           slack = current_slack;
00200           seq = se;
00201           int64 hmin, hmax;
00202           se->ActiveHorizonRange(&hmin, &hmax);
00203           best_hmin = hmin;
00204         } else if (current_slack == slack) {
00205           int64 hmin, hmax;
00206           se->ActiveHorizonRange(&hmin, &hmax);
00207           if (hmin < best_hmin) {
00208             seq = se;
00209             best_hmin = hmin;
00210           }
00211         }
00212       }
00213     }
00214     if (seq != NULL) {
00215       seq->ComputePossibleRanks();
00216       int index = -1;
00217       int64 smin = kint64max;
00218       for (int i = 0; i < seq->size(); ++i) {
00219         if (seq->PossibleFirst(i)) {
00220           IntervalVar* const t = seq->Interval(i);
00221           if (t->MayBePerformed() && t->StartMin() < smin) {
00222             index = i;
00223             smin = t->StartMin();
00224           }
00225         }
00226       }
00227       if (index == -1) {
00228         s->Fail();
00229       }
00230       CHECK_NE(-1, index);
00231       return s->RevAlloc(new TryRankFirst(seq, index));
