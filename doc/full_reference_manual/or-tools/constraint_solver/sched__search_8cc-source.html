00076     : vars_(new IntervalVar*[size]),
00077       size_(size),
00078       markers_(new int64[size]) {
00079     memcpy(vars_.get(), vars, sizeof(*vars) * size);
00080     for (int i = 0; i < size_; ++i) {
00081       markers_[i] = kint64min;
00082     }
00083   }
00084 
00085   virtual ~SetTimesForward() {}

00087   virtual Decision* Next(Solver* const s) {
00088     int64 best_est = kint64max;
00089     int64 best_lct = kint64max;
00090     int support = -1;
00091     int refuted = 0;
00092     for (int i = 0; i < size_; ++i) {
00093       IntervalVar* const v = vars_[i];
00094       if (v->MayBePerformed() && v->StartMax() > v->StartMin()) {
00095         if (v->StartMin() >= markers_[i] &&
00096                   (v->StartMin() < best_est ||
00097                    (v->StartMin() == best_est && v->EndMax() < best_lct))) {
00098           best_est = v->StartMin();
00099           best_lct = v->EndMax();
00100           support = i;
00101         } else {
00102           refuted++;
00103         }
00104       }
00105     }
00106     // TODO(user) : remove this crude quadratic loop with
00107     // reversibles range reduction.
00108     if (support == -1) {
00109       if (refuted == 0) {
00110         return NULL;
00111       } else {
00112         s->Fail();
00113       }
00114     }
00115     return s->RevAlloc(new ScheduleOrPostpone(vars_[support],
00116                                               vars_[support]->StartMin(),
00117                                               &markers_[support]));
00118   }

00183 class RankFirstSequenceVars : public DecisionBuilder {
00184  public:
00185   RankFirstSequenceVars(const SequenceVar* const * sequences, int size)
00186       : sequences_(new SequenceVar*[size]), size_(size) {
00187     memcpy(sequences_.get(), sequences, size_ * sizeof(*sequences));
00188   }
00189 
00190   virtual ~RankFirstSequenceVars() {}
00191 
00192   virtual Decision* Next(Solver* const s) {
00193     SequenceVar* best_sequence = NULL;
00194     int64 best_slack = kint64max;
00195     int64 best_ahmin = kint64max;
00196     std::vector<int> best_possible_firsts;
00197     for (int i = 0; i < size_; ++i) {
00198       SequenceVar* const candidate_sequence = sequences_[i];
00199       if (candidate_sequence->NotRanked() > 0) {
00200         std::vector<int> candidate_possible_firsts;
00201         candidate_sequence->ComputePossibleFirsts(&candidate_possible_firsts);
00202         // No possible first, failing.
00203         if (candidate_possible_firsts.size() == 0) {
00204           s->Fail();
00205         }
00206         // Only 1 candidate, and non optional: ranking without branching.
00207         if (candidate_possible_firsts.size() == 1 &&
00208             candidate_sequence->Interval(
00209                 candidate_possible_firsts.back())->MustBePerformed()) {
00210           candidate_sequence->RankFirst(candidate_possible_firsts[0]);
00211           continue;
00212         }
00213 
00214         // Evaluating the sequence.
00215         int64 hmin, hmax, dmin, dmax;
00216         candidate_sequence->HorizonRange(&hmin, &hmax);
00217         candidate_sequence->DurationRange(&dmin, &dmax);
00218         int64 ahmin, ahmax;
00219         candidate_sequence->ActiveHorizonRange(&ahmin, &ahmax);
00220         const int64 current_slack = (hmax - hmin - dmax);
00221         if (current_slack < best_slack ||
00222             (current_slack == best_slack && ahmin < best_ahmin)) {
00223           best_slack = current_slack;
00224           best_sequence = candidate_sequence;
00225           best_possible_firsts = candidate_possible_firsts;
00226           best_ahmin = ahmin;
00227         }
00228       }
00229     }
00230     if (best_sequence != NULL) {
00231       int best_interval = -1;
00232       int64 best_start_min = kint64max;
00233       for (int index = 0; index < best_possible_firsts.size(); ++index) {
00234         const int candidate = best_possible_firsts[index];
00235         IntervalVar* const interval = best_sequence->Interval(candidate);
00236         if (interval->StartMin() < best_start_min) {
00237           best_interval = candidate;
00238           best_start_min = interval->StartMin();
00239         }
00240       }
00241       if (best_interval == -1) {
00242         s->Fail();
00243       }
00244       CHECK_NE(-1, best_interval);
00245       return s->RevAlloc(new TryRankFirst(best_sequence, best_interval));
00246     }
00247     return NULL;
00248   }
