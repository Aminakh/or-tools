03226 // Propagates set_min on left * right, left and right >= 0.
03227 void SetPosPosMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03228   DCHECK_GE(left->Min(), 0);
03229   DCHECK_GE(right->Min(), 0);
03230   const int64 lmax = left->Max();
03231   const int64 rmax = right->Max();
03232   if (m > lmax * rmax) {
03233     left->solver()->Fail();
03234   }
03235   if (m > left->Min() * right->Min()) {
03236     // Ok for m == 0 due to left and right being positive
03237     if (0 != rmax) {
03238       left->SetMin(PosIntDivUp(m, rmax));
03239     }
03240     if (0 != lmax) {
03241       right->SetMin(PosIntDivUp(m, lmax));
03242     }
03243   }
03244 }

03246 // Propagates set_max on left * right, left and right >= 0.
03247 void SetPosPosMaxExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03248   DCHECK_GE(left->Min(), 0);
03249   DCHECK_GE(right->Min(), 0);
03250   const int64 lmin = left->Min();
03251   const int64 rmin = right->Min();
03252   if (m < lmin * rmin) {
03253     left->solver()->Fail();
03254   }
03255   if (m < left->Max() * right->Max()) {
03256     if (0 != lmin) {
03257       right->SetMax(PosIntDivDown(m, lmin));
03258     }
03259     if (0 != rmin) {
03260       left->SetMax(PosIntDivDown(m, rmin));
03261     }

03263   }

03266 // Propagates set_min on left * right, left >= 0, right across 0.
03267 void SetPosGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03268   DCHECK_GE(left->Min(), 0);
03269   DCHECK_GT(right->Max(), 0);
03270   DCHECK_LT(right->Min(), 0);
03271   const int64 lmax = left->Max();
03272   const int64 rmax = right->Max();
03273   if (m > lmax * rmax) {
03274     left->solver()->Fail();
03275   }
03276   DCHECK_GT(left->Max(), 0);
03277   if (m > 0) {  // We deduce right > 0.
03278     left->SetMin(PosIntDivUp(m, rmax));
03279     right->SetMin(PosIntDivUp(m, lmax));
03280   } else if (m == 0) {
03281     const int64 lmin = left->Min();
03282     if (lmin > 0) {
03283       right->SetMin(0);
03284     }
03285   } else {  // m < 0
03286     const int64 lmin = left->Min();
03287     if (0 != lmin) {  // We cannot deduce anything if 0 is in the domain.
03288       right->SetMin(-PosIntDivDown(-m, lmin));
03289     }
03290   }
03291 }

03293 // Propagates set_min on left * right, left and right across 0.
03294 void SetGenGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03295   DCHECK_LT(left->Min(), 0);
03296   DCHECK_GT(left->Max(), 0);
03297   DCHECK_GT(right->Max(), 0);
03298   DCHECK_LT(right->Min(), 0);
03299   const int64 lmin = left->Min();
03300   const int64 lmax = left->Max();
03301   const int64 rmin = right->Min();
03302   const int64 rmax = right->Max();
03303   if (m > std::max(lmin * rmin, lmax * rmax)) {
03304     left->solver()->Fail();
03305   }
03306   if (m > lmin * rmin) {  // Must be positive section * positive section.
03307     left->SetMin(PosIntDivUp(m, rmax));
03308     right->SetMin(PosIntDivUp(m, lmax));
03309   } else if (m > lmax * rmax) {  // Negative section * negative section.
03310     left->SetMax(-PosIntDivUp(m, -rmin));
03311     right->SetMax(-PosIntDivUp(m, -lmin));
03312   }
03313 }

03315 void TimesSetMin(IntExpr* const left,
03316                  IntExpr* const right,
03317                  IntExpr* const minus_left,
03318                  IntExpr* const minus_right,
03319                  int m) {
03320   if (left->Min() >= 0) {
03321     if (right->Min() >= 0) {
03322       SetPosPosMinExpr(left, right, m);
03323     } else if (right->Max() <= 0) {
03324       SetPosPosMaxExpr(left, minus_right, -m);
03325     } else {  // right->Min() < 0 && right->Max() > 0
03326       SetPosGenMinExpr(left, right, m);
03327     }
03328   } else if (left->Max() <= 0) {
03329     if (right->Min() >= 0) {
03330       SetPosPosMaxExpr(right, minus_left, -m);
03331     } else if (right->Max() <= 0) {
03332       SetPosPosMinExpr(minus_left, minus_right, m);
03333     } else {  // right->Min() < 0 && right->Max() > 0
03334       SetPosGenMinExpr(minus_left, minus_right, m);
03335     }
03336   } else if (right->Min() >= 0) {  // left->Min() < 0 && left->Max() > 0
03337     SetPosGenMinExpr(right, left, m);
03338   } else if (right->Max() <= 0) {  // left->Min() < 0 && left->Max() > 0
03339     SetPosGenMinExpr(minus_right, minus_left, m);
03340   } else {  // left->Min() < 0 && left->Max() > 0 &&
03341             // right->Min() < 0 && right->Max() > 0
03342     SetGenGenMinExpr(left, right, m);
03343   }

03764 IntExpr* Solver::MakeDiv(IntExpr* const numerator, IntExpr* const denominator) {
03765   // Both numerator and denominator are positive.
03766   // Denominator needs to be != 0.
03767   AddConstraint(MakeGreater(denominator, 0));
03768   IntVar* const result = MakeIntVar(0, numerator->Max());
03769   IntExpr* const product = MakeProd(denominator, result);
03770   // The integer division result = numerator / denominator means
03771   // numerator = result * denominator + quotient, with quotient < denominator.
03772   // This translates into:
03773   //     numerator >= denominator * result
03774   AddConstraint(MakeGreaterOrEqual(numerator->Var(), product->Var()));
03775   //     numerator < denominator * result + denominator
03776   IntExpr* const product_up = MakeSum(product, denominator);
03777   AddConstraint(MakeLess(numerator->Var(), product_up->Var()));
03778   return result;
03779 }

04281 // ----- Convex Piecewise -----
04282 
04283 // This class is a very simple convex piecewise linear function.  The
04284 // argument of the function is the expression.  Between early_date and
04285 // late_date, the value of the function is 0.  Before early date, it
04286 // is affine and the cost is early_cost * (early_date - x). After
04287 // late_date, the cost is late_cost * (x - late_date).
04288 
04289 class SimpleConvexPiecewiseExpr : public BaseIntExpr {
04290  public:
04291   SimpleConvexPiecewiseExpr(Solver* const s, IntVar* const e,
04292                             int64 ec, int64 ed,
04293                             int64 ld, int64 lc)
04294       : BaseIntExpr(s),
04295     var_(e),
04296     early_cost_(ec),
04297     early_date_(ec == 0 ? kint64min : ed),
04298     late_date_(lc == 0 ? kint64max : ld),
04299     late_cost_(lc) {
04300         DCHECK_GE(ec, 0LL);
04301         DCHECK_GE(lc, 0LL);
04302         DCHECK_GE(ld, ed);
04303 
04304         // If the penalty is 0, we can push the "confort zone or zone
04305         // of no cost towards infinity.
04306 }
04307   virtual ~SimpleConvexPiecewiseExpr() {}
04308   virtual int64 Min() const {
04309     const int64 vmin = var_->Min();
04310     const int64 vmax = var_->Max();
04311     if (vmin >= late_date_) {
04312       return (vmin - late_date_) * late_cost_;
04313     } else if (vmax <= early_date_) {
04314       return (early_date_ - vmax) * early_cost_;
04315     } else {
04316       return 0LL;
04317     }
04318   }
04319   virtual void SetMin(int64 m) {
04320     if (m <= 0) {
04321       return;
04322     }
04323     const int64 vmin = var_->Min();
04324     const int64 vmax = var_->Max();
04325     const int64 rb = (late_cost_ == 0 ?
04326                       vmax :
04327                       late_date_ + PosIntDivUp(m , late_cost_) - 1);
04328     const int64 lb = (early_cost_ == 0 ?
04329                       vmin :
04330                       early_date_ - PosIntDivUp(m , early_cost_) + 1);
04331     var_->RemoveInterval(lb, rb);
04332   }
04333   virtual int64 Max() const {
04334     const int64 vmin = var_->Min();
04335     const int64 vmax = var_->Max();
04336     const int64 mr = vmax > late_date_ ? (vmax - late_date_) * late_cost_ : 0;
04337     const int64 ml = vmin < early_date_
04338         ? (early_date_ - vmin) * early_cost_
04339         : 0;
04340     return max(mr, ml);
04341   }
04342   virtual void SetMax(int64 m) {
04343     if (m < 0) {
04344       solver()->Fail();
04345     }
04346     if (late_cost_ != 0LL) {
04347       const int64 rb = late_date_ + PosIntDivDown(m, late_cost_);
04348       if (early_cost_ != 0LL) {
04349         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
04350         var_->SetRange(lb, rb);
04351       } else {
04352         var_->SetMax(rb);
04353       }
04354     } else {
04355       if (early_cost_ != 0LL) {
04356         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
04357         var_->SetMin(lb);
04358       }
04359     }
04360   }
04361   virtual string name() const {
04362     return StringPrintf(
04363         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
04364         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
04365         var_->name().c_str(),
04366         early_cost_, early_date_, late_date_, late_cost_);
04367   }
04368   virtual string DebugString() const {
04369     return StringPrintf(
04370         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
04371         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
04372         var_->DebugString().c_str(),
04373         early_cost_, early_date_, late_date_, late_cost_);
04374   }
04375   virtual void WhenRange(Demon* d) {
04376     var_->WhenRange(d);
04377   }
04378  private:
04379   IntVar* const var_;
04380   const int64 early_cost_;
04381   const int64 early_date_;
04382   const int64 late_date_;
04383   const int64 late_cost_;
04384 };

04388                                          int64 late_date, int64 late_cost) {
04389   return RevAlloc(new SimpleConvexPiecewiseExpr(this, e,
04390                                                 early_cost, early_date,
04391                                                 late_date, late_cost));
