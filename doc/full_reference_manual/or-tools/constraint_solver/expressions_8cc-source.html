03368 // Propagates set_min on left * right, left and right >= 0.
03369 void SetPosPosMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03370   DCHECK_GE(left->Min(), 0);
03371   DCHECK_GE(right->Min(), 0);
03372   const int64 lmax = left->Max();
03373   const int64 rmax = right->Max();
03374   if (m > lmax * rmax) {
03375     left->solver()->Fail();
03376   }
03377   if (m > left->Min() * right->Min()) {
03378     // Ok for m == 0 due to left and right being positive
03379     if (0 != rmax) {
03380       left->SetMin(PosIntDivUp(m, rmax));
03381     }
03382     if (0 != lmax) {
03383       right->SetMin(PosIntDivUp(m, lmax));
03384     }
03385   }
03386 }

03389 void SetPosPosMaxExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03390   DCHECK_GE(left->Min(), 0);
03391   DCHECK_GE(right->Min(), 0);
03392   const int64 lmin = left->Min();
03393   const int64 rmin = right->Min();
03394   if (m < lmin * rmin) {
03395     left->solver()->Fail();
03396   }
03397   if (m < left->Max() * right->Max()) {
03398     if (0 != lmin) {
03399       right->SetMax(PosIntDivDown(m, lmin));
03400     }
03401     if (0 != rmin) {
03402       left->SetMax(PosIntDivDown(m, rmin));
03403     }
03404     // else do nothing: 0 is supporting any value from other expr.
03405   }
03406 }

03408 // Propagates set_min on left * right, left >= 0, right across 0.
03409 void SetPosGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03410   DCHECK_GE(left->Min(), 0);
03411   DCHECK_GT(right->Max(), 0);
03412   DCHECK_LT(right->Min(), 0);
03413   const int64 lmax = left->Max();
03414   const int64 rmax = right->Max();
03415   if (m > lmax * rmax) {
03416     left->solver()->Fail();
03417   }
03418   DCHECK_GT(left->Max(), 0);
03419   if (m > 0) {  // We deduce right > 0.
03420     left->SetMin(PosIntDivUp(m, rmax));
03421     right->SetMin(PosIntDivUp(m, lmax));
03422   } else if (m == 0) {
03423     const int64 lmin = left->Min();
03424     if (lmin > 0) {
03425       right->SetMin(0);
03426     }
03427   } else {  // m < 0
03428     const int64 lmin = left->Min();
03429     if (0 != lmin) {  // We cannot deduce anything if 0 is in the domain.
03430       right->SetMin(-PosIntDivDown(-m, lmin));
03431     }
03432   }
03433 }

03435 // Propagates set_min on left * right, left and right across 0.
03436 void SetGenGenMinExpr(IntExpr* const left, IntExpr* const right, int64 m) {
03437   DCHECK_LT(left->Min(), 0);
03438   DCHECK_GT(left->Max(), 0);
03439   DCHECK_GT(right->Max(), 0);
03440   DCHECK_LT(right->Min(), 0);
03441   const int64 lmin = left->Min();
03442   const int64 lmax = left->Max();
03443   const int64 rmin = right->Min();
03444   const int64 rmax = right->Max();
03445   if (m > std::max(lmin * rmin, lmax * rmax)) {
03446     left->solver()->Fail();
03447   }
03448   if (m > lmin * rmin) {  // Must be positive section * positive section.
03449     left->SetMin(PosIntDivUp(m, rmax));
03450     right->SetMin(PosIntDivUp(m, lmax));
03451   } else if (m > lmax * rmax) {  // Negative section * negative section.
03452     left->SetMax(-PosIntDivUp(m, -rmin));
03453     right->SetMax(-PosIntDivUp(m, -lmin));
03454   }
03455 }

03458                  IntExpr* const right,
03459                  IntExpr* const minus_left,
03460                  IntExpr* const minus_right,
03461                  int m) {
03462   if (left->Min() >= 0) {
03463     if (right->Min() >= 0) {
03464       SetPosPosMinExpr(left, right, m);
03465     } else if (right->Max() <= 0) {
03466       SetPosPosMaxExpr(left, minus_right, -m);
03467     } else {  // right->Min() < 0 && right->Max() > 0
03468       SetPosGenMinExpr(left, right, m);
03469     }
03470   } else if (left->Max() <= 0) {
03471     if (right->Min() >= 0) {
03472       SetPosPosMaxExpr(right, minus_left, -m);
03473     } else if (right->Max() <= 0) {
03474       SetPosPosMinExpr(minus_left, minus_right, m);
03475     } else {  // right->Min() < 0 && right->Max() > 0
03476       SetPosGenMinExpr(minus_left, minus_right, m);
03477     }
03478   } else if (right->Min() >= 0) {  // left->Min() < 0 && left->Max() > 0
03479     SetPosGenMinExpr(right, left, m);
03480   } else if (right->Max() <= 0) {  // left->Min() < 0 && left->Max() > 0
03481     SetPosGenMinExpr(minus_right, minus_left, m);
03482   } else {  // left->Min() < 0 && left->Max() > 0 &&
03483             // right->Min() < 0 && right->Max() > 0
03484     SetGenGenMinExpr(left, right, m);
03485   }

03906   }
03907   if (r->Bound()) {
03908     return MakeProd(l, r->Min());
03909   }
03910   if (l == r) {
03911     return MakeSquare(l);
03912   }
03913   CHECK_EQ(this, l->solver());
03914   CHECK_EQ(this, r->solver());
03915   if (l->IsVar() &&
03916       reinterpret_cast<IntVar*>(l)->VarType() == BOOLEAN_VAR) {

03923           new TimesBooleanIntExpr(this, reinterpret_cast<BooleanVar*>(l), r));
03924     }

03926   if (r->IsVar() &&
03927       reinterpret_cast<IntVar*>(r)->VarType() == BOOLEAN_VAR) {
03928     if (l->Min() >= 0) {
03929       return RevAlloc(
03930           new TimesBooleanPosIntExpr(this,

03932     } else {
03933       return RevAlloc(
03934           new TimesBooleanIntExpr(this, reinterpret_cast<BooleanVar*>(r), l));
03935     }

03937   if (l->Min() >= 0 && r->Min() >= 0) {
03938     return RevAlloc(new TimesIntPosExpr(this, l, r));

03944 IntExpr* Solver::MakeDiv(IntExpr* const numerator, IntExpr* const denominator) {
03945   // Both numerator and denominator are positive.
03946   // Denominator needs to be != 0.
03947   AddConstraint(MakeGreater(denominator, 0));
03948   IntVar* const result = MakeIntVar(0, numerator->Max());
03949   IntExpr* const product = MakeProd(denominator, result);
03950   // The integer division result = numerator / denominator means
03951   // numerator = result * denominator + quotient, with quotient < denominator.
03952   // This translates into:
03953   //     numerator >= denominator * result
03954   AddConstraint(MakeGreaterOrEqual(numerator->Var(), product->Var()));
03955   //     numerator < denominator * result + denominator
03956   IntExpr* const product_up = MakeSum(product, denominator);
03957   AddConstraint(MakeLess(numerator->Var(), product_up->Var()));
03958   return result;
03959 }

04531 // ----- Convex Piecewise -----
04532 
04533 // This class is a very simple convex piecewise linear function.  The
04534 // argument of the function is the expression.  Between early_date and
04535 // late_date, the value of the function is 0.  Before early date, it
04536 // is affine and the cost is early_cost * (early_date - x). After
04537 // late_date, the cost is late_cost * (x - late_date).
04538 
04539 class SimpleConvexPiecewiseExpr : public BaseIntExpr {
04540  public:
04541   SimpleConvexPiecewiseExpr(Solver* const s, IntVar* const e,
04542                             int64 ec, int64 ed,
04543                             int64 ld, int64 lc)
04544       : BaseIntExpr(s),
04545     var_(e),
04546     early_cost_(ec),
04547     early_date_(ec == 0 ? kint64min : ed),
04548     late_date_(lc == 0 ? kint64max : ld),
04549     late_cost_(lc) {
04550         DCHECK_GE(ec, 0LL);
04551         DCHECK_GE(lc, 0LL);
04552         DCHECK_GE(ld, ed);
04553 
04554         // If the penalty is 0, we can push the "confort zone or zone
04555         // of no cost towards infinity.
04556 }
04557   virtual ~SimpleConvexPiecewiseExpr() {}
04558   virtual int64 Min() const {
04559     const int64 vmin = var_->Min();
04560     const int64 vmax = var_->Max();
04561     if (vmin >= late_date_) {
04562       return (vmin - late_date_) * late_cost_;
04563     } else if (vmax <= early_date_) {
04564       return (early_date_ - vmax) * early_cost_;
04565     } else {
04566       return 0LL;
04567     }
04568   }
04569   virtual void SetMin(int64 m) {
04570     if (m <= 0) {
04571       return;
04572     }
04573     const int64 vmin = var_->Min();
04574     const int64 vmax = var_->Max();
04575     const int64 rb = (late_cost_ == 0 ?
04576                       vmax :
04577                       late_date_ + PosIntDivUp(m , late_cost_) - 1);
04578     const int64 lb = (early_cost_ == 0 ?
04579                       vmin :
04580                       early_date_ - PosIntDivUp(m , early_cost_) + 1);
04581     var_->RemoveInterval(lb, rb);
04582   }
04583   virtual int64 Max() const {
04584     const int64 vmin = var_->Min();
04585     const int64 vmax = var_->Max();
04586     const int64 mr = vmax > late_date_ ? (vmax - late_date_) * late_cost_ : 0;
04587     const int64 ml = vmin < early_date_
04588         ? (early_date_ - vmin) * early_cost_
04589         : 0;
04590     return std::max(mr, ml);
04591   }
04592   virtual void SetMax(int64 m) {
04593     if (m < 0) {
04594       solver()->Fail();
04595     }
04596     if (late_cost_ != 0LL) {
04597       const int64 rb = late_date_ + PosIntDivDown(m, late_cost_);
04598       if (early_cost_ != 0LL) {
04599         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
04600         var_->SetRange(lb, rb);
04601       } else {
04602         var_->SetMax(rb);
04603       }
04604     } else {
04605       if (early_cost_ != 0LL) {
04606         const int64 lb = early_date_ - PosIntDivDown(m, early_cost_);
04607         var_->SetMin(lb);
04608       }
04609     }
04610   }
04611   virtual string name() const {
04612     return StringPrintf(
04613         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
04614         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
04615         var_->name().c_str(),
04616         early_cost_, early_date_, late_date_, late_cost_);
04617   }
04618   virtual string DebugString() const {
04619     return StringPrintf(
04620         "ConvexPiecewiseExpr(%s, ec = %" GG_LL_FORMAT "d, ed = %"
04621         GG_LL_FORMAT "d, ld = %" GG_LL_FORMAT "d, lc = %" GG_LL_FORMAT "d)",
04622         var_->DebugString().c_str(),
04623         early_cost_, early_date_, late_date_, late_cost_);
04624   }
04625   virtual void WhenRange(Demon* d) {
04626     var_->WhenRange(d);
04627   }
04628 
04629   virtual void Accept(ModelVisitor* const visitor) const {
04630     visitor->BeginVisitIntegerExpression(ModelVisitor::kConvexPiecewise, this);
04631     visitor->VisitIntegerExpressionArgument(ModelVisitor::kExpressionArgument,
04632                                             var_);
04633     visitor->VisitIntegerArgument(ModelVisitor::kEarlyCostArgument,
04634                                   early_cost_);
04635     visitor->VisitIntegerArgument(ModelVisitor::kEarlyDateArgument,
04636                                   early_date_);
04637     visitor->VisitIntegerArgument(ModelVisitor::kLateCostArgument,
04638                                   late_cost_);
04639     visitor->VisitIntegerArgument(ModelVisitor::kLateDateArgument,
04640                                   late_date_);
04641     visitor->EndVisitIntegerExpression(ModelVisitor::kConvexPiecewise, this);
