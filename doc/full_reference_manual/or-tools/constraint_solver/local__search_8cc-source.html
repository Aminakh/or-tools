00309 // A local search operator that compares the current assignment with a target
00310 // one, and that generates neighbors corresponding to a single variable being
00311 // changed from its current value to its target value.
00312 class MoveTowardTargetLS: public IntVarLocalSearchOperator {
00313  public:
00314   explicit MoveTowardTargetLS(const std::vector<IntVar*>& variables,
00315                               const std::vector<int64>& target_values)
00316       : IntVarLocalSearchOperator(variables.data(), variables.size()),
00317         target_(target_values),
00318         // Initialize variable_index_ at the number of the of variables minus
00319         // one, so that the first to be tried (after one increment) is the one
00320         // of index 0.
00321         variable_index_(Size() - 1) {
00322     CHECK_EQ(target_values.size(), variables.size()) << "Illegal arguments.";
00323   }

00325   virtual ~MoveTowardTargetLS() {}

00327   // Make a neighbor assigning one variable to its target value.
00328   virtual bool MakeNextNeighbor(Assignment* delta, Assignment* deltadelta) {
00329     CHECK_NOTNULL(delta);
00330     while (num_var_since_last_start_ < Size()) {
00331       ++num_var_since_last_start_;
00332       variable_index_ = (variable_index_ + 1) % Size();
00333       const int64 target_value = target_.at(variable_index_);
00334       const int64 current_value = OldValue(variable_index_);
00335       if (current_value != target_value) {
00336         RevertChanges(false);
00337         SetValue(variable_index_, target_value);
00338         if (ApplyChanges(delta, deltadelta)) {
00339           return true;
00340         }
00341       }
00342     }
00343     return false;
00344   }

00346  private:
00347   virtual void OnStart() {
00348     // Do not change the value of variable_index_: this way, we keep going from
00349     // where we last modified something. This is because we expect that most
00350     // often, the variables we have just checked are less likely to be able
00351     // to be changed to their target values than the ones we have not yet
00352     // checked.
00353     //
00354     // Consider the case where oddly indexed variables can be assigned to their
00355     // target values (no matter in what order they are considered), while even
00356     // indexed ones cannot. Restarting at index 0 each time an odd-indexed
00357     // variable is modified will cause a total of Theta(n^2) neighbors to be
00358     // generated, while not restarting will produce only Theta(n) neighbors.
00359     CHECK_GE(variable_index_, 0);
00360     CHECK_LT(variable_index_, Size());
00361     num_var_since_last_start_ = 0;
00362   }

01190 bool TSPLns::MakeNeighbor() {
01191   const int64 base_node = BaseNode(0);
01192   if (IsPathEnd(base_node)) {
01193     return false;
01194   }
01195   std::vector<int64> nodes;
01196   for (int64 node = StartNode(0); !IsPathEnd(node); node = Next(node)) {
01197     nodes.push_back(node);
01198   }
01199   if (nodes.size() <= tsp_size_) {
01200     return false;
01201   }
01202   // Randomly select break nodes (final nodes of a meta-node, after which
01203   // an arc is relaxed.
01204   hash_set<int64> breaks_set;
01205   // Always add base node to break nodes (diversification)
01206   breaks_set.insert(base_node);
01207   while (breaks_set.size() < tsp_size_) {
01208     const int64 one_break = nodes[rand_.Uniform(nodes.size())];
01209     if (!ContainsKey(breaks_set, one_break)) {
01210       breaks_set.insert(one_break);
01211     }
01212   }
01213   CHECK_EQ(breaks_set.size(), tsp_size_);
01214   // Setup break node indexing and internal meta-node cost (cost of partial
01215   // route starting at first node of the meta-node and ending at its last node);
01216   // this cost has to be added to the TSP matrix cost in order to respect the
01217   // triangle inequality.
01218   std::vector<int> breaks;
01219   std::vector<int64> meta_node_costs;
01220   int64 cost = 0;
01221   int64 node = StartNode(0);
01222   int64 node_path = Path(node);
01223   while (!IsPathEnd(node)) {
01224     int64 next = Next(node);
01225     if (ContainsKey(breaks_set, node)) {
01226       breaks.push_back(node);
01227       meta_node_costs.push_back(cost);
01228       cost = 0;
01229     } else {
01230       cost += evaluator_->Run(node, next, node_path);
01231     }
01232     node = next;
01233   }
01234   meta_node_costs[0] += cost;
01235   CHECK_EQ(breaks.size(), tsp_size_);
01236   // Setup TSP cost matrix
01237   CHECK_EQ(meta_node_costs.size(), tsp_size_);
01238   for (int i = 0; i < tsp_size_; ++i) {
01239     cost_[i][0] = meta_node_costs[i]
01240         + evaluator_->Run(breaks[i], Next(breaks[tsp_size_ - 1]), node_path);
01241     for (int j = 1; j < tsp_size_; ++j) {
01242       cost_[i][j] = meta_node_costs[i]
01243           + evaluator_->Run(breaks[i], Next(breaks[j - 1]), node_path);
01244     }
01245     cost_[i][i] = 0;
01246   }
01247   // Solve TSP and inject solution in delta (only if it leads to a new solution)
01248   hamiltonian_path_solver_.ChangeCostMatrix(cost_);
01249   std::vector<PathNodeIndex> path;
01250   hamiltonian_path_solver_.TravelingSalesmanPath(&path);
01251   bool nochange = true;
01252   for (int i = 0; i < path.size() - 1; ++i) {
01253     if (path[i] != i) {
01254       nochange = false;
01255       break;
01256     }
01257   }
01258   if (nochange) {
01259     return false;
01260   }
01261   CHECK_EQ(0, path[path.size() - 1]);
01262   for (int i = 0; i < tsp_size_ - 1; ++i) {
01263     SetNext(breaks[path[i]], OldNext(breaks[path[i + 1] - 1]), node_path);
01264   }
01265   SetNext(breaks[path[tsp_size_ - 1]],
01266           OldNext(breaks[tsp_size_ - 1]),
01267           node_path);
01268   return true;
01269 }

01327 void NearestNeighbors::ComputeNearest(int row) {
01328   // Find size_ nearest neighbors for row of index 'row'.
01329   const int path = path_operator_.Path(row);
01330   const IntVar* var = path_operator_.Var(row);
01331   const int64 var_min = var->Min();
01332   const int var_size = var->Max() - var_min + 1;
01333   scoped_array<int> neighbors(new int[var_size]);
01334   scoped_array<int64> row_data(new int64[var_size]);
01335   for (int i = 0; i < var_size; ++i) {
01336     const int index = i + var_min;
01337     neighbors[i] = index;
01338     row_data[i] = evaluator_->Run(row, index, path);
01339   }
01340 
01341   if (var_size > size_) {
01342     int start = 0;
01343     int end = var_size;
01344     int size = size_;
01345     while (size > 0) {
01346       int index = (end - start) / 2;
01347       Pivot(start, end, neighbors.get(), row_data.get(), &index);
01348       if (index - start >= size) {
01349         end = index;
01350       } else {
01351         start = index + 1;
01352         size -= start;
01353       }
01354     }
01355   }
01356 
01357   // Setup global neighbor matrix for row row_index
01358   for (int i = 0; i < std::min(size_, var_size); ++i) {
01359     neighbors_[row].push_back(neighbors[i]);
01360     std::sort(neighbors_[row].begin(), neighbors_[row].end());
01361   }
01362 }

01364 void NearestNeighbors::Pivot(int start,
01365                              int end,
01366                              int* neighbors,
01367                              int64* row,
01368                              int* index) {
01369   Swap(start, *index, neighbors, row);
01370   int j = start;
01371   for (int i = start + 1; i < end; ++i) {
01372     if (row[i] < row[j]) {
01373       Swap(j, i, neighbors, row);
01374       ++j;
01375       Swap(i, j, neighbors, row);
01376     }
01377   }
01378   *index = j;
01379 }

01381 void NearestNeighbors::Swap(int i, int j, int* neighbors, int64* row) {
01382   const int64 row_i = row[i];
01383   const int neighbor_i = neighbors[i];
01384   row[i] = row[j];
01385   neighbors[i] = neighbors[j];
01386   row[j] = row_i;
01387   neighbors[j] = neighbor_i;
01388 }

01440 bool LinKernighan::MakeNeighbor() {
01441   marked_.clear();
01442   int64 node = BaseNode(0);
01443   if (IsPathEnd(node)) return false;
01444   int64 path = Path(node);
01445   int64 base = node;
01446   int64 next = Next(node);
01447   if (IsPathEnd(next)) return false;
01448   int64 out = -1;
01449   int64 gain = 0;
01450   marked_.insert(node);
01451   if (topt_) {  // Try a 3opt first
01452     if (InFromOut(node, next, &out, &gain)) {
01453       marked_.insert(next);
01454       marked_.insert(out);
01455       const int64 node1 = out;
01456       if (IsPathEnd(node1)) return false;
01457       const int64 next1 = Next(node1);
01458       if (IsPathEnd(next1)) return false;
01459       if (InFromOut(node1, next1, &out, &gain)) {
01460         marked_.insert(next1);
01461         marked_.insert(out);
01462         if (MoveChain(out, node1, node)) {
01463           const int64 next_out = Next(out);
01464           int64 in_cost = evaluator_->Run(node, next_out, path);
01465           int64 out_cost = evaluator_->Run(out, next_out, path);
01466           if (gain - in_cost + out_cost > 0)
01467             return true;
01468           node = out;
01469           if (IsPathEnd(node)) {
01470             return false;
01471           }
01472           next = next_out;
01473           if (IsPathEnd(next)) {
01474             return false;
01475           }
01476         } else {
01477           return false;
01478         }
01479       } else {
01480         return false;
01481       }
01482     } else {
01483       return false;
01484     }
01485   }
01486   // Try 2opts
01487   while (InFromOut(node, next, &out, &gain)) {
01488     marked_.insert(next);
01489     marked_.insert(out);
01490     int64 chain_last;
01491     if (!ReverseChain(node, out, &chain_last)) {
01492       return false;
01493     }
01494     int64 in_cost = evaluator_->Run(base, chain_last, path);
01495     int64 out_cost = evaluator_->Run(chain_last, out, path);
01496     if (gain - in_cost + out_cost > 0) {
01497       return true;
01498     }
01499     node = chain_last;
01500     if (IsPathEnd(node)) {
01501       return false;
01502     }
01503     next = out;
01504     if (IsPathEnd(next)) {
01505       return false;
01506     }
01507   }
01508   return false;
01509 }

01511 const int LinKernighan::kNeighbors = 5 + 1;
01512 
01513 bool LinKernighan::InFromOut(int64 in_i, int64 in_j, int64* out, int64* gain) {
01514   const std::vector<int>& nexts = neighbors_.Neighbors(in_j);
01515   int64 best_gain = kint64min;
01516   int64 path = Path(in_i);
01517   int64 out_cost = evaluator_->Run(in_i, in_j, path);
01518   const int64 current_gain = *gain + out_cost;
01519   for (int k = 0; k < nexts.size(); ++k) {
01520     const int64 next = nexts[k];
01521     if (next != in_j) {
01522       int64 in_cost = evaluator_->Run(in_j, next, path);
01523       int64 new_gain = current_gain - in_cost;
01524       if (new_gain > 0
01525           && next != Next(in_j)
01526           && marked_.count(in_j) == 0
01527           && marked_.count(next) == 0) {
01528         if (best_gain < new_gain) {
01529           *out = next;
01530           best_gain = new_gain;
01531         }
01532       }
01533     }
01534   }
01535   *gain = best_gain;
01536   return (best_gain > kint64min);
01537 }

02207 bool ObjectiveFilter::Accept(const Assignment* delta,
02208                              const Assignment* deltadelta) {
02209   if (delta == NULL) {
02210     return false;
02211   }
02212   int64 value = 0;
02213   if (!deltadelta->Empty()) {
02214     if (!incremental_) {
02215       value = Evaluate(delta, old_value_, cache_, true);
02216     } else {
02217       value = Evaluate(deltadelta, old_delta_value_, delta_cache_, true);
02218     }
02219     incremental_ = true;
02220   } else {
02221     if (incremental_) {
02222       for (int i = 0; i < primary_vars_size_; ++i) {
02223         delta_cache_[i] = cache_[i];
02224       }
02225       old_delta_value_ = old_value_;
02226     }
02227     incremental_ = false;
02228     value = Evaluate(delta, old_value_, cache_, false);
02229   }
02230   old_delta_value_ = value;
02231   int64 var_min = objective_->Min();
02232   int64 var_max = objective_->Max();
02233   if (delta->Objective() == objective_) {
02234     var_min = std::max(var_min, delta->ObjectiveMin());
02235     var_max = std::min(var_max, delta->ObjectiveMax());
02236   }
02237   switch (filter_enum_) {
02238     case Solver::LE: {
02239       return value <= var_max;
02240     }
02241     case Solver::GE: {
02242       return value >= var_min;
02243     }
02244     case Solver::EQ: {
02245       return value <= var_max && value >= var_min;
02246     }
02247     default: {
02248       LOG(ERROR) << "Unknown local search filter enum value";
02249       return false;
02250     }
02251   }
02252 }

02254 void ObjectiveFilter::OnSynchronize() {
02255   op_->Init();
02256   for (int i = 0; i < primary_vars_size_; ++i) {
02257     const int64 obj_value = SynchronizedElementValue(i);
02258     cache_[i] = obj_value;
02259     delta_cache_[i] = obj_value;
02260     op_->Update(obj_value);
02261   }
02262   old_value_ = op_->value();
02263   old_delta_value_ = old_value_;
02264   incremental_ = false;
02265 }
02266 
02267 int64 ObjectiveFilter::Evaluate(const Assignment* delta,
02268                                 int64 current_value,
02269                                 const int64* const out_values,
02270                                 bool cache_delta_values) {
02271   if (current_value == kint64max) return current_value;
02272   op_->set_value(current_value);
02273   const Assignment::IntContainer& container = delta->IntVarContainer();
02274   const int size = container.Size();
02275   for (int i = 0; i < size; ++i) {
02276     const IntVarElement& new_element = container.Element(i);
02277     const IntVar* var = new_element.Var();
02278     int64 index = -1;
02279     if (FindIndex(var, &index) && index < primary_vars_size_) {
02280       op_->Remove(out_values[index]);
02281       int64 obj_value = 0LL;
02282       if (EvaluateElementValue(container, index, &i, &obj_value)) {
02283         op_->Update(obj_value);
02284         if (cache_delta_values) {
02285           delta_cache_[index] = obj_value;
02286         }
02287       }
02288     }
02289   }
02290   return op_->value();
02291 }

02613       ls_operator_(ls_operator),
02614       sub_decision_builder_(sub_decision_builder),
02615       limit_(NULL),
02616       original_limit_(limit),
02617       neighbor_found_(false),
02618       filters_(filters) {
02619   CHECK(NULL != assignment);
02620   CHECK(NULL != ls_operator);
02621 
02622   // If limit is NULL, default limit is 1 solution
02623   if (NULL == limit) {
02624     Solver* const solver = assignment_->solver();
02625     limit_ = solver->MakeLimit(kint64max, kint64max, kint64max, 1);

02631 Decision* FindOneNeighbor::Next(Solver* const solver) {
02632   CHECK(NULL != solver);
02633 
02634   if (original_limit_ != NULL) {
02635     limit_->Copy(original_limit_);
02636   }
02637 
02638   if (!neighbor_found_) {
02639     // Only called on the first call to Next(), reference_assignment_ has not
02640     // been synced with assignment_ yet
02641 
02642     // Keeping the code in case a performance problem forces us to
02643     // use the old code with a zero test on pool_.
02644     // reference_assignment_->Copy(assignment_);
02645     pool_->Initialize(assignment_);
02646     SynchronizeAll();
02647   }
02648 
02649   {
02650     // Another assignment is needed to apply the delta
02651     Assignment* assignment_copy = solver->RevAlloc(
02652             new Assignment(reference_assignment_.get()));
02653     int counter = 0;
02654 
02655     DecisionBuilder* restore =
02656         solver->MakeRestoreAssignment(assignment_copy);
02657     if (sub_decision_builder_) {
02658       restore = solver->Compose(restore, sub_decision_builder_);
02659     }
02660     Assignment* delta = solver->RevAlloc(new Assignment(solver));
02661     Assignment* deltadelta = solver->RevAlloc(new Assignment(solver));
02662     while (true) {
02663       delta->Clear();
02664       deltadelta->Clear();
02665       solver->TopPeriodicCheck();
02666       if (++counter >= FLAGS_cp_local_search_sync_frequency &&
02667           pool_->SyncNeeded(reference_assignment_.get())) {
02668         // TODO(user) : SyncNeed(assignment_) ?
02669         counter = 0;
02670         SynchronizeAll();
02671       }
02672 
02673       if (!limit_->Check()
02674           && ls_operator_->MakeNextNeighbor(delta, deltadelta)) {
02675         solver->neighbors_ += 1;
02676         // All filters must be called for incrementality reasons.
02677         // Empty deltas must also be sent to incremental filters; can be needed
02678         // to resync filters on non-incremental (empty) moves.
02679         // TODO(user): Don't call both if no filter is incremental and one
02680         // of them returned false.
02681         const bool mh_filter = solver->AcceptDelta(delta, deltadelta);
02682         const bool move_filter = FilterAccept(delta, deltadelta);
02683         if (mh_filter && move_filter) {
02684           solver->filtered_neighbors_ += 1;
02685           assignment_copy->Copy(reference_assignment_.get());
02686           assignment_copy->Copy(delta);
02687           if (solver->NestedSolve(restore, false)) {
02688             solver->accepted_neighbors_ += 1;
02689             assignment_->Store();
02690             neighbor_found_ = true;
02691             return NULL;
02692           }
02693         }
02694       } else {
02695         if (neighbor_found_) {
02696           solver->AcceptNeighbor();
02697           // Keeping the code in case a performance problem forces us to
02698           // use the old code with a zero test on pool_.
02699           //          reference_assignment_->Copy(assignment_);
02700           pool_->RegisterNewSolution(assignment_);
02701           SynchronizeAll();
02702         } else {
02703           break;
02704         }
02705       }
02706     }
02707   }
02708   solver->Fail();
02709   return NULL;
02710 }

02712 bool FindOneNeighbor::FilterAccept(const Assignment* delta,
02713                                    const Assignment* deltadelta) {
02714   bool ok = true;
02715   for (int i = 0; i < filters_.size(); ++i) {
02716     if (filters_[i]->IsIncremental()) {
02717       ok = filters_[i]->Accept(delta, deltadelta) && ok;
02718     } else {
02719       ok = ok && filters_[i]->Accept(delta, deltadelta);
02720     }
02721   }
02722   return ok;
02723 }

02726   pool_->GetNextSolution(reference_assignment_.get());
02727   neighbor_found_ = false;
02728   limit_->Init();
02729   ls_operator_->Start(reference_assignment_.get());
02730   SynchronizeFilters(reference_assignment_.get());
02731 }

02897 Decision* LocalSearch::Next(Solver* const solver) {
02898   CHECK(NULL != solver);
02899   CHECK_LT(0, nested_decisions_.size());
02900   if (!has_started_) {
02901     nested_decision_index_ = 0;
02902     solver->SaveAndSetValue(&has_started_, true);
02903   } else if (nested_decision_index_ < 0) {
02904     solver->Fail();
02905   }
02906   NestedSolveDecision* decision = nested_decisions_[nested_decision_index_];
02907   const int state = decision->state();
02908   switch (state) {
02909     case NestedSolveDecision::DECISION_FAILED: {
02910       if (!solver->LocalOptimum()) {
02911         nested_decision_index_ = -1;  // Stop the search
02912       }
02913       solver->Fail();
02914       return NULL;
02915     }
02916     case NestedSolveDecision::DECISION_PENDING: {
02917       const int depth = solver->SearchDepth();
02918       if (depth < FLAGS_cp_local_search_balanced_tree_depth) {
02919         return solver->balancing_decision();
02920       } else if (depth > FLAGS_cp_local_search_balanced_tree_depth) {
02921         solver->Fail();
02922       }
02923       return decision;
02924     }
02925     case NestedSolveDecision::DECISION_FOUND: {
02926       // Next time go to next decision
02927       if (nested_decision_index_ + 1 < nested_decisions_.size()) {
02928         ++nested_decision_index_;
02929       }
02930       return NULL;
02931     }
02932     default: {
02933       LOG(ERROR) << "Unknown local search state";
02934       return NULL;
02935     }
02936   }
02937   return NULL;
02938 }
