00898 // Backtrack is implemented using 3 primitives:
00899 // CP_TRY to start searching
00900 // CP_DO_FAIL to signal a failure. The program will continue on the CP_ON_FAIL
00901 // primitive.
00902 // CP_FAST_BACKTRACK protects an implementation of backtrack using
00903 // setjmp/longjmp.  The clean portable way is to use exceptions,
00904 // unfortunately, it can be much slower.  Thus we use ideas from
00905 // Prolog, CP/CLP implementations, continuations in C and implement failing
00906 // and backtracking using setjmp/longjmp.
00907 #define CP_FAST_BACKTRACK
00908 #if defined(CP_FAST_BACKTRACK)
00909 // We cannot use a method/function for this as we would lose the
00910 // context in the setjmp implementation.
00911 #define CP_TRY(search)                                              \
00912   CHECK(!search->jmpbuf_filled_) << "Fail() called outside search"; \
00913   search->jmpbuf_filled_ = true;                                    \
00914   if (setjmp(search->fail_buffer_) == 0)
00915 #define CP_ON_FAIL else
00916 #define CP_DO_FAIL(search) longjmp(search->fail_buffer_, 1)
00917 #else  // CP_FAST_BACKTRACK
00918 class FailException {};
00919 #define CP_TRY(search)                                                 \
00920   CHECK(!search->jmpbuf_filled_) << "Fail() called outside search";    \
00921   search->jmpbuf_filled_ = true;                                       \
00922   try
00923 #define CP_ON_FAIL catch(FailException&)
00924 #define CP_DO_FAIL(search) throw FailException()
00925 #endif  // CP_FAST_BACKTRACK

01612 void Solver::ProcessConstraints() {
01613   // Both constraints_list_ and additional_constraints_list_ are used in
01614   // a FIFO way.
01615   if (FLAGS_cp_print_model) {
01616     ModelVisitor* const visitor = MakePrintModelVisitor();
01617     Accept(visitor);
01618   }
01619   if (FLAGS_cp_model_stats) {
01620     ModelVisitor* const visitor = MakeStatisticsModelVisitor();
01621     Accept(visitor);
01622   }
01623 
01624   // Clear state before processing constraints.
01625   const int constraints_size = constraints_list_.size();
01626   additional_constraints_list_.clear();
01627   additional_constraints_parent_list_.clear();
01628 
01629   // Process constraints from the model.
01630   for (constraint_index_ = 0;
01631        constraint_index_ < constraints_size;
01632        ++constraint_index_) {
01633     Constraint* const constraint = constraints_list_[constraint_index_];
01634     if (parameters_.profile_level != SolverParameters::NO_PROFILING) {
01635       DemonMonitorStartInitialPropagation(demon_monitor_, constraint);
01636     }
01637     constraint->PostAndPropagate();
01638     if (parameters_.profile_level != SolverParameters::NO_PROFILING) {
01639       DemonMonitorEndInitialPropagation(demon_monitor_, constraint);
01640     }
01641   }
01642   CHECK_EQ(constraints_list_.size(), constraints_size);
01643 
01644   // Process nested constraints added during the previous step.
01645   for (int additional_constraint_index_ = 0;
01646        additional_constraint_index_ < additional_constraints_list_.size();
01647        ++additional_constraint_index_) {
01648     Constraint* const constraint =
01649         additional_constraints_list_[additional_constraint_index_];
01650     if (parameters_.profile_level != SolverParameters::NO_PROFILING) {
01651       DemonMonitorStartInitialPropagation(demon_monitor_, constraint);
01652     }
01653     constraint->PostAndPropagate();
01654     if (parameters_.profile_level != SolverParameters::NO_PROFILING) {
01655       DemonMonitorEndInitialPropagation(demon_monitor_, constraint);
01656     }
01657   }
01658 }

01809 // Backtrack to the last open right branch in the search tree.
01810 // It returns true in case the search tree has been completely explored.
01811 bool Solver::BacktrackOneLevel(Decision** fail_decision) {
01812   bool no_more_solutions = false;
01813   bool end_loop = false;
01814   while (!end_loop) {
01815     StateInfo info;
01816     MarkerType t = PopState(&info);
01817     switch (t) {
01818       case SENTINEL:
01819         CHECK_EQ(info.ptr_info, this) << "Wrong sentinel found";
01820         CHECK((info.int_info == ROOT_NODE_SENTINEL && searches_.size() == 1) ||
01821               (info.int_info == INITIAL_SEARCH_SENTINEL &&
01822                searches_.size() > 1));
01823         searches_.back()->sentinel_pushed_--;
01824         no_more_solutions = true;
01825         end_loop = true;
01826         break;
01827       case SIMPLE_MARKER:
01828         LOG(ERROR)
01829             << "Simple markers should not be encountered during search";
01830         break;
01831       case CHOICE_POINT:
01832         if (info.int_info == 0) {  // was left branch
01833           (*fail_decision) = reinterpret_cast<Decision*>(info.ptr_info);
01834           end_loop = true;
01835           searches_.back()->set_search_depth(info.depth);
01836           searches_.back()->set_search_left_depth(info.left_depth);
01837         }
01838         break;
01839       case REVERSIBLE_ACTION: {
01840         Action* d = reinterpret_cast<Action*>(info.ptr_info);
01841         d->Run(this);
01842         break;
01843       }
01844     }
01845   }
01846   Search* const search = searches_.back();
01847   search->EndFail();
01848   CallFailHooks();
01849   fail_stamp_++;
01850   if (no_more_solutions) {
01851     search->NoMoreSolutions();
01852   }
01853   return no_more_solutions;
01854 }

01857   StateInfo info(this, magic_code);
01858   PushState(SENTINEL, info);

01860   if (magic_code != SOLVER_CTOR_SENTINEL) {
01861     searches_.back()->sentinel_pushed_++;
01862   }
01863   const int pushed = searches_.back()->sentinel_pushed_;
01864   DCHECK((magic_code == SOLVER_CTOR_SENTINEL) ||

01867 }

01869 void Solver::RestartSearch() {
01870   Search* const search = searches_.back();
01871   CHECK_NE(0, search->sentinel_pushed_);
01872   if (searches_.size() == 1) {  // top level.
01873     if (search->sentinel_pushed_ > 1) {
01874       BacktrackToSentinel(ROOT_NODE_SENTINEL);
01875     }
01876     CHECK_EQ(1, search->sentinel_pushed_);
01877     PushSentinel(ROOT_NODE_SENTINEL);
01878     state_ = IN_SEARCH;
01879   } else {
01880     CHECK_EQ(IN_SEARCH, state_);
01881     if (search->sentinel_pushed_ > 0) {
01882       BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
01883     }
01884     CHECK_EQ(0, search->sentinel_pushed_);
01885     PushSentinel(INITIAL_SEARCH_SENTINEL);
01886   }
01887 
01888   if (parameters_.profile_level != SolverParameters::NO_PROFILING) {
01889     CHECK_NOTNULL(demon_monitor_);
01890     DemonMonitorRestartSearch(demon_monitor_);
01891   }
01892 
01893   search->RestartSearch();
01894 }

01896 // Backtrack to the initial search sentinel.
01897 // Does not change the state, this should be done by the caller.
01898 void Solver::BacktrackToSentinel(int magic_code) {
01899   Search* const search = searches_.back();
01900   bool end_loop = search->sentinel_pushed_ == 0;
01901   while (!end_loop) {
01902     StateInfo info;
01903     MarkerType t = PopState(&info);
01904     switch (t) {
01905       case SENTINEL: {
01906         CHECK_EQ(info.ptr_info, this) << "Wrong sentinel found";
01907         CHECK_GE(--search->sentinel_pushed_, 0);
01908         search->set_search_depth(0);
01909         search->set_search_left_depth(0);
01910 
01911         if (info.int_info == magic_code) {
01912           end_loop = true;
01913         }
01914         break;
01915       }
01916       case SIMPLE_MARKER:
01917         break;
01918       case CHOICE_POINT:
01919         break;
01920       case REVERSIBLE_ACTION: {
01921         Demon* d = reinterpret_cast<Demon*>(info.ptr_info);
01922         d->Run(this);
01923         break;
01924       }
01925     }
01926   }
01927   fail_stamp_++;
01928 }

01930 // Closes the current search without backtrack.
01931 void Solver::JumpToSentinelWhenNested() {
01932   CHECK_GT(searches_.size(), 1) << "calling JumpToSentinel from top level";
01933   Search* c = searches_.back();
01934   Search* p = searches_[searches_.size() - 2];
01935   bool found = false;
01936   while (!c->marker_stack_.empty()) {
01937     StateMarker* m = c->marker_stack_.back();
01938     if (m->type_ == REVERSIBLE_ACTION) {
01939       p->marker_stack_.push_back(m);
01940     } else {
01941       if (m->type_ == SENTINEL) {
01942         CHECK_EQ(c->marker_stack_.size(), 1) << "Sentinel found too early";
01943         found = true;
01944       }
01945       delete m;
01946     }
01947     c->marker_stack_.pop_back();
01948   }
01949   c->set_search_depth(0);
01950   c->set_search_left_depth(0);
01951   CHECK_EQ(found, true) << "Sentinel not found";
01952 }

01986 // Search for the next solution in the search tree.
01987 bool Solver::NextSolution() {
01988   Search* const search = searches_.back();
01989   Decision* fd = NULL;
01990   const bool top_level = (searches_.size() == 1);
01991 
01992   if (top_level && state_ == OUTSIDE_SEARCH && !search->decision_builder()) {
01993     LOG(WARNING) << "NextSolution() called without a NewSearch before";
01994     return false;
01995   }
01996 
01997   if (top_level) {  // Manage top level state.
01998     switch (state_) {
01999       case PROBLEM_INFEASIBLE:
02000         return false;
02001       case NO_MORE_SOLUTIONS:
02002         return false;
02003       case AT_SOLUTION: {
02004         if (BacktrackOneLevel(&fd)) {  // No more solutions.
02005           state_ = NO_MORE_SOLUTIONS;
02006           return false;
02007         }
02008         state_ = IN_SEARCH;
02009         break;
02010       }
02011       case OUTSIDE_SEARCH: {
02012         state_ = IN_ROOT_NODE;
02013         search->BeginInitialPropagation();
02014         CP_TRY(search) {
02015           ProcessConstraints();
02016           search->EndInitialPropagation();
02017           PushSentinel(ROOT_NODE_SENTINEL);
02018           state_ = IN_SEARCH;
02019           search->ClearBuffer();
02020         } CP_ON_FAIL {
02021           queue_->AfterFailure();
02022           BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02023           state_ = PROBLEM_INFEASIBLE;
02024           return false;
02025         }
02026         break;
02027       }
02028       case IN_SEARCH:  // Usually after a RestartSearch
02029         break;
02030       case IN_ROOT_NODE:
02031         LOG(FATAL) << "Should not happen";
02032         break;
02033     }
02034   }
02035 
02036   volatile bool finish = false;
02037   volatile bool result = false;
02038   DecisionBuilder* const db = search->decision_builder();
02039 
02040   while (!finish) {
02041     CP_TRY(search) {
02042       if (fd != NULL) {
02043         StateInfo i1(fd,
02044                      1,
02045                      search->search_depth(),
02046                      search->left_search_depth());  // 1 for right branch
02047         PushState(CHOICE_POINT, i1);
02048         search->RefuteDecision(fd);
02049         branches_++;
02050         fd->Refute(this);
02051         search->AfterDecision(fd, false);
02052         search->RightMove();
02053         fd = NULL;
02054       }
02055       Decision* d = NULL;
02056       for (;;) {
02057         search->BeginNextDecision(db);
02058         d = db->Next(this);
02059         search->EndNextDecision(db, d);
02060         if (d == fail_decision_) {
02061           Fail();  // fail now instead of after 2 branches.
02062         }
02063         if (d != NULL) {
02064           DecisionModification modification = search->ModifyDecision();
02065           switch (modification) {
02066             case SWITCH_BRANCHES: {
02067               d = RevAlloc(new ReverseDecision(d));
02068             }  // We reverse the decision and fall through the normal code.
02069             case NO_CHANGE: {
02070               decisions_++;
02071               StateInfo i2(d,
02072                            0,
02073                            search->search_depth(),
02074                            search->left_search_depth());  // 0 for left branch
02075               PushState(CHOICE_POINT, i2);
02076               search->ApplyDecision(d);
02077               branches_++;
02078               d->Apply(this);
02079               search->AfterDecision(d, true);
02080               search->LeftMove();
02081               break;
02082             }
02083             case KEEP_LEFT: {
02084               search->ApplyDecision(d);
02085               d->Apply(this);
02086               search->AfterDecision(d, true);
02087               break;
02088             }
02089             case KEEP_RIGHT: {
02090               search->RefuteDecision(d);
02091               d->Refute(this);
02092               search->AfterDecision(d, false);
02093               break;
02094             }
02095             case KILL_BOTH: {
02096               Fail();
02097             }
02098           }
02099         } else {
02100           break;
02101         }
02102       }
02103       if (search->AcceptSolution()) {
02104         search->IncrementSolutionCounter();
02105         if (!search->AtSolution() || !CurrentlyInSolve()) {
02106           result = true;
02107           finish = true;
02108         } else {
02109           Fail();
02110         }
02111       } else {
02112         Fail();
02113       }
02114     } CP_ON_FAIL {
02115       queue_->AfterFailure();
02116       if (search->should_finish()) {
02117         fd = NULL;
02118         BacktrackToSentinel(top_level ?
02119                             ROOT_NODE_SENTINEL :
02120                             INITIAL_SEARCH_SENTINEL);
02121         result = false;
02122         finish = true;
02123         search->set_should_finish(false);
02124         search->set_should_restart(false);
02125         // We do not need to push back the sentinel as we are exiting anyway.
02126       } else if (search->should_restart()) {
02127         fd = NULL;
02128         BacktrackToSentinel(top_level ?
02129                             ROOT_NODE_SENTINEL :
02130                             INITIAL_SEARCH_SENTINEL);
02131         search->set_should_finish(false);
02132         search->set_should_restart(false);
02133         PushSentinel(top_level ? ROOT_NODE_SENTINEL : INITIAL_SEARCH_SENTINEL);
02134         search->RestartSearch();
02135       } else {
02136         if (BacktrackOneLevel(&fd)) {  // no more solutions.
02137           result = false;
02138           finish = true;
02139         }
02140       }
02141     }
02142   }
02143   if (result) {
02144     search->ClearBuffer();
02145   }
02146   if (top_level) {  // Manage state after NextSolution().
02147     state_ = (result ? AT_SOLUTION : NO_MORE_SOLUTIONS);
02148   }
02149   return result;
02150 }

02152 void Solver::EndSearch() {
02153   CHECK_EQ(1, searches_.size());
02154   Search* const search = searches_.back();
02155   BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02156   search->ExitSearch();
02157   search->Clear();
02158   state_ = OUTSIDE_SEARCH;
02159 }

02161 bool Solver::CheckAssignment(Assignment* const solution) {
02162   CHECK(solution);
02163   if (state_ == IN_SEARCH || state_ == IN_ROOT_NODE) {
02164     LOG(FATAL) << "Use NestedSolve() inside search";
02165   }
02166   // Check state and go to OUTSIDE_SEARCH.
02167   Search* const search = searches_.back();
02168   search->set_created_by_solve(false);  // default behavior.
02169 
02170   BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02171   state_ = OUTSIDE_SEARCH;
02172 
02173   // Push monitors and enter search.
02174   search->EnterSearch();
02175 
02176   // Push sentinel and set decision builder.
02177   DCHECK_EQ(1, searches_.size());
02178   PushSentinel(INITIAL_SEARCH_SENTINEL);
02179   search->BeginInitialPropagation();
02180   CP_TRY(search) {
02181     state_ = IN_ROOT_NODE;
02182     DecisionBuilder * const restore = MakeRestoreAssignment(solution);
02183     restore->Next(this);
02184     ProcessConstraints();
02185     search->EndInitialPropagation();
02186     BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02187     search->ClearBuffer();
02188     state_ = OUTSIDE_SEARCH;
02189     return true;
02190   } CP_ON_FAIL {
02191     const int index = constraint_index_ < constraints_list_.size() ?
02192         constraint_index_ :
02193         additional_constraints_parent_list_[additional_constraint_index_];
02194     Constraint* const ct = constraints_list_[index];
02195     if (ct->name().empty()) {
02196       LOG(INFO) << "Failing constraint = " << ct->DebugString();
02197     } else {
02198       LOG(INFO) << "Failing constraint = " << ct->name() << ":"
02199                 << ct->DebugString();
02200     }
02201     queue_->AfterFailure();
02202     BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02203     state_ = PROBLEM_INFEASIBLE;
02204     return false;
02205   }
02206 }

02248 bool Solver::NestedSolve(DecisionBuilder* const db,
02249                          bool restore,
02250                          SearchMonitor* const * monitors,
02251                          int size) {
02252   Search new_search(this);
02253   for (int i = 0; i < size; ++i) {
02254     new_search.push_monitor(monitors[i]);
02255   }
02256   std::vector<SearchMonitor*> extras;
02257   db->AppendMonitors(this, &extras);
02258   for (ConstIter<std::vector<SearchMonitor*> > it(extras); !it.at_end(); ++it) {
02259     new_search.push_monitor(*it);
02260   }
02261   searches_.push_back(&new_search);
02262   searches_.back()->set_created_by_solve(true);  // Overwrites default.
02263   new_search.EnterSearch();
02264   PushSentinel(INITIAL_SEARCH_SENTINEL);
02265   new_search.set_decision_builder(db);
02266   bool res = NextSolution();
02267   if (res) {
02268     if (restore) {
02269       BacktrackToSentinel(INITIAL_SEARCH_SENTINEL);
02270     } else {
02271       JumpToSentinelWhenNested();
02272     }
02273   }
02274   new_search.ExitSearch();
02275   new_search.Clear();
02276   searches_.pop_back();
02277   return res;
02278 }
