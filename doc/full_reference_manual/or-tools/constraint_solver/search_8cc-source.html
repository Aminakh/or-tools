00757 IntVar* PathSelector::Select(Solver* const s, int64* id) {
00758   *id = first_;

00760     return NULL;
00761   }
00762   int count = 0;
00763   while (vars_[*id]->Bound()) {

00766       return NULL;

00778 bool PathSelector::UpdateIndex(int64* index) const {

00782     }

00787 // Pick an unbound variable to which no other variable can point: it will be
00788 // a good start for a path. If none is found pick the first unbound one.
00789 bool PathSelector::FindPathStart(int64* index) const {
00790   // Pick path start
00791   for (int i = size_ - 1; i >= 0; --i) {
00792     if (!vars_[i]->Bound()) {
00793       bool has_possible_prev = false;
00794       for (int j = 0; j < size_; ++j) {
00795         if (vars_[j]->Contains(i)) {
00796           has_possible_prev = true;
00797           break;
00798         }
00799       }
00800       if (!has_possible_prev) {
00801         *index = i;
00802         return true;
00803       }
00804     }
00805   }
00806   // Pick first unbound
00807   for (int i = 0; i < size_; ++i) {
00808     if (!vars_[i]->Bound()) {
00809       *index = i;
00810       return true;
00811     }
00812   }
00813   return false;
00814 }

00856   const int64 span = v->Max() - v->Min() + 1;
00857   const int64 size = v->Size();
00858   Solver* const s = v->solver();
00859   if (size > span / 4) {  // Dense enough, we can try to find the
00860     // value randomly.
00861     for (;;) {
00862       const int64 value = v->Min() + s->Rand64(span);
00863       if (v->Contains(value)) {
00864         return value;
00865       }
00866     }
00867   } else {  // Not dense enough, we will count.
00868     int64 index = s->Rand64(size);
00869     if (index <= size / 2) {
00870       for (int64 i = v->Min(); i <= v->Max(); ++i) {
00871         if (v->Contains(i)) {
00872           if (--index == 0) {
00873             return i;
00874           }
00875         }
00876       }
00877       CHECK_LE(index, 0);
00878     } else {
00879       for (int64 i = v->Max(); i > v->Min(); --i) {
00880         if (v->Contains(i)) {
00881           if (--index == 0) {
00882             return i;
00883           }
00884         }
00885       }
00886       CHECK_LE(index, 0);
00887     }
00888   }
00889   return 0LL;

02596 void TabuSearch::ApplyDecision(Decision* const d) {
02597   Solver* const s = solver();
02598   if (d == s->balancing_decision()) {
02599     return;
02600   }
02601   // Aspiration criterion
02602   // Accept a neighbor if it improves the best solution found so far
02603   IntVar* aspiration = s->MakeBoolVar();
02604   if (maximize_) {
02605     s->AddConstraint(s->MakeIsGreaterOrEqualCstCt(objective_,
02606                                                   best_ + step_,
02607                                                   aspiration));
02608   } else {
02609     s->AddConstraint(s->MakeIsLessOrEqualCstCt(objective_,
02610                                                best_ - step_,
02611                                                aspiration));
02612   }
02613 
02614   // Tabu criterion
02615   // A variable in the "keep" list must keep its value, a variable in the
02616   // "forbid" list must not take its value in the list. The tabu criterion is
02617   // softened by the tabu factor which gives the number of violations to
02618   // the tabu criterion which is tolerated; a factor of 1 means no violations
02619   // allowed, a factor of 0 means all violations allowed.
02620   std::vector<IntVar*> tabu_vars;
02621   for (ConstIter<TabuList > it(keep_tabu_list_); !it.at_end(); ++it) {
02622     IntVar* tabu_var = s->MakeBoolVar();
02623     Constraint* keep_cst =
02624         s->MakeIsEqualCstCt((*it).var_, (*it).value_, tabu_var);
02625     s->AddConstraint(keep_cst);
02626     tabu_vars.push_back(tabu_var);
02627   }
02628   for (ConstIter<TabuList > it(forbid_tabu_list_); !it.at_end(); ++it) {
02629     IntVar* tabu_var = s->MakeBoolVar();
02630     Constraint* forbid_cst =
02631         s->MakeIsDifferentCstCt((*it).var_, (*it).value_, tabu_var);
02632     s->AddConstraint(forbid_cst);
02633     tabu_vars.push_back(tabu_var);
02634   }
02635   if (tabu_vars.size() > 0) {
02636     IntVar* tabu = s->MakeBoolVar();
02637     s->AddConstraint(s->MakeIsGreaterOrEqualCstCt(s->MakeSum(tabu_vars)->Var(),
02638                                                   tabu_vars.size() *
02639                                                   tabu_factor_,
02640                                                   tabu));
02641     s->AddConstraint(s->MakeGreaterOrEqual(s->MakeSum(aspiration, tabu), 1LL));
02642   }
02643 
02644   // Go downhill to the next local optimum
02645   if (maximize_) {
02646     const int64 bound = (current_ > kint64min) ? current_ + step_ : current_;
02647     s->AddConstraint(s->MakeGreaterOrEqual(objective_, bound));
02648   } else {
02649     const int64 bound = (current_ < kint64max) ? current_ - step_ : current_;
02650     s->AddConstraint(s->MakeLessOrEqual(objective_, bound));
02651   }
02652 
02653   // Avoid cost plateau's which lead to tabu cycles
02654   if (found_initial_solution_) {
02655     s->AddConstraint(s->MakeNonEquality(objective_, last_));
02656   }
02657 }

02659 bool TabuSearch::AtSolution() {
02660   if (!Metaheuristic::AtSolution()) {
02661     return false;
02662   }
02663   found_initial_solution_ = true;
02664   last_ = current_;
02665 
02666   // New solution found: add new assignments to tabu lists; this is only
02667   // done after the first local optimum (stamp_ != 0)
02668   if (0 != stamp_) {
02669     for (int i = 0; i < size_; ++i) {
02670       IntVar* const var = vars_[i];
02671       const int64 old_value = assignment_.Value(var);
02672       const int64 new_value = var->Value();
02673       if (old_value != new_value) {
02674         VarValue keep_value(var, new_value, stamp_);
02675         keep_tabu_list_.push_front(keep_value);
02676         VarValue forbid_value(var, old_value, stamp_);
02677         forbid_tabu_list_.push_front(forbid_value);
02678       }
02679     }
02680   }
02681   assignment_.Store();
02682 
02683   return true;
02684 }

02688   if (maximize_) {
02689     current_ = kint64min;
02690   } else {
02691     current_ = kint64max;
02692   }
02693   return true;
02694 }

02699   }

02987 GuidedLocalSearch::GuidedLocalSearch(Solver* const s,
02988                                      IntVar* objective,
02989                                      bool maximize,
02990                                      int64 step,
02991                                      const IntVar* const* vars,
02992                                      int size,
02993                                      double penalty_factor)
02994     : Metaheuristic(s, maximize, objective, step),
02995       penalized_objective_(NULL),
02996       assignment_(s),
02997       assignment_penalized_value_(0),
02998       old_penalized_value_(0),

03004     vars_.reset(new IntVar*[size_]);
03005     memcpy(vars_.get(), vars, size_ * sizeof(*vars));
03006     assignment_.Add(vars_.get(), size);
03007     current_penalized_values_.reset(new int64[size]);
03008     delta_cache_.reset(new int64[size]);
03009     memset(current_penalized_values_.get(), 0,
03010            size_ * sizeof(*current_penalized_values_.get()));
03011   }
03012   for (int i = 0; i < size; ++i) {
03013     indices_[vars_[i]] = i;

03022 // Add the following constraint (includes aspiration criterion):
03023 // if minimizing,
03024 //      objective =< Max(current penalized cost - penalized_objective - step,
03025 //                       best solution cost - step)
03026 // if maximizing,
03027 //      objective >= Min(current penalized cost - penalized_objective + step,
03028 //                       best solution cost + step)
03029 void GuidedLocalSearch::ApplyDecision(Decision* const d) {
03030   if (d == solver()->balancing_decision()) {
03031     return;
03032   }
03033   std::vector<IntVar*> elements;
03034   assignment_penalized_value_ = 0;
03035   if (penalties_->HasValues()) {
03036     for (int i = 0; i < size_; ++i) {
03037       IntExpr* expr = MakeElementPenalty(i);
03038       elements.push_back(expr->Var());
03039       const int64 penalty = AssignmentElementPenalty(assignment_, i);
03040       current_penalized_values_[i] = penalty;
03041       delta_cache_[i] = penalty;
03042       assignment_penalized_value_ += penalty;
03043     }
03044     old_penalized_value_ = assignment_penalized_value_;
03045     incremental_ = false;
03046     penalized_objective_ = solver()->MakeSum(elements)->Var();
03047     if (maximize_) {
03048       IntExpr* min_pen_exp = solver()->MakeDifference(current_ + step_,
03049                                                       penalized_objective_);
03050       IntVar* min_exp = solver()->MakeMin(min_pen_exp, best_ + step_)->Var();
03051       solver()->AddConstraint(solver()->MakeGreaterOrEqual(objective_,
03052                                                            min_exp));
03053     } else {
03054       IntExpr* max_pen_exp = solver()->MakeDifference(current_ - step_,
03055                                                       penalized_objective_);
03056       IntVar* max_exp = solver()->MakeMax(max_pen_exp, best_ - step_)->Var();
03057       solver()->AddConstraint(solver()->MakeLessOrEqual(objective_, max_exp));
03058     }
03059   } else {
03060     penalized_objective_ = NULL;
03061     if (maximize_) {
03062       const int64 bound = (current_ > kint64min) ? current_ + step_ : current_;
03063       objective_->SetMin(bound);
03064     } else {
03065       const int64 bound = (current_ < kint64max) ? current_ - step_ : current_;
03066       objective_->SetMax(bound);
03067     }
03068   }
03069 }
03070 
03071 bool GuidedLocalSearch::AtSolution() {
03072   if (!Metaheuristic::AtSolution()) {
03073     return false;
03074   }
03075   if (penalized_objective_ != NULL) {  // In case no move has been found
03076     current_ += penalized_objective_->Value();
03077   }
03078   assignment_.Store();
03079   return true;
03080 }

03082 // GLS filtering; compute the penalized value corresponding to the delta and
03083 // modify objective bound accordingly.
03084 bool GuidedLocalSearch::AcceptDelta(Assignment* delta, Assignment* deltadelta) {
03085   if ((delta != NULL || deltadelta != NULL) && penalties_->HasValues()) {
03086     int64 penalty = 0;
03087     if (!deltadelta->Empty()) {
03088       if (!incremental_) {
03089         penalty = Evaluate(delta,
03090                            assignment_penalized_value_,
03091                            current_penalized_values_.get(),
03092                            true);
03093       } else {
03094         penalty = Evaluate(deltadelta,
03095                            old_penalized_value_,
03096                            delta_cache_.get(),
03097                            true);
03098       }
03099       incremental_ = true;
03100     } else {
03101       if (incremental_) {
03102         for (int i = 0; i < size_; ++i) {
03103           delta_cache_[i] = current_penalized_values_[i];
03104         }
03105         old_penalized_value_ = assignment_penalized_value_;
03106       }
03107       incremental_ = false;
03108       penalty = Evaluate(delta,
03109                          assignment_penalized_value_,
03110                          current_penalized_values_.get(),
03111                          false);
03112     }
03113     old_penalized_value_ = penalty;
03114     if (!delta->HasObjective()) {
03115       delta->AddObjective(objective_);
03116     }
03117     if (delta->Objective() == objective_) {
03118       if (maximize_) {
03119         delta->SetObjectiveMin(std::max(std::min(current_ + step_ - penalty,
03120                                                  best_ + step_),
03121                                         delta->ObjectiveMin()));
03122       } else {
03123         delta->SetObjectiveMax(std::min(std::max(current_ - step_ - penalty,
03124                                                  best_ - step_),
03125                                         delta->ObjectiveMax()));
03126       }
03127     }
03128   }
03129   return true;
03130 }

03132 int64 GuidedLocalSearch::Evaluate(const Assignment* delta,
03133                                   int64 current_penalty,
03134                                   const int64* const out_values,
03135                                   bool cache_delta_values) {
03136   int64 penalty = current_penalty;
03137   const Assignment::IntContainer& container = delta->IntVarContainer();
03138   const int size = container.Size();
03139   for (int i = 0; i < size; ++i) {
03140     const IntVarElement& new_element = container.Element(i);
03141     const IntVar* var = new_element.Var();
03142     int64 index = -1;
03143     if (FindCopy(indices_, var, &index)) {
03144       penalty -= out_values[index];
03145       int64 new_penalty = 0;
03146       if (EvaluateElementValue(container, index, &i, &new_penalty)) {
03147         penalty += new_penalty;
03148         if (cache_delta_values) {
03149           delta_cache_[index] = new_penalty;
03150         }
03151       }
03152     }
03153   }
03154   return penalty;
03155 }

03157 // Penalize all the most expensive arcs (var, value) according to their utility:
03158 // utility(i, j) = cost(i, j) / (1 + penalty(i, j))
03159 bool GuidedLocalSearch::LocalOptimum() {
03160   std::vector<std::pair<Arc, double> > utility(size_);
03161   for (int i = 0; i < size_; ++i) {
03162     const int64 var_value = assignment_.Value(vars_[i]);
03163     const int64 value =
03164         (var_value != i) ? AssignmentPenalty(assignment_, i, var_value) : 0;
03165     const Arc arc(i, var_value);
03166     const int64 penalty = penalties_->Value(arc);
03167     utility[i] = std::pair<Arc, double>(arc, value / (penalty + 1.0));
03168   }
03169   Comparator comparator;
03170   std::sort(utility.begin(), utility.end(), comparator);
03171   int64 utility_value = utility[0].second;
03172   penalties_->Increment(utility[0].first);
03173   for (int i = 1;
03174        i < utility.size() && utility_value == utility[i].second;
03175        ++i) {
03176     penalties_->Increment(utility[i].first);
03177   }
03178   if (maximize_) {
03179     current_ = kint64min;
03180   } else {
03181     current_ = kint64max;
03182   }
03183   return true;
03184 }

03378 // Penalized value for (i, j) = penalty_factor_ * penalty(i, j) * cost (i, j)
03379 int64 TernaryGuidedLocalSearch::PenalizedValue(int64 i, int64 j, int64 k) {
03380   const Arc arc(i, j);
03381   const int64 penalty = penalties_->Value(arc);
03382   if (penalty != 0) {  // objective_function_->Run(i, j, k) can be costly
03383     const int64 penalized_value =
03384         penalty_factor_ * penalty * objective_function_->Run(i, j, k);
03385     if (maximize_) {
03386       return -penalized_value;
03387     } else {
03388       return penalized_value;
03389     }
03390   } else {
03391     return 0;
03392   }
03393 }

03400   int hint_index = *container_index + 1;

03402       && secondary_var == container.Element(hint_index).Var()) {
03403     *container_index = hint_index;
03404     return container.Element(hint_index).Value();
03405   } else {

03670 bool RegularLimit::CheckTime() {
03671   const int64 kMaxSkip = 100;
03672   const int64 kCheckWarmupIterations = 100;
03673   ++check_count_;
03674   if (wall_time_ != kint64max && next_check_ <= check_count_) {
03675     Solver* const s = solver();
03676     int64 time_delta = s->wall_time() -  wall_time_offset_;
03677     if (smart_time_check_
03678         && check_count_ > kCheckWarmupIterations
03679         && time_delta > 0) {
03680       int64 approximate_calls = (wall_time_ * check_count_) / time_delta;
03681       next_check_ = check_count_ + std::min(kMaxSkip, approximate_calls);
03682     }
03683     return time_delta >= wall_time_;
03684   } else {
03685     return false;
03686   }
03687 }

04102   // Luby Strategy
04103 int64 NextLuby(int i) {
04104   DCHECK_GT(i, 0);
04105   DCHECK_LT(i, kint32max);
04106   int64 power;
04107 
04108   // let's find the least power of 2 >= (i+1).
04109   power = 2;
04110   // Cannot overflow, because bounded by kint32max + 1.
04111   while (power < (i + 1)) {
04112     power <<= 1;
04113   }
04114   if (power == i + 1) {
04115     return (power / 2);
04116   }
04117   return NextLuby(i - (power / 2) + 1);
04118 }

04189 // ---------- Symmetry Breaking ----------
04190 
04191 // The symmetry manager maintains a list of problem symmetries.  Each
04192 // symmetry is called on each decision and should return a term
04193 // representing the boolean status of the symmetrical decision.
04194 // e.g. : the decision is x == 3, the symmetrical decision is y == 5
04195 // then the symmetry breaker should use
04196 // AddIntegerVariableEqualValueClause(y, 5).  Once this is done, upon
04197 // refutation, for each symmetry breaker, the system adds a constraint
04198 // that will forbid the symmetrical variation of the current explored
04199 // search tree. This constraint can be expressed very simply just by
04200 // keeping the list of current symmetrical decisions.
04201 //
04202 // This is called Symmetry Breaking During Search (Ian Gent, Barbara
04203 // Smith, ECAI 2000).
04204 // http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.42.3788&rep=rep1&type=pdf  // NOLINT
04205 class SymmetryManager : public SearchMonitor {
04206  public:
04207   SymmetryManager(Solver* const s,
04208                   SymmetryBreaker* const * visitors,
04209                   int size)
04210       : SearchMonitor(s),
04211         visitors_(new SymmetryBreaker*[size]),
04212         clauses_(new SimpleRevFIFO<IntVar*>[size]),
04213         decisions_(new SimpleRevFIFO<Decision*>[size]),
04214         directions_(new SimpleRevFIFO<bool>[size]),  // false = left.
04215         size_(size) {
04216     CHECK_GT(size, 0);
04217     memcpy(visitors_.get(), visitors, size_ * sizeof(*visitors));
04218     for (int i = 0; i < size_; ++i) {
04219       visitors_[i]->set_symmetry_manager_and_index(this, i);
04220     }
04221   }
04222 
04223   virtual ~SymmetryManager() {}
04224 
04225   virtual void EndNextDecision(DecisionBuilder* const db, Decision* const d) {
04226     if (d) {
04227       for (int i = 0; i < size_; ++i) {
04228         const void* const last = clauses_[i].Last();
04229         d->Accept(visitors_[i]);
04230         if (last != clauses_[i].Last()) {
04231           // Synchroneous push of decision as marker.
04232           decisions_[i].Push(solver(), d);
04233           directions_[i].Push(solver(), false);
04234         }
04235       }
04236     }
04237   }
04238 
04239   virtual void RefuteDecision(Decision* d) {
04240     for (int i = 0; i < size_; ++i) {
04241       if (decisions_[i].Last() != NULL && decisions_[i].LastValue() == d) {
04242         CheckSymmetries(i);
04243       }
04244     }
04245   }
04246 
04247   // TODO(user) : Improve speed, cache previous min and build them
04248   // incrementally.
04249   void CheckSymmetries(int index) {
04250     SimpleRevFIFO<IntVar*>::Iterator tmp(&clauses_[index]);
04251     SimpleRevFIFO<bool>::Iterator tmp_dir(&directions_[index]);
04252     Constraint* ct = NULL;
04253     {
04254       std::vector<IntVar*> guard;
04255       // keep the last entry for later, if loop doesn't exit.
04256       ++tmp;
04257       ++tmp_dir;
04258       while (tmp.ok()) {
04259         IntVar* const term = *tmp;
04260         if (!*tmp_dir) {
04261           if (term->Max() == 0) {
04262             // Premise is wrong. The clause will never apply.
04263             return;
04264           }
04265           if (term->Min() == 0) {
04266             DCHECK_EQ(1, term->Max());
04267             // Premise may be true. Adding to guard vector.
04268             guard.push_back(term);
04269           }
04270         }
04271         ++tmp;
04272         ++tmp_dir;
04273       }
04274       guard.push_back(clauses_[index].LastValue());
04275       directions_[index].SetLastValue(true);
04276       // Given premises: xi = ai
04277       // and a term y != b
04278       // The following is equivalent to
04279       // And(xi == a1) => y != b.
04280       ct = solver()->MakeEquality(solver()->MakeMin(guard), Zero());
04281     }
04282     DCHECK(ct != NULL);
04283     solver()->AddConstraint(ct);
04284   }
04285 
04286   void AddTermToClause(SymmetryBreaker* const visitor, IntVar* const term) {
04287     clauses_[visitor->index_in_symmetry_manager()].Push(solver(), term);
04288   }

04292   scoped_array<SimpleRevFIFO<IntVar*> > clauses_;
04293   scoped_array<SimpleRevFIFO<Decision*> > decisions_;
04294   scoped_array<SimpleRevFIFO<bool> > directions_;
04295   const int size_;
04296 };
