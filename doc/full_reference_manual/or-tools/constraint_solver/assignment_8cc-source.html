00259 template <class V, class E>
00260 void IdToElementMap(AssignmentContainer<V, E>* container,
00261                     hash_map<string, E*>* id_to_element_map) {
00262   CHECK(id_to_element_map != NULL);
00263   id_to_element_map->clear();
00264   for (int i = 0; i < container->Size(); ++i) {
00265     E& element = container->MutableElement(i);
00266     const V* const var = element.Var();
00267     const string& name = var->name();
00268     if (name.empty()) {
00269       LOG(INFO) << "Cannot save/load variables with empty name"
00270                 << "; variable will be ignored";
00271     } else if (ContainsKey(*id_to_element_map, name)) {
00272       LOG(INFO) << "Cannot save/load variables with duplicate names: " << name
00273                 << "; variable will be ignored";
00274     } else {
00275       (*id_to_element_map)[name] = &element;
00276     }
00277   }
00278 }

00341   }
00342   fast_load =
00343       (interval_var_container_.Size()

00766   int_var_container_.Copy(assignment->int_var_container_);
00767   interval_var_container_.Copy(assignment->interval_var_container_);
00768   if (assignment->HasObjective() && HasObjective()) {
00769     IntVarElement* obj_element = assignment->obj_element_;
00770     obj_element_->SetRange(obj_element->Min(), obj_element->Max());
00771     if (obj_element->Activated())
00772       obj_element_->Activate();
00773     else
00774       obj_element_->Deactivate();
00775   }
