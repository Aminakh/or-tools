01393   // Creates a Tabu Search monitor.
01394   // In the context of local search the behavior is similar to MakeOptimize(),
01395   // creating an objective in a given sense. The behavior differs once a local
01396   // optimum is reached: thereafter solutions which degrade the value of the
01397   // objective are allowed if they are not "tabu". A solution is "tabu" if it

03638 template <class V, class E> class AssignmentContainer {
03639  public:
03640   AssignmentContainer() {}
03641   E& Add(V* const var) {
03642     CHECK_NOTNULL(var);
03643     int index = -1;
03644     if (!Find(var, &index)) {
03645       return FastAdd(var);
03646     } else {
03647       return elements_[index];
03648     }
03649   }
03650   // Adds element without checking its presence in the container.
03651   E& FastAdd(V* const var) {
03652     DCHECK(var != NULL);
03653     E e(var);
03654     elements_.push_back(e);
03655     if (elements_map_.size() != 0) {
03656       CopyToMap();
03657     }
03658     return elements_.back();
03659   }
03660   void Clear() {
03661     elements_.clear();
03662     elements_map_.clear();
03663   }
03664   bool Empty() const {
03665     return elements_.empty();
03666   }
03667   // Copies intersection of containers.
03668   void Copy(const AssignmentContainer<V, E>& container) {
03669     for (int i = 0; i < container.elements_.size(); ++i) {
03670       const E& element = container.elements_[i];
03671       const V* const var = element.Var();
03672       int index = -1;
03673       if (i < elements_.size() && elements_[i].Var() == var) {
03674         index = i;
03675       } else if (!Find(var, &index)) {
03676         continue;
03677       }
03678       DCHECK_GE(index, 0);
03679       E& local_element(elements_[index]);
03680       local_element.Copy(element);
03681       if (element.Activated()) {
03682         local_element.Activate();
03683       } else {
03684         local_element.Deactivate();
03685       }
03686     }
03687   }
03688   bool Contains(const V* const var) const {
03689     int index;
03690     return Find(var, &index);
03691   }
03692   E& MutableElement(const V* const var) {
03693     int index = -1;
03694     const bool found = Find(var, &index);
03695     DCHECK(found);
03696     return MutableElement(index);
03697   }
03698   const E& Element(const V* const var) const {
03699     int index = -1;
03700     const bool found = Find(var, &index);
03701     DCHECK(found);
03702     return Element(index);
03703   }
03704   const std::vector<E>& elements() const { return elements_; }
03705   E& MutableElement(int index) { return elements_[index]; }
03706   const E& Element(int index) const { return elements_[index]; }
03707   int Size() const { return elements_.size(); }
03708   void Store() {
03709     for (int i = 0; i < elements_.size(); ++i) {
03710       elements_[i].Store();
03711     }
03712   }
03713   void Restore() {
03714     for (int i = 0; i < elements_.size(); ++i) {
03715       E& element = elements_[i];
03716       if (element.Activated()) {
03717         element.Restore();
03718       }
03719     }
03720   }
03721 
03722  private:
03723   void CopyToMap() {
03724     for (int i = elements_map_.size(); i < elements_.size(); ++i) {
03725       elements_map_[elements_[i].Var()] = i;
03726     }
03727   }
03728   bool Find(const V* const var, int* index) const {
03729     if (elements_map_.size() == 0) {
03730       const_cast<AssignmentContainer<V, E>*>(this)->CopyToMap();
03731     }
03732     DCHECK_EQ(elements_map_.size(), elements_.size());
03733     return FindCopy(elements_map_, var, index);
03734   }
03735 
03736   std::vector<E> elements_;
03737   hash_map<const V*, int> elements_map_;
03738 };
