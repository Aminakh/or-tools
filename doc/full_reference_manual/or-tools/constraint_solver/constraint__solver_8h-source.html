03104 template <class V, class E> class AssignmentContainer {
03105  public:
03106   AssignmentContainer() {}
03107   E& Add(V* const var) {
03108     CHECK_NOTNULL(var);
03109     int index = -1;
03110     if (!Find(var, &index)) {
03111       return FastAdd(var);
03112     } else {
03113       return elements_[index];
03114     }
03115   }
03116   // Adds element without checking its presence in the container.
03117   E& FastAdd(V* const var) {
03118     DCHECK(var != NULL);
03119     E e(var);
03120     elements_.push_back(e);
03121     if (elements_map_.size() != 0) {
03122       CopyToMap();
03123     }
03124     return elements_.back();
03125   }
03126   void Clear() {
03127     elements_.clear();
03128     elements_map_.clear();
03129   }
03130   bool Empty() const {
03131     return elements_.empty();
03132   }
03133   // Copies intersection of containers.
03134   void Copy(const AssignmentContainer<V, E>& container) {
03135     for (int i = 0; i < container.elements_.size(); ++i) {
03136       const E& element = container.elements_[i];
03137       const V* const var = element.Var();
03138       int index = -1;
03139       if (i < elements_.size() && elements_[i].Var() == var) {
03140         index = i;
03141       } else if (!Find(var, &index)) {
03142         continue;
03143       }
03144       DCHECK_GE(index, 0);
03145       E& local_element(elements_[index]);
03146       local_element.Copy(element);
03147       if (element.Activated()) {
03148         local_element.Activate();
03149       } else {
03150         local_element.Deactivate();
03151       }
03152     }
03153   }
03154   bool Contains(const V* const var) const {
03155     int index;
03156     return Find(var, &index);
03157   }
03158   E& MutableElement(const V* const var) {
03159     int index = -1;
03160     const bool found = Find(var, &index);
03161     DCHECK(found);
03162     return MutableElement(index);
03163   }
03164   const E& Element(const V* const var) const {
03165     int index = -1;
03166     const bool found = Find(var, &index);
03167     DCHECK(found);
03168     return Element(index);
03169   }
03170   E& MutableElement(int index) { return elements_[index]; }
03171   const E& Element(int index) const { return elements_[index]; }
03172   int Size() const { return elements_.size(); }
03173   void Store() {
03174     for (int i = 0; i < elements_.size(); ++i) {
03175       elements_[i].Store();
03176     }
03177   }
03178   void Restore() {
03179     for (int i = 0; i < elements_.size(); ++i) {
03180       E& element = elements_[i];
03181       if (element.Activated()) {
03182         element.Restore();
03183       }
03184     }
03185   }
03186  private:
03187   void CopyToMap() {
03188     for (int i = elements_map_.size(); i < elements_.size(); ++i) {
03189       elements_map_[elements_[i].Var()] = i;
03190     }
03191   }
03192   bool Find(const V* const var, int* index) const {
03193     if (elements_map_.size() == 0) {
03194       const_cast<AssignmentContainer<V, E>*>(this)->CopyToMap();
03195     }
03196     DCHECK_EQ(elements_map_.size(), elements_.size());
03197     return FindCopy(elements_map_, var, index);
03198   }
03199 
03200   vector<E> elements_;
03201   hash_map<const V*, int> elements_map_;
03202 };
