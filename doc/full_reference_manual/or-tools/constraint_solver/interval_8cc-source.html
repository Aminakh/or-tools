00093 // An IntervalVar that passes all function calls to an underlying interval
00094 // variable as long as it is not prohibited, and that interprets prohibited
00095 // intervals as intervals of duration 0 that must be executed between
00096 // [kMinValidValue and kMaxValidValue].

00098 // Such interval variables have a very similar behavior to others.

00102 // values returned by the start/end getters. For example, during a given
00103 // propagation, three successive calls to StartMin could return,
00104 // in this order, 1, 2, and kMinValidValue.

00175 // An interval variable that wraps around an underlying one, relaxing the max
00176 // start and end. Relaxing means making unbounded when optional.
00177 //
00178 // More precisely, such an interval variable behaves as follows:
00179 // * When the underlying must be performed, this interval variable behaves
00180 //     exactly as the underlying;
00181 // * When the underlying may or may not be performed, this interval variable
00182 //     behaves like the underlying, except that it is unbounded on the max side;
00183 // * When the underlying cannot be performed, this interval variable is of
00184 //      duration 0 and must be performed in an interval unbounded on both sides.
00185 //
00186 // This class is very useful to implement propagators that may only modify
00187 // the start min or end min.

00672 void FixedDurationIntervalVar::Process() {
00673   CHECK(!in_process_);
00674   in_process_ = true;
00675   new_start_min_ = start_min_;
00676   new_start_max_ = start_max_;
00677   new_performed_ = performed_;
00678   set_queue_action_on_fail(&cleaner_);
00679   if (performed_ != 0) {
00680     if (start_min_ == start_max_) {
00681       for (SimpleRevFIFO<Demon*>::Iterator it(&start_bound_demons_);
00682            it.ok();
00683            ++it) {
00684         Enqueue(*it);
00685       }
00686     }
00687     if (start_min_ != old_start_min_ || start_max_ != old_start_max_) {
00688       for (SimpleRevFIFO<Demon*>::Iterator it(&start_range_demons_);
00689            it.ok();
00690            ++it) {
00691         Enqueue(*it);
00692       }
00693     }
00694   }
00695   if (old_performed_ != performed_) {
00696     for (SimpleRevFIFO<Demon*>::Iterator it(&performed_bound_demons_);
00697          it.ok();
00698          ++it) {
00699       Enqueue(*it);
00700     }
00701   }
00702   ProcessDemonsOnQueue();
00703   clear_queue_action_on_fail();
00704   ClearInProcess();
00705   old_start_min_ = start_min_;
00706   old_start_max_ = start_max_;
00707   old_performed_ = performed_;
00708   if (start_min_ < new_start_min_) {
00709     SetStartMin(new_start_min_);
00710   }
00711   if (start_max_ > new_start_max_) {
00712     SetStartMax(new_start_max_);
00713   }
00714   if (new_performed_ != performed_) {
00715     CHECK_NE(2, new_performed_);
00716     SetPerformed(new_performed_);
00717   }
00718 }

00867     }

00869     solver()->Fail();
00870   }
00871 }
