00087 // An IntervalVar that passes all function calls to an underlying interval

00092 // Such interval variables have a very similar behavior to others.

00097 // propagation, three successive calls to StartMin could return,
00098 // in this order, 1, 2, and kMinValidValue.

00101 // IntervalVarRelaxedMax and IntervalVarRelaxedMin classes below.

00166 // An interval variable that wraps around an underlying one, relaxing the max
00167 // start and end. Relaxing means making unbounded when optional.
00168 //
00169 // More precisely, such an interval variable behaves as follows:
00170 // * When the underlying must be performed, this interval variable behaves
00171 //     exactly as the underlying;
00172 // * When the underlying may or may not be performed, this interval variable
00173 //     behaves like the underlying, except that it is unbounded on the max side;
00174 // * When the underlying cannot be performed, this interval variable is of
00175 //      duration 0 and must be performed in an interval unbounded on both sides.
00176 //
00177 // This class is very useful to implement propagators that may only modify
00178 // the start min or end min.
00179 class IntervalVarRelaxedMax : public AlwaysPerformedIntervalVarWrapper {
00180  public:
00181   explicit IntervalVarRelaxedMax(IntervalVar* const t)
00182   : AlwaysPerformedIntervalVarWrapper(t) {}
00183   virtual ~IntervalVarRelaxedMax() {}
00184   virtual int64 StartMax() const {
00185     // It matters to use DurationMin() and not underlying()->DurationMin() here.
00186     return underlying()->MustBePerformed() ?
00187         underlying()->StartMax() : (kMaxValidValue - DurationMin());
00188   }
00189   virtual void SetStartMax(int64 m) {
00190     LOG(FATAL)
00191         << "Calling SetStartMax on a IntervalVarRelaxedMax is not supported, "
00192         << "as it seems there is no legitimate use case.";
00193   }
00194   virtual int64 EndMax() const {
00195     return underlying()->MustBePerformed() ?
00196         underlying()->EndMax() : kMaxValidValue;
00197   }
00198   virtual void SetEndMax(int64 m) {
00199     LOG(FATAL)
00200         << "Calling SetEndMax on a IntervalVarRelaxedMax is not supported, "
00201         << "as it seems there is no legitimate use case.";
00202   }

00649 void FixedDurationIntervalVar::Process() {
00650   CHECK(!in_process_);
00651   in_process_ = true;
00652   new_start_min_ = start_min_;
00653   new_start_max_ = start_max_;
00654   new_performed_ = performed_;
00655   set_queue_action_on_fail(&cleaner_);
00656   if (performed_ != 0) {
00657     if (start_min_ == start_max_) {
00658       for (SimpleRevFIFO<Demon*>::Iterator it(&start_bound_demons_);
00659            it.ok();
00660            ++it) {
00661         Enqueue(*it);
00662       }
00663     }
00664     if (start_min_ != old_start_min_ || start_max_ != old_start_max_) {
00665       for (SimpleRevFIFO<Demon*>::Iterator it(&start_range_demons_);
00666            it.ok();
00667            ++it) {
00668         Enqueue(*it);
00669       }
00670     }
00671   }
00672   if (old_performed_ != performed_) {
00673     for (SimpleRevFIFO<Demon*>::Iterator it(&performed_bound_demons_);
00674          it.ok();
00675          ++it) {
00676       Enqueue(*it);
00677     }
00678   }
00679   ProcessDemonsOnQueue();
00680   clear_queue_action_on_fail();
00681   ClearInProcess();
00682   old_start_min_ = start_min_;
00683   old_start_max_ = start_max_;
00684   old_performed_ = performed_;
00685   if (start_min_ < new_start_min_) {
00686     SetStartMin(new_start_min_);
00687   }
00688   if (start_max_ > new_start_max_) {
00689     SetStartMax(new_start_max_);
00690   }
00691   if (new_performed_ != performed_) {
00692     CHECK_NE(2, new_performed_);
00693     SetPerformed(new_performed_);
00694   }
00695 }

00847   }
