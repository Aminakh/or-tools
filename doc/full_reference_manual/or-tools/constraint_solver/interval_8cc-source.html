00091 // An IntervalVar that passes all function calls to an underlying interval
00092 // variable as long as it is not prohibited, and that interprets prohibited
00093 // intervals as intervals of duration 0 that must be executed between

00096 // Such interval variables have a very similar behavior to others.

00101 // propagation, three successive calls to StartMin could return,
00102 // in this order, 1, 2, and kMinValidValue.

00105 // IntervalVarRelaxedMax and IntervalVarRelaxedMin classes below.

00172 // An interval variable that wraps around an underlying one, relaxing the max
00173 // start and end. Relaxing means making unbounded when optional.
00174 //
00175 // More precisely, such an interval variable behaves as follows:
00176 // * When the underlying must be performed, this interval variable behaves
00177 //     exactly as the underlying;
00178 // * When the underlying may or may not be performed, this interval variable
00179 //     behaves like the underlying, except that it is unbounded on the max side;
00180 // * When the underlying cannot be performed, this interval variable is of
00181 //      duration 0 and must be performed in an interval unbounded on both sides.
00182 //
00183 // This class is very useful to implement propagators that may only modify
00184 // the start min or end min.

00700 void FixedDurationIntervalVar::Process() {
00701   CHECK(!in_process_);
00702   in_process_ = true;
00703   new_start_min_ = start_min_;
00704   new_start_max_ = start_max_;
00705   new_performed_ = performed_;
00706   set_queue_action_on_fail(&cleaner_);
00707   if (performed_ != 0) {
00708     if (start_min_ == start_max_) {
00709       for (SimpleRevFIFO<Demon*>::Iterator it(&start_bound_demons_);
00710            it.ok();
00711            ++it) {
00712         Enqueue(*it);
00713       }
00714     }
00715     if (start_min_ != old_start_min_ || start_max_ != old_start_max_) {
00716       for (SimpleRevFIFO<Demon*>::Iterator it(&start_range_demons_);
00717            it.ok();
00718            ++it) {
00719         Enqueue(*it);
00720       }
00721     }
00722   }
00723   if (old_performed_ != performed_) {
00724     for (SimpleRevFIFO<Demon*>::Iterator it(&performed_bound_demons_);
00725          it.ok();
00726          ++it) {
00727       Enqueue(*it);
00728     }
00729   }
00730   ProcessDemonsOnQueue();
00731   clear_queue_action_on_fail();
00732   ClearInProcess();
00733   old_start_min_ = start_min_;
00734   old_start_max_ = start_max_;
00735   old_performed_ = performed_;
00736   if (start_min_ < new_start_min_) {
00737     SetStartMin(new_start_min_);
00738   }
00739   if (start_max_ > new_start_max_) {
00740     SetStartMax(new_start_max_);
00741   }
00742   if (new_performed_ != performed_) {
00743     CHECK_NE(2, new_performed_);
00744     SetPerformed(new_performed_);
00745   }
00746 }
