00382 void NoCycle::NextBound(int index) {
00383   if (active_[index]->Min() == 0) return;
00384   const int64 next = nexts_[index]->Value();
00385   const int64 chain_start = starts_[index];
00386   const int64 chain_end = !sink_handler_->Run(next) ? ends_[next] : next;
00387   Solver* const s = solver();
00388   s->SaveAndSetValue(&ends_[chain_start], chain_end);
00389   if (!sink_handler_->Run(chain_end)) {
00390     s->SaveAndSetValue(&starts_[chain_end], chain_start);
00391     nexts_[chain_end]->RemoveValue(chain_start);
00392     if (!assume_paths_) {
00393       for (int i = 0; i < size_; ++i) {
00394         int64 current = i;
00395         bool found = (current == chain_end);
00396         // Counter to detect implicit cycles.
00397         int count = 0;
00398         while (!found
00399                && count < size_
00400                && !sink_handler_->Run(current)
00401                && nexts_[current]->Bound()) {
00402           current = nexts_[current]->Value();
00403           found = (current == chain_end);
00404           ++count;
00405         }
00406         if (found) {
00407           nexts_[chain_end]->RemoveValue(i);
00408         }
00409       }
00410     }
00411   }

00414 // For each variable, find a path connecting to a sink. Starts partial paths
00415 // from the sinks down to all unconnected variables. If some variables remain
00416 // unconnected, fail. Resulting paths are used as supports.
00417 void NoCycle::ComputeSupports() {
00418   scoped_array<int64> supported(new int64[size_]);
00419   int64 support_count = 0;
00420   for (int i = 0; i < size_; ++i) {
00421     if (nexts_[i]->Bound()) {
00422       supported[support_count] = i;
00423       outbound_supports_[i] = nexts_[i]->Value();
00424       ++support_count;
00425     } else {
00426       outbound_supports_[i] = -1;
00427     }
00428   }
00429   if (size_ == support_count) {
00430     return;
00431   }
00432   const int size = sinks_.size();
00433   for (int i = 0; i < size_; ++i) {
00434     const IntVar* next = nexts_[i];
00435     if (!nexts_[i]->Bound()) {
00436       for (int j = 0; j < size; ++j) {
00437         if (next->Contains(sinks_[j])) {
00438           supported[support_count] = i;
00439           outbound_supports_[i] = sinks_[j];
00440           ++support_count;
00441           break;
00442         }
00443       }
00444     }
00445   }
00446   for (int i = 0; i < support_count && support_count < size_; ++i) {
00447     const int64 supported_i = supported[i];
00448     for (int j = 0; j < size_; ++j) {
00449       if (outbound_supports_[j] < 0 && nexts_[j]->Contains(supported_i)) {
00450         supported[support_count] = j;
00451         outbound_supports_[j] = supported_i;
00452         ++support_count;
00453       }
00454     }
00455   }
00456   if (size_ != support_count) {
00457     supported.reset(NULL);
00458     for (int i = 0; i < size_; ++i) {
00459       if (outbound_supports_[i] < 0) {
00460         active_[i]->SetMax(0);
00461       }
00462     }
00463   }
00464 }

00639                                                &PathCumul::ActiveBound,
00640                                                "ActiveBound",
00641                                                i);
00642     active_[i]->WhenBound(active_demon);
00643     Demon* transit_demon = MakeConstraintDemon1(solver(),
00644                                                 this,
00645                                                 &PathCumul::TransitRange,
00646                                                 "TransitRange",
00647                                                 i);
00648     transits_[i]->WhenRange(transit_demon);
00649   }
00650   for (int i = 0; i < cumul_size_; ++i) {
00651     IntVar* cumul = cumuls_[i];
00652     Demon* d = MakeConstraintDemon1(solver(),
00653                                     this,
00654                                     &PathCumul::CumulRange,
00655                                     "CumulRange",
00656                                     i);
00657     cumul->WhenRange(d);
00658   }
00659 }

00661 void PathCumul::ActiveBound(int index) {
00662   if (nexts_[index]->Bound()) {
00663     NextBound(index);
00664   }
00665 }
00666 
00667 void PathCumul::NextBound(int index) {
00668   if (active_[index]->Min() == 0) return;
00669   const int64 next = nexts_[index]->Value();
00670   IntVar* cumul = cumuls_[index];
00671   IntVar* cumul_next = cumuls_[next];
00672   IntVar* transit = transits_[index];
00673   cumul_next->SetMin(cumul->Min() + transit->Min());
00674   const int64 cumul_next_max = cumul->Max() + transit->Max();
00675   if (cumul_next_max >= 0) {  // detect integer overflow
00676     cumul_next->SetMax(cumul_next_max);
00677   }
00678   cumul->SetMin(cumul_next->Min() - transit->Max());
00679   cumul->SetMax(cumul_next->Max() - transit->Min());
00680   transit->SetMin(cumul_next->Min() - cumul->Max());
00681   transit->SetMax(cumul_next->Max() - cumul->Min());
00682   if (prevs_[next] < 0) {
00683     solver()->SaveAndSetValue(&prevs_[next], index);
00684   }
00685 }
00686 
00687 void PathCumul::CumulRange(int index) {
00688   if (index < size_) {
00689     if (nexts_[index]->Bound()) {
00690       NextBound(index);
00691     } else {
00692       UpdateSupport(index);
00693     }
00694   }
00695   if (prevs_[index] >= 0) {
00696     NextBound(prevs_[index]);
00697   } else {
00698     for (int i = 0; i < size_; ++i) {
00699       if (index == supports_[i]) {
00700         UpdateSupport(i);
00701       }
00702     }
00703   }
00704 }
00705 
00706 void PathCumul::TransitRange(int index) {
00707   if (nexts_[index]->Bound()) {
00708     NextBound(index);
00709   } else {
00710     UpdateSupport(index);
00711   }
00712   if (prevs_[index] >= 0) {
00713     NextBound(prevs_[index]);
00714   } else {
00715     for (int i = 0; i < size_; ++i) {
00716       if (index == supports_[i]) {
00717         UpdateSupport(i);
00718       }
00719     }
00720   }
00721 }

00723 bool PathCumul::AcceptLink(int i, int j) const {
00724   const IntVar* const cumul_i = cumuls_[i];
00725   const IntVar* const cumul_j = cumuls_[j];
00726   const IntVar* const transit_i = transits_[i];
00727   return transit_i->Min() <= cumul_j->Max() - cumul_i->Min()
00728       && cumul_j->Min() - cumul_i->Max() <= transit_i->Max();
00729 }
00730 
00731 void PathCumul::UpdateSupport(int index) {
00732   int support = supports_[index];
00733   if (support < 0 || !AcceptLink(index, support)) {
00734     IntVar* var = nexts_[index];
00735     for (int i = var->Min(); i <= var->Max(); ++i) {
00736       if (i != support && AcceptLink(index, i)) {
00737         supports_[index] = i;
00738         return;
00739       }
00740     }
00741     active_[index]->SetMax(0);
00742   }
00743 }
