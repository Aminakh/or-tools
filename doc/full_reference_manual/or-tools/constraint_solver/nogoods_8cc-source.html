00065 // IntVar == int64 specialization.
00066 class IntegerVariableNoGoodTerm : public NoGoodTerm {
00067  public:
00068   IntegerVariableNoGoodTerm(IntVar* const var, int64 value, bool assign)
00069       : integer_variable_(var), value_(value), assign_(assign) {
00070     CHECK_NOTNULL(integer_variable_);
00071   }
00072 
00073   virtual TermStatus Evaluate() const {
00074     if (!integer_variable_->Contains(value_)) {
00075       return assign_ ? ALWAYS_FALSE : ALWAYS_TRUE;
00076     } else if (integer_variable_->Bound()) {
00077       return assign_ ? ALWAYS_TRUE : ALWAYS_FALSE;
00078     } else {
00079       return UNDECIDED;
00080     }
00081   }
00082 
00083   virtual void Refute() {
00084     if (assign_) {
00085       integer_variable_->RemoveValue(value_);
00086     } else {
00087       integer_variable_->SetValue(value_);
00088     }
00089   }
00090 
00091   virtual string DebugString() const {
00092     return StringPrintf("(%s %s %lld)",
00093                         integer_variable_->name().c_str(),
00094                         assign_ ? "==" : "!=",
00095                         value_);
00096   }
00097 
00098   IntVar* integer_variable() const { return integer_variable_; }
00099   int64 value() const { return value_; }
00100   bool assign() const { return assign_; }
00101 
00102  private:
00103   IntVar* const integer_variable_;
00104   const int64 value_;
00105   const bool assign_;
00106   DISALLOW_COPY_AND_ASSIGN(IntegerVariableNoGoodTerm);
00107 };

00124 void NoGood::Apply(Solver* const solver) {
00125   NoGoodTerm* first_undecided = NULL;
00126   for (int i = 0; i < terms_.size(); ++i) {
00127     switch (terms_[i]->Evaluate()) {
00128       case NoGoodTerm::ALWAYS_TRUE: {
00129         break;
00130       }
00131       case NoGoodTerm::ALWAYS_FALSE: {
00132         return;
00133       }
00134       case NoGoodTerm::UNDECIDED: {
00135         if (first_undecided == NULL) {
00136           first_undecided = terms_[i];
00137         } else {
00138           // more than one undecided, we cannot deduce anything.
00139           return;
00140         }
00141         break;
00142       }
00143     }
00144   }
00145   if (first_undecided == NULL && terms_.size() > 0) {
00146     solver->Fail();
00147   }
00148   if (first_undecided != NULL) {
00149     first_undecided->Refute();
00150   }
00151 }
