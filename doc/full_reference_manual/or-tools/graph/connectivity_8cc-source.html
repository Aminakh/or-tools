00024 void ConnectedComponents::Init(NodeIndex min_index, NodeIndex max_index) {
00025   CHECK_LT(1, max_index - min_index);
00026   CHECK_LE(0, min_index);
00027   min_index_ = min_index;
00028   max_index_ = max_index;
00029   class_.Reserve(min_index, max_index);
00030   class_size_.Reserve(min_index, max_index);
00031   for (NodeIndex node = min_index; node <= max_index; ++node) {
00032     class_.Set(node, node);
00033   }
00034   class_size_.Assign(1);
00035 }

00037 void ConnectedComponents::AddArc(NodeIndex tail, NodeIndex head) {
00038   max_seen_index_ = std::max(max_seen_index_, tail);
00039   max_seen_index_ = std::max(max_seen_index_, head);
00040   NodeIndex tail_class = CompressPath(tail);
00041   NodeIndex head_class = CompressPath(head);
00042   if (tail_class != head_class) {
00043     MergeClasses(tail_class, head_class);

00047 void ConnectedComponents::AddGraph(const StarGraph& graph) {
00048   Init(StarGraph::kFirstNode, graph.num_nodes());
00049   for (StarGraph::ArcIterator arc_it(graph); arc_it.Ok(); arc_it.Next()) {
00050     const ArcIndex arc = arc_it.Index();
00051     AddArc(graph.Tail(arc), graph.Head(arc));
00052   }
00053   max_seen_index_ = graph.num_nodes() - 1;
00054 }

00056 NodeIndex ConnectedComponents::CompressPath(NodeIndex node) {
00057   DCHECK_LE(min_index_, node);
00058   DCHECK_GE(max_index_, node);
00059   while (node != class_[node]) {
00060     DCHECK_LE(min_index_, class_[node]);
00061     DCHECK_GE(max_index_, class_[node]);
00062     DCHECK_LE(min_index_, class_[class_[node]]);
00063     DCHECK_GE(max_index_, class_[class_[node]]);
00064     class_.Set(node, class_[class_[node]]);
00065     node = class_[node];
00066   }
00067   return node;
00068 }

00070 NodeIndex ConnectedComponents::GetClassRepresentative(NodeIndex node) {
00071   DCHECK_LE(min_index_, node);
00072   DCHECK_GE(max_index_, node);
00073   NodeIndex representative = node;
00074   while (class_[representative] != representative) {
00075     representative = class_[representative];
00076   }
00077   // The following line is an optimization that brings about 20% when using
00078   // Union-Find for counting the number of connected components with
00079   // GetNumberOfConnectedComponents.
00080   class_.Set(node, representative);
00081   return representative;
00082 }

00084 NodeIndex ConnectedComponents::GetNumberOfConnectedComponents() {
00085   std::vector<bool> seen(max_index_);
00086   NodeIndex number = 0;
00087   for (NodeIndex node = min_index_; node <= max_seen_index_; ++node) {
00088     NodeIndex representative = GetClassRepresentative(node);
00089     if (!seen[representative]) {
00090       seen[representative] = true;
00091       ++number;
00092     }
00093   }
00094   return number;
00095 }

00097 void ConnectedComponents::MergeClasses(NodeIndex node1, NodeIndex node2) {
00098   // It's faster (~10%) to swap the two values and have a single piece of
00099   // code for merging the classes.
00100   DCHECK_LE(min_index_, node1);
00101   DCHECK_GE(max_index_, node1);
00102   DCHECK_LE(min_index_, node2);
00103   DCHECK_GE(max_index_, node2);
00104   if (class_size_[node1] < class_size_[node2]) {
00105     std::swap(node1, node2);
00106   }
00107   class_.Set(node2, node1);
00108   class_size_.Set(node1, class_size_[node1] + class_size_[node2]);
00109 }
