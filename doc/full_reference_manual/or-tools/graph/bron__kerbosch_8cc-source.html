00041   for (int i = 0; i < input_candidate_size && index != 0; ++i) {
00042     int p = input_candidates[i];
00043     int count = 0;
00044     int position = 0;
00045 
00046     // Count disconnections.
00047     for (int j = input_size; j < input_candidate_size && count < index; ++j) {
00048       if (!graph->Run(p, input_candidates[j])) {
00049         count++;
00050         // Save position of potential candidate.
00051         position = j;
00052       }
00053     }
00054

00056     if (count < index) {
00057       pivot = p;
00058       index = count;
00059 
00060       if (i < input_size) {
00061         start = position;
00062       } else {
00063         start = i;
00064         // pre increment obvious candidate.
00065         pre_increment = 1;
00066       }
00067     }
00068   }
00069 
00070   // If fixed point initially chosen from candidates then
00071   // number of diconnections will be preincreased by one
00072   // Backtracking step for all nodes in the candidate list CD.
00073   for (int nod = index + pre_increment; nod >= 1; nod--) {
00074     // Swap.
00075     int selected = input_candidates[start];
00076     input_candidates[start] = input_candidates[input_size];
00077     input_candidates[input_size] = selected;

00081 
00082     for (int i = 0; i < input_size; ++i) {

00085       }
00086     }
00087

00089     actual_size = actual_candidate_size;
00090 
00091     for (int i = input_size + 1; i < input_candidate_size; ++i) {
00092       if (graph->Run(selected, input_candidates[i])) {
00093         actual_candidates[actual_size++] = input_candidates[i];
00094       }
00095     }
00096 
00097     // Add to "actual relevant nodes".
00098     actual->push_back(selected);
00099 
00100     // We have found a maximum clique.
00101     if (actual_size == 0) {
00102       *stop = callback->Run(*actual);
00103     } else {
00104       if (actual_candidate_size < actual_size) {
00105          Search(graph, callback, actual_candidates.data(),
00106                 actual_candidate_size, actual_size, actual, stop);
00107         if (*stop) {
00108           return;
00109         }
00110       }
00111     }
00112 
00113     // move node from MD to ND
00114     // Remove from compsub
00115     actual->pop_back();
00116 
00117     // Add to "nod"
00118     input_size++;
00119 
00120     if (nod > 1) {
00121       // Select a candidate disgraph to the fixed point
00122       start = input_size;
00123       while (graph->Run(pivot, input_candidates[start])) {
00124         start++;
00125       }
00126     }
