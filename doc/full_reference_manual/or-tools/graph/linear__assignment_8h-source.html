00014 // An implementation of a cost-scaling push-relabel algorithm for the
00015 // assignment problem (minimum-cost perfect bipartite matching), from
00016 // the paper of Goldberg and Kennedy (1995).
00017 //
00018 // This implementation finds the minimum-cost perfect assignment in
00019 // the given graph with integral edge weights set through the
00020 // SetArcCost method.
00021 //
00022 // Example usage:
00023 //
00024 // #include "graph/ebert_graph.h"
00025 // #include "graph/linear_assignment.h"
00026 // ...
00027 // ::operations_research::NodeIndex num_nodes = ...;
00028 // ::operations_research::NodeIndex num_left_nodes = num_nodes / 2;
00029 // // Define a num_nodes/2 by num_nodes/2 assignment problem:
00030 // ::operations_research::ArcIndex num_forward_arcs = ...;
00031 // ::operations_research::ForwardStarGraph g(num_nodes, num_arcs);
00032 // ::operations_research::LinearSumAssignment<
00033 //     ::operations_research::ForwardStarGraph> a(g, num_left_nodes);
00034 // for (int i = 0; i < num_forward_arcs; ++i) {
00035 //   ::operations_research::NodeIndex this_arc_head = ...;
00036 //   ::operations_research::NodeIndex this_arc_tail = ...;
00037 //   ::operations_research::CostValue this_arc_cost = ...;
00038 //   ::operations_research::ArcIndex this_arc_index =
00039 //       g.AddArc(this_arc_tail, this_arc_head);
00040 //   a.SetArcCost(this_arc_index, this_arc_cost);
00041 //  }
00042 //  // Compute the optimum assignment.
00043 //  bool success = a.ComputeAssignment();
00044 //  // Retrieve the cost of the optimum assignment.
00045 //  CostValue optimum_cost = a.GetCost();
00046 //  // Retrieve the node-node correspondence of the optimum assignment and the

00059 // In the following, we consider a bipartite graph
00060 //   G = (V = X union Y, E subset XxY),
00061 // where V denodes the set of nodes (vertices) in the graph, E denotes
00062 // the set of arcs (edges), n = |V| denotes the number of nodes in the
00063 // graph, and m = |E| denotes the number of arcs in the graph.
00064 //
00065 // The set of nodes is divided into two parts, X and Y, and every arc
00066 // must go between a node of X and a node of Y. With each arc is
00067 // associated a cost c(v, w). A matching M is a subset of E with the
00068 // property that no two arcs in M have a head or tail node in common,
00069 // and a perfect matching is a matching that touches every node in the
00070 // graph. The cost of a matching M is the sum of the costs of all the
00071 // arcs in M.
00072 //
00073 // The assignment problem is to find a perfect matching of minimum
00074 // cost in the given bipartite graph. The present algorithm reduces
00075 // the assignment problem to an instance of the minimum-cost flow
00076 // problem and takes advantage of special properties of the resulting
00077 // minimum-cost flow problem to solve it efficiently using a
00078 // push-relabel method. For more information about minimum-cost flow
00079 // see or-tools/graph/min_cost_flow.h
00080 //
00081 // The method used here is the cost-scaling approach for the
00082 // minimum-cost circulation problem as described in [Goldberg and
00083 // Tarjan] with some technical modifications:
00084 // 1. For efficiency, we solve a transportation problem instead of
00085 //    minimum-cost circulation. We might revisit this decision if it
00086 //    is important to handle problems in which no perfect matching
00087 //    exists.
00088 // 2. We use a modified "asymmetric" notion of epsilon-optimality in
00089 //    which left-to-right residual arcs are required to have reduced
00090 //    cost bounded below by zero and right-to-left residual arcs are
00091 //    required to have reduced cost bounded below by -epsilon. For
00092 //    each residual arc direction, the reduced-cost threshold for
00093 //    admissibility is epsilon/2 above the threshold for epsilon
00094 //    optimality.
00095 // 3. We do not limit the applicability of the relabeling operation to
00096 //    nodes with excess. Instead we use the double-push operation
00097 //    (discussed in the Goldberg and Kennedy CSA paper and Kennedy's
00098 //    thesis) which relabels right-side nodes just *after* they have
00099 //    been discharged.
00100 // The above differences are explained in detail in [Kennedy's thesis]
00101 // and explained not quite as cleanly in [Goldberg and Kennedy's CSA
00102 // paper]. But note that the thesis explanation uses a value of
00103 // epsilon that's double what we use here.
00104 //
00105 // Some definitions:
00106 //   Active: A node is called active when it has excess. It is
00107 //     eligible to be pushed from. In this implementation, every active
00108 //     node is on the left side of the graph where prices are determined
00109 //     implicitly, so no left-side relabeling is necessary before
00110 //     pushing from an active node. We do, however, need to compute
00111 //     the implications for price changes on the affected right-side
00112 //     nodes.
00113 //   Admissible: A residual arc (one that can carry more flow) is
00114 //     called admissible when its reduced cost is small enough. We can
00115 //     push additional flow along such an arc without violating
00116 //     epsilon-optimality. In the case of a left-to-right residual
00117 //     arc, the reduced cost must be at most epsilon/2. In the case of
00118 //     a right-to-left residual arc, the reduced cost must be at most
00119 //     -epsilon/2. The careful reader will note that these thresholds
00120 //     are not used explicitly anywhere in this implementation, and
00121 //     the reason is the implicit pricing of left-side nodes.
00122 //   Reduced cost: Essentially an arc's reduced cost is its
00123 //     complementary slackness. In push-relabel algorithms this is
00124 //       c_p(v, w) = p(v) + c(v, w) - p(w),
00125 //     where p() is the node price function and c(v, w) is the cost of
00126 //     the arc from v to w. See min_cost_flow.h for more details.
00127 //   Partial reduced cost: We maintain prices implicitly for left-side
00128 //     nodes in this implementation, so instead of reduced costs we
00129 //     work with partial reduced costs, defined as
00130 //       c'_p(v, w) = c(v, w) - p(w).
00131 //
00132 // We check at initialization time for the possibility of arithmetic
00133 // overflow and warn if the given costs are too large. In many cases
00134 // the bound we use to trigger the warning is pessimistic so the given
00135 // problem can often be solved even if we warn that overflow is
00136 // possible.
00137 //
00138 // We don't use the interface from
00139 // operations_research/algorithms/hungarian.h because we want to be
00140 // able to express sparse problems efficiently.
00141 //
00142 // When asked to solve the given assignment problem we return a
00143 // boolean to indicate whether the given problem was feasible.
00144 //
00145 // References:
00146 // [ Goldberg and Kennedy's CSA paper ] A. V. Goldberg and R. Kennedy,
00147 // "An Efficient Cost Scaling Algorithm for the Assignment Problem."
00148 // Mathematical Programming, Vol. 71, pages 153-178, December 1995.
00149 //
00150 // [ Goldberg and Tarjan ] A. V. Goldberg and R. E. Tarjan, "Finding
00151 // Minimum-Cost Circulations by Successive Approximation." Mathematics
00152 // of Operations Research, Vol. 15, No. 3, pages 430-466, August 1990.
00153 //
00154 // [ Kennedy's thesis ] J. R. Kennedy, Jr., "Solving Unweighted and
00155 // Weighted Bipartite Matching Problems in Theory and Practice."
00156 // Stanford University Doctoral Dissertation, Department of Computer
00157 // Science, 1995.
00158 //
00159 // [ Burkard et al. ] R. Burkard, M. Dell'Amico, S. Martello, "Assignment
00160 // Problems", SIAM, 2009, ISBN: 978-0898716634,
00161 // http://www.amazon.com/dp/0898716632/

00495   // A lower bound on the price of any node at any time throughout the
00496   // computation. A price below this level proves infeasibility.
00497   //
00498   // The value of this lower bound is determined according to the
00499   // following sketch: Suppose the price decrease of every node in the
00500   // iteration with epsilon_ == x is bounded by B(x) which is
00501   // proportional to x. Then the total price decrease of every node
00502   // across all iterations is bounded above by
00503   //   B(C/alpha) + B(C/alpha^2) + ... + B(kMinEpsilon)
00504   //   == B(C/alpha) * alpha / (alpha - 1)
00505   //   == B(C) / (alpha - 1).
00506   // Therefore we set price_lower_bound_ = -ceil(B(C) / (alpha - 1))
00507   // where B() is the expression that determines
00508   // price_reduction_bound_, discussed below.
00509   CostValue price_lower_bound_;
00510 
00511   // An upper bound on the amount that a single node's price can
00512   // decrease in a single scaling iteration. In each iteration, this
00513   // value corresponds to B(epsilon_) in the comments describing
00514   // price_lower_bound_ above. Exceeding this amount of price decrease
00515   // in one iteration proves that there is some excess that cannot
00516   // reach a deficit, i.e., that the problem is infeasible.
00517   //
00518   // Let v be a node with excess and suppose P is a simple residual
00519   // path P from v to some node w with deficit such that reverse(P) is
00520   // residual at the beginning of this iteration (such a path is
00521   // guaranteed to exist by feasibility -- see lemma 5.7 in Goldberg
00522   // and Tarjan). We have c_p(P) = p(v) + c(P) - p(w) and of those
00523   // three terms, only p(v) may have changed during this iteration
00524   // because w has a deficit and nodes with deficits are not
00525   // relabeled. Assuming without loss of generality that p == 0 and
00526   // c_p == c at the beginning of this iteration, we seek a bound on
00527   // simply
00528   //   p(v) = c_p(P) - c(P).
00529   // Let arc a lie on P.
00530   // Case 1: a is a forward arc. Then c_p(a) >= 0 and the reverse of a
00531   //         was residual when this iteration began. By
00532   //         approximate optimality at the end of the prior iteration,
00533   //         c(a) < alpha * epsilon. So
00534   //           c_p(a) - c(a) > -alpha * epsilon_.
00535   // Case 2: a is a reverse arc. Then c_p(a) >= -epsilon_ and the
00536   //         reverse of a was residual when this iteration began. By
00537   //         approximate optimality at the end of the prior iteration,
00538   //         c(a) < 0. So
00539   //           c_p(a) - c(a) > -epsilon_.
00540   // Nodes with excess are only on the left and nodes with deficit are
00541   // only on the right; there are at most n - 1 arcs on the path P,
00542   // making up at most (n-1)/2 left-right-left arc pairs. Each
00543   // pair's contribution to c_p(P) - c(P) is bounded below by
00544   // most (n-1)/2 of those are forward arcs and (n-2)/2 of them are reverse
00545   // arcs, so
00546   //   p(v) = c_p(P) - c(P)
00547   //        > (n-1)/2 * (-alpha * epsilon_ - epsilon_)
00548   //        = -(n-1)/2 * epsilon_ * (1 + alpha).
00549   // So we set
00550   //   price_reduction_bound_ = ceil((n-1)/2 * epsilon * (1 + alpha)).
00551   CostValue price_reduction_bound_;
00552 
00553   // Computes the value of price_reduction_bound_ for an iteration,
00554   // given the new value of epsilon_, on the assumption that the value
00555   // of epsilon_ for the previous iteration was no more than a factor
00556   // of alpha_ times the new value. Because the expression computed
00557   // here is used in at least one place where we want an additional
00558   // factor in the denominator, we take that factor as an argument.
00559   //
00560   // Avoids overflow in computing the bound.
00561   inline CostValue PriceChangeBound(CostValue extra_divisor,
00562                                     bool* in_range) const {
00563     const CostValue n = graph_.num_nodes();
00564     // We work in double-precision floating point to determine whether
00565     // we'll overflow the integral CostValue type's range of
00566     // representation. Switching between integer and double is a
00567     // rather expensive operation, but we do this only once per
00568     // scaling iteration, so we can afford it rather than resort to
00569     // complex and subtle tricks within the bounds of integer
00570     // arithmetic.
00571     //
00572     // To understand the values of numerator and denominator here, you
00573     // will want to read the comments above about price_lower_bound_
00574     // and price_reduction_bound_, and have a pencil handy. :-)
00575     const double numerator = (static_cast<double>(n - 1) *
00576                               static_cast<double>(epsilon_ * (1 + alpha_)));
00577     const double denominator = static_cast<double>(2 * extra_divisor);
00578     const double quotient = numerator / denominator;
00579     const double limit =
00580         static_cast<double>(std::numeric_limits<CostValue>::max());
00581     if (quotient > limit) {
00582       // Our integer computations could overflow.
00583       if (in_range != NULL) *in_range = false;
00584       return std::numeric_limits<CostValue>::max();
00585     } else {
00586       if (in_range != NULL) *in_range = true;
00587       return static_cast<CostValue>(quotient);
00588     }
00589   }
00590 
00591   // A scaled record of the largest arc-cost magnitude we've been
00592   // given during problem setup. This is used to set the initial value
00593   // of epsilon_, which in turn is used not only as the error
00594   // parameter but also to determine whether we risk arithmetic
00595   // overflow during the algorithm.
00596   CostValue largest_scaled_cost_magnitude_;
00597 
00598   // The total excess in the graph. Given our asymmetric definition of
00599   // epsilon-optimality and our use of the double-push operation, this
00600   // equals the number of unmatched left-side nodes.
00601   NodeIndex total_excess_;
00602 
00603   // Indexed by node index, the price_ values are maintained only for
00604   // right-side nodes.
00605   CostArray price_;
00606 
00607   // Indexed by left-side node index, the matched_arc_ array gives the
00608   // arc index of the arc matching any given left-side node, or
00609   // GraphType::kNilArc if the node is unmatched.
00610   ArcIndexArray matched_arc_;
00611 
00612   // Indexed by right-side node index, the matched_node_ array gives
00613   // the node index of the left-side node matching any given
00614   // right-side node, or GraphType::kNilNode if the right-side node is
00615   // unmatched.
00616   NodeIndexArray matched_node_;
00617 
00618   // The array of arc costs as given in the problem definition, except
00619   // that they are scaled up by the number of nodes in the graph so we
00620   // can use integer arithmetic throughout.
00621   CostArray scaled_arc_cost_;
00622 
00623   // The container of active nodes (i.e., unmatched nodes). This can
00624   // be switched easily between ActiveNodeStack and ActiveNodeQueue
00625   // for experimentation.
00626   scoped_ptr<ActiveNodeContainerInterface> active_nodes_;
00627 
00628   // Statistics giving the overall numbers of various operations the
00629   // algorithm performs.
00630   Stats total_stats_;
00631 
00632   // Statistics giving the numbers of various operations the algorithm
00633   // has performed in the current iteration.
00634   Stats iteration_stats_;
00635 
00636   DISALLOW_COPY_AND_ASSIGN(LinearSumAssignment);
00637 };

00753 template <typename GraphType>
00754 bool LinearSumAssignment<GraphType>::UpdateEpsilon() {
00755   // There are some somewhat subtle issues around using integer
00756   // arithmetic to compute successive values of epsilon_, the error
00757   // parameter.
00758   //
00759   // First, the value of price_reduction_bound_ is chosen under the
00760   // assumption that it is truly an upper bound on the amount by which
00761   // a node's price can change during the current iteration. The value
00762   // of this bound in turn depends on the assumption that the flow
00763   // computed by the previous iteration was
00764   // (epsilon_ * alpha_)-optimal. If epsilon_ decreases by more than a
00765   // factor of alpha_ due to truncating arithmetic, that bound might
00766   // not hold, and the consequence is that BestArcAndGap could return
00767   // an overly cautious admissibility gap in the case where a
00768   // left-side node has only one incident arc. This is not a big deal
00769   // at all. At worst it could lead to a few extra relabelings.
00770   //
00771   // Second, it is not a problem currently, but in the future if we
00772   // use an arc-fixing heuristic, we cannot permit truncating integer
00773   // division to decrease epsilon_ by a factor greater than alpha_
00774   // because our bounds on price changes (and hence on the reduced
00775   // cost of an arc that might ever become admissible in the future)
00776   // depend on the ratio between the values of the error parameter at
00777   // successive iterations. One consequence will be that we might
00778   // occasionally do an extra iteration today in the interest of being
00779   // able to "price arcs out" (which we don't do today). Today we
00780   // simply use truncating integer division, but note that this will
00781   // have to change if we ever price arcs out.
00782   //
00783   // Since neither of those issues is a problem today, we simply use
00784   // truncating integer arithmetic. But future changes might
00785   // necessitate rounding epsilon upward in the division.
00786   epsilon_ = std::max(epsilon_ / alpha_, kMinEpsilon);
00787   VLOG(3) << "Updated: epsilon_ == " << epsilon_;
00788   price_reduction_bound_ = PriceChangeBound(1, NULL);
00789   DCHECK_GT(price_reduction_bound_, 0);
00790   // For today we always return true; in the future updating epsilon
00791   // in sophisticated ways could conceivably detect infeasibility.
00792   return true;
00793 }

00829 // There exists a price function such that the admissible arcs at the
00830 // beginning of an iteration are exactly the reverse arcs of all
00831 // matching arcs. Saturating all admissible arcs with respect to that
00832 // price function therefore means simply unmatching every matched
00833 // node.
00834 //
00835 // In the future we will price out arcs, which will reduce the set of
00836 // nodes we unmatch here. If a matching arc is priced out, we will not
00837 // unmatch its endpoints since that element of the matching is
00838 // guaranteed not to change.
00839 template <typename GraphType>
00840 void LinearSumAssignment<GraphType>::SaturateNegativeArcs() {
00841   total_excess_ = 0;
00842   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
00843        node_it.Ok();
00844        node_it.Next()) {
00845     const NodeIndex node = node_it.Index();
00846     if (IsActive(node)) {
00847       // This can happen in the first iteration when nothing is
00848       // matched yet.
00849       total_excess_ += 1;
00850     } else {
00851       // We're about to create a unit of excess by unmatching these nodes.
00852       total_excess_ += 1;
00853       const NodeIndex mate = GetMate(node);
00854       matched_arc_.Set(node, GraphType::kNilArc);
00855       matched_node_.Set(mate, GraphType::kNilNode);
00856     }
00857   }
00858 }

00860 // Returns true for success, false for infeasible.
00861 template <typename GraphType>
00862 bool LinearSumAssignment<GraphType>::DoublePush(NodeIndex source) {
00863   DCHECK_GT(num_left_nodes_, source);
00864   DCHECK(IsActive(source));
00865   ImplicitPriceSummary summary = BestArcAndGap(source);
00866   const ArcIndex best_arc = summary.first;
00867   const CostValue gap = summary.second;
00868   // Now we have the best arc incident to source, i.e., the one with
00869   // minimum reduced cost. Match that arc, unmatching its head if
00870   // necessary.
00871   if (best_arc == GraphType::kNilArc) {
00872     return false;
00873   }
00874   const NodeIndex new_mate = Head(best_arc);
00875   const NodeIndex to_unmatch = matched_node_[new_mate];
00876   if (to_unmatch != GraphType::kNilNode) {
00877     // Unmatch new_mate from its current mate, pushing the unit of
00878     // flow back to a node on the left side as a unit of excess.
00879     matched_arc_.Set(to_unmatch, GraphType::kNilArc);
00880     active_nodes_->Add(to_unmatch);
00881     // This counts as a double push.
00882     iteration_stats_.double_pushes_ += 1;
00883   } else {
00884     // We are about to increase the cardinality of the matching.
00885     total_excess_ -= 1;
00886     // This counts as a single push.
00887     iteration_stats_.pushes_ += 1;
00888   }
00889   matched_arc_.Set(source, best_arc);
00890   matched_node_.Set(new_mate, source);
00891   // Finally, relabel new_mate.
00892   iteration_stats_.relabelings_ += 1;
00893   price_.Set(new_mate, price_[new_mate] - gap - epsilon_);
00894   return price_[new_mate] >= price_lower_bound_;
00895 }

00897 template <typename GraphType>
00898 bool LinearSumAssignment<GraphType>::Refine() {
00899   SaturateNegativeArcs();
00900   InitializeActiveNodeContainer();
00901   while (total_excess_ > 0) {
00902     // Get an active node (i.e., one with excess == 1) and discharge
00903     // it using DoublePush.
00904     const NodeIndex node = active_nodes_->Get();
00905     if (!DoublePush(node)) {
00906       // Infeasibility detected.
00907       return false;
00908     }
00909   }
00910   DCHECK(active_nodes_->Empty());
00911   iteration_stats_.refinements_ += 1;
00912   return true;
00913 }

00915 // Computes best_arc, the minimum reduced-cost arc incident to
00916 // left_node and admissibility_gap, the amount by which the reduced
00917 // cost of best_arc must be increased to make it equal in reduced cost
00918 // to another residual arc incident to left_node.
00919 //
00920 // Precondition: left_node is unmatched. This allows us to simplify
00921 // the code. The debug-only counterpart to this routine is
00922 // LinearSumAssignment::ImplicitPrice() and it does not assume this
00923 // precondition.
00924 //
00925 // This function is large enough that our suggestion that the compiler
00926 // inline it might be pointless.
00927 template <typename GraphType>
00928 inline typename LinearSumAssignment<GraphType>::ImplicitPriceSummary
00929 LinearSumAssignment<GraphType>::BestArcAndGap(NodeIndex left_node) const {
00930     DCHECK(IsActive(left_node));
00931   DCHECK_GT(epsilon_, 0);
00932   // During any scaling iteration, the price of an active node
00933   // decreases by at most price_reduction_bound_ and all left-side
00934   // nodes are made active at the beginning of Refine(), so the bound
00935   // applies to all left-side nodes.
00936   typename GraphType::OutgoingArcIterator arc_it(graph_, left_node);
00937   ArcIndex best_arc = arc_it.Index();
00938   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00939   // We choose second_min_partial_reduced_cost so that in the case of
00940   // the largest possible gap (which results from a left-side node
00941   // with only a single incident residual arc), the corresponding
00942   // right-side node will be relabeled by an amount that exactly
00943   // matches price_reduction_bound_. The overall price_lower_bound_ is
00944   // computed tightly enough that if we relabel by an amount even
00945   // epsilon_ greater than that, we can incorrectly conclude
00946   // infeasibility in DoublePush().
00947   CostValue second_min_partial_reduced_cost =
00948       min_partial_reduced_cost + price_reduction_bound_ - epsilon_;
00949   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00950     const ArcIndex arc = arc_it.Index();
00951     const CostValue partial_reduced_cost = PartialReducedCost(arc);
00952     if (partial_reduced_cost < second_min_partial_reduced_cost) {
00953       if (partial_reduced_cost < min_partial_reduced_cost) {
00954         best_arc = arc;
00955         second_min_partial_reduced_cost = min_partial_reduced_cost;
00956         min_partial_reduced_cost = partial_reduced_cost;
00957       } else {
00958         second_min_partial_reduced_cost = partial_reduced_cost;
00959       }
00960     }
00961   }
00962   const CostValue gap =
00963       second_min_partial_reduced_cost - min_partial_reduced_cost;
00964   DCHECK_GE(gap, 0);
00965   return std::make_pair(best_arc, gap);
00966 }

00968 // Only for debugging.
00969 template <typename GraphType> inline CostValue
00970 LinearSumAssignment<GraphType>::ImplicitPrice(NodeIndex left_node) const {
00971   DCHECK_GT(num_left_nodes_, left_node);
00972   DCHECK_GT(epsilon_, 0);
00973   typename GraphType::OutgoingArcIterator arc_it(graph_, left_node);
00974   // If the input problem is feasible, it is always the case that
00975   // arc_it.Ok(), i.e., that there is at least one arc incident to
00976   // left_node.
00977   DCHECK(arc_it.Ok());
00978   ArcIndex best_arc = arc_it.Index();
00979   if (best_arc == matched_arc_[left_node]) {
00980     arc_it.Next();
00981     if (arc_it.Ok()) {
00982       best_arc = arc_it.Index();
00983     }
00984   }
00985   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00986   if (!arc_it.Ok()) {
00987     // Only one arc is incident to left_node, and the node is
00988     // currently matched along that arc, which must be the case in any
00989     // feasible solution. Therefore we implicitly price this node so
00990     // low that we will never consider unmatching it.
00991     return -(min_partial_reduced_cost + price_reduction_bound_);
00992   }
00993   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00994     const ArcIndex arc = arc_it.Index();
00995     if (arc != matched_arc_[left_node]) {
00996       const CostValue partial_reduced_cost = PartialReducedCost(arc);
00997       if (partial_reduced_cost < min_partial_reduced_cost) {
00998         min_partial_reduced_cost = partial_reduced_cost;
00999       }
01000     }
01001   }
01002   return -min_partial_reduced_cost;
01003 }

01018 // Only for debugging.
01019 template <typename GraphType>
01020 bool LinearSumAssignment<GraphType>::EpsilonOptimal() const {
01021   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
01022        node_it.Ok();
01023        node_it.Next()) {
01024     const NodeIndex left_node = node_it.Index();
01025     // Get the implicit price of left_node and make sure the reduced
01026     // costs of left_node's incident arcs are in bounds.
01027     CostValue left_node_price = ImplicitPrice(left_node);
01028     for (typename GraphType::OutgoingArcIterator arc_it(graph_, left_node);
01029          arc_it.Ok();
01030          arc_it.Next()) {
01031       const ArcIndex arc = arc_it.Index();
01032       const CostValue reduced_cost =
01033           left_node_price + PartialReducedCost(arc);
01034       // Note the asymmetric definition of epsilon-optimality that we
01035       // use because it means we can saturate all admissible arcs in
01036       // the beginning of Refine() just by unmatching all matched
01037       // nodes.
01038       if (matched_arc_[left_node] == arc) {
01039         // The reverse arc is residual. Epsilon-optimality requires
01040         // that the reduced cost of the forward arc be at most
01041         // epsilon_.
01042         if (reduced_cost > epsilon_) {
01043           return false;
01044         }
01045       } else {
01046         // The forward arc is residual. Epsilon-optimality requires
01047         // that the reduced cost of the forward arc be at least zero.
01048         if (reduced_cost < 0) {
01049           return false;
01050         }
01051       }
01052     }
01053   }
01054   return true;
01055 }

01057 template <typename GraphType>
01058 bool LinearSumAssignment<GraphType>::FinalizeSetup() {
01059   epsilon_ = largest_scaled_cost_magnitude_;
01060   VLOG(2) << "Largest given cost magnitude: " <<
01061       largest_scaled_cost_magnitude_ / cost_scaling_factor_;
01062   // Initialize left-side node-indexed arrays.
01063   typename GraphType::NodeIterator node_it(graph_);
01064   for (; node_it.Ok(); node_it.Next()) {
01065     const NodeIndex node = node_it.Index();
01066     if (node >= num_left_nodes_) {
01067       break;
01068     }
01069     matched_arc_.Set(node, GraphType::kNilArc);
01070   }
01071   // Initialize right-side node-indexed arrays. Example: prices are
01072   // stored only for right-side nodes.
01073   for (; node_it.Ok(); node_it.Next()) {
01074     const NodeIndex node = node_it.Index();
01075     price_.Set(node, 0);
01076     matched_node_.Set(node, GraphType::kNilNode);
01077   }
01078   bool in_range;
01079   price_lower_bound_ = -PriceChangeBound(alpha_ - 1, &in_range);
01080   DCHECK_LE(price_lower_bound_, 0);
01081   if (!in_range) {
01082     LOG(WARNING) << "Price change bound exceeds range of representable "
01083                  << "costs; arithmetic overflow is not ruled out.";
01084   }
01085   return in_range;
01086 }

01088 template <typename GraphType>
01089 void LinearSumAssignment<GraphType>::ReportAndAccumulateStats() {
01090   total_stats_.Add(iteration_stats_);
01091   VLOG(3) << "Iteration stats: " << iteration_stats_.StatsString();
01092   iteration_stats_.Clear();
01093 }

01095 template <typename GraphType>
01096 bool LinearSumAssignment<GraphType>::ComputeAssignment() {
01097   // Note: FinalizeSetup() might have been called already by white-box
01098   // test code or by a client that wants to react to the possibility
01099   // of overflow before solving the given problem, but FinalizeSetup()
01100   // is idempotent and reasonably fast, so we call it unconditionally
01101   // here.
01102   FinalizeSetup();
01103   bool ok = graph_.num_nodes() == 2 * num_left_nodes_;
01104   DCHECK(!ok || EpsilonOptimal());
01105   while (ok && epsilon_ > kMinEpsilon) {
01106     ok &= UpdateEpsilon();
01107     ok &= Refine();
01108     ReportAndAccumulateStats();
01109     DCHECK(!ok || EpsilonOptimal());
01110     DCHECK(!ok || AllMatched());
01111   }
01112   success_ = ok;
01113   VLOG(1) << "Overall stats: " << total_stats_.StatsString();
01114   return ok;
01115 }

01124        node_it.Ok();
