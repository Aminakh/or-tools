00014 // An implementation of a cost-scaling push-relabel algorithm for the
00015 // assignment problem (minimum-cost perfect bipartite matching), from
00016 // the paper of Goldberg and Kennedy (1995).
00017 //
00018 // This implementation finds the minimum-cost perfect assignment in
00019 // the given graph with integral edge weights set through the
00020 // SetArcCost method.
00021 //
00022 // Example usage:
00023 //
00024 // #include "graph/ebert_graph.h"
00025 // #include "graph/linear_assignment.h"
00026 // ...
00027 // ::operations_research::NodeIndex num_nodes = ...;
00028 // ::operations_research::NodeIndex num_left_nodes = num_nodes / 2;
00029 // // Define a num_nodes/2 by num_nodes/2 assignment problem:
00030 // ::operations_research::ArcIndex num_forward_arcs = ...;
00031 // ::operations_research::StarGraph g(num_nodes, num_arcs);
00032 // ::operations_research::LinearSumAssignment a(g, num_left_nodes);
00033 // for (int i = 0; i < num_forward_arcs; ++i) {
00034 //   ::operations_research::NodeIndex this_arc_head = ...;
00035 //   ::operations_research::NodeIndex this_arc_tail = ...;
00036 //   ::operations_research::CostValue this_arc_cost = ...;
00037 //   ::operations_research::ArcIndex this_arc_index =
00038 //       g.AddArc(this_arc_tail, this_arc_head);
00039 //   a.SetArcCost(this_arc_index, this_arc_cost);
00040 //  }
00041 //  // Compute the optimum assignment.
00042 //  bool success = a.ComputeAssignment();
00043 //  // Retrieve the cost of the optimum assignment.
00044 //  CostValue optimum_cost = a.GetCost();
00045 //  // Retrieve the node-node correspondence of the optimum assignment and the
00046 //  // cost of each node pairing.
00047 //  for (::operations_research::LinearSumAssignment::BipartiteLeftNodeIterator
00048 //         node_it(a);

00052 //    ::operations_research::NodeIndex right_node = a.GetMate(left_node);

00058 // In the following, we consider a bipartite graph
00059 //   G = (V = X union Y, E subset XxY),
00060 // where V denodes the set of nodes (vertices) in the graph, E denotes
00061 // the set of arcs (edges), n = |V| denotes the number of nodes in the
00062 // graph, and m = |E| denotes the number of arcs in the graph.
00063 //
00064 // The set of nodes is divided into two parts, X and Y, and every arc
00065 // must go between a node of X and a node of Y. With each arc is
00066 // associated a cost c(v, w). A matching M is a subset of E with the
00067 // property that no two arcs in M have a head or tail node in common,
00068 // and a perfect matching is a matching that touches every node in the
00069 // graph. The cost of a matching M is the sum of the costs of all the
00070 // arcs in M.
00071 //
00072 // The assignment problem is to find a perfect matching of minimum
00073 // cost in the given bipartite graph. The present algorithm reduces
00074 // the assignment problem to an instance of the minimum-cost flow
00075 // problem and takes advantage of special properties of the resulting
00076 // minimum-cost flow problem to solve it efficiently using a
00077 // push-relabel method. For more information about minimum-cost flow
00078 // see or-tools/graph/min_cost_flow.h
00079 //
00080 // The method used here is the cost-scaling approach for the
00081 // minimum-cost circulation problem as described in [Goldberg and
00082 // Tarjan] with some technical modifications:
00083 // 1. For efficiency, we solve a transportation problem instead of
00084 //    minimum-cost circulation. We might revisit this decision if it
00085 //    is important to handle problems in which no perfect matching
00086 //    exists.
00087 // 2. We use a modified "asymmetric" notion of epsilon-optimality in
00088 //    which left-to-right residual arcs are required to have reduced
00089 //    cost bounded below by zero and right-to-left residual arcs are
00090 //    required to have reduced cost bounded below by -epsilon. For
00091 //    each residual arc direction, the reduced-cost threshold for
00092 //    admissibility is epsilon/2 above the threshold for epsilon
00093 //    optimality.
00094 // 3. We do not limit the applicability of the relabeling operation to
00095 //    nodes with excess. Instead we use the double-push operation
00096 //    (discussed in the Goldberg and Kennedy CSA paper and Kennedy's
00097 //    thesis) which relabels right-side nodes just *after* they have
00098 //    been discharged.
00099 // The above differences are explained in detail in [Kennedy's thesis]
00100 // and explained not quite as cleanly in [Goldberg and Kennedy's CSA
00101 // paper]. But note that the thesis explanation uses a value of
00102 // epsilon that's double what we use here.
00103 //
00104 // Some definitions:
00105 //   Active: A node is called active when it has excess. It is
00106 //     eligible to be pushed from. In this implementation, every active
00107 //     node is on the left side of the graph where prices are determined
00108 //     implicitly, so no left-side relabeling is necessary before
00109 //     pushing from an active node. We do, however, need to compute
00110 //     the implications for price changes on the affected right-side
00111 //     nodes.
00112 //   Admissible: A residual arc (one that can carry more flow) is
00113 //     called admissible when its reduced cost is small enough. We can
00114 //     push additional flow along such an arc without violating
00115 //     epsilon-optimality. In the case of a left-to-right residual
00116 //     arc, the reduced cost must be at most epsilon/2. In the case of
00117 //     a right-to-left residual arc, the reduced cost must be at most
00118 //     -epsilon/2. The careful reader will note that these thresholds
00119 //     are not used explicitly anywhere in this implementation, and
00120 //     the reason is the implicit pricing of left-side nodes.
00121 //   Reduced cost: Essentially an arc's reduced cost is its
00122 //     complementary slackness. In push-relabel algorithms this is
00123 //       c_p(v, w) = p(v) + c(v, w) - p(w),
00124 //     where p() is the node price function and c(v, w) is the cost of
00125 //     the arc from v to w. See min_cost_flow.h for more details.
00126 //   Partial reduced cost: We maintain prices implicitly for left-side
00127 //     nodes in this implementation, so instead of reduced costs we
00128 //     work with partial reduced costs, defined as
00129 //       c'_p(v, w) = c(v, w) - p(w).
00130 //
00131 // We check at initialization time for the possibility of arithmetic
00132 // overflow and warn if the given costs are too large. In many cases
00133 // the bound we use to trigger the warning is pessimistic so the given
00134 // problem can often be solved even if we warn that overflow is
00135 // possible.
00136 //
00137 // We don't use the interface from
00138 // operations_research/algorithms/hungarian.h because we want to be
00139 // able to express sparse problems efficiently.
00140 //
00141 // When asked to solve the given assignment problem we return a
00142 // boolean to indicate whether the given problem was feasible.
00143 //
00144 // References:
00145 // [ Goldberg and Kennedy's CSA paper ] A. V. Goldberg and R. Kennedy,
00146 // "An Efficient Cost Scaling Algorithm for the Assignment Problem."
00147 // Mathematical Programming, Vol. 71, pages 153-178, December 1995.
00148 //
00149 // [ Goldberg and Tarjan ] A. V. Goldberg and R. E. Tarjan, "Finding
00150 // Minimum-Cost Circulations by Successive Approximation." Mathematics
00151 // of Operations Research, Vol. 15, No. 3, pages 430-466, August 1990.
00152 //
00153 // [ Kennedy's thesis ] J. R. Kennedy, Jr., "Solving Unweighted and
00154 // Weighted Bipartite Matching Problems in Theory and Practice."
00155 // Stanford University Doctoral Dissertation, Department of Computer
00156 // Science, 1995.

00496   // A lower bound on the price of any node at any time throughout the
00497   // computation. A price below this level proves infeasibility.
00498   //
00499   // The value of this lower bound is determined according to the
00500   // following sketch: Suppose the price decrease of every node in the
00501   // iteration with epsilon_ == x is bounded by B(x) which is
00502   // proportional to x. Then the total price decrease of every node
00503   // across all iterations is bounded above by
00504   //   B(C/alpha) + B(C/alpha^2) + ... + B(kMinEpsilon)
00505   //   == B(C/alpha) * alpha / (alpha - 1)
00506   //   == B(C) / (alpha - 1).
00507   // Therefore we set price_lower_bound_ = -ceil(B(C) / (alpha - 1))
00508   // where B() is the expression that determines
00509   // price_reduction_bound_, discussed below.
00510   CostValue price_lower_bound_;
00511 
00512   // An upper bound on the amount that a single node's price can
00513   // decrease in a single scaling iteration. In each iteration, this
00514   // value corresponds to B(epsilon_) in the comments describing
00515   // price_lower_bound_ above. Exceeding this amount of price decrease
00516   // in one iteration proves that there is some excess that cannot
00517   // reach a deficit, i.e., that the problem is infeasible.
00518   //
00519   // Let v be a node with excess and suppose P is a simple residual
00520   // path P from v to some node w with deficit such that reverse(P) is
00521   // residual at the beginning of this iteration (such a path is
00522   // guaranteed to exist by feasibility -- see lemma 5.7 in Goldberg
00523   // and Tarjan). We have c_p(P) = p(v) + c(P) - p(w) and of those
00524   // three terms, only p(v) may have changed during this iteration
00525   // because w has a deficit and nodes with deficits are not
00526   // relabeled. Assuming without loss of generality that p == 0 and
00527   // c_p == c at the beginning of this iteration, we seek a bound on
00528   // simply
00529   //   p(v) = c_p(P) - c(P).
00530   // Let arc a lie on P.
00531   // Case 1: a is a forward arc. Then c_p(a) >= 0 and the reverse of a
00532   //         was residual when this iteration began. By
00533   //         approximate optimality at the end of the prior iteration,
00534   //         c(a) < alpha * epsilon. So
00535   //           c_p(a) - c(a) > -alpha * epsilon_.
00536   // Case 2: a is a reverse arc. Then c_p(a) >= -epsilon_ and the
00537   //         reverse of a was residual when this iteration began. By
00538   //         approximate optimality at the end of the prior iteration,
00539   //         c(a) < 0. So
00540   //           c_p(a) - c(a) > -epsilon_.
00541   // Nodes with excess are only on the left and nodes with deficit are
00542   // only on the right; there are at most n - 1 arcs on the path P,
00543   // making up at most (n-1)/2 left-right-left arc pairs. Each
00544   // pair's contribution to c_p(P) - c(P) is bounded below by
00545   // most (n-1)/2 of those are forward arcs and (n-2)/2 of them are reverse
00546   // arcs, so
00547   //   p(v) = c_p(P) - c(P)
00548   //        > (n-1)/2 * (-alpha * epsilon_ - epsilon_)
00549   //        = -(n-1)/2 * epsilon_ * (1 + alpha).
00550   // So we set
00551   //   price_reduction_bound_ = ceil((n-1)/2 * epsilon * (1 + alpha)).
00552   CostValue price_reduction_bound_;
00553 
00554   // Computes the value of price_reduction_bound_ for an iteration,
00555   // given the new value of epsilon_, on the assumption that the value
00556   // of epsilon_ for the previous iteration was no more than a factor
00557   // of alpha_ times the new value. Because the expression computed
00558   // here is used in at least one place where we want an additional
00559   // factor in the denominator, we take that factor as an argument.
00560   //
00561   // Avoids overflow in computing the bound.
00562   inline CostValue PriceChangeBound(CostValue extra_divisor,
00563                                     bool* in_range) const {
00564     const CostValue n = graph_.num_nodes();
00565     // We work in double-precision floating point to determine whether
00566     // we'll overflow the integral CostValue type's range of
00567     // representation. Switching between integer and double is a
00568     // rather expensive operation, but we do this only once per
00569     // scaling iteration, so we can afford it rather than resort to
00570     // complex and subtle tricks within the bounds of integer
00571     // arithmetic.
00572     //
00573     // To understand the values of numerator and denominator here, you
00574     // will want to read the comments above about price_lower_bound_
00575     // and price_reduction_bound_, and have a pencil handy. :-)
00576     const double numerator = (static_cast<double>(n - 1) *
00577                               static_cast<double>(epsilon_ * (1 + alpha_)));
00578     const double denominator = static_cast<double>(2 * extra_divisor);
00579     const double quotient = numerator / denominator;
00580     const double limit =
00581         static_cast<double>(std::numeric_limits<CostValue>::max());
00582     if (quotient > limit) {
00583       // Our integer computations could overflow.
00584       if (in_range != NULL) *in_range = false;
00585       return std::numeric_limits<CostValue>::max();
00586     } else {
00587       if (in_range != NULL) *in_range = true;
00588       return static_cast<CostValue>(quotient);
00589     }
00590   }
00591 
00592   // A scaled record of the largest arc-cost magnitude we've been
00593   // given during problem setup. This is used to set the initial value
00594   // of epsilon_, which in turn is used not only as the error
00595   // parameter but also to determine whether we risk arithmetic
00596   // overflow during the algorithm.
00597   CostValue largest_scaled_cost_magnitude_;
00598 
00599   // The total excess in the graph. Given our asymmetric definition of
00600   // epsilon-optimality and our use of the double-push operation, this
00601   // equals the number of unmatched left-side nodes.
00602   NodeIndex total_excess_;
00603 
00604   // Indexed by node index, the price_ values are maintained only for
00605   // right-side nodes. These are kept as int64 values instead of
00606   // CostValues because we scale the initial arc costs up by the
00607   // number of nodes in order to use integer arithmetic everywhere,
00608   // and such scaling up increases the risk of overflow.
00609   CostArray price_;
00610 
00611   // Indexed by left-side node index, the matched_arc_ array gives the
00612   // arc index of the arc matching any given left-side node, or
00613   // StarGraph::kNilArc if the node is unmatched.
00614   ArcIndexArray matched_arc_;

00618   // right-side node, or StarGraph::kNilNode if the right-side node is

00620   NodeIndexArray matched_node_;
