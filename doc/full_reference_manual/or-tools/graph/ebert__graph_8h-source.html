00017 // An implementation (with some improvements) of the star-representation of a
00018 // graph as described in J. Ebert, "A versatile data structure for
00019 // edge-oriented graph algorithms." Communications of the ACM 30(6):513-519
00020 // (June 1987). http://portal.acm.org/citation.cfm?id=214769
00021 // Both forward- and backward-star representations are contained in this
00022 // representation.
00023 
00024 // The graph is represented with three arrays.
00025 // Let n be the number of nodes and m be the number of arcs.
00026 // Let i be an integer in [0..m-1], denoting the index of an arc.
00027 //  * node_[i] contains the end-node of arc i,
00028 //  * node_[-i-1] contains the start-node of arc i.
00029 // Note that in two's-complement arithmetic, -i-1 = ~i.
00030 // Consequently:
00031 //  * node_[~i] contains the start-node of the arc reverse to arc i,
00032 //  * node_[i] contains the end-node of the arc reverse to arc i.
00033 //  Note that if arc (u, v) is defined, then the data structure also stores
00034 //  (v, u).
00035 //  Arc ~i thus denotes the arc reverse to arc i.
00036 //  This is what makes this representation useful for undirected graphs,
00037 //  and for implementing algorithms like two-directions shortest-path.
00038 //
00039 // Now, for an integer u in [0..n-1] denoting the index of a node:
00040 //  * first_incident_arc_[u] denotes the first arc in the adjacency list of u.
00041 //  * going from an arc i, the adjacency list can be traversed using
00042 //    j = next_adjacent_arc_[i].
00043 //
00044 // This implementation has the following benefits:
00045 //  * It is able to handle both directed or undirected graphs.
00046 //  * Being based on indices, it is easily serializable. Only the contents
00047 //    of the node_ array needs to be stored.
00048 //  * The sizes of node indices and arc indices can be specified in number of
00049 //    bytes. For example, it is possible to store indices on 5 bytes or 40 bits,
00050 //    which seems like a good compromise for the architectures of 2010. This
00051 //    represents a 37.5% savings compared to a pointer-based implementation.
00052 //    Taking into account the fact that no extra pointer to the reverse arc is
00053 //    needed, only 80 bits are needed to store an arc instead of 192 bits if a
00054 //    pointer-based representation were used.
00055 //  * The representation can be recomputed if edges have been loaded from
00056 //    external memory or if edges have been re-ordered.
00057 //  * The memory consumption is: 2 * m * NodeIndexSize
00058 //                             + 2 * m * ArcIndexSize
00059 //                             + n * ArcIndexSize
00060 //
00061 // This implementation differs from the implementation described in [Ebert 1987]
00062 // in the following respects:
00063 //  * arcs are represented using an (i, ~i) approach, whereas Ebert used
00064 //    (i, -i). Indices for direct arcs thus start at 0, in a fashion that is
00065 //    compatible with the index numbering in C and C++. Note that we also tested
00066 //    a (2*i, 2*i+1) storage pattern, which did not show any speed benefit, and
00067 //    made the use of the API much more difficult.
00068 //  * because of this, the 'nil' values for nodes and arcs are not 0, as Ebert
00069 //    first described. The value for the 'nil' node is set to -1, while the
00070 //    value for the 'nil' arc is set to the smallest integer representable with
00071 //    ArcIndexSize bytes.
00072 //  * it is possible to add arcs to the graph, with AddArc, in a much simpler
00073 //    way than described by Ebert.
00074 //  * TODO(user) it is possible to group all the outgoing (resp. incoming) arcs
00075 //    of a node to allow to traverse the outgoing (resp. incoming) arcs in
00076 //    O(out_degree(node)) (resp. O(in_degree(node))) instead of O(degree(node)).
00077 //  * TODO(user) it is possible to implement arc deletion and garbage collection
00078 //    in an efficient (relatively) manner. For the time being we haven't seen an
00079 //    application to this.

00585   ArcIndex LookUpArc(const NodeIndex tail, const NodeIndex head) const {
00586     for (ArcIndex arc = FirstOutgoingArc(tail);
00587          arc != kNilArc;
00588          arc = NextOutgoingArc(arc)) {
00589       if (Head(arc) == head) {
00590         return arc;
00591       }
00592     }
00593     return kNilArc;
00594   }
00595 
00596   // Returns the tail or start-node of arc if it is positive
00597   // (i.e. it is taken in the direction it was entered in the graph),
00598   // and the head or end-node otherwise. 'This' in Ebert's paper.
00599   NodeIndex DirectArcTail(const ArcIndex arc) const {
00600     return Tail(DirectArc(arc));
00601   }
00602 
00603   // Returns the head or end-node of arc if it is positive
00604   // (i.e. it is taken in the direction it was entered in the graph),
00605   // and the tail or start-node otherwise. 'That' in Ebert's paper.
00606   NodeIndex DirectArcHead(const ArcIndex arc) const {
00607     return Head(DirectArc(arc));
00608   }
00609 
00610   // Returns the arc in normal/direct direction.
00611   ArcIndex DirectArc(const ArcIndex arc) const {
00612     DCHECK(CheckArcValidity(arc));
00613     return std::max(arc, Opposite(arc));
00614   }
00615 
00616   // Returns the arc in reverse direction.
00617   ArcIndex ReverseArc(const ArcIndex arc) const {
00618     DCHECK(CheckArcValidity(arc));
00619     return std::min(arc, Opposite(arc));
00620   }
00621 
00622   // Returns the opposite arc, i.e the direct arc is the arc is in reverse
00623   // direction, and the reverse arc if the arc is direct.
00624   ArcIndex Opposite(const ArcIndex arc) const {
00625     const ArcIndex opposite = ~arc;
00626     DCHECK(CheckArcValidity(arc));
00627     DCHECK(CheckArcValidity(opposite));
00628     return opposite;
00629   }
00630 
00631   // Returns true if the arc is direct.
00632   bool IsDirect(const ArcIndex arc) const {
00633     DCHECK(CheckArcBounds(arc));
00634     return arc != kNilArc && arc >= 0;
00635   }
00636 
00637   // Returns true if the arc is in the reverse direction.
00638   bool IsReverse(const ArcIndex arc) const {
00639     DCHECK(CheckArcBounds(arc));
00640     return arc != kNilArc && arc < 0;
00641   }
00642 
00643   // Returns true if arc is incident to node.
00644   bool IsIncident(ArcIndex arc, NodeIndex node) const {
00645     return IsIncoming(arc, node) || IsOutgoing(arc, node);
00646   }
00647 
00648   // Returns true if arc is incoming to node.
00649   bool IsIncoming(ArcIndex arc, NodeIndex node) const {
00650     return DirectArcHead(arc) == node;
00651   }
00652 
00653   // Returns true if arc is outgoing from node.
00654   bool IsOutgoing(ArcIndex arc, NodeIndex node) const {
00655     return DirectArcTail(arc) == node;
00656   }
00657 
00658   // Recreates the next_adjacent_arc_ and first_incident_arc_ variables from
00659   // the array node_ in O(n + m) time.
00660   // This is useful if node_ array has been sorted according to a given
00661   // criterion, for example.
00662   void BuildRepresentation() {
00663     first_incident_arc_.Assign(kNilArc);
00664     for (ArcIndex arc = kFirstArc; arc <  max_num_arcs_; ++arc) {
00665       Attach(arc);
00666     }
00667     representation_clean_ = true;
00668   }
00669 
00670   // Returns a debug string containing all the information contained in the
00671   // data structure in raw form.
00672   string DebugString() const {
00673     DCHECK(representation_clean_);
00674     string result = "Arcs:(node, next arc) :\n";
00675     for (ArcIndex arc = -num_arcs_; arc < num_arcs_; ++arc) {
00676       result += " " + ArcDebugString(arc) + ":(" + NodeDebugString(node_[arc])
00677               + "," + ArcDebugString(next_adjacent_arc_[arc]) + ")\n";
00678     }
00679     result += "Node:First arc :\n";
00680     for (NodeIndex node = kFirstNode; node < num_nodes_; ++node) {
00681       result += " " + NodeDebugString(node) + ":"
00682               + ArcDebugString(first_incident_arc_[node]) + "\n";
00683     }
00684     return result;
00685   }
00686 
00687   string NodeDebugString(const NodeIndex node) const {
00688     if (node == kNilNode) {
00689       return "NilNode";
00690     } else {
00691       return StringPrintf("%lld", static_cast<int64>(node));
00692     }
00693   }
00694 
00695   string ArcDebugString(const ArcIndex arc) const {
00696     if (arc == kNilArc) {
00697       return "NilArc";
00698     } else {
00699       return StringPrintf("%lld", static_cast<int64>(arc));
00700     }
00701   }

00703  private:
00704   // Returns kNilNode if the graph has no nodes or node if it has at least one
00705   // node. Useful for initializing iterators correctly in the case of empty
00706   // graphs.
00707   NodeIndex StartNode(NodeIndex node) const {
00708     return num_nodes_ == 0 ? kNilNode : node;
00709   }
00710 
00711   // Returns kNilArc if the graph has no arcs arc if it has at least one arc.
00712   // Useful for initializing iterators correctly in the case of empty graphs.
00713   ArcIndex StartArc(ArcIndex arc) const {
00714     return num_arcs_ == 0 ? kNilArc : arc;
00715   }
00716 
00717   // Returns the first outgoing arc for node.
00718   ArcIndex FirstOutgoingArc(const NodeIndex node) const {
00719     DCHECK(CheckNodeValidity(node));
00720     return FindNextOutgoingArc(FirstIncidentArc(node));
00721   }
00722 
00723   // Returns the outgoing arc following the argument in the adjacency list.
00724   ArcIndex NextOutgoingArc(const ArcIndex arc) const {
00725     DCHECK(CheckArcValidity(arc));
00726     return FindNextOutgoingArc(NextAdjacentArc(DirectArc(arc)));
00727   }
00728 
00729   // Returns the first incoming arc for node.
00730   ArcIndex FirstIncomingArc(const NodeIndex node) const {
00731     DCHECK_LE(kFirstNode, node);
00732     DCHECK_GE(max_num_nodes_, node);
00733     return FindNextIncomingArc(FirstIncidentArc(node));
00734   }
00735 
00736   // Returns the incoming arc following the argument in the adjacency list.
00737   ArcIndex NextIncomingArc(const ArcIndex arc) const {
00738     DCHECK(CheckArcValidity(arc));
00739     return FindNextIncomingArc(NextAdjacentArc(ReverseArc(arc)));
00740   }
00741 
00742   // Returns the first arc in node's incidence list.
00743   ArcIndex FirstIncidentArc(const NodeIndex node) const {
00744     DCHECK(representation_clean_);
00745     DCHECK(CheckNodeValidity(node));
00746     return first_incident_arc_[node];
00747   }
00748 
00749   // Returns the next arc following the passed argument in its adjacency list.
00750   ArcIndex NextAdjacentArc(const ArcIndex arc) const {
00751     DCHECK(representation_clean_);
00752     DCHECK(CheckArcValidity(arc));
00753     return next_adjacent_arc_[arc];
00754   }
00755 
00756   // Returns the node following the argument in the graph.
00757   // Returns kNilNode (= end) if the range of nodes has been exhausted.
00758   // It is called by NodeIterator::Next() and as such does not expect do be
00759   // passed an argument equal to kNilNode.
00760   // This is why the return line is simplified from
00761   // return (node == kNilNode || next_node >= num_nodes_)
00762   //        ? kNilNode : next_node;
00763   // to
00764   // return next_node < num_nodes_ ? next_node : kNilNode;
00765   NodeIndex NextNode(const NodeIndex node) const {
00766     DCHECK(CheckNodeValidity(node));
00767     const NodeIndex next_node = node + 1;
00768     return next_node < num_nodes_ ? next_node : kNilNode;
00769   }
00770 
00771   // Returns the arc following the argument in the graph.
00772   // Returns kNilArc (= end) if the range of arcs has been exhausted.
00773   // It is called by ArcIterator::Next() and as such does not expect do be
00774   // passed an argument equal to kNilArc.
00775   // This is why the return line is simplified from
00776   // return ( arc == kNilArc || next_arc >= num_arcs_) ? kNilArc : next_arc;
00777   // to
00778   // return next_arc < num_arcs_ ? next_arc : kNilArc;
00779   ArcIndex NextArc(const ArcIndex arc) const {
00780     DCHECK(CheckArcValidity(arc));
00781     const ArcIndex next_arc = arc + 1;
00782     return next_arc < num_arcs_ ? next_arc : kNilArc;
00783   }
00784 
00785   // Using the SetTail() method implies that the BuildRepresentation()
00786   // method must be called to restore consistency before the graph is
00787   // used.
00788   void SetTail(const ArcIndex arc, const NodeIndex tail) {
00789     representation_clean_ = false;
00790     node_.Set(Opposite(arc), tail);
00791   }
00792 
00793   // Using the SetHead() method implies that the BuildRepresentation()
00794   // method must be called to restore consistency before the graph is
00795   // used.
00796   void SetHead(const ArcIndex arc, const NodeIndex head) {
00797     representation_clean_ = false;
00798     node_.Set(arc, head);
00799   }
00800 
00801   // Utility method to attach a new arc.
00802   void Attach(ArcIndex arc) {
00803     DCHECK(CheckArcValidity(arc));
00804     const NodeIndex tail = node_[Opposite(arc)];
00805     DCHECK(CheckNodeValidity(tail));
00806     next_adjacent_arc_.Set(arc, first_incident_arc_[tail]);
00807     first_incident_arc_.Set(tail, arc);
00808     const NodeIndex head = node_[arc];
00809     DCHECK(CheckNodeValidity(head));
00810     next_adjacent_arc_.Set(Opposite(arc), first_incident_arc_[head]);
00811     first_incident_arc_.Set(head, Opposite(arc));
00812   }
00813 
00814   // Utility method that finds the next outgoing arc.
00815   ArcIndex FindNextOutgoingArc(ArcIndex arc) const {
00816     DCHECK(CheckArcBounds(arc));
00817     while (IsReverse(arc)) {
00818       arc = NextAdjacentArc(arc);
00819       DCHECK(CheckArcBounds(arc));
00820     }
00821     return arc;
00822   }
00823 
00824   // Utility method that finds the next incoming arc.
00825   ArcIndex FindNextIncomingArc(ArcIndex arc) const {
00826     DCHECK(CheckArcBounds(arc));
00827     while (IsDirect(arc)) {
00828       arc = NextAdjacentArc(arc);
00829       DCHECK(CheckArcBounds(arc));
00830     }
00831     return arc;
00832   }
00833 
00834   // The maximum number of nodes that the graph can hold.
00835   NodeIndex max_num_nodes_;
00836 
00837   // The maximum number of arcs that the graph can hold.
00838   ArcIndex max_num_arcs_;
00839 
00840   // The maximum index of the node currently held by the graph.
00841   NodeIndex num_nodes_;
00842 
00843   // The current number of arcs held by the graph.
00844   ArcIndex num_arcs_;
00845 
00846   // Array of node indices. node_[i] contains the tail node of arc i.
00847   PackedArray<NodeIndexSize> node_;
00848 
00849   // Array of next indices.
00850   // next_adjacent_arc_[i] contains the next arc in the adjacency list of arc i.
00851   PackedArray<ArcIndexSize> next_adjacent_arc_;
00852 
00853   // Array of arc indices. first_incident_arc_[i] contains the first arc
00854   // incident to node i.
00855   PackedArray<ArcIndexSize> first_incident_arc_;
00856 
00857   // Flag to indicate that BuildRepresentation() needs to be called
00858   // before the adjacency lists are examined. Only for DCHECK in debug
00859   // builds.
00860   bool representation_clean_;
00861 };
