00017 // An implementation (with some improvements) of the star-representation of a
00018 // graph as described in J. Ebert, "A versatile data structure for
00019 // edge-oriented graph algorithms." Communications of the ACM 30(6):513-519
00020 // (June 1987). http://portal.acm.org/citation.cfm?id=214769
00021 // Both forward- and backward-star representations are contained in this
00022 // representation.
00023 
00024 // The graph is represented with three arrays.
00025 // Let n be the number of nodes and m be the number of arcs.
00026 // Let i be an integer in [0..m-1], denoting the index of an arc.
00027 //  * node_[i] contains the end-node of arc i,
00028 //  * node_[-i-1] contains the start-node of arc i.
00029 // Note that in two's-complement arithmetic, -i-1 = ~i.
00030 // Consequently:
00031 //  * node_[~i] contains the start-node of the arc reverse to arc i,
00032 //  * node_[i] contains the end-node of the arc reverse to arc i.
00033 //  Note that if arc (u, v) is defined, then the data structure also stores
00034 //  (v, u).
00035 //  Arc ~i thus denotes the arc reverse to arc i.
00036 //  This is what makes this representation useful for undirected graphs,
00037 //  and for implementing algorithms like two-directions shortest-path.
00038 //
00039 // Now, for an integer u in [0..n-1] denoting the index of a node:
00040 //  * first_incident_arc_[u] denotes the first arc in the adjacency list of u.
00041 //  * going from an arc i, the adjacency list can be traversed using
00042 //    j = next_adjacent_arc_[i].
00043 //
00044 // This implementation has the following benefits:
00045 //  * It is able to handle both directed or undirected graphs.
00046 //  * Being based on indices, it is easily serializable. Only the contents
00047 //    of the node_ array needs to be stored.
00048 //  * The sizes of node indices and arc indices can be specified in number of
00049 //    bytes. For example, it is possible to store indices on 5 bytes or 40 bits,
00050 //    which seems like a good compromise for the architectures of 2010. This
00051 //    represents a 37.5% savings compared to a pointer-based implementation.
00052 //    Taking into account the fact that no extra pointer to the reverse arc is
00053 //    needed, only 80 bits are needed to store an arc instead of 192 bits if a
00054 //    pointer-based representation were used.
00055 //  * The representation can be recomputed if edges have been loaded from
00056 //    external memory or if edges have been re-ordered.
00057 //  * The memory consumption is: 2 * m * NodeIndexSize
00058 //                             + 2 * m * ArcIndexSize
00059 //                             + n * ArcIndexSize
00060 //
00061 // This implementation differs from the implementation described in [Ebert 1987]
00062 // in the following respects:
00063 //  * arcs are represented using an (i, ~i) approach, whereas Ebert used
00064 //    (i, -i). Indices for direct arcs thus start at 0, in a fashion that is
00065 //    compatible with the index numbering in C and C++. Note that we also tested
00066 //    a (2*i, 2*i+1) storage pattern, which did not show any speed benefit, and
00067 //    made the use of the API much more difficult.
00068 //  * because of this, the 'nil' values for nodes and arcs are not 0, as Ebert
00069 //    first described. The value for the 'nil' node is set to -1, while the
00070 //    value for the 'nil' arc is set to the smallest integer representable with
00071 //    ArcIndexSize bytes.
00072 //  * it is possible to add arcs to the graph, with AddArc, in a much simpler
00073 //    way than described by Ebert.
00074 //  * TODO(user) it is possible to group all the outgoing (resp. incoming) arcs
00075 //    of a node to allow to traverse the outgoing (resp. incoming) arcs in
00076 //    O(out_degree(node)) (resp. O(in_degree(node))) instead of O(degree(node)).
00077 //  * TODO(user) it is possible to implement arc deletion and garbage collection
00078 //    in an efficient (relatively) manner. For the time being we haven't seen an
00079 //    application to this.

00584          arc = NextOutgoingArc(arc)) {
00585       if (Head(arc) == head) {
00586         return arc;
00587       }
00588     }
00589     return kNilArc;
00590   }
00591 
00592   // Returns the tail or start-node of arc if it is positive
00593   // (i.e. it is taken in the direction it was entered in the graph),
00594   // and the head or end-node otherwise. 'This' in Ebert's paper.
00595   NodeIndex DirectArcTail(const ArcIndex arc) const {
00596     return Tail(DirectArc(arc));
00597   }
00598 
00599   // Returns the head or end-node of arc if it is positive
00600   // (i.e. it is taken in the direction it was entered in the graph),
00601   // and the tail or start-node otherwise. 'That' in Ebert's paper.
00602   NodeIndex DirectArcHead(const ArcIndex arc) const {
00603     return Head(DirectArc(arc));
00604   }
00605 
00606   // Returns the arc in normal/direct direction.
00607   ArcIndex DirectArc(const ArcIndex arc) const {
00608     DCHECK(CheckArcValidity(arc));
00609     return std::max(arc, Opposite(arc));
00610   }
00611 
00612   // Returns the arc in reverse direction.
00613   ArcIndex ReverseArc(const ArcIndex arc) const {
00614     DCHECK(CheckArcValidity(arc));
00615     return std::min(arc, Opposite(arc));
00616   }
00617 
00618   // Returns the opposite arc, i.e the direct arc is the arc is in reverse
00619   // direction, and the reverse arc if the arc is direct.
00620   ArcIndex Opposite(const ArcIndex arc) const {
00621     const ArcIndex opposite = ~arc;
00622     DCHECK(CheckArcValidity(arc));
00623     DCHECK(CheckArcValidity(opposite));
00624     return opposite;
00625   }
00626 
00627   // Returns true if the arc is direct.
00628   bool IsDirect(const ArcIndex arc) const {
00629     DCHECK(CheckArcBounds(arc));
00630     return arc != kNilArc && arc >= 0;
00631   }
00632 
00633   // Returns true if the arc is in the reverse direction.
00634   bool IsReverse(const ArcIndex arc) const {
00635     DCHECK(CheckArcBounds(arc));
00636     return arc != kNilArc && arc < 0;
00637   }
00638 
00639   // Returns true if arc is incident to node.
00640   bool IsIncident(ArcIndex arc, NodeIndex node) const {
00641     return IsIncoming(arc, node) || IsOutgoing(arc, node);
00642   }
00643 
00644   // Returns true if arc is incoming to node.
00645   bool IsIncoming(ArcIndex arc, NodeIndex node) const {
00646     return DirectArcHead(arc) == node;
00647   }
00648 
00649   // Returns true if arc is outgoing from node.
00650   bool IsOutgoing(ArcIndex arc, NodeIndex node) const {
00651     return DirectArcTail(arc) == node;
00652   }
00653 
00654   // Recreates the next_adjacent_arc_ and first_incident_arc_ variables from
00655   // the array node_ in O(n + m) time.
00656   // This is useful if node_ array has been sorted according to a given
00657   // criterion, for example.
00658   void BuildRepresentation() {
00659     first_incident_arc_.Assign(kNilArc);
00660     for (ArcIndex arc = kFirstArc; arc <  max_num_arcs_; ++arc) {
00661       Attach(arc);
00662     }
00663     representation_clean_ = true;
00664   }
00665 
00666   // Returns a debug string containing all the information contained in the
00667   // data structure in raw form.
00668   string DebugString() const {
00669     DCHECK(representation_clean_);
00670     string result = "Arcs:(node, next arc) :\n";
00671     for (ArcIndex arc = -num_arcs_; arc < num_arcs_; ++arc) {
00672       result += " " + ArcDebugString(arc) + ":(" + NodeDebugString(node_[arc])
00673               + "," + ArcDebugString(next_adjacent_arc_[arc]) + ")\n";
00674     }
00675     result += "Node:First arc :\n";
00676     for (NodeIndex node = kFirstNode; node < num_nodes_; ++node) {
00677       result += " " + NodeDebugString(node) + ":"
00678               + ArcDebugString(first_incident_arc_[node]) + "\n";
00679     }
00680     return result;
00681   }
00682 
00683   string NodeDebugString(const NodeIndex node) const {
00684     if (node == kNilNode) {
00685       return "NilNode";
00686     } else {
00687       return StringPrintf("%lld", static_cast<int64>(node));
00688     }
00689   }
00690 
00691   string ArcDebugString(const ArcIndex arc) const {
00692     if (arc == kNilArc) {
00693       return "NilArc";
00694     } else {
00695       return StringPrintf("%lld", static_cast<int64>(arc));
00696     }
00697   }

00699  private:
00700   // Returns kNilNode if the graph has no nodes or node if it has at least one
00701   // node. Useful for initializing iterators correctly in the case of empty
00702   // graphs.
00703   NodeIndex StartNode(NodeIndex node) const {
00704     return num_nodes_ == 0 ? kNilNode : node;
00705   }
00706 
00707   // Returns kNilArc if the graph has no arcs arc if it has at least one arc.
00708   // Useful for initializing iterators correctly in the case of empty graphs.
00709   ArcIndex StartArc(ArcIndex arc) const {
00710     return num_arcs_ == 0 ? kNilArc : arc;
00711   }
00712 
00713   // Returns the first outgoing arc for node.
00714   ArcIndex FirstOutgoingArc(const NodeIndex node) const {
00715     DCHECK(CheckNodeValidity(node));
00716     return FindNextOutgoingArc(FirstIncidentArc(node));
00717   }
00718 
00719   // Returns the outgoing arc following the argument in the adjacency list.
00720   ArcIndex NextOutgoingArc(const ArcIndex arc) const {
00721     DCHECK(CheckArcValidity(arc));
00722     return FindNextOutgoingArc(NextAdjacentArc(DirectArc(arc)));
00723   }
00724 
00725   // Returns the first incoming arc for node.
00726   ArcIndex FirstIncomingArc(const NodeIndex node) const {
00727     DCHECK_LE(kFirstNode, node);
00728     DCHECK_GE(max_num_nodes_, node);
00729     return FindNextIncomingArc(FirstIncidentArc(node));
00730   }
00731 
00732   // Returns the incoming arc following the argument in the adjacency list.
00733   ArcIndex NextIncomingArc(const ArcIndex arc) const {
00734     DCHECK(CheckArcValidity(arc));
00735     return FindNextIncomingArc(NextAdjacentArc(ReverseArc(arc)));
00736   }
00737 
00738   // Returns the first arc in node's incidence list.
00739   ArcIndex FirstIncidentArc(const NodeIndex node) const {
00740     DCHECK(representation_clean_);
00741     DCHECK(CheckNodeValidity(node));
00742     return first_incident_arc_[node];
00743   }
00744 
00745   // Returns the next arc following the passed argument in its adjacency list.
00746   ArcIndex NextAdjacentArc(const ArcIndex arc) const {
00747     DCHECK(representation_clean_);
00748     DCHECK(CheckArcValidity(arc));
00749     return next_adjacent_arc_[arc];
00750   }
00751 
00752   // Returns the node following the argument in the graph.
00753   // Returns kNilNode (= end) if the range of nodes has been exhausted.
00754   // It is called by NodeIterator::Next() and as such does not expect do be
00755   // passed an argument equal to kNilNode.
00756   // This is why the return line is simplified from
00757   // return (node == kNilNode || next_node >= num_nodes_)
00758   //        ? kNilNode : next_node;
00759   // to
00760   // return next_node < num_nodes_ ? next_node : kNilNode;
00761   NodeIndex NextNode(const NodeIndex node) const {
00762     DCHECK(CheckNodeValidity(node));
00763     const NodeIndex next_node = node + 1;
00764     return next_node < num_nodes_ ? next_node : kNilNode;
00765   }
00766 
00767   // Returns the arc following the argument in the graph.
00768   // Returns kNilArc (= end) if the range of arcs has been exhausted.
00769   // It is called by ArcIterator::Next() and as such does not expect do be
00770   // passed an argument equal to kNilArc.
00771   // This is why the return line is simplified from
00772   // return ( arc == kNilArc || next_arc >= num_arcs_) ? kNilArc : next_arc;
00773   // to
00774   // return next_arc < num_arcs_ ? next_arc : kNilArc;
00775   ArcIndex NextArc(const ArcIndex arc) const {
00776     DCHECK(CheckArcValidity(arc));
00777     const ArcIndex next_arc = arc + 1;
00778     return next_arc < num_arcs_ ? next_arc : kNilArc;
00779   }
00780 
00781   // Using the SetTail() method implies that the BuildRepresentation()
00782   // method must be called to restore consistency before the graph is
00783   // used.
00784   void SetTail(const ArcIndex arc, const NodeIndex tail) {
00785     representation_clean_ = false;
00786     node_.Set(Opposite(arc), tail);
00787   }
00788 
00789   // Using the SetHead() method implies that the BuildRepresentation()
00790   // method must be called to restore consistency before the graph is
00791   // used.
00792   void SetHead(const ArcIndex arc, const NodeIndex head) {
00793     representation_clean_ = false;
00794     node_.Set(arc, head);
00795   }
00796 
00797   // Utility method to attach a new arc.
00798   void Attach(ArcIndex arc) {
00799     DCHECK(CheckArcValidity(arc));
00800     const NodeIndex tail = node_[Opposite(arc)];
00801     DCHECK(CheckNodeValidity(tail));
00802     next_adjacent_arc_.Set(arc, first_incident_arc_[tail]);
00803     first_incident_arc_.Set(tail, arc);
00804     const NodeIndex head = node_[arc];
00805     DCHECK(CheckNodeValidity(head));
00806     next_adjacent_arc_.Set(Opposite(arc), first_incident_arc_[head]);
00807     first_incident_arc_.Set(head, Opposite(arc));
00808   }
00809 
00810   // Utility method that finds the next outgoing arc.
00811   ArcIndex FindNextOutgoingArc(ArcIndex arc) const {
00812     DCHECK(CheckArcBounds(arc));
00813     while (IsReverse(arc)) {
00814       arc = NextAdjacentArc(arc);
00815       DCHECK(CheckArcBounds(arc));
00816     }
00817     return arc;
00818   }
00819 
00820   // Utility method that finds the next incoming arc.
00821   ArcIndex FindNextIncomingArc(ArcIndex arc) const {
00822     DCHECK(CheckArcBounds(arc));
00823     while (IsDirect(arc)) {
00824       arc = NextAdjacentArc(arc);
00825       DCHECK(CheckArcBounds(arc));
00826     }
00827     return arc;
00828   }
00829 
00830   // The maximum number of nodes that the graph can hold.
00831   NodeIndex max_num_nodes_;
00832 
00833   // The maximum number of arcs that the graph can hold.
00834   ArcIndex max_num_arcs_;
00835 
00836   // The maximum index of the node currently held by the graph.
00837   NodeIndex num_nodes_;
00838 
00839   // The current number of arcs held by the graph.
00840   ArcIndex num_arcs_;
00841 
00842   // Array of node indices. node_[i] contains the tail node of arc i.
00843   PackedArray<NodeIndexSize> node_;
00844 
00845   // Array of next indices.
00846   // next_adjacent_arc_[i] contains the next arc in the adjacency list of arc i.
00847   PackedArray<ArcIndexSize> next_adjacent_arc_;
00848 
00849   // Array of arc indices. first_incident_arc_[i] contains the first arc
00850   // incident to node i.
00851   PackedArray<ArcIndexSize> first_incident_arc_;
00852 
00853   // Flag to indicate that BuildRepresentation() needs to be called
00854   // before the adjacency lists are examined. Only for DCHECK in debug
00855   // builds.
00856   bool representation_clean_;
00857 };
