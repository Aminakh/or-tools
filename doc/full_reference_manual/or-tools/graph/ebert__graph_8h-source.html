00017 // An implementation (with some improvements) of the star-representation of a
00018 // graph as described in J. Ebert, "A versatile data structure for
00019 // edge-oriented graph algorithms." Communications of the ACM 30(6):513-519
00020 // (June 1987). http://portal.acm.org/citation.cfm?id=214769
00021 // Both forward- and backward-star representations are contained in this
00022 // representation.
00023 
00024 // The graph is represented with three arrays.
00025 // Let n be the number of nodes and m be the number of arcs.
00026 // Let i be an integer in [0..m-1], denoting the index of an arc.
00027 //  * node_[i] contains the end-node of arc i,
00028 //  * node_[-i-1] contains the start-node of arc i.
00029 // Note that in two's-complement arithmetic, -i-1 = ~i.
00030 // Consequently:
00031 //  * node_[~i] contains the start-node of the arc reverse to arc i,
00032 //  * node_[i] contains the end-node of the arc reverse to arc i.
00033 //  Note that if arc (u, v) is defined, then the data structure also stores
00034 //  (v, u).
00035 //  Arc ~i thus denotes the arc reverse to arc i.
00036 //  This is what makes this representation useful for undirected graphs and for
00037 //  implementing algorithms like bi-directional shortest-path.
00038 //  Also note that the representation handles multi-graphs. If several arcs
00039 //  going from node u to node v are added to the graph, they will be handled as
00040 //  separate arcs.
00041 //
00042 // Now, for an integer u in [0..n-1] denoting the index of a node:
00043 //  * first_incident_arc_[u] denotes the first arc in the adjacency list of u.
00044 //  * going from an arc i, the adjacency list can be traversed using
00045 //    j = next_adjacent_arc_[i].
00046 //
00047 // This implementation has the following benefits:
00048 //  * It is able to handle both directed or undirected graphs.
00049 //  * Being based on indices, it is easily serializable. Only the contents
00050 //    of the node_ array needs to be stored.
00051 //  * The sizes of node indices and arc indices can be stored in 32 bits, while
00052 //    still allowing to go a bit further than the 4-gigabyte limitation
00053 //    (48 gigabytes for a pure graph, without capacities or costs.)
00054 //  * The representation can be recomputed if edges have been loaded from
00055 //  * The representation can be recomputed if edges have been loaded from
00056 //    external memory or if edges have been re-ordered.
00057 //  * The memory consumption is: 2 * m * sizeof(NodeIndexType)
00058 //                             + 2 * m * sizeof(ArcIndexType)
00059 //                             + n * sizeof(ArcIndexType)
00060 //
00061 // This implementation differs from the implementation described in [Ebert 1987]
00062 // in the following respects:
00063 //  * arcs are represented using an (i, ~i) approach, whereas Ebert used
00064 //    (i, -i). Indices for direct arcs thus start at 0, in a fashion that is
00065 //    compatible with the index numbering in C and C++. Note that we also tested
00066 //    a (2*i, 2*i+1) storage pattern, which did not show any speed benefit, and
00067 //    made the use of the API much more difficult.
00068 //  * because of this, the 'nil' values for nodes and arcs are not 0, as Ebert
00069 //    first described. The value for the 'nil' node is set to -1, while the
00070 //    value for the 'nil' arc is set to the smallest integer representable with
00071 //    ArcIndexSize bytes.
00072 //  * it is possible to add arcs to the graph, with AddArc, in a much simpler
00073 //    way than described by Ebert.
00074 //  * TODO(user) it is possible to group all the outgoing (resp. incoming) arcs
00075 //    of a node to allow to traverse the outgoing (resp. incoming) arcs in
00076 //    O(out_degree(node)) (resp. O(in_degree(node))) instead of O(degree(node)).
00077 //  * TODO(user) it is possible to implement arc deletion and garbage collection
00078 //    in an efficient (relatively) manner. For the time being we haven't seen an
00079 //    application to this.

00222   // Adds an arc to the graph and returns its index.
00223   // Returns kNilArc if the arc could not be added.
00224   // Note that for a given pair (tail, head) AddArc does not overwrite an
00225   // already-existing arc between tail and head: Another arc is created
00226   // instead. This makes it possible to handle multi-graphs.
00227   ArcIndexType AddArc(NodeIndexType tail, NodeIndexType head) {
00228     if (num_arcs_ >= max_num_arcs_
00229         || !IsNodeValid(tail) || !IsNodeValid(head)) {
00230       return kNilArc;
00231     }
00232     if (tail + 1 > num_nodes_) {
00233       num_nodes_ = tail + 1;   // std::max does not work on int16.

00236       num_nodes_ = head + 1;

00239     ++num_arcs_;

00586   // It is exported so that users of the EbertGraph class can use it.
00587   // To be used in a DCHECK.
00588   bool CheckArcValidity(const ArcIndexType arc) const {
00589     return (arc != kNilArc) && (arc >= -max_num_arcs_ && arc < max_num_arcs_);
00590   }
00591 
00592   // Utility function to check that a node index is within the bounds AND
00593   // different from kNilNode.
00594   // It is exported so that users of the EbertGraph class can use it.
00595   // To be used in a DCHECK.
00596   bool CheckNodeValidity(const NodeIndexType node) const {
00597     return node >= kFirstNode && node < max_num_nodes_;
00598   }
00599 
00600   // Returns the tail or start-node of arc.
00601   NodeIndexType Tail(const ArcIndexType arc) const {
00602     DCHECK(CheckArcValidity(arc));
00603     return node_[Opposite(arc)];
00604   }
00605 
00606   // Returns the head or end-node of arc.
00607   NodeIndexType Head(const ArcIndexType arc) const {
00608     DCHECK(CheckArcValidity(arc));
00609     return node_[arc];
00610   }
00611 
00612   // Returns the first arc going from tail to head, if it exists, or kNilArc
00613   // if such an arc does not exist.
00614   ArcIndexType LookUpArc(const NodeIndexType tail,
00615                          const NodeIndexType head) const {
00616     for (ArcIndexType arc = FirstOutgoingArc(tail);
00617          arc != kNilArc;
00618          arc = NextOutgoingArc(arc)) {
00619       if (Head(arc) == head) {
00620         return arc;
00621       }
00622     }
00623     return kNilArc;
00624   }
00625 
00626   // Returns the tail or start-node of arc if it is positive
00627   // (i.e. it is taken in the direction it was entered in the graph),
00628   // and the head or end-node otherwise. 'This' in Ebert's paper.
00629   NodeIndexType DirectArcTail(const ArcIndexType arc) const {
00630     return Tail(DirectArc(arc));
00631   }
00632 
00633   // Returns the head or end-node of arc if it is positive
00634   // (i.e. it is taken in the direction it was entered in the graph),
00635   // and the tail or start-node otherwise. 'That' in Ebert's paper.
00636   NodeIndexType DirectArcHead(const ArcIndexType arc) const {
00637     return Head(DirectArc(arc));
00638   }
00639 
00640   // Returns the arc in normal/direct direction.
00641   ArcIndexType DirectArc(const ArcIndexType arc) const {
00642     DCHECK(CheckArcValidity(arc));
00643     return std::max(arc, Opposite(arc));
00644   }
00645 
00646   // Returns the arc in reverse direction.
00647   ArcIndexType ReverseArc(const ArcIndexType arc) const {
00648     DCHECK(CheckArcValidity(arc));
00649     return std::min(arc, Opposite(arc));
00650   }
00651 
00652   // Returns the opposite arc, i.e the direct arc is the arc is in reverse
00653   // direction, and the reverse arc if the arc is direct.
00654   ArcIndexType Opposite(const ArcIndexType arc) const {
00655     const ArcIndexType opposite = ~arc;
00656     DCHECK(CheckArcValidity(arc));
00657     DCHECK(CheckArcValidity(opposite));
00658     return opposite;
00659   }
00660 
00661   // Returns true if the arc is direct.
00662   bool IsDirect(const ArcIndexType arc) const {
00663     DCHECK(CheckArcBounds(arc));
00664     return arc != kNilArc && arc >= 0;
00665   }
00666 
00667   // Returns true if the arc is in the reverse direction.
00668   bool IsReverse(const ArcIndexType arc) const {
00669     DCHECK(CheckArcBounds(arc));
00670     return arc != kNilArc && arc < 0;
00671   }
00672 
00673   // Returns true if arc is incident to node.
00674   bool IsIncident(ArcIndexType arc, NodeIndexType node) const {
00675     return IsIncoming(arc, node) || IsOutgoing(arc, node);
00676   }
00677 
00678   // Returns true if arc is incoming to node.
00679   bool IsIncoming(ArcIndexType arc, NodeIndexType node) const {
00680     return DirectArcHead(arc) == node;
00681   }
00682 
00683   // Returns true if arc is outgoing from node.
00684   bool IsOutgoing(ArcIndexType arc, NodeIndexType node) const {
00685     return DirectArcTail(arc) == node;
00686   }
00687 
00688   // Recreates the next_adjacent_arc_ and first_incident_arc_ variables from
00689   // the array node_ in O(n + m) time.
00690   // This is useful if node_ array has been sorted according to a given
00691   // criterion, for example.
00692   void BuildRepresentation() {
00693     first_incident_arc_.SetAll(kNilArc);
00694     for (ArcIndexType arc = kFirstArc; arc <  max_num_arcs_; ++arc) {
00695       Attach(arc);
00696     }
00697     representation_clean_ = true;
00698   }
00699 
00700   // Returns a debug string containing all the information contained in the
00701   // data structure in raw form.
00702   string DebugString() const {
00703     DCHECK(representation_clean_);
00704     string result = "Arcs:(node, next arc) :\n";
00705     for (ArcIndexType arc = -num_arcs_; arc < num_arcs_; ++arc) {
00706       result += " " + ArcDebugString(arc) + ":(" + NodeDebugString(node_[arc])
00707               + "," + ArcDebugString(next_adjacent_arc_[arc]) + ")\n";
00708     }
00709     result += "Node:First arc :\n";
00710     for (NodeIndexType node = kFirstNode; node < num_nodes_; ++node) {
00711       result += " " + NodeDebugString(node) + ":"
00712               + ArcDebugString(first_incident_arc_[node]) + "\n";
00713     }
00714     return result;
00715   }
00716 
00717   string NodeDebugString(const NodeIndexType node) const {
00718     if (node == kNilNode) {
00719       return "NilNode";
00720     } else {
00721       return StringPrintf("%lld", static_cast<int64>(node));
00722     }
00723   }
00724 
00725   string ArcDebugString(const ArcIndexType arc) const {
00726     if (arc == kNilArc) {
00727       return "NilArc";
00728     } else {
00729       return StringPrintf("%lld", static_cast<int64>(arc));
00730     }
00731   }

00733  private:
00734   // Returns kNilNode if the graph has no nodes or node if it has at least one
00735   // node. Useful for initializing iterators correctly in the case of empty
00736   // graphs.
00737   NodeIndexType StartNode(NodeIndexType node) const {
00738     return num_nodes_ == 0 ? kNilNode : node;
00739   }
00740 
00741   // Returns kNilArc if the graph has no arcs arc if it has at least one arc.
00742   // Useful for initializing iterators correctly in the case of empty graphs.
00743   ArcIndexType StartArc(ArcIndexType arc) const {
00744     return num_arcs_ == 0 ? kNilArc : arc;
00745   }
00746 
00747   // Returns the first outgoing arc for node.
00748   ArcIndexType FirstOutgoingArc(const NodeIndexType node) const {
00749     DCHECK(CheckNodeValidity(node));
00750     return FindNextOutgoingArc(FirstIncidentArc(node));
00751   }
00752 
00753   // Returns the outgoing arc following the argument in the adjacency list.
00754   ArcIndexType NextOutgoingArc(const ArcIndexType arc) const {
00755     DCHECK(CheckArcValidity(arc));
00756     DCHECK(IsDirect(arc));
00757     return FindNextOutgoingArc(NextAdjacentArc(arc));
00758   }
00759 
00760   // Returns the first incoming arc for node.
00761   ArcIndexType FirstIncomingArc(const NodeIndexType node) const {
00762     DCHECK_LE(kFirstNode, node);
00763     DCHECK_GE(max_num_nodes_, node);
00764     return FindNextIncomingArc(FirstIncidentArc(node));
00765   }
00766 
00767   // Returns the incoming arc following the argument in the adjacency list.
00768   ArcIndexType NextIncomingArc(const ArcIndexType arc) const {
00769     DCHECK(CheckArcValidity(arc));
00770     DCHECK(IsReverse(arc));
00771     return FindNextIncomingArc(NextAdjacentArc(arc));
00772   }
00773 
00774   // Returns the first arc in node's incidence list.
00775   ArcIndexType FirstIncidentArc(const NodeIndexType node) const {
00776     DCHECK(representation_clean_);
00777     DCHECK(CheckNodeValidity(node));
00778     return first_incident_arc_[node];
00779   }
00780 
00781   // Returns the next arc following the passed argument in its adjacency list.
00782   ArcIndexType NextAdjacentArc(const ArcIndexType arc) const {
00783     DCHECK(representation_clean_);
00784     DCHECK(CheckArcValidity(arc));
00785     return next_adjacent_arc_[arc];
00786   }
00787 
00788   // Returns the node following the argument in the graph.
00789   // Returns kNilNode (= end) if the range of nodes has been exhausted.
00790   // It is called by NodeIterator::Next() and as such does not expect do be
00791   // passed an argument equal to kNilNode.
00792   // This is why the return line is simplified from
00793   // return (node == kNilNode || next_node >= num_nodes_)
00794   //        ? kNilNode : next_node;
00795   // to
00796   // return next_node < num_nodes_ ? next_node : kNilNode;
00797   NodeIndexType NextNode(const NodeIndexType node) const {
00798     DCHECK(CheckNodeValidity(node));
00799     const NodeIndexType next_node = node + 1;
00800     return next_node < num_nodes_ ? next_node : kNilNode;
00801   }
00802 
00803   // Returns the arc following the argument in the graph.
00804   // Returns kNilArc (= end) if the range of arcs has been exhausted.
00805   // It is called by ArcIterator::Next() and as such does not expect do be
00806   // passed an argument equal to kNilArc.
00807   // This is why the return line is simplified from
00808   // return ( arc == kNilArc || next_arc >= num_arcs_) ? kNilArc : next_arc;
00809   // to
00810   // return next_arc < num_arcs_ ? next_arc : kNilArc;
00811   ArcIndexType NextArc(const ArcIndexType arc) const {
00812     DCHECK(CheckArcValidity(arc));
00813     const ArcIndexType next_arc = arc + 1;
00814     return next_arc < num_arcs_ ? next_arc : kNilArc;
00815   }
00816 
00817   // Using the SetTail() method implies that the BuildRepresentation()
00818   // method must be called to restore consistency before the graph is
00819   // used.
00820   void SetTail(const ArcIndexType arc, const NodeIndexType tail) {
00821     representation_clean_ = false;
00822     node_.Set(Opposite(arc), tail);
00823   }
00824 
00825   // Using the SetHead() method implies that the BuildRepresentation()
00826   // method must be called to restore consistency before the graph is
00827   // used.
00828   void SetHead(const ArcIndexType arc, const NodeIndexType head) {
00829     representation_clean_ = false;
00830     node_.Set(arc, head);
00831   }
00832 
00833   // Utility method to attach a new arc.
00834   void Attach(ArcIndexType arc) {
00835     DCHECK(CheckArcValidity(arc));
00836     const NodeIndexType tail = node_[Opposite(arc)];
00837     DCHECK(CheckNodeValidity(tail));
00838     next_adjacent_arc_.Set(arc, first_incident_arc_[tail]);
00839     first_incident_arc_.Set(tail, arc);
00840     const NodeIndexType head = node_[arc];
00841     DCHECK(CheckNodeValidity(head));
00842     next_adjacent_arc_.Set(Opposite(arc), first_incident_arc_[head]);
00843     first_incident_arc_.Set(head, Opposite(arc));
00844   }
00845 
00846   // Utility method that finds the next outgoing arc.
00847   ArcIndexType FindNextOutgoingArc(ArcIndexType arc) const {
00848     DCHECK(CheckArcBounds(arc));
00849     while (IsReverse(arc)) {
00850       arc = NextAdjacentArc(arc);
00851       DCHECK(CheckArcBounds(arc));
00852     }
00853     return arc;
00854   }
00855 
00856   // Utility method that finds the next incoming arc.
00857   ArcIndexType FindNextIncomingArc(ArcIndexType arc) const {
00858     DCHECK(CheckArcBounds(arc));
00859     while (IsDirect(arc)) {
00860       arc = NextAdjacentArc(arc);
00861       DCHECK(CheckArcBounds(arc));
00862     }
00863     return arc;
00864   }

00868 
00869   // The maximum number of arcs that the graph can hold.
00870   ArcIndexType max_num_arcs_;
00871 
00872   // The maximum index of the node currently held by the graph.
00873   NodeIndexType num_nodes_;
00874 
00875   // The current number of arcs held by the graph.
00876   ArcIndexType num_arcs_;
00877 
00878   // Array of node indices. node_[i] contains the tail node of arc i.
00879   ZVector<NodeIndexType> node_;
00880 
00881   // Array of next indices.
00882   // next_adjacent_arc_[i] contains the next arc in the adjacency list of arc i.
00883   ZVector<ArcIndexType> next_adjacent_arc_;
00884 
00885   // Array of arc indices. first_incident_arc_[i] contains the first arc
00886   // incident to node i.
00887   ZVector<ArcIndexType> first_incident_arc_;
00888 
00889   // Flag to indicate that BuildRepresentation() needs to be called

00892   bool representation_clean_;

00936 // Builds a directed line graph for 'graph' (see "directed line graph" in
00937 // http://en.wikipedia.org/wiki/Line_graph). Arcs of the original graph
00938 // become nodes and the new graph contains only nodes created from arcs in the
00939 // original graph (we use the notation (a->b) for these new nodes); the index
00940 // of the node (a->b) in the new graph is exactly the same as the index of the
00941 // arc a->b in the original graph.
00942 // An arc from node (a->b) to node (c->d) in the new graph is added if and only
00943 // if b == c in the original graph.
00944 // This method expects that 'line_graph' is an empty graph (it has no nodes
00945 // and no arcs).
00946 // Returns false on an error.
00947 template<typename NodeIndexType, typename ArcIndexType>
00948 bool BuildLineGraph(const EbertGraph<NodeIndexType, ArcIndexType>& graph,
00949                     EbertGraph<NodeIndexType, ArcIndexType>* const line_graph) {
00950   if (line_graph == NULL) {
00951     LOG(DFATAL) << "line_graph must not be NULL";
00952     return false;
00953   }
00954   if (line_graph->num_nodes() != 0) {
00955     LOG(DFATAL) << "line_graph must be empty";
00956     return false;
00957   }
00958   typedef EbertGraph<NodeIndexType, ArcIndexType> Graph;
00959   typedef typename Graph::ArcIterator ArcIterator;
00960   typedef typename Graph::OutgoingArcIterator OutgoingArcIterator;
00961   // Sizing then filling.
00962   const NodeIndexType num_nodes = graph.num_arcs();
00963   ArcIndexType num_arcs = 0;
00964   for (ArcIterator arc_iterator(graph);
00965        arc_iterator.Ok();
00966        arc_iterator.Next()) {
00967     const ArcIndexType arc = arc_iterator.Index();
00968     const NodeIndexType head = graph.Head(arc);
00969     for (OutgoingArcIterator iterator(graph, head);
00970          iterator.Ok();
00971          iterator.Next()) {
00972       ++num_arcs;
00973     }
00974   }
00975   line_graph->Reserve(num_nodes, num_arcs);
00976   for (ArcIterator arc_iterator(graph);
00977        arc_iterator.Ok();
00978        arc_iterator.Next()) {
00979     const ArcIndexType arc = arc_iterator.Index();
00980     const NodeIndexType head = graph.Head(arc);
00981     for (OutgoingArcIterator iterator(graph, head);
00982          iterator.Ok();
00983          iterator.Next()) {
00984       line_graph->AddArc(arc, iterator.Index());
00985     }
00986   }
00987   return true;
00988 }
