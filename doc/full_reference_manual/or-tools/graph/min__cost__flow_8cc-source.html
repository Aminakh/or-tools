00071 void MinCostFlow::SetArcCapacity(ArcIndex arc, FlowQuantity new_capacity) {
00072   DCHECK_LE(0, new_capacity);
00073   DCHECK(graph_->CheckArcValidity(arc));
00074   const FlowQuantity free_capacity = residual_arc_capacity_[arc];
00075   const FlowQuantity capacity_delta = new_capacity - Capacity(arc);
00076   VLOG(1) << "Changing capacity on arc " << arc
00077           << " from " << Capacity(arc) << " to " << new_capacity
00078           << ". Current free capacity = " << free_capacity;
00079   if (capacity_delta == 0) {
00080     return;  // Nothing to do.
00081   }
00082   status_ = NOT_SOLVED;
00083   if (free_capacity + capacity_delta >= 0) {
00084     // The above condition is true when one of two following holds:
00085     // 1/ (capacity_delta > 0), meaning we are increasing the capacity
00086     // 2/ (capacity_delta < 0 && free_capacity + capacity_delta >= 0)
00087     //    meaning we are reducing the capacity, but that the capacity
00088     //    reduction is not larger than the free capacity.
00089     residual_arc_capacity_.Set(arc, free_capacity + capacity_delta);
00090     DCHECK_LE(0, residual_arc_capacity_[arc]);
00091     VLOG(1) << "Now: capacity = " << Capacity(arc) << " flow = " << Flow(arc);
00092   } else {
00093     // We have to reduce the flow on the arc, and update the excesses
00094     // accordingly.
00095     const FlowQuantity flow = residual_arc_capacity_[Opposite(arc)];
00096     const FlowQuantity flow_excess = flow - new_capacity;
00097     VLOG(1) << "Flow value " << flow << " exceeds new capacity "
00098             << new_capacity << " by " << flow_excess;
00099     residual_arc_capacity_.Set(arc, 0);
00100     residual_arc_capacity_.Set(Opposite(arc), new_capacity);
00101     const NodeIndex tail = Tail(arc);
00102     node_excess_.Set(tail, node_excess_[tail] + flow_excess);
00103     const NodeIndex head = Head(arc);
00104     node_excess_.Set(head, node_excess_[head] - flow_excess);
00105     DCHECK_LE(0, residual_arc_capacity_[arc]);
00106     DCHECK_LE(0, residual_arc_capacity_[Opposite(arc)]);
00107     VLOG(1) << DebugString("After SetArcCapacity:", arc);
00108   }
00109 }

00111 bool MinCostFlow::CheckInputConsistency() const {
00112   FlowQuantity total_supply = 0;
00113   uint64 max_capacity = 0;  // uint64 because it is positive and will be used
00114                             // to check against FlowQuantity overflows.
00115   for (StarGraph::ArcIterator arc_it(*graph_); arc_it.Ok(); arc_it.Next()) {
00116     const ArcIndex arc = arc_it.Index();
00117     const uint64 capacity = static_cast<uint64>(residual_arc_capacity_[arc]);
00118     max_capacity = std::max(capacity, max_capacity);
00119   }
00120   uint64 total_flow = 0;   // uint64 for the same reason as max_capacity.
00121   for (StarGraph::NodeIterator node_it(*graph_); node_it.Ok(); node_it.Next()) {
00122     const NodeIndex node = node_it.Index();
00123     const FlowQuantity excess = node_excess_[node];
00124     total_supply += excess;
00125     if (excess > 0) {
00126       total_flow += excess;
00127       if (std::numeric_limits<FlowQuantity>::max() <
00128           max_capacity + total_flow) {
00129         LOG(DFATAL) << "Input consistency error: max capacity + flow exceed "
00130                     << "precision";
00131       }
00132     }
00133   }
00134   if (total_supply != 0) {
00135     LOG(DFATAL) << "Input consistency error: unbalanced problem";
00136   }
00137   return true;
00138 }

00140 bool MinCostFlow::CheckResult() const {

00142     const NodeIndex node = it.Index();
00143     if (node_excess_[node] != 0) {
00144       LOG(DFATAL) << "node_excess_[" << node << "] != 0";
00145       return false;

00147     for (StarGraph::IncidentArcIterator

00149       const ArcIndex arc = arc_it.Index();
00150       bool ok = true;
00151       if (residual_arc_capacity_[arc] < 0) {
00152         LOG(DFATAL) << "residual_arc_capacity_[" << arc << "] < 0";
00153         ok = false;

00156         LOG(DFATAL) << "residual_arc_capacity_[" << arc
00157                    << "] > 0 && ReducedCost(" << arc << ") < " << -epsilon_
00158                    << ". (epsilon_ = " << epsilon_;
00159         ok = false;
00160       }
00161       if (!ok) {
00162         LOG(DFATAL) << DebugString("CheckResult ", arc);

00164     }

00227 bool MinCostFlow::Solve() {
00228   status_ = NOT_SOLVED;
00229   if (FLAGS_min_cost_flow_check_balance && !CheckInputConsistency()) {
00230     status_ = UNBALANCED;
00231     return false;
00232   }
00233   if (FLAGS_min_cost_flow_check_costs && !CheckCostRange()) {
00234     status_ = BAD_COST_RANGE;
00235     return false;
00236   }
00237   node_potential_.Assign(0);
00238   ResetFirstAdmissibleArcs();
00239   ScaleCosts();
00240   Optimize();
00241   UnscaleCosts();
00242   if (status_ != OPTIMAL) {
00243     total_flow_cost_ = 0;
00244     return false;
00245   }
00246   if (FLAGS_min_cost_flow_check_result && !CheckResult()) {
00247     status_ = BAD_RESULT;
00248     return false;
00249   }
00250   total_flow_cost_ = 0;
00251   for (StarGraph::ArcIterator it(*graph_); it.Ok(); it.Next()) {
00252     const ArcIndex arc = it.Index();
00253     const FlowQuantity flow_on_arc = residual_arc_capacity_[Opposite(arc)];
00254     VLOG(1) << "Flow for arc " << arc << " = " << flow_on_arc
00255             << ", scaled cost = " << scaled_arc_unit_cost_[arc];
00256     total_flow_cost_ += scaled_arc_unit_cost_[arc] * flow_on_arc;
00257   }
00258   status_ = OPTIMAL;
00259   return true;
00260 }

00269 void MinCostFlow::ScaleCosts() {
00270   cost_scaling_factor_ = graph_->num_nodes() + 1;
00271   epsilon_ = 1LL;
00272   VLOG(1) << "Number of arcs in the graph = " << graph_->num_arcs();
00273   for (StarGraph::ArcIterator arc_it(*graph_); arc_it.Ok(); arc_it.Next()) {
00274     const ArcIndex arc = arc_it.Index();
00275     const CostValue cost = scaled_arc_unit_cost_[arc] * cost_scaling_factor_;
00276     scaled_arc_unit_cost_.Set(arc, cost);
00277     scaled_arc_unit_cost_.Set(Opposite(arc), -cost);
00278     epsilon_ = std::max(epsilon_, MathUtil::Abs(cost));
00279   }
00280   VLOG(1) << "Initial epsilon = " << epsilon_;
00281   VLOG(1) << "Cost scaling factor = " << cost_scaling_factor_;

00294 void MinCostFlow::Optimize() {
00295   const CostValue kEpsilonMin = 1LL;
00296   do {
00297     // Avoid epsilon_ == 0.
00298     epsilon_ = std::max(epsilon_ / alpha_, kEpsilonMin);

00323 void MinCostFlow::PushFlow(FlowQuantity flow, ArcIndex arc) {
00324   DCHECK_GT(residual_arc_capacity_[arc], 0);
00325   VLOG(1) << "PushFlow: pushing " << flow << " on arc " << arc
00326           << " from node " << Tail(arc) << " to node " << Head(arc);
00327   // Reduce the residual capacity on arc by flow.
00328   residual_arc_capacity_.Set(arc, residual_arc_capacity_[arc] - flow);
00329   // Increase the residual capacity on opposite arc by flow.
00330   const ArcIndex opposite = Opposite(arc);
00331   residual_arc_capacity_.Set(opposite, residual_arc_capacity_[opposite] + flow);
00332   // Update the excesses at the tail and head of the arc.
00333   const NodeIndex tail = Tail(arc);
00334   node_excess_.Set(tail, node_excess_[tail] - flow);
00335   const NodeIndex head = Head(arc);
00336   node_excess_.Set(head, node_excess_[head] + flow);
00337   VLOG(2) << DebugString("PushFlow: ", arc);
00338 }

00351 void MinCostFlow::Refine() {
00352   SaturateAdmissibleArcs();
00353   InitializeActiveNodeStack();
00354   while (!active_nodes_.empty()) {
00355     const NodeIndex node = active_nodes_.top();
00356     active_nodes_.pop();

00367 void MinCostFlow::Discharge(NodeIndex node) {
00368   DCHECK(IsActive(node));
00369   VLOG(1) << "Discharging node " << node << ", excess = " << node_excess_[node];
00370   while (IsActive(node)) {
00371     for (StarGraph::IncidentArcIterator arc_it(*graph_, node,
00372                                                first_admissible_arc_[node]);
00373          arc_it.Ok();
00374          arc_it.Next()) {
00375       const ArcIndex arc = arc_it.Index();
00376       VLOG(2) << DebugString("Discharge: considering", arc);
00377       if (IsAdmissible(arc)) {
00378         if (node_excess_[node] != 0) {
00379           VLOG(1) << "Discharge: calling PushFlow.";
00380           const NodeIndex head = Head(arc);
00381           const bool head_active_before_push = IsActive(head);
00382           const FlowQuantity delta = std::min(node_excess_[node],
00383                                               residual_arc_capacity_[arc]);
00384           PushFlow(delta, arc);
00385           if (IsActive(head) && !head_active_before_push) {
00386             active_nodes_.push(Head(arc));
00387           }
00388         }
00389         if (node_excess_[node] == 0) {
00390           first_admissible_arc_.Set(node, arc);  // arc may still be admissible.
00391           return;
00392         }
00393       }
00394     }
00395     Relabel(node);
00396     if (status_ == INFEASIBLE) {
00397       return;
00398     }
00399   }

00402 void MinCostFlow::Relabel(NodeIndex node) {
00403   DCHECK(CheckRelabelPrecondition(node));
00404   CostValue new_potential;
00405   if (FLAGS_min_cost_flow_fast_potential_update) {
00406     new_potential = node_potential_[node] - epsilon_;
00407   } else {
00408     new_potential = kint64min;
00409     for (StarGraph::IncidentArcIterator arc_it(*graph_, node);
00410          arc_it.Ok();
00411          arc_it.Next()) {
00412       ArcIndex arc = arc_it.Index();
00413       DCHECK_EQ(Tail(arc), node);
00414       if (residual_arc_capacity_[arc] > 0) {
00415         const NodeIndex head = Head(arc);
00416         new_potential = std::max(new_potential, node_potential_[head]
00417                                               - scaled_arc_unit_cost_[arc]
00418                                               - epsilon_);
00419         DCHECK_GE(0, new_potential);
00420       }
00421     }
00422     if (new_potential == kint64min) {
00423       // Note that this infeasibility detection is incomplete.
00424       // Only max flow can detect that a min-cost flow problem is infeasible.
00425       status_ = INFEASIBLE;
00426       return;
00427     }
00428   }
00429   VLOG(1) << "Relabel: node " << node << " from " << node_potential_[node]
00430           << " to " << new_potential;
00431   node_potential_.Set(node, new_potential);
00432   first_admissible_arc_.Set(node, GetFirstIncidentArc(node));
