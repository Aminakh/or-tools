00141 bool LinearSumAssignment::UpdateEpsilon() {
00142   // There are some somewhat subtle issues around using integer
00143   // arithmetic to compute successive values of epsilon_, the error
00144   // parameter.
00145   //
00146   // First, the value of price_reduction_bound_ is chosen under the
00147   // assumption that it is truly an upper bound on the amount by which
00148   // a node's price can change during the current iteration. The value
00149   // of this bound in turn depends on the assumption that the flow
00150   // computed by the previous iteration was
00151   // (epsilon_ * alpha_)-optimal. If epsilon_ decreases by more than a
00152   // factor of alpha_ due to truncating arithmetic, that bound might
00153   // not hold, and the consequence is that BestArcAndGap could return
00154   // an overly cautious admissibility gap in the case where a
00155   // left-side node has only one incident arc. This is not a big deal
00156   // at all. At worst it could lead to a few extra relabelings.
00157   //
00158   // Second, it is not a problem currently, but in the future if we
00159   // use an arc-fixing heuristic, we cannot permit truncating integer
00160   // division to decrease epsilon_ by a factor greater than alpha_
00161   // because our bounds on price changes (and hence on the reduced
00162   // cost of an arc that might ever become admissible in the future)
00163   // depend on the ratio between the values of the error parameter at
00164   // successive iterations. One consequence will be that we might
00165   // occasionally do an extra iteration today in the interest of being
00166   // able to "price arcs out" (which we don't do today). Today we
00167   // simply use truncating integer division, but note that this will
00168   // have to change if we ever price arcs out.
00169   //
00170   // Since neither of those issues is a problem today, we simply use
00171   // truncating integer arithmetic. But future changes might
00172   // necessitate rounding epsilon upward in the division.
00173   epsilon_ = std::max(epsilon_ / alpha_, kMinEpsilon);
00174   VLOG(3) << "Updated: epsilon_ == " << epsilon_;
00175   price_reduction_bound_ = PriceChangeBound(1, NULL);
00176   DCHECK_GT(price_reduction_bound_, 0);
00177   // For today we always return true; in the future updating epsilon
00178   // in sophisticated ways could conceivably detect infeasibility.
00179   return true;
00180 }

00201 // There exists a price function such that the admissible arcs at the
00202 // beginning of an iteration are exactly the reverse arcs of all
00203 // matching arcs. Saturating all admissible arcs with respect to that
00204 // price function therefore means simply unmatching every matched
00205 // node.
00206 //
00207 // In the future we will price out arcs, which will reduce the set of
00208 // nodes we unmatch here. If a matching arc is priced out, we will not
00209 // unmatch its endpoints since that element of the matching is
00210 // guaranteed not to change.
00211 void LinearSumAssignment::SaturateNegativeArcs() {
00212   total_excess_ = 0;
00213   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
00214        node_it.Ok();
00215        node_it.Next()) {
00216     const NodeIndex node = node_it.Index();
00217     if (IsActive(node)) {
00218       // This can happen in the first iteration when nothing is
00219       // matched yet.
00220       total_excess_ += 1;
00221     } else {
00222       // We're about to create a unit of excess by unmatching these nodes.
00223       total_excess_ += 1;
00224       const NodeIndex mate = GetMate(node);
00225       matched_.Set(mate, StarGraph::kNilArc);
00226       matched_.Set(node, StarGraph::kNilArc);
00227     }
00228   }
00229 }

00232 bool LinearSumAssignment::DoublePush(NodeIndex source) {
00233     DCHECK(IsActive(source));
00234   ImplicitPriceSummary summary = BestArcAndGap(source);
00235   const ArcIndex best_arc = summary.first;
00236   const CostValue gap = summary.second;
00237   // Now we have the best arc incident to source, i.e., the one with
00238   // minimum reduced cost. Match that arc, unmatching its head if
00239   // necessary.
00240   if (best_arc == StarGraph::kNilArc) {
00241     return false;
00242   }
00243   const NodeIndex new_mate = Head(best_arc);
00244   const ArcIndex to_unmatch = matched_[new_mate];
00245   if (to_unmatch != StarGraph::kNilArc) {
00246     // Unmatch new_mate from its current mate, pushing the unit of
00247     // flow back to a node on the left side as a unit of excess.
00248     const NodeIndex destination = Tail(to_unmatch);
00249     matched_.Set(destination, StarGraph::kNilArc);
00250     active_nodes_->Add(destination);
00251     // This counts as a double push.
00252     iteration_stats_.double_pushes_ += 1;
00253   } else {
00254     // We are about to increase the cardinality of the matching.
00255     total_excess_ -= 1;
00256     // This counts as a single push.
00257     iteration_stats_.pushes_ += 1;
00258   }
00259   matched_.Set(source, best_arc);
00260   matched_.Set(new_mate, best_arc);
00261   // Finally, relabel new_mate.
00262   iteration_stats_.relabelings_ += 1;
00263   price_.Set(new_mate, price_[new_mate] - gap - epsilon_);
00264   return price_[new_mate] >= price_lower_bound_;
00265 }

00267 bool LinearSumAssignment::Refine() {
00268   SaturateNegativeArcs();
00269   InitializeActiveNodeContainer();
00270   while (total_excess_ > 0) {
00271     // Get an active node (i.e., one with excess == 1) and discharge
00272     // it using DoublePush.
00273     const NodeIndex node = active_nodes_->Get();
00274     if (!DoublePush(node)) {
00275       // Infeasibility detected.
00276       return false;
00277     }
00278   }
00279   DCHECK(active_nodes_->Empty());
00280   iteration_stats_.refinements_ += 1;
00281   return true;
00282 }

00284 // Computes best_arc, the minimum reduced-cost arc incident to
00285 // left_node and admissibility_gap, the amount by which the reduced
00286 // cost of best_arc must be increased to make it equal in reduced cost
00287 // to another residual arc incident to left_node.
00288 //
00289 // Precondition: left_node is unmatched. This allows us to simplify
00290 // the code. The debug-only counterpart to this routine is
00291 // LinearSumAssignment::ImplicitPrice() and it does not assume this
00292 // precondition.
00293 //
00294 // This function is large enough that our suggestion that the compiler
00295 // inline it might be pointless.
00296 inline LinearSumAssignment::ImplicitPriceSummary
00297 LinearSumAssignment::BestArcAndGap(NodeIndex left_node) const {
00298     DCHECK(IsActive(left_node));
00299   DCHECK_GT(epsilon_, 0);
00300   // During any scaling iteration, the price of an active node
00301   // decreases by at most price_reduction_bound_ and all left-side
00302   // nodes are made active at the beginning of Refine(), so the bound
00303   // applies to all left-side nodes.
00304   StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00305   ArcIndex best_arc = arc_it.Index();
00306   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00307   CostValue second_min_partial_reduced_cost =
00308       min_partial_reduced_cost + price_reduction_bound_;
00309   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00310     const ArcIndex arc = arc_it.Index();
00311     const CostValue partial_reduced_cost = PartialReducedCost(arc);
00312     if (partial_reduced_cost < min_partial_reduced_cost) {
00313       best_arc = arc;
00314       second_min_partial_reduced_cost = min_partial_reduced_cost;
00315       min_partial_reduced_cost = partial_reduced_cost;
00316     } else if (partial_reduced_cost < second_min_partial_reduced_cost) {
00317       second_min_partial_reduced_cost = partial_reduced_cost;
00318     }
00319   }
00320   const CostValue gap =
00321       second_min_partial_reduced_cost - min_partial_reduced_cost;
00322   DCHECK_GE(gap, 0);
00323   return std::make_pair(best_arc, gap);
00324 }

00327 inline CostValue
00328 LinearSumAssignment::ImplicitPrice(NodeIndex left_node) const {
00329   DCHECK_GT(epsilon_, 0);
00330   StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00331   // If the input problem is feasible, it is always the case that
00332   // arc_it.Ok(), i.e., that there is at least one arc incident to
00333   // left_node.
00334   DCHECK(arc_it.Ok());
00335   ArcIndex best_arc = arc_it.Index();
00336   if (best_arc == matched_[left_node]) {
00337     arc_it.Next();
00338     if (arc_it.Ok()) {
00339       best_arc = arc_it.Index();
00340     }
00341   }
00342   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00343   if (!arc_it.Ok()) {
00344     // Only one arc is incident to left_node, and the node is
00345     // currently matched along that arc, which must be the case in any
00346     // feasible solution. Therefore we implicitly price this node so
00347     // low that we will never consider unmatching it.
00348     return -(min_partial_reduced_cost + price_reduction_bound_);
00349   }
00350   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00351     const ArcIndex arc = arc_it.Index();
00352     if (arc != matched_[left_node]) {
00353       const CostValue partial_reduced_cost = PartialReducedCost(arc);
00354       if (partial_reduced_cost < min_partial_reduced_cost) {
00355         min_partial_reduced_cost = partial_reduced_cost;
00356       }
00357     }
00358   }
00359   return -min_partial_reduced_cost;
00360 }

00375 bool LinearSumAssignment::EpsilonOptimal() const {
00376   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
00377        node_it.Ok();
00378        node_it.Next()) {
00379     const NodeIndex left_node = node_it.Index();
00380     // Get the implicit price of left_node and make sure the reduced
00381     // costs of left_node's incident arcs are in bounds.
00382     CostValue left_node_price = ImplicitPrice(left_node);
00383     for (StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00384          arc_it.Ok();
00385          arc_it.Next()) {
00386       const ArcIndex arc = arc_it.Index();
00387       const CostValue reduced_cost =
00388           left_node_price + PartialReducedCost(arc);
00389       // Note the asymmetric definition of epsilon-optimality that we
00390       // use because it means we can saturate all admissible arcs in
00391       // the beginning of Refine() just by unmatching all matched
00392       // nodes.
00393       if (matched_[left_node] == arc) {
00394         // The reverse arc is residual. Epsilon-optimality requires
00395         // that the reduced cost of the forward arc be at most
00396         // epsilon_.
00397         if (reduced_cost > epsilon_) {
00398           return false;
00399         }
00400       } else {
00401         // The forward arc is residual. Epsilon-optimality requires
00402         // that the reduced cost of the forward arc be at least zero.
00403         if (reduced_cost < 0) {
00404           return false;
00405         }
00406       }
00407     }
00408   }
00409   return true;
00410 }

00412 bool LinearSumAssignment::FinalizeSetup() {
00413   epsilon_ = largest_scaled_cost_magnitude_;
00414   VLOG(2) << "Largest given cost magnitude: " <<
00415       largest_scaled_cost_magnitude_ / cost_scaling_factor_;
00416   // Initialize left-side node-indexed arrays.
00417   StarGraph::NodeIterator node_it(graph_);
00418   for (; node_it.Ok(); node_it.Next()) {
00419     const NodeIndex node = node_it.Index();
00420     if (node >= num_left_nodes_) {
00421       break;
00422     }
00423     matched_.Set(node, StarGraph::kNilArc);
00424   }
00425   // Initialize right-side node-indexed arrays. Example: prices are
00426   // stored only for right-side nodes.
00427   for (; node_it.Ok(); node_it.Next()) {
00428     const NodeIndex node = node_it.Index();
00429     price_.Set(node, 0);
00430     matched_.Set(node, StarGraph::kNilArc);
00431   }
00432   bool in_range;
00433   price_lower_bound_ = -PriceChangeBound(alpha_ - 1, &in_range);
00434   DCHECK_LE(price_lower_bound_, 0);
00435   if (!in_range) {
00436     LOG(WARNING) << "Price change bound exceeds range of representable "
00437                  << "costs; arithmetic overflow is not ruled out.";
00438   }
00439   return in_range;
00440 }

00448 bool LinearSumAssignment::ComputeAssignment() {
00449   // Note: FinalizeSetup() might have been called already by white-box
00450   // test code or by a client that wants to react to the possibility
00451   // of overflow before solving the given problem, but FinalizeSetup()
00452   // is idempotent and reasonably fast, so we call it unconditionally
00453   // here.
00454   FinalizeSetup();
00455   bool ok = graph_.num_nodes() == 2 * num_left_nodes_;
00456   DCHECK(!ok || EpsilonOptimal());
00457   while (ok && epsilon_ > kMinEpsilon) {
00458     ok &= UpdateEpsilon();
00459     ok &= Refine();
00460     ReportAndAccumulateStats();
00461     DCHECK(!ok || EpsilonOptimal());
00462     DCHECK(!ok || AllMatched());
00463   }
00464   success_ = ok;
00465   return ok;
