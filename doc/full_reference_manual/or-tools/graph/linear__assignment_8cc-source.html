00143 bool LinearSumAssignment::UpdateEpsilon() {
00144   // There are some somewhat subtle issues around using integer
00145   // arithmetic to compute successive values of epsilon_, the error
00146   // parameter.
00147   //
00148   // First, the value of price_reduction_bound_ is chosen under the
00149   // assumption that it is truly an upper bound on the amount by which
00150   // a node's price can change during the current iteration. The value
00151   // of this bound in turn depends on the assumption that the flow
00152   // computed by the previous iteration was
00153   // (epsilon_ * alpha_)-optimal. If epsilon_ decreases by more than a
00154   // factor of alpha_ due to truncating arithmetic, that bound might
00155   // not hold, and the consequence is that BestArcAndGap could return
00156   // an overly cautious admissibility gap in the case where a
00157   // left-side node has only one incident arc. This is not a big deal
00158   // at all. At worst it could lead to a few extra relabelings.
00159   //
00160   // Second, it is not a problem currently, but in the future if we
00161   // use an arc-fixing heuristic, we cannot permit truncating integer
00162   // division to decrease epsilon_ by a factor greater than alpha_
00163   // because our bounds on price changes (and hence on the reduced
00164   // cost of an arc that might ever become admissible in the future)
00165   // depend on the ratio between the values of the error parameter at
00166   // successive iterations. One consequence will be that we might
00167   // occasionally do an extra iteration today in the interest of being
00168   // able to "price arcs out" (which we don't do today). Today we
00169   // simply use truncating integer division, but note that this will
00170   // have to change if we ever price arcs out.
00171   //
00172   // Since neither of those issues is a problem today, we simply use
00173   // truncating integer arithmetic. But future changes might
00174   // necessitate rounding epsilon upward in the division.
00175   epsilon_ = std::max(epsilon_ / alpha_, kMinEpsilon);
00176   VLOG(3) << "Updated: epsilon_ == " << epsilon_;
00177   price_reduction_bound_ = PriceChangeBound(1, NULL);
00178   DCHECK_GT(price_reduction_bound_, 0);
00179   // For today we always return true; in the future updating epsilon
00180   // in sophisticated ways could conceivably detect infeasibility.
00181   return true;
00182 }

00214 // There exists a price function such that the admissible arcs at the
00215 // beginning of an iteration are exactly the reverse arcs of all
00216 // matching arcs. Saturating all admissible arcs with respect to that
00217 // price function therefore means simply unmatching every matched
00218 // node.
00219 //
00220 // In the future we will price out arcs, which will reduce the set of
00221 // nodes we unmatch here. If a matching arc is priced out, we will not
00222 // unmatch its endpoints since that element of the matching is
00223 // guaranteed not to change.
00224 void LinearSumAssignment::SaturateNegativeArcs() {
00225   total_excess_ = 0;
00226   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
00227        node_it.Ok();
00228        node_it.Next()) {
00229     const NodeIndex node = node_it.Index();
00230     if (IsActive(node)) {
00231       // This can happen in the first iteration when nothing is
00232       // matched yet.
00233       total_excess_ += 1;
00234     } else {
00235       // We're about to create a unit of excess by unmatching these nodes.
00236       total_excess_ += 1;
00237       const NodeIndex mate = GetMate(node);
00238       matched_arc_.Set(node, StarGraph::kNilArc);
00239       matched_node_.Set(mate, StarGraph::kNilNode);
00240     }
00241   }
00242 }

00244 // Returns true for success, false for infeasible.
00245 bool LinearSumAssignment::DoublePush(NodeIndex source) {
00246   DCHECK_GT(num_left_nodes_, source);
00247   DCHECK(IsActive(source));
00248   ImplicitPriceSummary summary = BestArcAndGap(source);
00249   const ArcIndex best_arc = summary.first;
00250   const CostValue gap = summary.second;
00251   // Now we have the best arc incident to source, i.e., the one with
00252   // minimum reduced cost. Match that arc, unmatching its head if
00253   // necessary.
00254   if (best_arc == StarGraph::kNilArc) {
00255     return false;
00256   }
00257   const NodeIndex new_mate = Head(best_arc);
00258   const NodeIndex to_unmatch = matched_node_[new_mate];
00259   if (to_unmatch != StarGraph::kNilNode) {
00260     // Unmatch new_mate from its current mate, pushing the unit of
00261     // flow back to a node on the left side as a unit of excess.
00262     matched_arc_.Set(to_unmatch, StarGraph::kNilArc);
00263     active_nodes_->Add(to_unmatch);
00264     // This counts as a double push.
00265     iteration_stats_.double_pushes_ += 1;
00266   } else {
00267     // We are about to increase the cardinality of the matching.
00268     total_excess_ -= 1;
00269     // This counts as a single push.
00270     iteration_stats_.pushes_ += 1;
00271   }
00272   matched_arc_.Set(source, best_arc);
00273   matched_node_.Set(new_mate, source);
00274   // Finally, relabel new_mate.
00275   iteration_stats_.relabelings_ += 1;
00276   price_.Set(new_mate, price_[new_mate] - gap - epsilon_);
00277   return price_[new_mate] >= price_lower_bound_;
00278 }

00280 bool LinearSumAssignment::Refine() {
00281   SaturateNegativeArcs();
00282   InitializeActiveNodeContainer();
00283   while (total_excess_ > 0) {
00284     // Get an active node (i.e., one with excess == 1) and discharge
00285     // it using DoublePush.
00286     const NodeIndex node = active_nodes_->Get();
00287     if (!DoublePush(node)) {
00288       // Infeasibility detected.
00289       return false;
00290     }
00291   }
00292   DCHECK(active_nodes_->Empty());
00293   iteration_stats_.refinements_ += 1;
00294   return true;

00297 // Computes best_arc, the minimum reduced-cost arc incident to
00298 // left_node and admissibility_gap, the amount by which the reduced
00299 // cost of best_arc must be increased to make it equal in reduced cost
00300 // to another residual arc incident to left_node.
00301 //
00302 // Precondition: left_node is unmatched. This allows us to simplify
00303 // the code. The debug-only counterpart to this routine is
00304 // LinearSumAssignment::ImplicitPrice() and it does not assume this
00305 // precondition.
00306 //
00307 // This function is large enough that our suggestion that the compiler
00308 // inline it might be pointless.
00309 inline LinearSumAssignment::ImplicitPriceSummary
00310 LinearSumAssignment::BestArcAndGap(NodeIndex left_node) const {
00311     DCHECK(IsActive(left_node));
00312   DCHECK_GT(epsilon_, 0);
00313   // During any scaling iteration, the price of an active node
00314   // decreases by at most price_reduction_bound_ and all left-side
00315   // nodes are made active at the beginning of Refine(), so the bound
00316   // applies to all left-side nodes.
00317   StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00318   ArcIndex best_arc = arc_it.Index();
00319   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00320   CostValue second_min_partial_reduced_cost =
00321       min_partial_reduced_cost + price_reduction_bound_;
00322   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00323     const ArcIndex arc = arc_it.Index();
00324     const CostValue partial_reduced_cost = PartialReducedCost(arc);
00325     if (partial_reduced_cost < min_partial_reduced_cost) {
00326       best_arc = arc;
00327       second_min_partial_reduced_cost = min_partial_reduced_cost;
00328       min_partial_reduced_cost = partial_reduced_cost;
00329     } else if (partial_reduced_cost < second_min_partial_reduced_cost) {
00330       second_min_partial_reduced_cost = partial_reduced_cost;
00331     }
00332   }
00333   const CostValue gap =
00334       second_min_partial_reduced_cost - min_partial_reduced_cost;
00335   DCHECK_GE(gap, 0);
00336   return std::make_pair(best_arc, gap);
00337 }

00340 inline CostValue
00341 LinearSumAssignment::ImplicitPrice(NodeIndex left_node) const {
00342   DCHECK_GT(num_left_nodes_, left_node);
00343   DCHECK_GT(epsilon_, 0);
00344   StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00345   // If the input problem is feasible, it is always the case that
00346   // arc_it.Ok(), i.e., that there is at least one arc incident to
00347   // left_node.
00348   DCHECK(arc_it.Ok());
00349   ArcIndex best_arc = arc_it.Index();
00350   if (best_arc == matched_arc_[left_node]) {
00351     arc_it.Next();
00352     if (arc_it.Ok()) {
00353       best_arc = arc_it.Index();
00354     }
00355   }
00356   CostValue min_partial_reduced_cost = PartialReducedCost(best_arc);
00357   if (!arc_it.Ok()) {
00358     // Only one arc is incident to left_node, and the node is
00359     // currently matched along that arc, which must be the case in any
00360     // feasible solution. Therefore we implicitly price this node so
00361     // low that we will never consider unmatching it.
00362     return -(min_partial_reduced_cost + price_reduction_bound_);
00363   }
00364   for (arc_it.Next(); arc_it.Ok(); arc_it.Next()) {
00365     const ArcIndex arc = arc_it.Index();
00366     if (arc != matched_arc_[left_node]) {
00367       const CostValue partial_reduced_cost = PartialReducedCost(arc);
00368       if (partial_reduced_cost < min_partial_reduced_cost) {
00369         min_partial_reduced_cost = partial_reduced_cost;
00370       }
00371     }
00372   }
00373   return -min_partial_reduced_cost;
00374 }

00389 bool LinearSumAssignment::EpsilonOptimal() const {
00390   for (BipartiteLeftNodeIterator node_it(graph_, num_left_nodes_);
00391        node_it.Ok();
00392        node_it.Next()) {
00393     const NodeIndex left_node = node_it.Index();
00394     // Get the implicit price of left_node and make sure the reduced
00395     // costs of left_node's incident arcs are in bounds.
00396     CostValue left_node_price = ImplicitPrice(left_node);
00397     for (StarGraph::OutgoingArcIterator arc_it(graph_, left_node);
00398          arc_it.Ok();
00399          arc_it.Next()) {
00400       const ArcIndex arc = arc_it.Index();
00401       const CostValue reduced_cost =
00402           left_node_price + PartialReducedCost(arc);
00403       // Note the asymmetric definition of epsilon-optimality that we
00404       // use because it means we can saturate all admissible arcs in
00405       // the beginning of Refine() just by unmatching all matched
00406       // nodes.
00407       if (matched_arc_[left_node] == arc) {
00408         // The reverse arc is residual. Epsilon-optimality requires
00409         // that the reduced cost of the forward arc be at most
00410         // epsilon_.
00411         if (reduced_cost > epsilon_) {
00412           return false;
00413         }
00414       } else {
00415         // The forward arc is residual. Epsilon-optimality requires
00416         // that the reduced cost of the forward arc be at least zero.
00417         if (reduced_cost < 0) {
00418           return false;
00419         }
00420       }
00421     }
00422   }
00423   return true;
00424 }

00426 bool LinearSumAssignment::FinalizeSetup() {
00427   epsilon_ = largest_scaled_cost_magnitude_;
00428   VLOG(2) << "Largest given cost magnitude: " <<
00429       largest_scaled_cost_magnitude_ / cost_scaling_factor_;
00430   // Initialize left-side node-indexed arrays.
00431   StarGraph::NodeIterator node_it(graph_);
00432   for (; node_it.Ok(); node_it.Next()) {
00433     const NodeIndex node = node_it.Index();
00434     if (node >= num_left_nodes_) {
00435       break;
00436     }
00437     matched_arc_.Set(node, StarGraph::kNilArc);
00438   }
00439   // Initialize right-side node-indexed arrays. Example: prices are
00440   // stored only for right-side nodes.
00441   for (; node_it.Ok(); node_it.Next()) {
00442     const NodeIndex node = node_it.Index();
00443     price_.Set(node, 0);
00444     matched_node_.Set(node, StarGraph::kNilNode);
00445   }
00446   bool in_range;
00447   price_lower_bound_ = -PriceChangeBound(alpha_ - 1, &in_range);
00448   DCHECK_LE(price_lower_bound_, 0);
00449   if (!in_range) {
00450     LOG(WARNING) << "Price change bound exceeds range of representable "
00451                  << "costs; arithmetic overflow is not ruled out.";
00452   }
00453   return in_range;

00462 bool LinearSumAssignment::ComputeAssignment() {
00463   // Note: FinalizeSetup() might have been called already by white-box
00464   // test code or by a client that wants to react to the possibility
00465   // of overflow before solving the given problem, but FinalizeSetup()
00466   // is idempotent and reasonably fast, so we call it unconditionally
00467   // here.
00468   FinalizeSetup();
00469   bool ok = graph_.num_nodes() == 2 * num_left_nodes_;
00470   DCHECK(!ok || EpsilonOptimal());
00471   while (ok && epsilon_ > kMinEpsilon) {
00472     ok &= UpdateEpsilon();
00473     ok &= Refine();
00474     ReportAndAccumulateStats();
00475     DCHECK(!ok || EpsilonOptimal());
00476     DCHECK(!ok || AllMatched());
00477   }
00478   success_ = ok;
00479   VLOG(1) << "Overall stats: " << total_stats_.StatsString();
00480   return ok;
