00069 void MaxFlow::SetArcCapacity(ArcIndex arc, FlowQuantity new_capacity) {
00070   DCHECK_LE(0, new_capacity);
00071   DCHECK(graph_->CheckArcValidity(arc));
00072   const FlowQuantity free_capacity = residual_arc_capacity_[arc];
00073   const FlowQuantity capacity_delta = new_capacity - Capacity(arc);
00074   VLOG(1) << "Changing capacity on arc " << arc
00075           << " from " << Capacity(arc) << " to " << new_capacity
00076           << ". Current free capacity = " << free_capacity;
00077   if (capacity_delta == 0) {
00078     return;  // Nothing to do.
00079   }
00080   status_ = NOT_SOLVED;
00081   if (free_capacity + capacity_delta >= 0) {
00082     // The above condition is true if one of the two conditions is true:
00083     // 1/ (capacity_delta > 0), meaning we are increasing the capacity
00084     // 2/ (capacity_delta < 0 && free_capacity + capacity_delta >= 0)
00085     //    meaning we are reducing the capacity, but that the capacity
00086     //    reduction is not larger than the free capacity.
00087     residual_arc_capacity_.Set(arc, free_capacity + capacity_delta);
00088     DCHECK_LE(0, residual_arc_capacity_[arc]);
00089     VLOG(1) << "Now: capacity = " << Capacity(arc) << " flow = " << Flow(arc);
00090   } else {
00091     // We have to reduce the flow on the arc, and update the excesses
00092     // accordingly.
00093     const FlowQuantity flow = residual_arc_capacity_[Opposite(arc)];
00094     const FlowQuantity flow_excess = flow - new_capacity;
00095     VLOG(1) << "Flow value " << flow << " exceeds new capacity "
00096             << new_capacity << " by " << flow_excess;
00097     residual_arc_capacity_.Set(arc, 0);
00098     residual_arc_capacity_.Set(Opposite(arc), new_capacity);
00099     const NodeIndex head = Head(arc);
00100     node_excess_.Set(head, node_excess_[head] + flow_excess);
00101     DCHECK_LE(0, residual_arc_capacity_[arc]);
00102     DCHECK_LE(0, residual_arc_capacity_[Opposite(arc)]);
00103     VLOG(1) << DebugString("After SetArcCapacity:", arc);
00104   }
00105 }

00107 bool MaxFlow::CheckResult() const {
00108   bool ok = true;
00109   for (StarGraph::NodeIterator node_it(*graph_); node_it.Ok(); node_it.Next()) {
00110     const NodeIndex node = node_it.Index();
00111     if (node != source_ && node != sink_) {
00112       if (node_excess_[node] != 0) {
00113         LOG(DFATAL) << "node_excess_[" << node << "] = " << node_excess_[node]
00114                     << " != 0";
00115         ok = false;
00116       }
00117     }
00118   }
00119   for (StarGraph::ArcIterator arc_it(*graph_); arc_it.Ok(); arc_it.Next()) {
00120     const ArcIndex arc = arc_it.Index();
00121     const ArcIndex opposite = Opposite(arc);
00122     const FlowQuantity direct_capacity = residual_arc_capacity_[arc];
00123     const FlowQuantity opposite_capacity = residual_arc_capacity_[opposite];
00124     if (direct_capacity < 0) {
00125       LOG(DFATAL) << "residual_arc_capacity_[" << arc << "] = "
00126                   << direct_capacity << " != 0";
00127       ok = false;
00128     }
00129     if (opposite_capacity < 0) {
00130       LOG(DFATAL) << "residual_arc_capacity_[" << opposite << "] = "
00131                   << opposite_capacity << " != 0";
00132       ok = false;
00133     }
00134     // The initial capacity of the direct arcs is non-negative.
00135     if (direct_capacity + opposite_capacity < 0) {
00136       LOG(DFATAL) << "initial capacity [" << arc << "] = "
00137                   << direct_capacity + opposite_capacity << " < 0";
00138       ok = false;
00139     }
00140   }
00141   return ok;

00169 bool MaxFlow::Solve() {
00170   status_ = NOT_SOLVED;
00171   if (FLAGS_max_flow_check_input && !CheckInputConsistency()) {
00172     status_ = BAD_INPUT;
00173     return false;
00174   }
00175   InitializePreflow();
00176   ResetFirstAdmissibleArcs();
00177   Refine();
00178   if (FLAGS_max_flow_check_result && !CheckResult()) {
00179     status_ = BAD_RESULT;
00180     return false;
00181   }
00182   total_flow_ = 0;
00183   for (StarGraph::OutgoingArcIterator arc_it(*graph_, source_);
00184        arc_it.Ok();
00185        arc_it.Next()) {
00186     const ArcIndex arc = arc_it.Index();
00187     total_flow_ += Flow(arc);
00188   }
00189   status_ = OPTIMAL;
00190   return true;
00191 }

00200 void MaxFlow::InitializePreflow() {
00201   // InitializePreflow() clears the whole flow that could have been computed
00202   // by a previous Solve(). This is not optimal in terms of complexity.
00203   // TODO(user): find a way to make the re-solving incremental (not an obvious
00204   // task, and there has not been a lot of literature on the subject.)
00205   node_potential_.Assign(0);
00206   node_excess_.Assign(0);
00207   for (StarGraph::ArcIterator arc_it(*graph_); arc_it.Ok(); arc_it.Next()) {
00208     const ArcIndex arc = arc_it.Index();
00209     // Reset the residual capacities of direct arc to their initial values.
00210     residual_arc_capacity_.Set(arc, Capacity(arc));
00211     // The initial residual capacities on reverse arcs are set to 0.
00212     residual_arc_capacity_.Set(Opposite(arc), 0);
00213   }
00214   // The initial height of the source is equal to the number of nodes.
00215   node_potential_.Set(source_, graph_->num_nodes());
00216   for (StarGraph::OutgoingArcIterator arc_it(*graph_, source_);
00217        arc_it.Ok();
00218        arc_it.Next()) {
00219     const ArcIndex arc = arc_it.Index();
00220     const FlowQuantity arc_capacity = Capacity(arc);
00221     // Saturate arcs outgoing from the source. This is not really a PushFlow,
00222     // since the preconditions for PushFlow are not (yet) met, and we do not
00223     // need to update the excess at the source.
00224     residual_arc_capacity_.Set(arc, 0);
00225     residual_arc_capacity_.Set(Opposite(arc), arc_capacity);
00226     node_excess_.Set(Head(arc), arc_capacity);
00227     VLOG(1) << DebugString("InitializePreflow:", arc);
00228   }
00229 }

00231 void MaxFlow::PushFlow(FlowQuantity flow, ArcIndex arc) {
00232   DCHECK_GT(residual_arc_capacity_[arc], 0);
00233   DCHECK_GT(node_excess_[Tail(arc)], 0);
00234   VLOG(1) << "PushFlow: pushing " << flow << " on arc " << arc
00235           << " from node " << Tail(arc) << " to node " << Head(arc);
00236   // Reduce the residual capacity on arc by flow.
00237   residual_arc_capacity_.Set(arc, residual_arc_capacity_[arc] - flow);
00238   // Increase the residual capacity on opposite arc by flow.
00239   const ArcIndex opposite = Opposite(arc);
00240   residual_arc_capacity_.Set(opposite, residual_arc_capacity_[opposite] + flow);
00241   // Update the excesses at the tail and head of the arc.
00242   const NodeIndex tail = Tail(arc);
00243   node_excess_.Set(tail, node_excess_[tail] - flow);
00244   const NodeIndex head = Head(arc);
00245   node_excess_.Set(head, node_excess_[head] + flow);
00246   VLOG(2) << DebugString("PushFlow: ", arc);
00247 }

00272 void MaxFlow::Discharge(NodeIndex node) {
00273   DCHECK(IsActive(node));
00274   VLOG(1) << "Discharging node " << node << ", excess = " << node_excess_[node];
00275   while (IsActive(node)) {
00276     for (StarGraph::IncidentArcIterator arc_it(*graph_, node,
00277                                                first_admissible_arc_[node]);
00278          arc_it.Ok();
00279          arc_it.Next()) {
00280       const ArcIndex arc = arc_it.Index();
00281       VLOG(2) << DebugString("Discharge: considering", arc);
00282       if (IsAdmissible(arc)) {
00283         if (node_excess_[node] != 0) {
00284           VLOG(1) << "Discharge: calling PushFlow.";
00285           const NodeIndex head = Head(arc);
00286           const bool head_active_before_push = IsActive(head);
00287           const FlowQuantity delta = std::min(node_excess_[node],
00288                                               residual_arc_capacity_[arc]);
00289           PushFlow(delta, arc);
00290           if (IsActive(head) && !head_active_before_push) {
00291             active_nodes_.push(Head(arc));
00292           }
00293         }
00294         if (node_excess_[node] == 0) {
00295           first_admissible_arc_.Set(node, arc);  // arc may still be admissible.
00296           return;
00297         }
00298       }
00299     }
00300     Relabel(node);
00301   }
00302 }

00304 void MaxFlow::Relabel(NodeIndex node) {
00305   DCHECK(CheckRelabelPrecondition(node));
00306   CostValue min_height = node_potential_[node];
00307   for (StarGraph::IncidentArcIterator arc_it(*graph_, node);
00308        arc_it.Ok();
00309        arc_it.Next()) {
00310     const ArcIndex arc = arc_it.Index();
00311     DCHECK_EQ(Tail(arc), node);
00312     if (residual_arc_capacity_[arc] > 0) {
00313       // Update min_height only for arcs with available capacity.
00314       min_height = std::min(min_height, node_potential_[Head(arc)]);
00315     }
00316   }
00317   VLOG(1) << "Relabel: height(" << node << ") relabeled from "
00318           << node_potential_[node] << " to " << min_height + 1;
00319   node_potential_.Set(node, min_height + 1);
00320   first_admissible_arc_.Set(node, GetFirstIncidentArc(node));
00321 }
