00054 bool MaxFlow::CheckResult() const {
00055   for (StarGraph::NodeIterator node_it(graph_); node_it.Ok(); node_it.Next()) {
00056     const NodeIndex node = node_it.Index();
00057     if (node != source_ && node != sink_) {
00058       CHECK_EQ(0, node_excess_[node]) << " node = " << node;
00059     }
00060   }
00061   for (StarGraph::ArcIterator arc_it(graph_); arc_it.Ok(); arc_it.Next()) {
00062     const ArcIndex arc = arc_it.Index();
00063     const ArcIndex opposite = Opposite(arc);
00064     CHECK_GE(residual_arc_capacity_[arc], 0);
00065     CHECK_GE(residual_arc_capacity_[opposite], 0);
00066     // The initial capacity of the direct arcs is non-negative.
00067     CHECK_GE(residual_arc_capacity_[arc] + residual_arc_capacity_[opposite], 0);
00068   }
00069   return true;

00129 void MaxFlow::InitializePreflow() {
00130   // The initial height of the source is equal to the number of nodes.
00131   node_potential_.Set(source_, graph_.num_nodes());
00132   for (StarGraph::OutgoingArcIterator arc_it(graph_, source_);
00133        arc_it.Ok();
00134        arc_it.Next()) {
00135     const ArcIndex arc = arc_it.Index();
00136     const FlowQuantity arc_capacity = residual_arc_capacity_[arc];
00137     // Saturate arcs outgoing from the source. This is not really a PushFlow,
00138     // since the preconditions for PushFlow are not (yet) met, and we do not
00139     // need to update the excess at the source.
00140     residual_arc_capacity_.Set(arc, 0);
00141     residual_arc_capacity_.Set(Opposite(arc), arc_capacity);
00142     node_excess_.Set(Head(arc), arc_capacity);
00143     VLOG(1) << DebugString("InitializePreflow:", arc);
00144   }
00145 }

00147 void MaxFlow::PushFlow(FlowQuantity flow, ArcIndex arc) {
00148   DCHECK_GT(residual_arc_capacity_[arc], 0);
00149   DCHECK_GT(node_excess_[Tail(arc)], 0);
00150   VLOG(1) << "PushFlow: pushing " << flow << " on arc " << arc
00151           << " from node " << Tail(arc) << " to node " << Head(arc);
00152   // Reduce the residual capacity on arc by flow.
00153   residual_arc_capacity_.Set(arc, residual_arc_capacity_[arc] - flow);
00154   // Increase the residual capacity on opposite arc by flow.
00155   const ArcIndex opposite = Opposite(arc);
00156   residual_arc_capacity_.Set(opposite, residual_arc_capacity_[opposite] + flow);
00157   // Update the excesses at the tail and head of the arc.
00158   const NodeIndex tail = Tail(arc);
00159   node_excess_.Set(tail, node_excess_[tail] - flow);
00160   const NodeIndex head = Head(arc);
00161   node_excess_.Set(head, node_excess_[head] + flow);
00162   VLOG(2) << DebugString("PushFlow: ", arc);
00163 }

00188 void MaxFlow::Discharge(NodeIndex node) {
00189   DCHECK(IsActive(node));
00190   VLOG(1) << "Discharging node " << node << ", excess = " << node_excess_[node];
00191   while (IsActive(node)) {
00192     for (StarGraph::IncidentArcIterator arc_it(graph_, node,
00193                                                first_admissible_arc_[node]);
00194          arc_it.Ok();
00195          arc_it.Next()) {
00196       const ArcIndex arc = arc_it.Index();
00197       VLOG(2) << DebugString("Discharge: considering", arc);
00198       if (IsAdmissible(arc)) {
00199         if (node_excess_[node] != 0) {
00200           VLOG(1) << "Discharge: calling PushFlow.";
00201           const NodeIndex head = Head(arc);
00202           const bool head_active_before_push = IsActive(head);
00203           const FlowQuantity delta = min(node_excess_[node],
00204                                          residual_arc_capacity_[arc]);
00205           PushFlow(delta, arc);
00206           if (IsActive(head) && !head_active_before_push) {
00207             active_nodes_.push(Head(arc));
00208           }
00209         }
00210         if (node_excess_[node] == 0) {
00211           first_admissible_arc_.Set(node, arc);  // arc may still be admissible.
00212           return;
00213         }
00214       }
00215     }
00216     Relabel(node);
00217   }

00220 void MaxFlow::Relabel(NodeIndex node) {
00221   DCHECK(CheckRelabelPrecondition(node));
00222   CostValue min_height = node_potential_[node];
00223   for (StarGraph::IncidentArcIterator arc_it(graph_, node);
00224        arc_it.Ok();
00225        arc_it.Next()) {
00226     const ArcIndex arc = arc_it.Index();
00227     DCHECK_EQ(Tail(arc), node);
00228     if (residual_arc_capacity_[arc] > 0) {
00229       // Update min_height only for arcs with available capacity.
00230       min_height = min(min_height, node_potential_[Head(arc)]);
00231     }
00232   }
00233   VLOG(1) << "Relabel: height(" << node << ") relabeled from "
00234           << node_potential_[node] << " to " << min_height + 1;
00235   node_potential_.Set(node, min_height + 1);
00236   first_admissible_arc_.Set(node, GetFirstIncidentArc(node));
00237 }
