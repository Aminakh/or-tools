00031 // An array class for storing arrays of signed integers on NumBytes bytes,
00032 // The range of indices is specified at the construction of the object.
00033 // The minimum and maximum indices are inclusive.
00034 // Think of the Pascal syntax array[min_index..max_index] of ...
00035 // For example PackedArray<5>(-100000,100000) will store 200001 signed
00036 // integers on 5 bytes or 40 bits, and the possible range of indices will be
00037 // -100000..100000.
00038 //
00039 // There is no penalty for using this class for integer sizes of 1,2,4,8 bytes.
00040 // For other sizes the write time penalty ranges from 20% to 100% (for a 7-byte
00041 // integer.) The read time penalty is about 50% for integer sizes different
00042 // from 1,2,4,8 bytes.
00043 //
00044 // WARNING: The implementation of this class (member functions Set and Value)
00045 // assumes the underlying architecture is little-endian.
00046 // TODO(user): make the implementation big-endian compatible.
00047 
00048 #if __BYTE_ORDER != __LITTLE_ENDIAN
00049 #error "The implementation of PackedArray assumes a little-endian architecture."
00050 #endif

00056 template<class T, int NumBytes> class PackedArrayAllocator {
00057  public:
00058   PackedArrayAllocator()
00059       : base_(NULL),
00060         min_index_(0),
00061         max_index_(0),
00062         size_in_bytes_(0),
00063         storage_() {}
00064   // Reserves memory for new minimum and new maximum indices.
00065   // Returns true if the memory could be reserved.
00066   // Never shrinks the memory allocated.
00067   bool Reserve(int64 new_min_index, int64 new_max_index) {
00068     DCHECK_LE(new_min_index, new_max_index);
00069     if (base_ != NULL
00070         && new_min_index >= min_index_
00071         && new_max_index <= max_index_) {
00072       return true;
00073     }
00074     DCHECK(base_ == NULL || new_min_index <= min_index_);
00075     DCHECK(base_ == NULL || new_max_index >= max_index_);
00076     const uint64 max_uint64 = std::numeric_limits<uint64>::max();
00077     const uint64 new_size = new_max_index - new_min_index + 1;
00078     DCHECK_GT(max_uint64 / NumBytes, new_size);
00079     // We need to pad the array to allow reading the last element as an int64.
00080     const uint64 new_size_in_bytes = new_size * NumBytes
00081                                    + sizeof(int64) - NumBytes; // NOLINT
00082     byte* new_storage = new byte[new_size_in_bytes];
00083     if (new_storage == NULL) {
00084       return false;
00085     }
00086 
00087     byte* const new_base = new_storage - new_min_index * NumBytes;
00088     if (base_ != NULL) {
00089       byte* const destination = new_base + min_index_ * NumBytes;
00090       memcpy(destination, storage_.get(), size_in_bytes_);
00091     }
00092 
00093     base_ = reinterpret_cast<T*>(new_base);
00094     min_index_ = new_min_index;
00095     max_index_ = new_max_index;
00096     size_in_bytes_ = new_size_in_bytes;
00097     storage_.reset(reinterpret_cast<T*>(new_storage));
00098     return true;
00099   }
00100 
00101   int64 min_index() const { return min_index_; }
00102 
00103   int64 max_index() const { return max_index_; }
00104 
00105   T* Base() const {
00106     DCHECK(base_ != NULL);
00107     return base_;
00108   }
00109 
00110  private:
00111   // Pointer to the element indexed by zero in the array.
00112   T*                 base_;
00113 
00114   // Minimum index for the array.
00115   int64              min_index_;
00116 
00117   // Maximum index for the array.
00118   int64              max_index_;
00119 
00120   // The size of the array in bytes.
00121   uint64             size_in_bytes_;
00122 
00123   // Storage memory for the array.
00124   scoped_array<T> storage_;

00131   // The constructor for PackedArray takes a mininum index and a maximum index.
00132   // These can be positive or negative, and the value for minimum_index

00149   // Returns the value stored at index.
00150   // This assumes that the underlying architecture is little-endian. In
00151   // particular, we select the value to return from the low-address NumBytes
00152   // bytes of the int64 beginning at the location determined by the given index.
00153   // To extract those NumBytes from the int64 loaded from memory, we choose the
00154   // bytes of low arithmetic significance. Code for a big-endian system would
00155   // need to choose the bytes of high arithmetic significance.
00156   int64 Value(int64 index) const {
00157     DCHECK_LE(allocator_.min_index(), index);
00158     DCHECK_GE(allocator_.max_index(), index);
00159     int64 value =
00160         *reinterpret_cast<int64*>(allocator_.Base() + index * NumBytes);
00161     const int shift = (sizeof(value) - NumBytes) * CHAR_BIT;
00162     value <<= shift;  // These two lines are
00163     value >>= shift;  // for sign extension.
00164     return value;
00165   }
00166 
00167 #if !defined(SWIG)
00168   // Shortcut for returning the value stored at index.
00169   int64 operator[](int64 index) const {
00170     return Value(index);
00171   }
00172 #endif

00174 // Let n by the number of bytes (i.e. sizeof) of type.
00175 // PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE writes the first n lowest
00176 // significant bytes of value at address. It then shifts value by n bytes to
00177 // the right, and advances address by n bytes.
00178 // This assumes that the underlying architecture is little-endian, since
00179 // we write the lowest n bytes for value. Code for a big-endian system would
00180 // need to write the bytes of high arithmetic significance.
00181 #define PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(type, address, value)  \
00182 {                                                                             \
00183   const int64 kSize = sizeof(type);                                           \
00184   if (NumBytes & kSize) {                                                     \
00185     const int64 k1 = GG_LONGLONG(1);                                          \
00186     const int64 kSizeInBits = sizeof(type) * CHAR_BIT;                        \
00187     const int64 kMask = (k1 << kSizeInBits) - 1;                              \
00188     reinterpret_cast<type*>(address)[0] = value & kMask;                      \
00189     value >>= kSizeInBits;                                                    \
00190     address += kSize;                                                         \
00191   }                                                                           \
00192 }
00193 
00194   // Sets to value the content of the array at index.
00195   void Set(int64 index, int64 value) {
00196     DCHECK_LE(allocator_.min_index(), index);
00197     DCHECK_GE(allocator_.max_index(), index);
00198     DCHECK_LE(kMinInteger, value);
00199     DCHECK_GE(kMaxInteger, value);
00200     byte* current = allocator_.Base() + index * NumBytes;
00201     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int32, current, value);
00202     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int16, current, value);
00203     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(byte, current, value);
00204   }
00205 
00206   // Reserves memory for new minimum and new maximum indices.
00207   // Never shrinks the memory allocated.
00208   bool Reserve(int64 new_min_index, int64 new_max_index) {
00209     return allocator_.Reserve(new_min_index, new_max_index);
00210   }
00211 
00212   // Sets all the elements in the array to value. Set is bypassed to maximize
00213   // performance.
00214   void Assign(int64 value) {
00215     DCHECK_LE(kMinInteger, value);
00216     DCHECK_GE(kMaxInteger, value);
00217     const byte* end = allocator_.Base() + allocator_.max_index() * NumBytes;
00218     byte* current = allocator_.Base() + allocator_.min_index() * NumBytes;
00219     while (current <= end) {
00220       int64 v = value;  // v is going to be modified by the following macro.
00221       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int32, current, v);
00222       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int16, current, v);
00223       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(byte, current, v);
00224     }
00225   }
00226 #undef PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE

00228  private:
00229   // The bitmask corresponding to all the bits in Numbytes bytes set.
00230   static const uint64 kBitMask = (GG_ULONGLONG(1) << (CHAR_BIT * NumBytes)) - 1;
00231 
00232   // The maximum signed integer representable with NumBytes bytes.
00233   static const int64 kMaxInteger = kBitMask >> 1;
00234 
00235   // The minimum signed integer representable with NumBytes bytes.
00236   static const int64 kMinInteger = ~kMaxInteger;
00237 
00238   PackedArrayAllocator<byte, NumBytes> allocator_;
