00031 // An array class for storing arrays of signed integers on NumBytes bytes,
00032 // The range of indices is specified at the construction of the object.
00033 // The minimum and maximum indices are inclusive.
00034 // Think of the Pascal syntax array[min_index..max_index] of ...
00035 // For example PackedArray<5>(-100000,100000) will store 200001 signed
00036 // integers on 5 bytes or 40 bits, and the possible range of indices will be
00037 // -100000..100000.
00038 //
00039 // There is no penalty for using this class for integer sizes of 1,2,4,8 bytes.
00040 // For other sizes the write time penalty ranges from 20% to 100% (for a 7-byte
00041 // integer.) The read time penalty is about 50% for integer sizes different
00042 // from 1,2,4,8 bytes.
00043 //
00044 // WARNING: The implementation of this class (member functions Set and Value)
00045 // assumes the underlying architecture is little-endian.
00046 // TODO(user): make the implementation big-endian compatible.
00047 
00048 #if __BYTE_ORDER != __LITTLE_ENDIAN
00049 #error "The implementation of PackedArray assumes a little-endian architecture."
00050 #endif

00056 template<class T, int NumBytes> class PackedArrayAllocator {
00057  public:
00058   PackedArrayAllocator()
00059       : base_(NULL),
00060         min_index_(0),
00061         max_index_(0),
00062         size_in_bytes_(0),
00063         storage_() {}
00064   // Reserves memory for new minimum and new maximum indices.
00065   // Never shrinks the memory allocated.
00066   void Reserve(int64 new_min_index, int64 new_max_index) {
00067     DCHECK_LE(new_min_index, new_max_index);
00068     if (base_ != NULL
00069         && new_min_index >= min_index_
00070         && new_max_index <= max_index_) {
00071       return;
00072     }
00073     DCHECK(base_ == NULL || new_min_index <= min_index_);
00074     DCHECK(base_ == NULL || new_max_index >= max_index_);
00075     const uint64 max_uint64 = std::numeric_limits<uint64>::max();
00076     const uint64 new_size = new_max_index - new_min_index + 1;
00077     DCHECK_GT(max_uint64 / NumBytes, new_size);
00078     // We need to pad the array to allow reading the last element as an int64.
00079     const uint64 new_size_in_bytes = new_size * NumBytes
00080                                    + sizeof(int64) - NumBytes; // NOLINT
00081     byte* new_storage = new byte[new_size_in_bytes];
00082 
00083     byte* const new_base = new_storage - new_min_index * NumBytes;
00084     if (base_ != NULL) {
00085       byte* const destination = new_base + min_index_ * NumBytes;
00086       memcpy(destination, storage_.get(), size_in_bytes_);
00087     }
00088 
00089     base_ = reinterpret_cast<T*>(new_base);
00090     min_index_ = new_min_index;
00091     max_index_ = new_max_index;
00092     size_in_bytes_ = new_size_in_bytes;
00093     storage_.reset(reinterpret_cast<T*>(new_storage));
00094   }
00095 
00096   int64 min_index() const { return min_index_; }
00097 
00098   int64 max_index() const { return max_index_; }
00099 
00100   T* Base() const {
00101     DCHECK(base_ != NULL);
00102     return base_;
00103   }
00104

00106   // Pointer to the element indexed by zero in the array.
00107   T*                 base_;
00108 
00109   // Minimum index for the array.
00110   int64              min_index_;
00111 
00112   // Maximum index for the array.
00113   int64              max_index_;
00114 
00115   // The size of the array in bytes.
00116   uint64             size_in_bytes_;

00123  public:

00126   // The constructor for PackedArray takes a mininum index and a maximum index.
00127   // These can be positive or negative, and the value for minimum_index
00128   // and for maximun_index can be set and read (i.e. the bounds are inclusive.)

00141   // Returns the value stored at index.
00142   // This assumes that the underlying architecture is little-endian. In
00143   // particular, we select the value to return from the low-address NumBytes
00144   // bytes of the int64 beginning at the location determined by the given index.
00145   // To extract those NumBytes from the int64 loaded from memory, we choose the
00146   // bytes of low arithmetic significance. Code for a big-endian system would
00147   // need to choose the bytes of high arithmetic significance.
00148   int64 Value(int64 index) const {
00149     DCHECK_LE(allocator_.min_index(), index);
00150     DCHECK_GE(allocator_.max_index(), index);
00151     int64 value =
00152         *reinterpret_cast<int64*>(allocator_.Base() + index * NumBytes);
00153     const int shift = (sizeof(value) - NumBytes) * CHAR_BIT;
00154     value <<= shift;  // These two lines are
00155     value >>= shift;  // for sign extension.
00156     return value;
00157   }
00158 
00159 #if !defined(SWIG)
00160   // Shortcut for returning the value stored at index.
00161   int64 operator[](int64 index) const {
00162     return Value(index);
00163   }
00164 #endif

00166 // Let n by the number of bytes (i.e. sizeof) of type.
00167 // PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE writes the first n lowest
00168 // significant bytes of value at address. It then shifts value by n bytes to
00169 // the right, and advances address by n bytes.
00170 // This assumes that the underlying architecture is little-endian, since
00171 // we write the lowest n bytes for value. Code for a big-endian system would
00172 // need to write the bytes of high arithmetic significance.
00173 #define PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(type, address, value)  \
00174 {                                                                             \
00175   const int64 kSize = sizeof(type);                                           \
00176   if (NumBytes & kSize) {                                                     \
00177     const int64 k1 = GG_LONGLONG(1);                                          \
00178     const int64 kSizeInBits = sizeof(type) * CHAR_BIT;                        \
00179     const int64 kMask = (k1 << kSizeInBits) - 1;                              \
00180     reinterpret_cast<type*>(address)[0] = value & kMask;                      \
00181     value >>= kSizeInBits;                                                    \
00182     address += kSize;                                                         \
00183   }                                                                           \
00184 }
00185 
00186   // Sets to value the content of the array at index.
00187   void Set(int64 index, int64 value) {
00188     DCHECK_LE(allocator_.min_index(), index);
00189     DCHECK_GE(allocator_.max_index(), index);
00190     DCHECK_LE(kMinInteger, value);
00191     DCHECK_GE(kMaxInteger, value);
00192     byte* current = allocator_.Base() + index * NumBytes;
00193     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int32, current, value);
00194     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int16, current, value);
00195     PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(byte, current, value);
00196   }
00197 
00198   // Reserves memory for new minimum and new maximum indices.
00199   // Never shrinks the memory allocated.
00200   void Reserve(int64 new_min_index, int64 new_max_index) {
00201     allocator_.Reserve(new_min_index, new_max_index);
00202   }
00203 
00204   // Sets all the elements in the array to value. Set is bypassed to maximize
00205   // performance.
00206   void Assign(int64 value) {
00207     DCHECK_LE(kMinInteger, value);
00208     DCHECK_GE(kMaxInteger, value);
00209     const byte* end = allocator_.Base() + allocator_.max_index() * NumBytes;
00210     byte* current = allocator_.Base() + allocator_.min_index() * NumBytes;
00211     while (current <= end) {
00212       int64 v = value;  // v is going to be modified by the following macro.
00213       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int32, current, v);
00214       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(int16, current, v);
00215       PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE(byte, current, v);
00216     }
00217   }
00218 #undef PACKED_ARRAY_WRITE_IF_ENOUGH_BYTES_AND_ADVANCE
00219 
00220  private:
00221   // The bitmask corresponding to all the bits in Numbytes bytes set.
00222   static const uint64 kBitMask = (GG_ULONGLONG(1) << (CHAR_BIT * NumBytes)) - 1;
00223 
00224   // The maximum signed integer representable with NumBytes bytes.
00225   static const int64 kMaxInteger = kBitMask >> 1;
00226 
00227   // The minimum signed integer representable with NumBytes bytes.
00228   static const int64 kMinInteger = ~kMaxInteger;
00229 
00230   PackedArrayAllocator<byte, NumBytes> allocator_;
00231 };
