00026 // This class is used to store constant copies of int64 arrays inside
00027 // constraints or expression. When constructed with a C array or a
00028 // vector, The const int array will make a internal copy and own that
00029 // copy. It will not take ownership of the vector/array which can be
00030 // deleted afterwards. This follows the semantics of constraints and
00031 // expressions which store a read-only copy of the data.
00032 //
00033 // Its goals are:
00034 // - to unify the construction code across the optimization libraries.
00035 // - to provide one code to scan these arrays and compute given properties like
00036 //   monotonicity.
00037 //
00038 // As const int arrays provide scanning capabilities, the code inside
00039 // a constraint and its factory looks like:
00040 //
00041 // IntExpr* MakeMyExpr(const std::vector<int64>& values) {
00042 //   ConstIntArray copy(values);
00043 //   if (copy.Status(ConstIntArray::IS_INCREASING)) {
00044 //     return new MyOptimizedExpr(copy.Release());
00045 //   } else {
00046 //    return new MyGenericExpr(copy.Release());
00047 //   }
00048 // }
00049 //
00050 // With:
00051 // class MyOptimizedExpr : IntExpr {
00052 //  public:
00053 //   MyOptimizedExpr(std::vector<int64>* data) : values_(data) {}
00054 //  private:
00055 //   ConstIntArray values_;
00056 // };
