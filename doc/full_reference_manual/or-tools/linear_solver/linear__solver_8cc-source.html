00361   // Allow names that conform to the LP and MPS format.

00367     return false;

00369   if (name.find_first_of(" +-*<>=:\\") != string::npos) {
00370     LOG(WARNING) << "Invalid name " << name
00371                  << ": contains forbidden character: +-*<>=:\\ space."
00372                  << " Will be unable to write model to file.";
00373     return false;
00374   }
00375   size_t first_occurrence = name.find_first_of(".0123456789");
00376   if (first_occurrence != string::npos && first_occurrence == 0) {
00377     LOG(WARNING) << "Invalid name " << name
00378                  << ": first character should not be . or a number."
00379                  << " Will be unable to write model to file.";
00380     return false;

00400 // Loads model from protocol buffer.
00401 MPSolver::LoadStatus MPSolver::LoadModel(const MPModelProto& model) {
00402   hash_map<string, MPVariable*> variables;
00403   for (int i = 0; i < model.variables_size(); ++i) {
00404     const MPVariableProto& var_proto = model.variables(i);
00405     const string& id = var_proto.id();
00406     if (!ContainsKey(variables, id)) {
00407       MPVariable* variable = MakeNumVar(var_proto.lb(), var_proto.ub(), id);
00408       variable->SetInteger(var_proto.integer());
00409       variables[id] = variable;
00410     } else {
00411       return MPSolver::DUPLICATE_VARIABLE_ID;
00412     }
00413   }
00414   for (int i = 0; i < model.constraints_size(); ++i) {
00415     const MPConstraintProto& ct_proto = model.constraints(i);
00416     const string& ct_id = ct_proto.has_id() ? ct_proto.id() : "";
00417     MPConstraint* const ct = MakeRowConstraint(ct_proto.lb(),
00418                                                ct_proto.ub(),
00419                                                ct_id);
00420     for (int j = 0; j < ct_proto.terms_size(); ++j) {
00421       const MPTermProto& term_proto = ct_proto.terms(j);
00422       const string& id = term_proto.variable_id();
00423       MPVariable* variable = FindPtrOrNull(variables, id);
00424       if (variable != NULL) {
00425         ct->AddTerm(variable, term_proto.coefficient());
00426       } else {
00427         return MPSolver::UNKNOWN_VARIABLE_ID;
00428       }
00429     }
00430   }
00431   for (int i = 0; i < model.objective_terms_size(); ++i) {
00432     const MPTermProto& term_proto = model.objective_terms(i);
00433     const string& id = term_proto.variable_id();
00434     MPVariable* variable = FindPtrOrNull(variables, id);
00435     if (variable != NULL) {
00436       AddObjectiveTerm(variable, term_proto.coefficient());
00437     } else {
00438       return MPSolver::UNKNOWN_VARIABLE_ID;
00439     }
00440   }
00441   SetOptimizationDirection(model.maximize());
00442   if (model.has_objective_offset()) {
00443     linear_objective_.SetOffset(model.objective_offset());
00444   }
00445   return MPSolver::NO_ERROR;

00448 // Exports model to protocol buffer.
00449 void MPSolver::ExportModel(MPModelProto* model) const {
00450   CHECK_NOTNULL(model);
00451   if (model->variables_size() > 0 ||
00452       model->has_maximize() ||
00453       model->objective_terms_size() > 0 ||
00454       model->constraints_size() > 0 ||
00455       model->has_name() ||
00456       model->has_objective_offset()) {
00457     LOG(WARNING) << "The model protocol buffer is not empty, "
00458                  << "it will be overwritten.";
00459     model->clear_variables();
00460     model->clear_maximize();
00461     model->clear_objective_terms();
00462     model->clear_constraints();
00463     model->clear_name();
00464   }
00465 
00466   // Variables need non-empty names for the model protocol buffer, so
00467   // we may need to create names.
00468   hash_map<MPVariable*, string> valid_variable_names;
00469 
00470   // Variables
00471   for (int j = 0; j < variables_.size(); ++j) {
00472     MPVariable* const var = variables_[j];
00473     MPVariableProto* const variable_proto = model->add_variables();
00474     valid_variable_names[var] = CreateValidVariableName(*var, j);
00475     variable_proto->set_id(valid_variable_names[var]);
00476     variable_proto->set_lb(var->lb());
00477     variable_proto->set_ub(var->ub());
00478     variable_proto->set_integer(var->integer());
00479   }
00480 
00481   // Constraints
00482   for (int i = 0; i < constraints_.size(); ++i) {
00483     MPConstraint* const constraint = constraints_[i];
00484     MPConstraintProto* const constraint_proto = model->add_constraints();
00485     // Constraint names need to be non-empty.
00486     string valid_constraint_name = CreateValidConstraintName(*constraint, i);
00487     constraint_proto->set_id(valid_constraint_name);
00488     constraint_proto->set_lb(constraint->lb());
00489     constraint_proto->set_ub(constraint->ub());
00490     for (ConstIter<hash_map<MPVariable*, double> >
00491              it(constraint->coefficients_);
00492          !it.at_end(); ++it) {
00493       MPVariable* const var = it->first;
00494       const double coef = it->second;
00495       MPTermProto* const term = constraint_proto->add_terms();
00496       term->set_variable_id(valid_variable_names[var]);
00497       term->set_coefficient(coef);
00498     }
00499   }
00500 
00501   // Objective
00502   for (hash_map<MPVariable*, double>::const_iterator it =
00503            linear_objective_.coefficients_.begin();
00504        it != linear_objective_.coefficients_.end();
00505        ++it) {
00506     MPVariable* const var = it->first;
00507     const double coef = it->second;
00508     MPTermProto* const term = model->add_objective_terms();
00509     term->set_variable_id(valid_variable_names[var]);
00510     term->set_coefficient(coef);
00511   }
00512   model->set_maximize(Maximization());

00705 // Compute the size of the constraint with the largest number of
00706 // coefficients with index in [min_constraint_index,
00707 // max_constraint_index)
00708 int MPSolver::ComputeMaxConstraintSize(int min_constraint_index,
00709                                        int max_constraint_index) const {
00710   int max_constraint_size = 0;
00711   DCHECK_GE(min_constraint_index, 0);
00712   DCHECK_LE(max_constraint_index, constraints_.size());
00713   for (int i = min_constraint_index; i < max_constraint_index; ++i) {
00714     MPConstraint* const ct = constraints_[i];
00715     if (ct->coefficients_.size() > max_constraint_size) {
00716       max_constraint_size = ct->coefficients_.size();
00717     }
00718   }
00719   return max_constraint_size;
00720 }

00722 bool MPSolver::HasInfeasibleConstraints() const {
00723   bool hasInfeasibleConstraints = false;
00724   for (int i = 0; i < constraints_.size(); ++i) {
00725     if (constraints_[i]->lb() > constraints_[i]->ub()) {
00726       LOG(WARNING) << "Constraint " << constraints_[i]->name()
00727                    << " (" << i << ") has contradictory bounds:"
00728                    << " lower bound = " << constraints_[i]->lb()
00729                    << " upper bound = " << constraints_[i]->ub();
00730       hasInfeasibleConstraints = true;
00731     }
00732   }
00733   return hasInfeasibleConstraints;
00734 }

00788 // Extracts model stored in MPSolver
00789 void MPSolverInterface::ExtractModel() {
00790   switch (sync_status_) {
00791     case MUST_RELOAD: {
00792       ExtractNewVariables();
00793       ExtractNewConstraints();
00794       ExtractObjective();
00795 
00796       last_constraint_index_ = solver_->constraints_.size();
00797       last_variable_index_ = solver_->variables_.size();
00798       sync_status_ = MODEL_SYNCHRONIZED;
00799       break;
00800     }
00801     case MODEL_SYNCHRONIZED: {
00802       // Everything has already been extracted.
00803       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00804       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00805       break;
00806     }
00807     case SOLUTION_SYNCHRONIZED: {
00808       // Nothing has changed since last solve.
00809       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00810       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00811       break;
00812     }
00813   }
00814 }

00816 void MPSolverInterface::ResetExtractionInformation() {
00817   sync_status_ = MUST_RELOAD;
00818   last_constraint_index_ = 0;
00819   last_variable_index_ = 0;
00820   for (int j = 0; j < solver_->variables_.size(); ++j) {
00821     MPVariable* const var = solver_->variables_[j];
00822     var->set_index(kNoIndex);
00823   }
00824   for (int i = 0; i < solver_->constraints_.size(); ++i) {
00825     MPConstraint* const ct = solver_->constraints_[i];
00826     ct->set_index(kNoIndex);
00827   }
00828 }
