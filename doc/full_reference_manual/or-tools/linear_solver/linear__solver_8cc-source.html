00349   // Allow names that conform to the LP and MPS format.
00350   const int kMaxNameLength = 255;
00351   if (name.size() > kMaxNameLength) {
00352     LOG(WARNING) << "Invalid name " << name

00354                  << " Will be unable to write model to file.";
00355     return false;
00356   }
00357   if (name.find_first_of(" +-*<>=:\\") != string::npos) {
00358     LOG(WARNING) << "Invalid name " << name
00359                  << ": contains forbidden character: +-*<>=:\\ space."
00360                  << " Will be unable to write model to file.";
00361     return false;
00362   }
00363   size_t first_occurrence = name.find_first_of(".0123456789");
00364   if (first_occurrence != string::npos && first_occurrence == 0) {
00365     LOG(WARNING) << "Invalid name " << name
00366                  << ": first character should not be . or a number."
00367                  << " Will be unable to write model to file.";
00368     return false;
00369   }
00370   return true;

00388 // Loads model from protocol buffer.
00389 MPSolver::LoadStatus MPSolver::LoadModel(const MPModelProto& model) {
00390   hash_map<string, MPVariable*> variables;
00391   for (int i = 0; i < model.variables_size(); ++i) {
00392     const MPVariableProto& var_proto = model.variables(i);
00393     const string& id = var_proto.id();
00394     if (!ContainsKey(variables, id)) {
00395       MPVariable* variable = MakeNumVar(var_proto.lb(), var_proto.ub(), id);
00396       variable->SetInteger(var_proto.integer());
00397       variables[id] = variable;
00398     } else {
00399       return MPSolver::DUPLICATE_VARIABLE_ID;
00400     }
00401   }
00402   for (int i = 0; i < model.constraints_size(); ++i) {
00403     const MPConstraintProto& ct_proto = model.constraints(i);
00404     const string& ct_id = ct_proto.has_id() ? ct_proto.id() : "";
00405     MPConstraint* const ct = MakeRowConstraint(ct_proto.lb(),
00406                                                ct_proto.ub(),
00407                                                ct_id);
00408     for (int j = 0; j < ct_proto.terms_size(); ++j) {
00409       const MPTermProto& term_proto = ct_proto.terms(j);
00410       const string& id = term_proto.variable_id();
00411       MPVariable* variable = FindPtrOrNull(variables, id);
00412       if (variable != NULL) {
00413         ct->AddTerm(variable, term_proto.coefficient());
00414       } else {
00415         return MPSolver::UNKNOWN_VARIABLE_ID;
00416       }
00417     }
00418   }
00419   for (int i = 0; i < model.objective_terms_size(); ++i) {
00420     const MPTermProto& term_proto = model.objective_terms(i);
00421     const string& id = term_proto.variable_id();
00422     MPVariable* variable = FindPtrOrNull(variables, id);
00423     if (variable != NULL) {
00424       AddObjectiveTerm(variable, term_proto.coefficient());
00425     } else {
00426       return MPSolver::UNKNOWN_VARIABLE_ID;
00427     }
00428   }
00429   SetOptimizationDirection(model.maximize());
00430   if (model.has_objective_offset()) {
00431     linear_objective_.SetOffset(model.objective_offset());
00432   }
00433   return MPSolver::NO_ERROR;
00434 }

00436 // Exports model to protocol buffer.
00437 void MPSolver::ExportModel(MPModelProto* model) const {
00438   CHECK_NOTNULL(model);
00439   if (model->variables_size() > 0 ||
00440       model->has_maximize() ||
00441       model->objective_terms_size() > 0 ||
00442       model->constraints_size() > 0 ||
00443       model->has_name() ||
00444       model->has_objective_offset()) {
00445     LOG(WARNING) << "The model protocol buffer is not empty, "
00446                  << "it will be overwritten.";
00447     model->clear_variables();
00448     model->clear_maximize();
00449     model->clear_objective_terms();
00450     model->clear_constraints();
00451     model->clear_name();
00452   }
00453 
00454   // Variables need non-empty names for the model protocol buffer, so
00455   // we may need to create names.
00456   hash_map<MPVariable*, string> valid_variable_names;
00457 
00458   // Variables
00459   for (int j = 0; j < variables_.size(); ++j) {
00460     MPVariable* const var = variables_[j];
00461     MPVariableProto* const variable_proto = model->add_variables();
00462     valid_variable_names[var] = CreateValidVariableName(*var, j);
00463     variable_proto->set_id(valid_variable_names[var]);
00464     variable_proto->set_lb(var->lb());
00465     variable_proto->set_ub(var->ub());
00466     variable_proto->set_integer(var->integer());
00467   }
00468 
00469   // Constraints
00470   for (int i = 0; i < constraints_.size(); ++i) {
00471     MPConstraint* const constraint = constraints_[i];
00472     MPConstraintProto* const constraint_proto = model->add_constraints();
00473     // Constraint names need to be non-empty.
00474     string valid_constraint_name = CreateValidConstraintName(*constraint, i);
00475     constraint_proto->set_id(valid_constraint_name);
00476     constraint_proto->set_lb(constraint->lb());
00477     constraint_proto->set_ub(constraint->ub());
00478     for (ConstIter<hash_map<MPVariable*, double> >
00479              it(constraint->coefficients_);
00480          !it.at_end(); ++it) {
00481       MPVariable* const var = it->first;
00482       const double coef = it->second;
00483       MPTermProto* const term = constraint_proto->add_terms();
00484       term->set_variable_id(valid_variable_names[var]);
00485       term->set_coefficient(coef);
00486     }
00487   }
00488 
00489   // Objective
00490   for (hash_map<MPVariable*, double>::const_iterator it =
00491            linear_objective_.coefficients_.begin();
00492        it != linear_objective_.coefficients_.end();
00493        ++it) {
00494     MPVariable* const var = it->first;
00495     const double coef = it->second;
00496     MPTermProto* const term = model->add_objective_terms();
00497     term->set_variable_id(valid_variable_names[var]);
00498     term->set_coefficient(coef);
00499   }
00500   model->set_maximize(Maximization());
00501   model->set_objective_offset(linear_objective_.offset_);
00502 }

00693 // Compute the size of the constraint with the largest number of
00694 // coefficients with index in [min_constraint_index,
00695 // max_constraint_index)
00696 int MPSolver::ComputeMaxConstraintSize(int min_constraint_index,
00697                                        int max_constraint_index) const {
00698   int max_constraint_size = 0;
00699   DCHECK_GE(min_constraint_index, 0);
00700   DCHECK_LE(max_constraint_index, constraints_.size());
00701   for (int i = min_constraint_index; i < max_constraint_index; ++i) {
00702     MPConstraint* const ct = constraints_[i];
00703     if (ct->coefficients_.size() > max_constraint_size) {
00704       max_constraint_size = ct->coefficients_.size();
00705     }
00706   }
00707   return max_constraint_size;
00708 }

00710 bool MPSolver::HasInfeasibleConstraints() const {
00711   bool hasInfeasibleConstraints = false;
00712   for (int i = 0; i < constraints_.size(); ++i) {
00713     if (constraints_[i]->lb() > constraints_[i]->ub()) {
00714       LOG(WARNING) << "Constraint " << constraints_[i]->name()
00715                    << " (" << i << ") has contradictory bounds:"
00716                    << " lower bound = " << constraints_[i]->lb()
00717                    << " upper bound = " << constraints_[i]->ub();
00718       hasInfeasibleConstraints = true;
00719     }
00720   }
00721   return hasInfeasibleConstraints;
00722 }

00776 // Extracts model stored in MPSolver
00777 void MPSolverInterface::ExtractModel() {
00778   switch (sync_status_) {
00779     case MUST_RELOAD: {
00780       ExtractNewVariables();
00781       ExtractNewConstraints();
00782       ExtractObjective();
00783 
00784       last_constraint_index_ = solver_->constraints_.size();
00785       last_variable_index_ = solver_->variables_.size();
00786       sync_status_ = MODEL_SYNCHRONIZED;
00787       break;
00788     }
00789     case MODEL_SYNCHRONIZED: {
00790       // Everything has already been extracted.
00791       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00792       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00793       break;
00794     }
00795     case SOLUTION_SYNCHRONIZED: {
00796       // Nothing has changed since last solve.
00797       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00798       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00799       break;
00800     }
00801   }
00802 }

00804 void MPSolverInterface::ResetExtractionInformation() {
00805   sync_status_ = MUST_RELOAD;
00806   last_constraint_index_ = 0;
00807   last_variable_index_ = 0;
00808   for (int j = 0; j < solver_->variables_.size(); ++j) {
00809     MPVariable* const var = solver_->variables_[j];
00810     var->set_index(kNoIndex);
00811   }
00812   for (int i = 0; i < solver_->constraints_.size(); ++i) {
00813     MPConstraint* const ct = solver_->constraints_[i];
00814     ct->set_index(kNoIndex);
00815   }
00816 }
