00372 bool MPSolver::CheckNameValidity(const string& name) {
00373   // Allow names that conform to the LP and MPS format.
00374   const int kMaxNameLength = 255;
00375   if (name.size() > kMaxNameLength) {
00376     LOG(WARNING) << "Invalid name " << name
00377                  << ": length > " << kMaxNameLength << "."
00378                  << " Will be unable to write model to file.";
00379     return false;
00380   }
00381   if (name.find_first_of(" +-*<>=:\\") != string::npos) {
00382     LOG(WARNING) << "Invalid name " << name
00383                  << ": contains forbidden character: +-*<>=:\\ space."
00384                  << " Will be unable to write model to file.";
00385     return false;
00386   }
00387   size_t first_occurrence = name.find_first_of(".0123456789");
00388   if (first_occurrence != string::npos && first_occurrence == 0) {
00389     LOG(WARNING) << "Invalid name " << name
00390                  << ": first character should not be . or a number."
00391                  << " Will be unable to write model to file.";
00392     return false;
00393   }
00394   return true;

00413 MPSolver::LoadStatus MPSolver::LoadModel(const MPModelProto& model) {
00414   hash_map<string, MPVariable*> variables;
00415   for (int i = 0; i < model.variables_size(); ++i) {
00416     const MPVariableProto& var_proto = model.variables(i);
00417     const string& id = var_proto.id();
00418     if (!ContainsKey(variables, id)) {
00419       MPVariable* variable = MakeNumVar(var_proto.lb(), var_proto.ub(), id);
00420       variable->SetInteger(var_proto.integer());
00421       variables[id] = variable;
00422     } else {
00423       return MPSolver::DUPLICATE_VARIABLE_ID;
00424     }
00425   }
00426   for (int i = 0; i < model.constraints_size(); ++i) {
00427     const MPConstraintProto& ct_proto = model.constraints(i);
00428     const string& ct_id = ct_proto.has_id() ? ct_proto.id() : "";
00429     MPConstraint* const ct = MakeRowConstraint(ct_proto.lb(),
00430                                                ct_proto.ub(),
00431                                                ct_id);
00432     for (int j = 0; j < ct_proto.terms_size(); ++j) {
00433       const MPTermProto& term_proto = ct_proto.terms(j);
00434       const string& id = term_proto.variable_id();
00435       MPVariable* variable = FindPtrOrNull(variables, id);
00436       if (variable != NULL) {
00437         ct->AddTerm(variable, term_proto.coefficient());
00438       } else {
00439         return MPSolver::UNKNOWN_VARIABLE_ID;
00440       }
00441     }
00442   }
00443   for (int i = 0; i < model.objective_terms_size(); ++i) {
00444     const MPTermProto& term_proto = model.objective_terms(i);
00445     const string& id = term_proto.variable_id();
00446     MPVariable* variable = FindPtrOrNull(variables, id);
00447     if (variable != NULL) {
00448       AddObjectiveTerm(variable, term_proto.coefficient());
00449     } else {
00450       return MPSolver::UNKNOWN_VARIABLE_ID;
00451     }
00452   }
00453   SetOptimizationDirection(model.maximize());
00454   if (model.has_objective_offset()) {
00455     linear_objective_->SetOffset(model.objective_offset());
00456   }
00457   return MPSolver::NO_ERROR;
00458 }

00460 void MPSolver::ExportModel(MPModelProto* model) const {
00461   CHECK_NOTNULL(model);
00462   if (model->variables_size() > 0 ||
00463       model->has_maximize() ||
00464       model->objective_terms_size() > 0 ||
00465       model->constraints_size() > 0 ||
00466       model->has_name() ||
00467       model->has_objective_offset()) {
00468     LOG(WARNING) << "The model protocol buffer is not empty, "
00469                  << "it will be overwritten.";
00470     model->clear_variables();
00471     model->clear_maximize();
00472     model->clear_objective_terms();
00473     model->clear_constraints();
00474     model->clear_name();
00475   }
00476 
00477   // Variables need non-empty names for the model protocol buffer, so
00478   // we may need to create names.
00479   hash_map<MPVariable*, string> valid_variable_names;
00480 
00481   // Variables
00482   for (int j = 0; j < variables_.size(); ++j) {
00483     MPVariable* const var = variables_[j];
00484     MPVariableProto* const variable_proto = model->add_variables();
00485     valid_variable_names[var] = CreateValidVariableName(*var, j);
00486     variable_proto->set_id(valid_variable_names[var]);
00487     variable_proto->set_lb(var->lb());
00488     variable_proto->set_ub(var->ub());
00489     variable_proto->set_integer(var->integer());
00490   }
00491 
00492   // Constraints
00493   for (int i = 0; i < constraints_.size(); ++i) {
00494     MPConstraint* const constraint = constraints_[i];
00495     MPConstraintProto* const constraint_proto = model->add_constraints();
00496     // Constraint names need to be non-empty.
00497     string valid_constraint_name = CreateValidConstraintName(*constraint, i);
00498     constraint_proto->set_id(valid_constraint_name);
00499     constraint_proto->set_lb(constraint->lb());
00500     constraint_proto->set_ub(constraint->ub());
00501     for (ConstIter<hash_map<MPVariable*, double> >
00502              it(constraint->coefficients_);
00503          !it.at_end(); ++it) {
00504       MPVariable* const var = it->first;
00505       const double coef = it->second;
00506       MPTermProto* const term = constraint_proto->add_terms();
00507       term->set_variable_id(valid_variable_names[var]);
00508       term->set_coefficient(coef);
00509     }
00510   }
00511 
00512   // Objective
00513   for (hash_map<MPVariable*, double>::const_iterator it =
00514            linear_objective_->coefficients_.begin();
00515        it != linear_objective_->coefficients_.end();
00516        ++it) {
00517     MPVariable* const var = it->first;
00518     const double coef = it->second;
00519     MPTermProto* const term = model->add_objective_terms();
00520     term->set_variable_id(valid_variable_names[var]);
00521     term->set_coefficient(coef);
00522   }
00523   model->set_maximize(Maximization());
00524   model->set_objective_offset(linear_objective_->offset_);

00555       break;

00713 int MPSolver::ComputeMaxConstraintSize(int min_constraint_index,
00714                                        int max_constraint_index) const {
00715   int max_constraint_size = 0;
00716   DCHECK_GE(min_constraint_index, 0);
00717   DCHECK_LE(max_constraint_index, constraints_.size());
00718   for (int i = min_constraint_index; i < max_constraint_index; ++i) {
00719     MPConstraint* const ct = constraints_[i];
00720     if (ct->coefficients_.size() > max_constraint_size) {
00721       max_constraint_size = ct->coefficients_.size();
00722     }
00723   }
00724   return max_constraint_size;
00725 }

00727 bool MPSolver::HasInfeasibleConstraints() const {
00728   bool hasInfeasibleConstraints = false;
00729   for (int i = 0; i < constraints_.size(); ++i) {
00730     if (constraints_[i]->lb() > constraints_[i]->ub()) {
00731       LOG(WARNING) << "Constraint " << constraints_[i]->name()
00732                    << " (" << i << ") has contradictory bounds:"
00733                    << " lower bound = " << constraints_[i]->lb()
00734                    << " upper bound = " << constraints_[i]->ub();
00735       hasInfeasibleConstraints = true;
00736     }
00737   }
00738   return hasInfeasibleConstraints;
00739 }

00797 void MPSolverInterface::ExtractModel() {
00798   switch (sync_status_) {
00799     case MUST_RELOAD: {
00800       ExtractNewVariables();
00801       ExtractNewConstraints();
00802       ExtractObjective();
00803 
00804       last_constraint_index_ = solver_->constraints_.size();
00805       last_variable_index_ = solver_->variables_.size();
00806       sync_status_ = MODEL_SYNCHRONIZED;
00807       break;
00808     }
00809     case MODEL_SYNCHRONIZED: {
00810       // Everything has already been extracted.
00811       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00812       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00813       break;
00814     }
00815     case SOLUTION_SYNCHRONIZED: {
00816       // Nothing has changed since last solve.
00817       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00818       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00819       break;
00820     }
00821   }
00822 }

00824 void MPSolverInterface::ResetExtractionInformation() {
00825   sync_status_ = MUST_RELOAD;
00826   last_constraint_index_ = 0;
00827   last_variable_index_ = 0;
00828   for (int j = 0; j < solver_->variables_.size(); ++j) {
00829     MPVariable* const var = solver_->variables_[j];
00830     var->set_index(kNoIndex);
00831   }
00832   for (int i = 0; i < solver_->constraints_.size(); ++i) {
00833     MPConstraint* const ct = solver_->constraints_[i];
00834     ct->set_index(kNoIndex);
00835   }
00836 }
