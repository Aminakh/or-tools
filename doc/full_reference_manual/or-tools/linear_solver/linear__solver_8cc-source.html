00323 bool MPSolver::CheckNameValidity(const string& name) {
00324   // Allow names that conform to the LP and MPS format.
00325   const int kMaxNameLength = 255;
00326   if (name.size() > kMaxNameLength) {
00327     LOG(WARNING) << "Invalid name " << name
00328                  << ": length > " << kMaxNameLength << "."
00329                  << " Will be unable to write model to file.";
00330     return false;
00331   }
00332   if (name.find_first_of(" +-*<>=:\\") != string::npos) {
00333     LOG(WARNING) << "Invalid name " << name
00334                  << ": contains forbidden character: +-*<>=:\\ space."
00335                  << " Will be unable to write model to file.";
00336     return false;
00337   }
00338   size_t first_occurrence = name.find_first_of(".0123456789");
00339   if (first_occurrence != string::npos && first_occurrence == 0) {
00340     LOG(WARNING) << "Invalid name " << name
00341                  << ": first character should not be . or a number."
00342                  << " Will be unable to write model to file.";
00343     return false;
00344   }
00345   return true;

00362 // ----- Load from protobuf -----
00363 MPSolver::LoadStatus MPSolver::Load(const MPModelProto& model) {
00364   hash_map<string, MPVariable*> variables;
00365   for (int i = 0; i < model.variables_size(); ++i) {
00366     const MPVariableProto& var_proto = model.variables(i);
00367     const string& id = var_proto.id();
00368     if (!ContainsKey(variables, id)) {
00369       MPVariable* variable = MakeNumVar(var_proto.lb(), var_proto.ub(), id);
00370       variable->SetInteger(var_proto.integer());
00371       variables[id] = variable;
00372     } else {
00373       return MPSolver::DUPLICATE_VARIABLE_ID;
00374     }
00375   }
00376   for (int i = 0; i < model.constraints_size(); ++i) {
00377     const MPConstraintProto& ct_proto = model.constraints(i);
00378     const string& ct_id = ct_proto.has_id() ? ct_proto.id() : "";
00379     MPConstraint* const ct = MakeRowConstraint(ct_proto.lb(),
00380                                                ct_proto.ub(),
00381                                                ct_id);
00382     for (int j = 0; j < ct_proto.terms_size(); ++j) {
00383       const MPTermProto& term_proto = ct_proto.terms(j);
00384       const string& id = term_proto.variable_id();
00385       MPVariable* variable = FindPtrOrNull(variables, id);
00386       if (variable != NULL) {
00387         ct->AddTerm(variable, term_proto.coefficient());
00388       } else {
00389         return MPSolver::UNKNOWN_VARIABLE_ID;
00390       }
00391     }
00392   }
00393   for (int i = 0; i < model.objective_terms_size(); ++i) {
00394     const MPTermProto& term_proto = model.objective_terms(i);
00395     const string& id = term_proto.variable_id();
00396     MPVariable* variable = FindPtrOrNull(variables, id);
00397     if (variable != NULL) {
00398       AddObjectiveTerm(variable, term_proto.coefficient());
00399     } else {
00400       return MPSolver::UNKNOWN_VARIABLE_ID;
00401     }
00402   }
00403   SetOptimizationDirection(model.maximize());
00404   return MPSolver::NO_ERROR;
00405 }
