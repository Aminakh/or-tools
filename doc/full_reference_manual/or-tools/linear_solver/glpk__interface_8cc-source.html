00067 void GLPKGatherInformationCallback(glp_tree* tree, void* info) {
00068   CHECK_NOTNULL(tree);
00069   CHECK_NOTNULL(info);
00070   GLPKInformation* glpk_info = reinterpret_cast<GLPKInformation*>(info);
00071   switch (glp_ios_reason(tree)) {
00072     // The best bound and the number of nodes change only when GLPK
00073     // branches, generates cuts or finds an integer solution.
00074     case GLP_ISELECT:
00075     case GLP_IROWGEN:
00076     case GLP_IBINGO: {
00077       // Get total number of nodes
00078       glp_ios_tree_size(tree, NULL, NULL, &glpk_info->num_all_nodes_);
00079       // Get best bound
00080       int node_id = glp_ios_best_node(tree);
00081       if (node_id > 0) {
00082         glpk_info->best_objective_bound_ = glp_ios_node_bound(tree, node_id);

00595       // GLP_UNDEF. So this is never (?) reached.  Return infeasible
00596       // in case GLPK returns a correct status in future versions.
00597       result_status_ = MPSolver::INFEASIBLE;
00598     } else {
00599       result_status_ = MPSolver::ABNORMAL;
00600       // GLPK does not have a status code for unbounded MIP models, so
00601       // we return an abnormal status instead.
00602     }
00603   } else {
00604     int tmp_status = glp_get_status(lp_);
00605     VLOG(1) << "gplk result status: " << tmp_status;
00606     if (tmp_status == GLP_OPT) {
00607       result_status_ = MPSolver::OPTIMAL;
00608     } else if (tmp_status == GLP_FEAS) {
00609       result_status_ = MPSolver::FEASIBLE;
00610     } else if (tmp_status == GLP_NOFEAS ||
00611                tmp_status == GLP_INFEAS) {
00612       // For infeasible problems, GLPK actually seems to return
00613       // GLP_UNDEF. So this is never (?) reached.  Return infeasible
00614       // in case GLPK returns a correct status in future versions.
00615       result_status_ = MPSolver::INFEASIBLE;
00616     } else if (tmp_status == GLP_UNBND) {
00617       // For unbounded problems, GLPK actually seems to return
00618       // GLP_UNDEF. So this is never (?) reached.  Return unbounded
00619       // in case GLPK returns a correct status in future versions.
00620       result_status_ = MPSolver::UNBOUNDED;
00621     } else {
00622       result_status_ = MPSolver::ABNORMAL;
00623     }
00624   }
00625 
00626   sync_status_ = SOLUTION_SYNCHRONIZED;
00627 
00628   return result_status_;
00629 }

00696 void GLPKInterface::ConfigureGLPKParameters(const MPSolverParameters& param) {
00697   if (mip_) {
00698     glp_init_iocp(&mip_param_);
00699     // Time limit
00700     if (solver_->time_limit()) {
00701       VLOG(1) << "Setting time limit = " << solver_->time_limit() << " ms.";
00702       mip_param_.tm_lim = solver_->time_limit();
00703     }
00704     // Initialize structures related to the callback.
00705     mip_param_.cb_func = GLPKGatherInformationCallback;
00706     mip_callback_info_->Reset(maximize_);
00707     mip_param_.cb_info = mip_callback_info_.get();
00708     // TODO(user): switch some cuts on? All cuts are off by default!?
00709   }
00710 
00711   // Configure LP parameters in all cases since they will be used to
00712   // solve the root LP in the MIP case.
00713   glp_init_smcp(&lp_param_);
00714   // Time limit
00715   if (solver_->time_limit()) {
00716     VLOG(1) << "Setting time limit = " << solver_->time_limit() << " ms.";
00717     lp_param_.tm_lim = solver_->time_limit();
00718   }
00719 
00720   // Should give a numerically better representation of the problem.
00721   glp_scale_prob(lp_, GLP_SF_AUTO);
00722 
00723   // Use advanced initial basis (options: standard / advanced / Bixby's).
00724   glp_adv_basis(lp_, NULL);
00725 
00726   // Set parameters specified by the user.
00727   SetParameters(param);
00728 }
