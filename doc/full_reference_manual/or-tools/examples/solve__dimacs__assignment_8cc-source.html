00039 CostValue BuildAndSolveHungarianInstance(
00040     const LinearSumAssignment& assignment) {
00041   const StarGraph& graph = assignment.Graph();
00042   typedef std::vector<double> HungarianRow;
00043   typedef std::vector<HungarianRow> HungarianProblem;
00044   HungarianProblem hungarian_cost;
00045   hungarian_cost.resize(assignment.NumLeftNodes());
00046   // First we have to find the biggest cost magnitude so we can
00047   // initialize the arc costs that aren't really there.
00048   CostValue largest_cost_magnitude = 0;
00049   for (StarGraph::ArcIterator arc_it(graph); arc_it.Ok(); arc_it.Next()) {
00050     ArcIndex arc = arc_it.Index();
00051     CostValue cost_magnitude = ::std::abs(assignment.ArcCost(arc));
00052     largest_cost_magnitude = ::std::max(largest_cost_magnitude, cost_magnitude);
00053   }
00054   double missing_arc_cost = static_cast<double>((assignment.NumLeftNodes() *
00055                                                  largest_cost_magnitude) +
00056                                                 1);
00057   for (HungarianProblem::iterator row = hungarian_cost.begin();
00058        row != hungarian_cost.end();
00059        ++row) {
00060     row->resize(assignment.NumNodes() - assignment.NumLeftNodes(),
00061                 missing_arc_cost);
00062   }
00063   for (StarGraph::ArcIterator arc_it(graph);
00064        arc_it.Ok();
00065        arc_it.Next()) {
00066     ArcIndex arc = arc_it.Index();
00067     NodeIndex tail = graph.Tail(arc) - StarGraph::kFirstNode;
00068     NodeIndex head = (graph.Head(arc) - assignment.NumLeftNodes() -
00069                       StarGraph::kFirstNode);
00070     double cost = static_cast<double>(assignment.ArcCost(arc));
00071     hungarian_cost[tail][head] = cost;
00072   }
00073   hash_map<int, int> result;
00074   hash_map<int, int> wish_this_could_be_null;
00075   WallTimer timer;
00076   VLOG(1) << "Beginning Hungarian method.";
00077   timer.Start();
00078   MinimizeLinearAssignment(hungarian_cost, &result, &wish_this_could_be_null);
00079   double elapsed = timer.GetInMs() / 1000.0;
00080   LOG(INFO) << "Hungarian result computed in " << elapsed << " seconds.";
00081   double result_cost = 0.0;
00082   for (int i = 0; i < assignment.NumLeftNodes(); ++i) {
00083     int mate = result[i];
00084     result_cost += hungarian_cost[i][mate];
00085   }
00086   return static_cast<CostValue>(result_cost);
00087 }

00089 void DisplayAssignment(const LinearSumAssignment& assignment) {
00090   for (LinearSumAssignment::BipartiteLeftNodeIterator node_it(assignment);
00091        node_it.Ok();
00092        node_it.Next()) {
00093     const NodeIndex left_node = node_it.Index();
00094     const ArcIndex matching_arc = assignment.GetAssignmentArc(left_node);
00095     const NodeIndex right_node = assignment.Head(matching_arc);
00096     VLOG(5) << "assigned (" << left_node << ", " << right_node << "): "
00097             << assignment.ArcCost(matching_arc);
00098   }
00099 }

00103 int solve_dimacs_assignment(int argc, char* argv[]) {
00104   string usage;
00105   if (argc < 1) {
00106     usage = StringPrintf(kUsageTemplate, "solve_dimacs_assignment");
00107   } else {
00108     usage = StringPrintf(kUsageTemplate, argv[0]);
00109   }
00110   google::SetUsageMessage(usage);
00111   google::ParseCommandLineFlags(&argc, &argv, true);
00112 
00113   if (argc < 2) {
00114     LOG(FATAL) << usage;
00115   }
00116   string error_message;
00117   LinearSumAssignment* assignment =
00118       ParseDimacsAssignment(argv[1], &error_message);
00119   if (assignment == NULL) {
00120     LOG(FATAL) << error_message;
00121   }
00122   if (!FLAGS_assignment_problem_output_file.empty()) {
00123     PrintDimacsAssignmentProblem(*assignment,
00124                                  FLAGS_assignment_problem_output_file);
00125   }
00126   CostValue hungarian_cost = 0.0;
00127   bool hungarian_solved = false;
00128   if (FLAGS_assignment_compare_hungarian) {
00129     hungarian_cost = BuildAndSolveHungarianInstance(*assignment);
00130     hungarian_solved = true;
00131   }
00132   WallTimer timer;
00133   timer.Start();
00134   bool success = assignment->ComputeAssignment();
00135   double elapsed = timer.GetInMs() / 1000.0;
00136   if (success) {
00137     CostValue cost = assignment->GetCost();
00138     DisplayAssignment(*assignment);
00139     LOG(INFO) << "Cost of optimum assignment: " << cost;
00140     LOG(INFO) << "Computed in " << elapsed << " seconds.";
00141     LOG(INFO) << assignment->StatsString();
00142     if (hungarian_solved && (cost != hungarian_cost)) {
00143       LOG(ERROR) << "Optimum cost mismatch: " << cost << " vs. "
00144                  << hungarian_cost << ".";
00145     }
00146   } else {
00147     LOG(WARNING) << "Given problem is infeasible.";
00148   }
00149   delete assignment;
00150   return 0;
00151 }
