00071     int possible = 0;
00072     int sure = 0;

00075       if (vars1_[i]->Min() == 1 && vars2_[i]->Min() == 1) {

00078       if (vars1_[i]->Max() == 1 && vars2_[i]->Max() == 1) {

00097           if (vars1_[i]->Max() == 1 && vars2_[i]->Max() == 1) {

00610   virtual bool Accept(const Assignment* delta, const Assignment* deltadelta) {
00611     const Assignment::IntContainer& solution_delta = delta->IntVarContainer();
00612     const int solution_delta_size = solution_delta.Size();
00613 
00614     // First we check we are not using LNS.
00615     for (int i = 0; i < solution_delta_size; ++i) {
00616       const IntVarElement& element = solution_delta.Element(i);
00617       if (!element.Activated()) {
00618         // Doodle filters are not robust to LNS, so skip filters when
00619         // some elements are not activated.
00620         VLOG(1) << "LNS";
00621         return true;
00622       }
00623     }
00624     VLOG(1) << "No LNS, size = " << solution_delta_size;
00625     backtrack_.clear();
00626     hash_set<int> touched_cards;
00627     for (int index = 0; index < solution_delta_size; ++index) {
00628       int64 touched_var = -1;
00629       FindIndex(solution_delta.Element(index).Var(), &touched_var);
00630       const int card = touched_var / num_symbols_;
00631       const int symbol = touched_var % num_symbols_;
00632       const int new_value = solution_delta.Element(index).Value();
00633       if (!ContainsKey(touched_cards, card)) {
00634         Save(card);
00635         touched_cards.insert(card);
00636       }
00637       if (new_value) {
00638         SetBit64(&cards_[card], symbol);
00639       } else {
00640         ClearBit64(&cards_[card], symbol);
00641       }
00642     }
00643 
00644     if (!CheckCards()) {
00645       Backtrack();
00646       DCHECK(CheckCards());
00647       VLOG(1) << "reject by size";
00648       return false;
00649     }
00650 
00651     hash_set<int> treated_cards;
00652     int cost_delta = 0;
00653     for (ConstIter<hash_set<int> > card_it(touched_cards);
00654          !card_it.at_end();
00655          ++card_it) {
00656       treated_cards.insert(*card_it);
00657       const uint64 bitset = cards_[*card_it];
00658       const vector<int>& row_cost = costs_[*card_it];
00659       for (int card = 0; card < num_cards_; ++card) {
00660         if (!ContainsKey(treated_cards, card)) {
00661           cost_delta +=
00662               Cost(BitCount64(bitset & cards_[card])) - row_cost[card];
00663         }
00664       }
00665     }
00666     Backtrack();
00667     if (cost_delta >= 0) {
00668       VLOG(1) << "reject";
00669     }
00670     return cost_delta < 0;
00671   }
00672  private:
00673   struct Undo {
00674     Undo(int c, uint64 b) : card(c), bitset(b) {}
00675     int card;
00676     uint64 bitset;
00677   };
00678 
00679   int Cost(int intersection_size) {
00680     return abs(intersection_size - 1);
00681   }
00682 
00683   void Backtrack() {
00684     for (int i = 0; i < backtrack_.size(); ++i) {
00685       cards_[backtrack_[i].card] = backtrack_[i].bitset;
00686     }
00687   }
00688 
00689   void Save(int card) {
00690     backtrack_.push_back(Undo(card, cards_[card]));
00691   }
00692 
00693   bool CheckCards() {
00694     for (int i = 0; i < num_cards_; ++i) {
00695       if (num_symbols_per_card_ != BitCount64(cards_[i])) {
00696         VLOG(1) << "card " << i << " has bitset of size "
00697                   << BitCount64(cards_[i]);
00698         return false;
00699       }
00700     }
00701     return true;
00702   }
00703 
00704   const int num_cards_;
00705   const int num_symbols_;
00706   const int num_symbols_per_card_;
00707   vector<uint64> cards_;
00708   vector<vector<int> > costs_;
00709   vector<Undo> backtrack_;
00710 };

00712 void SolveDobble(int num_cards, int num_symbols, int num_symbols_per_card) {
00713   LOG(INFO) << "Solving dobble assignment problem:";
00714   LOG(INFO) << "  - " << num_cards << " cards";
00715   LOG(INFO) << "  - " << num_symbols << " symbols";
00716   LOG(INFO) << "  - " << num_symbols_per_card << " symbols per card";
00717 
00718   Solver solver("dobble");
00719   vector<vector<IntVar*> > vars(num_cards);
00720   vector<IntVar*> all_vars;
00721   for (int card_index = 0; card_index < num_cards; ++card_index) {
00722     solver.MakeBoolVarArray(num_symbols,
00723                             StringPrintf("card_%i_", card_index),
00724                             &vars[card_index]);
00725     for (int symbol_index = 0;
00726          symbol_index < num_symbols;
00727          ++symbol_index) {
00728       all_vars.push_back(vars[card_index][symbol_index]);
00729     }
00730   }
00731   vector<IntVar*> slack_vars;
00732   for (int card1 = 0; card1 < num_cards; ++card1) {
00733     for (int card2 = 0; card2 < num_cards; ++card2) {
00734       if (card1 != card2) {
00735         slack_vars.push_back(AddIntersectionVar(&solver,
00736                                                 vars[card1],
00737                                                 vars[card2],
00738                                                 num_symbols_per_card));
00739       }
00740     }
00741   }
00742 
00743   for (int card = 0; card < num_cards; ++card) {
00744     solver.AddConstraint(solver.MakeSumEquality(vars[card],
00745                                                 num_symbols_per_card));
00746   }
00747 
00748   for (int symbol_index = 0; symbol_index < num_symbols; ++symbol_index) {
00749     vector<IntVar*> tmp;
00750     for (int card_index = 0; card_index < num_cards; ++card_index) {
00751       tmp.push_back(vars[card_index][symbol_index]);
00752      }
00753     solver.AddConstraint(solver.MakeSumEquality(tmp, num_symbols_per_card));
00754   }
00755 
00756   LOG(INFO) << "Solving with LNS";
00757   LOG(INFO) << "  - lns_size = " << FLAGS_lns_size;
00758   LOG(INFO) << "  - lns_limit = " << FLAGS_lns_limit;
00759   LOG(INFO) << "  - fail_limit = " << FLAGS_fail_limit;
00760 
00761   DecisionBuilder* const build_db = solver.MakePhase(all_vars,
00762                                                      Solver::CHOOSE_RANDOM,
00763                                                      Solver::ASSIGN_MAX_VALUE);
00764 
00765   const int kNhoodLimit = 1000;
00766   LocalSearchOperator* const switch_operator =
00767       solver.RevAlloc(new SwitchSymbols(all_vars.data(),
00768                                         all_vars.size(),
00769                                         num_cards,
00770                                         num_symbols,
00771                                         num_symbols_per_card));
00772   LocalSearchOperator* const cycle_operator =
00773       solver.RevAlloc(new CycleSymbols(all_vars.data(),
00774                                        all_vars.size(),
00775                                        num_cards,
00776                                        num_symbols,
00777                                        num_symbols_per_card));
00778   LocalSearchOperator* const long_cycle_operator_limited =
00779       solver.MakeNeighborhoodLimit(
00780           solver.RevAlloc(new CycleNeighborhood(all_vars.data(),
00781                                                 all_vars.size(),
00782                                                 FLAGS_lns_size,
00783                                                 num_cards,
00784                                                 num_symbols,
00785                                                 num_symbols_per_card)),
00786           kNhoodLimit);
00787   LocalSearchOperator* const long_cycle_operator_unlimited =
00788       solver.RevAlloc(new CycleNeighborhood(all_vars.data(),
00789                                             all_vars.size(),
00790                                             FLAGS_lns_size,
00791                                             num_cards,
00792                                             num_symbols,
00793                                             num_symbols_per_card));
00794   // LocalSearchOperator* const cross_lns_operator_limited =
00795   //     solver.MakeNeighborhoodLimit(
00796   //         solver.RevAlloc(new CrossLns(all_vars.data(),
00797   //                                      all_vars.size(),
00798   //                                      FLAGS_lns_size,
00799   //                                      num_cards,
00800   //                                      num_symbols,
00801   //                                      num_symbols_per_card)),
00802   //         kNhoodLimit);
00803   LocalSearchOperator* const card_lns_operator_limited =
00804       solver.MakeNeighborhoodLimit(
00805           solver.RevAlloc(new CardLns(all_vars.data(),
00806                                       all_vars.size(),
00807                                       FLAGS_lns_size,
00808                                       num_cards,
00809                                       num_symbols)),
00810           kNhoodLimit);
00811   LocalSearchOperator* const card_lns_operator_unlimited =
00812       solver.RevAlloc(new CardLns(all_vars.data(),
00813                                   all_vars.size(),
00814                                   FLAGS_lns_size,
00815                                   num_cards,
00816                                   num_symbols));
00817 
00818   vector<LocalSearchOperator*> operators;
00819   operators.push_back(switch_operator);
00820   operators.push_back(card_lns_operator_limited);
00821   operators.push_back(long_cycle_operator_limited);
00822   operators.push_back(cycle_operator);
00823   //  operators.push_back(card_lns_operator_unlimited);
00824   operators.push_back(long_cycle_operator_unlimited);
00825   LocalSearchOperator* const concat =
00826       solver.ConcatenateOperators(operators, true);
00827   SearchLimit* const lns_limit =
00828       solver.MakeLimit(kint64max, kint64max, FLAGS_lns_limit, kint64max);
00829   DecisionBuilder* const ls_db = solver.MakeSolveOnce(build_db, lns_limit);
00830   vector<LocalSearchFilter*> filters;
00831   filters.push_back(solver.RevAlloc(new DobbleFilter(all_vars.data(),
00832                                                      all_vars.size(),
00833                                                      num_cards,
00834                                                      num_symbols,
00835                                                      num_symbols_per_card)));
00836   LocalSearchPhaseParameters* const parameters =
00837       solver.MakeLocalSearchPhaseParameters(concat, ls_db, NULL, filters);
00838   DecisionBuilder* const final_db =
00839       solver.MakeLocalSearchPhase(all_vars, build_db, parameters);
00840 
00841   IntVar* const objective_var = solver.MakeSum(slack_vars)->Var();
00842 
00843   vector<SearchMonitor*> monitors;
00844   OptimizeVar* const optimize = solver.MakeMinimize(objective_var, 1);
00845   monitors.push_back(optimize);
00846   SearchMonitor* const log = solver.MakeSearchLog(100000, optimize);
00847   monitors.push_back(log);
00848   SearchLimit* const fail_limit =
00849       solver.MakeLimit(kint64max, kint64max, FLAGS_fail_limit, kint64max);
00850   monitors.push_back(fail_limit);
00851   solver.Solve(final_db, monitors);
00852 }
