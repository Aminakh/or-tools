00017 void BuildMixedIntegerProgrammingBoundedExample(
00018     MPSolver::OptimizationProblemType type) {
00019   MPSolver solver("Mixed Integer_Example", type);
00020   const double infinity = solver.infinity();
00021   MPVariable* const x1 = solver.MakeIntVar(0.0, infinity, "x1");
00022   MPVariable* const x2 = solver.MakeIntVar(0.0, infinity, "x2");
00023 
00024   // minimize x1 + 2 * x2
00025   solver.AddObjectiveTerm(x1, 1);
00026   solver.AddObjectiveTerm(x2, 2);
00027 
00028   // 2 * x2 + 3 * x1 >= 17
00029   MPConstraint* const c0 = solver.MakeRowConstraint(17, infinity);
00030   c0->AddTerm(x1, 3);
00031   c0->AddTerm(x2, 2);
00032 
00033   CHECK_EQ(MPSolver::OPTIMAL, solver.Solve());
00034   LOG(INFO) << "objective = " << solver.objective_value();
00035 }
