00014 // This model implements a simple jobshop problem.
00015 //
00016 // A jobshop is a standard scheduling problem where you must schedule a
00017 // set of jobs on a set of machines.  Each job is a sequence of tasks
00018 // (a task can only start when the preceding task finished), each of
00019 // which occupies a single specific machine during a specific
00020 // duration. Therefore, a job is simply given by a sequence of pairs
00021 // (machine id, duration).
00022 
00023 // The objective is to minimize the 'makespan', which is the duration
00024 // between the start of the first task (across all machines) and the
00025 // completion of the last task (across all machines).
00026 //
00027 // This will be modelled by sets of intervals variables (see class
00028 // IntervalVar in constraint_solver/constraint_solver.h), one per
00029 // task, representing the [start_time, end_time] of the task.  Tasks
00030 // in the same job will be linked by precedence constraints.  Tasks on
00031 // the same machine will be covered by Sequence constraints.
00032 //
00033 // Search will then be applied on the sequence constraints.

00140     static const char kWordDelimiters[] = " ";
00141     std::vector<string> words;

00143     switch (problem_type_) {
00144       case UNDEFINED: {
00145         if (words.size() == 2 && words[0] == "instance") {
00146           problem_type_ = JSSP;
00147           LOG(INFO) << "Reading jssp instance " << words[1];

00151           taillard_state_ = JOBS_READ;
00152           job_count_ = atoi32(words[0]);
00153           CHECK_GT(job_count_, 0);

00157       }

00170           CHECK_EQ(words.size(), machine_count_ * 2);
00171           for (int i = 0; i < machine_count_; ++i) {

00244 void Jobshop(const JobShopData& data) {
00245   Solver solver("jobshop");
00246   const int machine_count = data.machine_count();
00247   const int job_count = data.job_count();
00248   const int horizon = data.horizon();
00249 
00250   // ----- Creates all Intervals and vars -----
00251 
00252   // Stores all tasks attached interval variables per job.
00253   std::vector<std::vector<IntervalVar*> > jobs_to_tasks(job_count);
00254   // machines_to_tasks stores the same interval variables as above, but
00255   // grouped my machines instead of grouped by jobs.
00256   std::vector<std::vector<IntervalVar*> > machines_to_tasks(machine_count);
00257 
00258   // Creates all individual interval variables.
00259   for (int job_id = 0; job_id < job_count; ++job_id) {
00260     const std::vector<JobShopData::Task>& tasks = data.TasksOfJob(job_id);
00261     for (int task_index = 0; task_index < tasks.size(); ++task_index) {
00262       const JobShopData::Task& task = tasks[task_index];
00263       CHECK_EQ(job_id, task.job_id);
00264       const string name = StringPrintf("J%dM%dI%dD%d",
00265                                        task.job_id,
00266                                        task.machine_id,
00267                                        task_index,
00268                                        task.duration);
00269       IntervalVar* const one_task =
00270           solver.MakeFixedDurationIntervalVar(0,
00271                                               horizon,
00272                                               task.duration,
00273                                               false,
00274                                               name);
00275       jobs_to_tasks[task.job_id].push_back(one_task);
00276       machines_to_tasks[task.machine_id].push_back(one_task);
00277     }
00278   }
00279 
00280   // ----- Creates model -----
00281 
00282   // Creates precedences inside jobs.
00283   for (int job_id = 0; job_id < job_count; ++job_id) {
00284     const int task_count = jobs_to_tasks[job_id].size();
00285     for (int task_index = 0; task_index < task_count - 1; ++task_index) {
00286       IntervalVar* const t1 = jobs_to_tasks[job_id][task_index];
00287       IntervalVar* const t2 = jobs_to_tasks[job_id][task_index + 1];
00288       Constraint* const prec =
00289           solver.MakeIntervalVarRelation(t2, Solver::STARTS_AFTER_END, t1);
00290       solver.AddConstraint(prec);
00291     }
00292   }
00293 
00294   // Adds disjunctive constraints on unary resources.
00295   for (int machine_id = 0; machine_id < machine_count; ++machine_id) {
00296     solver.AddConstraint(
00297         solver.MakeDisjunctiveConstraint(machines_to_tasks[machine_id]));
00298   }
00299 
00300   // Creates sequences variables on machines. A sequence variable is a
00301   // dedicated variable whose job is to sequence interval variables.
00302   std::vector<SequenceVar*> all_sequences;
00303   for (int machine_id = 0; machine_id < machine_count; ++machine_id) {
00304     const string name = StringPrintf("Machine_%d", machine_id);
00305     SequenceVar* const sequence =
00306         solver.MakeSequenceVar(machines_to_tasks[machine_id], name);
00307     all_sequences.push_back(sequence);
00308   }
00309 
00310   // Creates array of end_times of jobs.
00311   std::vector<IntVar*> all_ends;
00312   for (int job_id = 0; job_id < job_count; ++job_id) {
00313     const int task_count = jobs_to_tasks[job_id].size();
00314     IntervalVar* const task = jobs_to_tasks[job_id][task_count - 1];
00315     all_ends.push_back(task->EndExpr()->Var());
00316   }
00317 
00318   // Objective: minimize the makespan (maximum end times of all tasks)
00319   // of the problem.
00320   IntVar* const objective_var = solver.MakeMax(all_ends)->Var();
00321   OptimizeVar* const objective_monitor = solver.MakeMinimize(objective_var, 1);
00322 
00323   // ----- Search monitors and decision builder -----
00324 
00325   // This decision builder will rank all tasks on all machines.
00326   DecisionBuilder* const sequence_phase =
00327       solver.MakePhase(all_sequences, Solver::SEQUENCE_DEFAULT);
00328 
00329   // After the ranking of tasks, the schedule is still loose and any
00330   // task can be postponed at will. But, because the problem is now a PERT
00331   // (http://en.wikipedia.org/wiki/Program_Evaluation_and_Review_Technique),
00332   // we can schedule each task at its earliest start time. This is
00333   // conveniently done by fixing the objective variable to its
00334   // minimum value.
00335   DecisionBuilder* const obj_phase =
00336       solver.MakePhase(objective_var,
00337                        Solver::CHOOSE_FIRST_UNBOUND,
00338                        Solver::ASSIGN_MIN_VALUE);
00339 
00340   // The main decision builder (ranks all tasks, then fixes the
00341   // objective_variable).
00342   DecisionBuilder* const main_phase =
00343       solver.Compose(sequence_phase, obj_phase);
00344 
00345   // Search log.
00346   const int kLogFrequency = 1000000;
00347   SearchMonitor* const search_log =
00348       solver.MakeSearchLog(kLogFrequency, objective_monitor);
00349 
00350   SearchLimit* limit = NULL;
00351   if (FLAGS_time_limit_in_ms > 0) {
00352     limit = solver.MakeTimeLimit(FLAGS_time_limit_in_ms);
00353   }
00354 
00355   // Search.
00356   solver.Solve(main_phase, search_log, objective_monitor, limit);
00357 }
