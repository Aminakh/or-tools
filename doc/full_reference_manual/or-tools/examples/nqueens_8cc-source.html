00051 class NQueenSymmetry : public SymmetryBreaker {
00052  public:
00053   NQueenSymmetry(Solver* const s, const vector<IntVar*>& vars)
00054       : solver_(s), vars_(vars), size_(vars.size()) {
00055     for (int i = 0; i < size_; ++i) {
00056       indices_[vars[i]] = i;
00057     }
00058   }
00059   virtual ~NQueenSymmetry() {}
00060   int Index(IntVar* const var) const {
00061     return FindWithDefault(indices_, var, -1);
00062   }
00063   IntVar* Var(int index) const {
00064     DCHECK_GE(index, 0);
00065     DCHECK_LT(index, size_);
00066     return vars_[index];
00067   }
00068   int size() const { return size_; }
00069   Solver* const solver() const { return solver_; }
00070  private:
00071   Solver* const solver_;
00072   const vector<IntVar*> vars_;
00073   map<IntVar*, int> indices_;
00074   const int size_;
00075 };

00168 void NQueens(int size) {
00169   CHECK_GE(size, 1);
00170   Solver s("nqueens");
00171 
00172   // model
00173   vector<IntVar*> queens;
00174   for (int i = 0; i < size; ++i) {
00175     queens.push_back(s.MakeIntVar(0, size - 1, StringPrintf("queen%04d", i)));
00176   }
00177   s.AddConstraint(s.MakeAllDifferent(queens, FLAGS_use_range));
00178 
00179   vector<IntVar*> vars(size);
00180   for (int i = 0; i < size; ++i) {
00181     vars[i] = s.MakeSum(queens[i], i)->Var();
00182   }
00183   s.AddConstraint(s.MakeAllDifferent(vars, FLAGS_use_range));
00184   for (int i = 0; i < size; ++i) {
00185     vars[i] = s.MakeSum(queens[i], -i)->Var();
00186   }
00187   s.AddConstraint(s.MakeAllDifferent(vars, FLAGS_use_range));
00188 
00189   SolutionCollector* const solution_counter = s.MakeAllSolutionCollector(NULL);
00190   SolutionCollector* const collector = s.MakeFirstSolutionCollector();
00191   collector->Add(queens);
00192   vector<SearchMonitor*> monitors;
00193   monitors.push_back(solution_counter);
00194   monitors.push_back(collector);
00195   DecisionBuilder* const db = s.MakePhase(queens,
00196                                           Solver::CHOOSE_FIRST_UNBOUND,
00197                                           Solver::ASSIGN_MIN_VALUE);
00198   if (FLAGS_use_symmetry) {
00199     vector<SymmetryBreaker*> breakers;
00200     NQueenSymmetry* sx = s.RevAlloc(new SX(&s, queens));
00201     breakers.push_back(sx);
00202     NQueenSymmetry* sy = s.RevAlloc(new SY(&s, queens));
00203     breakers.push_back(sy);
00204     NQueenSymmetry* sd1 = s.RevAlloc(new SD1(&s, queens));
00205     breakers.push_back(sd1);
00206     NQueenSymmetry* sd2 = s.RevAlloc(new SD2(&s, queens));
00207     breakers.push_back(sd2);
00208     NQueenSymmetry* r90 = s.RevAlloc(new R90(&s, queens));
00209     breakers.push_back(r90);
00210     NQueenSymmetry* r180 = s.RevAlloc(new R180(&s, queens));
00211     breakers.push_back(r180);
00212     NQueenSymmetry* r270 = s.RevAlloc(new R270(&s, queens));
00213     breakers.push_back(r270);
00214     SearchMonitor* symmetry_manager = s.MakeSymmetryManager(breakers);
00215     monitors.push_back(symmetry_manager);
00216   }
00217 
00218   for (int loop = 0; loop < FLAGS_nb_loops; ++loop) {
00219     s.Solve(db, monitors);  // go!
00220   }
00221 
00222   const int num_solutions = solution_counter->solution_count();
00223   if (num_solutions > 0 && size < kKnownSolutions) {
00224     int print_max = FLAGS_print_all ? num_solutions : FLAGS_print ? 1 : 0;
00225     for (int n = 0; n < print_max; ++n) {
00226       printf("--- solution #%d\n", n);
00227       for (int i = 0; i < size; ++i) {
00228         const int pos = static_cast<int>(collector->Value(n, queens[i]));
00229         for (int k = 0; k < pos; ++k) printf(" . ");
00230         printf("%2d ", i);
00231         for (int k = pos + 1; k < size; ++k) printf(" . ");
00232         printf("\n");
00233       }
00234     }
00235   }
00236   printf("========= number of solutions:%d\n", num_solutions);
00237   printf("          number of failures: %lld\n", s.failures());
00238   if (FLAGS_use_symmetry) {
00239     if (size - 1 < kKnownUniqueSolutions) {
00240       CHECK_EQ(num_solutions, kNumUniqueSolutions[size - 1] * FLAGS_nb_loops);
00241     } else {
00242       CHECK_GT(num_solutions, 0);
00243     }
00244   } else {
00245     if (size - 1 < kKnownSolutions) {
00246       CHECK_EQ(num_solutions, kNumSolutions[size - 1] * FLAGS_nb_loops);
00247     } else {
00248       CHECK_GT(num_solutions, 0);
00249     }
00250   }
00251 }
